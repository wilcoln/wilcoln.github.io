24  package org.hibernate.metamodel.source.hbm;
25  
26  import java.util.List;
27  
28  import org.hibernate.cfg.NamingStrategy;
29  import org.hibernate.internal.jaxb.JaxbRoot;
30  import org.hibernate.internal.jaxb.Origin;
31  import org.hibernate.internal.jaxb.mapping.hbm.EntityElement;
32  import org.hibernate.internal.jaxb.mapping.hbm.JaxbFetchProfileElement;
33  import org.hibernate.internal.jaxb.mapping.hbm.JaxbHibernateMapping;
34  import org.hibernate.internal.util.ValueHolder;
35  import org.hibernate.metamodel.domain.Type;
36  import org.hibernate.metamodel.source.MappingDefaults;
37  import org.hibernate.metamodel.source.MetaAttributeContext;
38  import org.hibernate.metamodel.source.MetadataImplementor;
39  import org.hibernate.metamodel.source.internal.OverriddenMappingDefaults;
40  import org.hibernate.service.ServiceRegistry;
41  
42  /**
43   * Aggregates together information about a mapping document.
44   * 
45   * @author Steve Ebersole
46   */
47  public class MappingDocument {
48  	private final JaxbRoot<JaxbHibernateMapping> hbmJaxbRoot;
49  	private final LocalBindingContextImpl mappingLocalBindingContext;
50  
51  	public MappingDocument(JaxbRoot<JaxbHibernateMapping> hbmJaxbRoot, MetadataImplementor metadata) {
52  		this.hbmJaxbRoot = hbmJaxbRoot;
53  		this.mappingLocalBindingContext = new LocalBindingContextImpl( metadata );
54  
55  	}
56  
57  	public JaxbHibernateMapping getMappingRoot() {
58  		return hbmJaxbRoot.getRoot();
59  	}
60  
61  	public Origin getOrigin() {
62  		return hbmJaxbRoot.getOrigin();
63  	}
64  
65  	public JaxbRoot<JaxbHibernateMapping> getJaxbRoot() {
66  		return hbmJaxbRoot;
67  	}
68  
69  	public HbmBindingContext getMappingLocalBindingContext() {
70  		return mappingLocalBindingContext;
71  	}
72  
73  	private class LocalBindingContextImpl implements HbmBindingContext {
74  		private final MetadataImplementor metadata;
75  		private final MappingDefaults localMappingDefaults;
76  		private final MetaAttributeContext metaAttributeContext;
77  
78  		private LocalBindingContextImpl(MetadataImplementor metadata) {
79  			this.metadata = metadata;
80  			this.localMappingDefaults = new OverriddenMappingDefaults(
81  					metadata.getMappingDefaults(),
82  					hbmJaxbRoot.getRoot().getPackage(),
83  					hbmJaxbRoot.getRoot().getSchema(),
84  					hbmJaxbRoot.getRoot().getCatalog(),
85  					null,
86  					null,
87  					hbmJaxbRoot.getRoot().getDefaultCascade(),
88  					hbmJaxbRoot.getRoot().getDefaultAccess(),
89  					hbmJaxbRoot.getRoot().isDefaultLazy()
90  			);
91  			if ( hbmJaxbRoot.getRoot().getMeta() == null || hbmJaxbRoot.getRoot().getMeta().isEmpty() ) {
92  				this.metaAttributeContext = new MetaAttributeContext( metadata.getGlobalMetaAttributeContext() );
93  			}
94  			else {
95  				this.metaAttributeContext = Helper.extractMetaAttributeContext(
96  						hbmJaxbRoot.getRoot().getMeta(),
97  						true,
98  						metadata.getGlobalMetaAttributeContext()
99  				);
100 			}
101 		}
102 
103 		@Override
104 		public ServiceRegistry getServiceRegistry() {
105 			return metadata.getServiceRegistry();
106 		}
107 
108 		@Override
109 		public NamingStrategy getNamingStrategy() {
110 			return metadata.getNamingStrategy();
111 		}
112 
113 		@Override
114 		public MappingDefaults getMappingDefaults() {
115 			return localMappingDefaults;
116 		}
117 
118 		@Override
119 		public MetadataImplementor getMetadataImplementor() {
120 			return metadata;
121 		}
122 
123 		@Override
124 		public <T> Class<T> locateClassByName(String name) {
125 			return metadata.locateClassByName( name );
126 		}
127 
128 		@Override
129 		public Type makeJavaType(String className) {
130 			return metadata.makeJavaType( className );
131 		}
132 
133 		@Override
134 		public ValueHolder<Class<?>> makeClassReference(String className) {
135 			return metadata.makeClassReference( className );
136 		}
137 
138 		@Override
139 		public boolean isAutoImport() {
140 			return hbmJaxbRoot.getRoot().isAutoImport();
141 		}
142 
143 		@Override
144 		public MetaAttributeContext getMetaAttributeContext() {
145 			return metaAttributeContext;
146 		}
147 
148 		@Override
149 		public Origin getOrigin() {
150 			return hbmJaxbRoot.getOrigin();
151 		}
152 
153 		@Override
154 		public String qualifyClassName(String unqualifiedName) {
155 			return Helper.qualifyIfNeeded( unqualifiedName, getMappingDefaults().getPackageName() );
156 		}
157 
158 		@Override
159 		public String determineEntityName(EntityElement entityElement) {
160 			return Helper.determineEntityName( entityElement, getMappingDefaults().getPackageName() );
161 		}
162 
163 		@Override
164 		public boolean isGloballyQuotedIdentifiers() {
165 			return metadata.isGloballyQuotedIdentifiers();
166 		}
167 
168 		@Override
169 		public void processFetchProfiles(List<JaxbFetchProfileElement> fetchProfiles, String containingEntityName) {
170 			// todo : this really needs to not be part of the context
171 		}
172 	}
173 }
