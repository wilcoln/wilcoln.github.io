31  /**
32   * For the scope method or constructor and super type hierarchy of it's declaring type,
33   * it is able to adapt type parameters.
34   *
35   * https://docs.oracle.com/javase/specs/jls/se8/html/jls-8.html#jls-8.4.4
36   * Where two methods or constructors M and N have the same type parameters {@link #hasSameMethodFormalTypeParameters(CtFormalTypeDeclarer)},
37   * a type mentioned in N can be adapted to the type parameters of M
38   */
39  public class MethodTypingContext extends AbstractTypingContext {
40  
41  	private CtFormalTypeDeclarer scopeMethod;
42  	private List<CtTypeReference<?>> actualTypeArguments;
43  	private ClassTypingContext classTypingContext;
44  
45  	public MethodTypingContext() {
46  	}
47  
48  	@Override
49  	public CtFormalTypeDeclarer getAdaptationScope() {
50  		return scopeMethod;
51  	}
52  
53  	public MethodTypingContext setMethod(CtMethod<?> method) {
54  		actualTypeArguments = getTypeReferences(method.getFormalCtTypeParameters());
55  		if (classTypingContext != null) {
56  			CtType<?> declType = method.getDeclaringType();
57  			if (declType == null) {
58  				throw new SpoonException("Cannot use method without declaring type as scope of method typing context");
59  			}
60  			if (classTypingContext.getAdaptationScope() != declType) {
61  				//the method is declared in different type. We have to adapt it to required classTypingContext
62  				if (classTypingContext.isSubtypeOf(declType.getReference()) == false) {
63  					throw new SpoonException("Cannot create MethodTypingContext for method declared in different ClassTypingContext");
64  				}
65  				/*
66  				 * The method is declared in an supertype of classTypingContext.
67  				 * Create virtual scope method by adapting generic types of supertype method to required scope
68  				 */
69  				Factory factory = method.getFactory();
70  				//create new scopeMethod, which is directly used during adaptation of it's parameters
71  				CtMethod<?> adaptedMethod = factory.Core().createMethod();
72  				adaptedMethod.setParent(classTypingContext.getAdaptationScope());
73  				adaptedMethod.setModifiers(method.getModifiers());
74  				adaptedMethod.setSimpleName(method.getSimpleName());
75  
76  				for (CtTypeParameter typeParam : method.getFormalCtTypeParameters()) {
77  					CtTypeParameter newTypeParam = typeParam.clone();
78  					newTypeParam.setSuperclass(adaptTypeForNewMethod(typeParam.getSuperclass()));
79  					adaptedMethod.addFormalCtTypeParameter(newTypeParam);
80  				}
81  				//now the formal type parameters of the scopeMethod are defined, so we can use adaptType of this MethodTypingContext
82  				scopeMethod = adaptedMethod;
83  				for (CtTypeReference<? extends Throwable> thrownType : method.getThrownTypes()) {
84  					adaptedMethod.addThrownType((CtTypeReference<Throwable>) adaptType(thrownType.clone()));
85  				}
86  				//adapt return type
87  				adaptedMethod.setType((CtTypeReference) adaptType(method.getType()));
88  				//adapt parameters
89  				List<CtParameter<?>> adaptedParams = new ArrayList<>(method.getParameters().size());
90  				for (CtParameter<?> parameter : method.getParameters()) {
91  					adaptedParams.add(factory.Executable().createParameter(null,
92  							adaptType(parameter.getType()),
93  							parameter.getSimpleName()));
94  				}
95  				adaptedMethod.setParameters(adaptedParams);
96  				method = adaptedMethod;
97  			}
98  		}
99  		scopeMethod = method;
100 		return this;
101 	}

103 	public MethodTypingContext setConstructor(CtConstructor<?> constructor) {
104 		actualTypeArguments = getTypeReferences(constructor.getFormalCtTypeParameters());
105 		//TODO may be the constructors have to be adapted too, same like in setScopeMethod??
106 		checkSameTypingContext(classTypingContext, constructor);
107 		scopeMethod = constructor;
108 		return this;
109 	}

111 	@Override
112 	public ClassTypingContext getEnclosingGenericTypeAdapter() {
113 		if (classTypingContext == null && scopeMethod != null) {
114 			classTypingContext = new ClassTypingContext(getScopeMethodDeclaringType());
115 		}
116 		return classTypingContext;
117 	}
118 
119 	public MethodTypingContext setClassTypingContext(ClassTypingContext classTypingContext) {
120 		checkSameTypingContext(classTypingContext, scopeMethod);
121 		this.classTypingContext = classTypingContext;
122 		return this;
123 	}

125 	public MethodTypingContext setInvocation(CtInvocation<?> invocation) {
126 		if (classTypingContext == null) {
127 			CtExpression<?> target = invocation.getTarget();
128 			if (target != null) {
129 				CtTypeReference<?> targetTypeRef = target.getType();
130 				if (targetTypeRef != null) {
131 					classTypingContext = new ClassTypingContext(targetTypeRef);
132 				}
133 			}
134 		}
135 		setExecutableReference(invocation.getExecutable());
136 		return this;
137 	}
138 
139 	public MethodTypingContext setExecutableReference(CtExecutableReference<?> execRef) {
140 		if (classTypingContext == null) {
141 			CtTypeReference<?> declaringTypeRef = execRef.getDeclaringType();
142 			if (declaringTypeRef != null) {
143 				classTypingContext = new ClassTypingContext(declaringTypeRef);
144 			}
145 		}
146 		CtExecutable<?> exec = execRef.getExecutableDeclaration();
147 		if (exec == null) {
148 			throw new SpoonException("Cannot create MethodTypingContext from CtExecutable of CtExecutableReference is null");
149 		}
150 		if (exec instanceof CtMethod<?>) {
151 			setMethod((CtMethod<?>) exec);
152 		} else if (exec instanceof CtConstructor<?>) {
153 			setConstructor((CtConstructor<?>) exec);
154 		} else {
155 			throw new SpoonException("Cannot create MethodTypingContext from " + exec.getClass().getName());
156 		}
157 		this.actualTypeArguments = execRef.getActualTypeArguments();
158 		return this;
159 	}

161 	/**
162 	 * Adapts `typeParam` to the {@link CtTypeReference}
163 	 * of scope of this {@link MethodTypingContext}
164 	 * In can be {@link CtTypeParameterReference} again - depending actual type arguments of this {@link MethodTypingContext}.
165 	 *
166 	 * @param typeParam to be resolved {@link CtTypeParameter}
167 	 * @return {@link CtTypeReference} or {@link CtTypeParameterReference} adapted to scope of this {@link MethodTypingContext}
168 	 *  or null if `typeParam` cannot be adapted to target `scope`
169 	 */
170 	@Override
171 	protected CtTypeReference<?> adaptTypeParameter(CtTypeParameter typeParam) {
172 		CtFormalTypeDeclarer typeParamDeclarer = typeParam.getTypeParameterDeclarer();
173 		if (typeParamDeclarer instanceof CtType<?>) {
174 			return getEnclosingGenericTypeAdapter().adaptType(typeParam);
175 		}
176 		//only method to method or constructor to constructor can be adapted
177 		if (typeParamDeclarer instanceof CtMethod<?>) {
178 			if ((scopeMethod instanceof CtMethod<?>) == false) {
179 				return null;
180 			}
181 		} else if (typeParamDeclarer instanceof CtConstructor<?>) {
182 			if ((scopeMethod instanceof CtConstructor<?>) == false) {
183 				return null;
184 			}
185 		} else {
186 			throw new SpoonException("Unexpected type parameter declarer");
187 		}
188 		//the typeParamDeclarer is method or constructor
189 		/*
190 		 *
191 		 * Two methods or constructors M and N have the same type parameters if both of the following are true:
192 		 * 1) M and N have same number of type parameters (possibly zero).
193 		 * 2) Where A1, ..., An are the type parameters of M and B1, ..., Bn are the type parameters of N, let T=[B1:=A1, ..., Bn:=An].
194 		 * Then, for all i (1 ≤ i ≤ n), the bound of Ai is the same type as T applied to the bound of Bi.
195 		 */
196 		if (hasSameMethodFormalTypeParameters(typeParamDeclarer) == false) {
197 			//the methods formal type parameters are different. We cannot adapt such parameters
198 			return null;
199 		}
200 		int typeParamPosition = typeParamDeclarer.getFormalCtTypeParameters().indexOf(typeParam);
201 		return actualTypeArguments.get(typeParamPosition);
202 	}

204 	private Set<CtFormalTypeDeclarer> checkingFormalTypeParamsOf = Collections.newSetFromMap(new IdentityHashMap<>(1));
205 
206 	/**
207 	 * https://docs.oracle.com/javase/specs/jls/se8/html/jls-8.html#jls-8.4.4
208 	 *
209 	 * Formal type parameters of method are same if
210 	 * 1) both methods have same number of formal type parameters
211 	 * 2) bounds of Formal type parameters are after adapting same types
212 	 *
213 	 * @return true if formal type parameters of method are same
214 	 */
215 	public boolean hasSameMethodFormalTypeParameters(CtFormalTypeDeclarer typeParamDeclarer) {
216 		List<CtTypeParameter> thisTypeParameters = scopeMethod.getFormalCtTypeParameters();
217 		List<CtTypeParameter> thatTypeParameters = typeParamDeclarer.getFormalCtTypeParameters();
218 		if (thisTypeParameters.size() != thatTypeParameters.size()) {
219 			return false;
220 		}
221 		if (checkingFormalTypeParamsOf.contains(typeParamDeclarer)) {
222 			//do not check isSameMethodFormalTypeParameter recursively
223 			return true;
224 		}
225 		try {
226 			checkingFormalTypeParamsOf.add(typeParamDeclarer);
227 			//the methods has same count of formal parameters
228 			//check that bounds of formal type parameters are same after adapting
229 			for (int i = 0; i < thisTypeParameters.size(); i++) {
230 				if (isSameMethodFormalTypeParameter(thisTypeParameters.get(i), thatTypeParameters.get(i)) == false) {
231 					return false;
232 				}
233 			}
234 		} finally {
235 			checkingFormalTypeParamsOf.remove(typeParamDeclarer);
236 		}
237 		return true;
238 	}

240 	private boolean isSameMethodFormalTypeParameter(CtTypeParameter scopeParam, CtTypeParameter superParam) {
241 		CtTypeReference<?> scopeBound = getBound(scopeParam);
242 		CtTypeReference<?> superBound = getBound(superParam);
243 		if (scopeBound.getActualTypeArguments().size() != superBound.getActualTypeArguments().size()) {
244 			return false;
245 		}
246 
247 		CtTypeReference<?> superBoundAdapted = adaptType(superBound);
248 		if (superBoundAdapted == null) {
249 			return false;
250 		}
251 		return scopeBound.getQualifiedName().equals(superBoundAdapted.getQualifiedName());
252 	}

254 	/*
255 	 * @param declared
256 	 * @param typeRef
257 	 * @return index of type parameter in formal type parameters of `declarer` if typeRef is reference refers type parameter of that declarer.
258 	 *  Returns -1 if it is not.
259 	 */
260 	private int getIndexOfTypeParam(CtFormalTypeDeclarer declarer, CtTypeReference<?> typeRef) {
261 		if (typeRef instanceof CtTypeParameterReference) {
262 			CtTypeParameter typeParam = ((CtTypeParameterReference) typeRef).getDeclaration();
263 			if (typeParam != null) {
264 				if (declarer == typeParam.getTypeParameterDeclarer()) {
265 					return declarer.getFormalCtTypeParameters().indexOf(typeParam);
266 				}
267 			}
268 		}
269 		return -1;
270 	}
271 
272 	private static CtTypeReference<?> getBound(CtTypeParameter typeParam) {
273 		CtTypeReference<?> bound = typeParam.getSuperclass();
274 		if (bound == null) {
275 			bound = typeParam.getFactory().Type().OBJECT;
276 		}
277 		return bound;
278 	}

280 	private CtType<?> getScopeMethodDeclaringType() {
281 		if (scopeMethod != null) {
282 			return scopeMethod.getDeclaringType();
283 		}
284 		throw new SpoonException("scopeMethod is not assigned");
285 	}
286 
287 	private CtTypeReference<?> adaptTypeForNewMethod(CtTypeReference<?> typeRef) {
288 		if (typeRef == null) {
289 			return null;
290 		}
291 		if (typeRef instanceof CtTypeParameterReference) {
292 			CtTypeParameterReference typeParamRef = (CtTypeParameterReference) typeRef;
293 			CtTypeParameter typeParam = typeParamRef.getDeclaration();
294 			if (typeParam == null) {
295 				throw new SpoonException("Declaration of the CtTypeParameter should not be null.");
296 			}
297 
298 			if (typeParam.getTypeParameterDeclarer() instanceof CtExecutable) {
299 				//the parameter is declared in scope of Method or Constructor
300 				return typeRef.clone();
301 			}
302 		}
303 		//it is not type reference of scopeMethod. Adapt it using classTypingContext
304 		return classTypingContext.adaptType(typeRef);
305 	}
306 
307 	private void checkSameTypingContext(ClassTypingContext ctc, CtFormalTypeDeclarer executable) {
308 		if (ctc != null && executable != null) {
309 			CtType<?> scope = executable.getDeclaringType();
310 			if (scope == null) {
311 				throw new SpoonException("Cannot use executable without declaring type as scope of method typing context");
312 			}
313 			if (scope != ctc.getAdaptationScope()) {
314 				throw new SpoonException("Declaring type of executable is not same like scope of classTypingContext provided for method typing context");
315 			}
316 		}
317 	}
