6   package spoon.support.visitor.java.internal;
7   
8   import spoon.reflect.declaration.CtAnnotation;
9   import spoon.reflect.declaration.CtField;
10  import spoon.reflect.declaration.CtParameter;
11  import spoon.reflect.declaration.CtVariable;
12  import spoon.reflect.path.CtRole;
13  import spoon.reflect.reference.CtTypeReference;
14  
15  import java.lang.annotation.Annotation;
16  
17  public class VariableRuntimeBuilderContext extends AbstractRuntimeBuilderContext {
18  	private CtVariable ctVariable;
19  
20  	public VariableRuntimeBuilderContext(CtField<?> ctField) {
21  		super(ctField);
22  		this.ctVariable = ctField;
23  	}
24  
25  	public VariableRuntimeBuilderContext(CtParameter<?> ctParameter) {
26  		super(ctParameter);
27  		this.ctVariable = ctParameter;
28  	}
29  
30  	@Override
31  	public void addAnnotation(CtAnnotation<Annotation> ctAnnotation) {
32  		ctVariable.addAnnotation(ctAnnotation);
33  	}
34  
35  	@Override
36  	public void addTypeReference(CtRole role, CtTypeReference<?> ctTypeReference) {
37  		switch (role) {
38  		case TYPE:
39  			ctVariable.setType(ctTypeReference);
40  			return;
41  		}
42  		super.addTypeReference(role, ctTypeReference);
43  	}
44  }
