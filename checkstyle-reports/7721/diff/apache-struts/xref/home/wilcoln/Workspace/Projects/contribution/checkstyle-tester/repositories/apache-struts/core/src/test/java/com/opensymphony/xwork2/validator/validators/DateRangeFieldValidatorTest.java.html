19  package com.opensymphony.xwork2.validator.validators;
20  
21  import com.opensymphony.xwork2.ActionContext;
22  import com.opensymphony.xwork2.TextProviderFactory;
23  import com.opensymphony.xwork2.XWorkTestCase;
24  import com.opensymphony.xwork2.util.ValueStack;
25  import com.opensymphony.xwork2.util.ValueStackFactory;
26  import com.opensymphony.xwork2.validator.DummyValidatorContext;
27  import com.opensymphony.xwork2.validator.ValidatorContext;
28  
29  import java.util.Calendar;
30  import java.util.Date;
31  import java.util.Locale;
32  
33  public class DateRangeFieldValidatorTest extends XWorkTestCase {
34  
35      private TextProviderFactory tpf;
36  
37      public void testPassValidation() throws Exception {
38          // given
39          ValidationAction action = prepareAction(createDate(2013, 6, 6));
40          ValidatorContext context = new DummyValidatorContext(action, tpf);
41          DateRangeFieldValidator validator = prepareValidator(action, context);
42  
43          // when
44          validator.validate(action);
45  
46          // then
47          assertTrue(context.getFieldErrors().size() == 0);
48      }
49  
50      public void testMinValidation() throws Exception {
51          // given
52          ValidationAction action = prepareAction(createDate(2012, Calendar.MARCH, 3));
53          ValidatorContext context = new DummyValidatorContext(action, tpf);
54          DateRangeFieldValidator validator = prepareValidator(action, context);
55  
56          // when
57          validator.validate(action);
58  
59          // then
60          assertTrue(context.getFieldErrors().size() == 1);
61          assertEquals("Max is 12.12.13, min is 01.01.13 but value is 03.03.12", context.getFieldErrors().get("dateRange").get(0));
62      }
63  
64      public void testMaxValidation() throws Exception {
65          // given
66          ValidationAction action = prepareAction(createDate(2014, Calendar.APRIL, 4));
67          ValidatorContext context = new DummyValidatorContext(action, tpf);
68          DateRangeFieldValidator validator = prepareValidator(action, context);
69  
70          // when
71          validator.validate(action);
72  
73          // then
74          assertTrue(context.getFieldErrors().size() == 1);
75          assertEquals("Max is 12.12.13, min is 01.01.13 but value is 04.04.14", context.getFieldErrors().get("dateRange").get(0));
76      }
77  
78      private ValidationAction prepareAction(Date range) {
79          ValidationAction action = new ValidationAction();
80          action.setDateMinValue(createDate(2013, Calendar.JANUARY, 1));
81          action.setDateMaxValue(createDate(2013, Calendar.DECEMBER, 12));
82          action.setDateRange(range);
83          return action;
84      }
85  
86      private Date createDate(int year, int month, int day) {
87          Calendar cal = Calendar.getInstance();
88          cal.set(year, month, day);
89          return cal.getTime();
90      }
91  
92      private DateRangeFieldValidator prepareValidator(ValidationAction action, ValidatorContext context) {
93          ValueStack valueStack = container.getInstance(ValueStackFactory.class).createValueStack();
94          valueStack.push(action);
95  
96          DateRangeFieldValidator validator = new DateRangeFieldValidator();
97          validator.setValueStack(valueStack);
98  
99          validator.setMaxExpression("${dateMaxValue}");
100         validator.setMinExpression("${dateMinValue}");
101         validator.setValidatorContext(context);
102         validator.setFieldName("dateRange");
103         validator.setDefaultMessage("Max is ${dateMaxValue}, min is ${dateMinValue} but value is ${dateRange}");
104 
105         return validator;
106     }
107 
108     @Override
109     public void setUp() throws Exception {
110         super.setUp();
111         ActionContext.getContext().setLocale(new Locale("DE"));
112         tpf = container.getInstance(TextProviderFactory.class);
113     }
114 
115 }
