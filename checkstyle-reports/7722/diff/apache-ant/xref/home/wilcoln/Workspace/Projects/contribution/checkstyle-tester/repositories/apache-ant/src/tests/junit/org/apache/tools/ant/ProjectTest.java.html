80      @Test
81      public void testResolveFile() {
82          if (Os.isFamily("netware") || Os.isFamily("dos")) {
83              assertEqualsIgnoreDriveCase(localize(File.separator),
84                  p.resolveFile("/", null).getPath());
85              assertEqualsIgnoreDriveCase(localize(File.separator),
86                  p.resolveFile("\\", null).getPath());
87              /*
88               * throw in drive letters
89               */
90              String driveSpec = "C:";
91              String driveSpecLower = "c:";
92              
93              assertEqualsIgnoreDriveCase(driveSpecLower + "\\",
94                           p.resolveFile(driveSpec + "/", null).getPath());
95              assertEqualsIgnoreDriveCase(driveSpecLower + "\\",
96                           p.resolveFile(driveSpec + "\\", null).getPath());
97              assertEqualsIgnoreDriveCase(driveSpecLower + "\\",
98                           p.resolveFile(driveSpecLower + "/", null).getPath());
99              assertEqualsIgnoreDriveCase(driveSpecLower + "\\",
100                          p.resolveFile(driveSpecLower + "\\", null).getPath());
101             /*
102              * promised to eliminate consecutive slashes after drive letter.
103              */
104             assertEqualsIgnoreDriveCase(driveSpec + "\\",
105                          p.resolveFile(driveSpec + "/////", null).getPath());
106             assertEqualsIgnoreDriveCase(driveSpec + "\\",
107                          p.resolveFile(driveSpec + "\\\\\\\\\\\\", null).getPath());
108         } else {
109             /*
110              * Start with simple absolute file names.
111              */
112             assertEquals(File.separator,
113                          p.resolveFile("/", null).getPath());
114             assertEquals(File.separator,
115                          p.resolveFile("\\", null).getPath());
116             /*
117              * drive letters are not used, just to be considered as normal
118              * part of a name
119              */
120             String driveSpec = "C:";
121             String udir = System.getProperty("user.dir") + File.separatorChar;
122             assertEquals(udir + driveSpec,
123                          p.resolveFile(driveSpec + "/", null).getPath());
124             assertEquals(udir + driveSpec,
125                          p.resolveFile(driveSpec + "\\", null).getPath());
126             String driveSpecLower = "c:";
127             assertEquals(udir + driveSpecLower,
128                          p.resolveFile(driveSpecLower + "/", null).getPath());
129             assertEquals(udir + driveSpecLower,
130                          p.resolveFile(driveSpecLower + "\\", null).getPath());
131         }
132         /*
133          * Now test some relative file name magic.
134          */
135         assertEquals(localize("/1/2/3/4"),
136                      p.resolveFile("4", new File(localize("/1/2/3"))).getPath());
137         assertEquals(localize("/1/2/3/4"),
138                      p.resolveFile("./4", new File(localize("/1/2/3"))).getPath());
139         assertEquals(localize("/1/2/3/4"),
140                      p.resolveFile(".\\4", new File(localize("/1/2/3"))).getPath());
141         assertEquals(localize("/1/2/3/4"),
142                      p.resolveFile("./.\\4", new File(localize("/1/2/3"))).getPath());
143         assertEquals(localize("/1/2/3/4"),
144                      p.resolveFile("../3/4", new File(localize("/1/2/3"))).getPath());
145         assertEquals(localize("/1/2/3/4"),
146                      p.resolveFile("..\\3\\4", new File(localize("/1/2/3"))).getPath());
147         assertEquals(localize("/1/2/3/4"),
148                      p.resolveFile("../../5/.././2/./3/6/../4", new File(localize("/1/2/3"))).getPath());
149         assertEquals(localize("/1/2/3/4"),
150                      p.resolveFile("..\\../5/..\\./2/./3/6\\../4", new File(localize("/1/2/3"))).getPath());
151 
152     }

154     /**
155      * adapt file separators to local conventions
156      */
157     private String localize(String path) {
158         path = root + path.substring(1);
159         return path.replace('\\', File.separatorChar).replace('/', File.separatorChar);
160     }

162     /**
163      * convenience method
164      * the drive letter is in lower case under cygwin
165      * calling this method allows tests where FileUtils.normalize
166      * is called via resolveFile to pass under cygwin
167      */
168     private void assertEqualsIgnoreDriveCase(String s1, String s2) {
169         if ((Os.isFamily("dos") || Os.isFamily("netware"))
170             && s1.length() >= 1 && s2.length() >= 1) {
171             StringBuffer sb1 = new StringBuffer(s1);
172             StringBuffer sb2 = new StringBuffer(s2);
173             sb1.setCharAt(0, Character.toUpperCase(s1.charAt(0)));
174             sb2.setCharAt(0, Character.toUpperCase(s2.charAt(0)));
175             assertEquals(sb1.toString(), sb2.toString());
176         } else {
177             assertEquals(s1, s2);
178         }
179     }

181     private void assertTaskDefFails(final Class taskClass,
182                                        final String message) {
183         final String dummyName = "testTaskDefinitionDummy";
184         try {
185             mbl.addBuildEvent(message, Project.MSG_ERR);
186             p.addTaskDefinition(dummyName, taskClass);
187             fail("expected BuildException(\""+message+"\", Project.MSG_ERR) when adding task " + taskClass);
188         }
189         catch(BuildException e) {
190             assertEquals(message, e.getMessage());
191             mbl.assertEmpty();
192             assertTrue(!p.getTaskDefinitions().containsKey(dummyName));
193         }
194     }

196     @Test
197     public void testAddTaskDefinition() {
198         p.addBuildListener(mbl);
199 
200         p.addTaskDefinition("Ok", DummyTaskOk.class);
201         assertEquals(DummyTaskOk.class, p.getTaskDefinitions().get("Ok"));
202         p.addTaskDefinition("OkNonTask", DummyTaskOkNonTask.class);
203         assertEquals(DummyTaskOkNonTask.class, p.getTaskDefinitions().get("OkNonTask"));
204         mbl.assertEmpty();
205 
206         assertTaskDefFails(DummyTaskPrivate.class,   DummyTaskPrivate.class   + " is not public");
207 
208         assertTaskDefFails(DummyTaskProtected.class,
209                            DummyTaskProtected.class + " is not public");
210 
211         assertTaskDefFails(DummyTaskPackage.class,   DummyTaskPackage.class   + " is not public");
212 
213         assertTaskDefFails(DummyTaskAbstract.class,  DummyTaskAbstract.class  + " is abstract");
214         assertTaskDefFails(DummyTaskInterface.class, DummyTaskInterface.class + " is abstract");
215 
216         assertTaskDefFails(DummyTaskWithoutDefaultConstructor.class, "No public no-arg constructor in " + DummyTaskWithoutDefaultConstructor.class);
217         assertTaskDefFails(DummyTaskWithoutPublicConstructor.class,  "No public no-arg constructor in " + DummyTaskWithoutPublicConstructor.class);
218 
219         assertTaskDefFails(DummyTaskWithoutExecute.class,       "No public execute() in " + DummyTaskWithoutExecute.class);
220         assertTaskDefFails(DummyTaskWithNonPublicExecute.class, "No public execute() in " + DummyTaskWithNonPublicExecute.class);
221 
222         mbl.addBuildEvent("return type of execute() should be void but was \"int\" in " + DummyTaskWithNonVoidExecute.class, Project.MSG_WARN);
223         p.addTaskDefinition("NonVoidExecute", DummyTaskWithNonVoidExecute.class);
224         mbl.assertEmpty();
225         assertEquals(DummyTaskWithNonVoidExecute.class, p.getTaskDefinitions().get("NonVoidExecute"));
226     }

269     @Test
270     public void testOutputDuringMessageLoggedIsSwallowed()
271         throws InterruptedException {
272         final String FOO = "foo", BAR = "bar";
273         p.addBuildListener(new BuildListener() {
274                 public void buildStarted(BuildEvent event) {}
275                 public void buildFinished(BuildEvent event) {}
276                 public void targetStarted(BuildEvent event) {}
277                 public void targetFinished(BuildEvent event) {}
278                 public void taskStarted(BuildEvent event) {}
279                 public void taskFinished(BuildEvent event) {}
280                 public void messageLogged(final BuildEvent actual) {
281                     assertEquals(FOO, actual.getMessage());
282                     // each of the following lines would cause an
283                     // infinite loop if the message wasn't swallowed
284                     System.err.println(BAR);
285                     System.out.println(BAR);
286                     p.log(BAR, Project.MSG_INFO);
287                 }
288             });
289         final boolean[] done = new boolean[] {false};
290         Thread t = new Thread() {
291                 public void run() {
292                     p.log(FOO, Project.MSG_INFO);
293                     done[0] = true;
294                 }
295             };
296         t.start();
297         t.join(2000);
298         assertTrue("Expected logging thread to finish successfully", done[0]);
299     }
