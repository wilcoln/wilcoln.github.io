2   package org.hibernate.test.annotations.manytomany;
3   import java.util.ArrayList;
4   import java.util.List;
5   import javax.persistence.Column;
6   import javax.persistence.Entity;
7   import javax.persistence.JoinColumn;
8   import javax.persistence.JoinTable;
9   import javax.persistence.ManyToMany;
10  import javax.persistence.PrimaryKeyJoinColumn;
11  
12  /**
13   * @author Emmanuel Bernard
14   */
15  @Entity
16  @PrimaryKeyJoinColumn(name = "inspector_id")
17  class InspectorPrefixes extends Inspector {
18  	@Column(name = "prefixes", nullable = false)
19  	private String prefixes;
20  	@ManyToMany()
21  	@JoinTable(name = "deserted_area",
22  			joinColumns = @JoinColumn(name = "inspector_name", referencedColumnName = "name"),
23  			inverseJoinColumns = @JoinColumn(name = "area_id", referencedColumnName = "id"))
24  	private List<Zone> desertedAreas = new ArrayList<Zone>();
25  
26  	@ManyToMany()
27  	@JoinTable(name = "inspector_prefixes_areas",
28  			joinColumns = @JoinColumn(name = "inspector_id", referencedColumnName = "inspector_id"),
29  			inverseJoinColumns = @JoinColumn(name = "area_id", referencedColumnName = "id"))
30  	private List<Zone> areas = new ArrayList<Zone>();
31  
32  	InspectorPrefixes() {
33  	}
34  
35  	InspectorPrefixes(String prefixes) {
36  		this.prefixes = prefixes;
37  	}
38  
39  	public String getPrefixes() {
40  		return this.prefixes;
41  	}
42  
43  	public List<Zone> getAreas() {
44  		return areas;
45  	}
46  
47  	public List<Zone> getDesertedAreas() {
48  		return desertedAreas;
49  	}
50  }
