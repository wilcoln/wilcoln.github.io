213     @Test
214     public void testDifferentWindowsDrive() {
215     	Assume.assumeTrue("Test with drive letters only run on windows", Os.isFamily("windows"));
216         // the lines below try to find a drive name different than the one containing the temp dir
217         // if the temp dir is C will try to use D
218         // if the temp dir is on D or other will try to use C
219         File tmpdir = new File(System.getProperty("java.io.tmpdir"));
220         String driveLetter = "C";
221         try {
222             String tmpCanonicalPath = tmpdir.getCanonicalPath();
223             driveLetter = tmpCanonicalPath.substring(0, 1).toUpperCase();
224         } catch (IOException ioe) {
225             System.out.println("exception happened getting canonical path of java.io.tmpdir : " + ioe.getMessage());
226         }
227         String altDriveLetter = null;
228         try {
229             if ("C".equals(driveLetter)) {
230                 altDriveLetter = "D";
231             } else {
232                 altDriveLetter = "C";
233             }
234             new java.io.File(altDriveLetter + ":/foo.txt").getCanonicalPath();
235         } catch (java.io.IOException e) {
236         	Assume.assumeNoException("Drive " + altDriveLetter + ": doesn't exist or is not ready", e);
237         }
238         buildRule.getProject().setProperty("altDriveLetter", altDriveLetter);
239         
240         try {
241 			buildRule.executeTarget("testDifferentDrive");
242 			fail("Build exception should have been thrown on no alternative drive");
243 		} catch (BuildException ex) {
244 			assertContains("No suitable relative path from ", ex.getMessage());
245 		}
246         
247         assertNull(buildRule.getProject().getProperty("cp"));
248     }
