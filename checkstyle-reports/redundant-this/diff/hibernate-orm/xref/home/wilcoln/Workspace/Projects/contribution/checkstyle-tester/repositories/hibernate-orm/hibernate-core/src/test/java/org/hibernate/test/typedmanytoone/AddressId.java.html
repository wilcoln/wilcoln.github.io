2   package org.hibernate.test.typedmanytoone;
3   import java.io.Serializable;
4   
5   /**
6    * @author Gavin King
7    */
8   public class AddressId implements Serializable {
9   	private String type;
10  	private String addressId;
11  	
12  	public AddressId(String type, String customerId) {
13  		this.addressId = customerId;
14  		this.type = type;
15  	}
16  	
17  	public AddressId() {}
18  	
19  	public String getType() {
20  		return type;
21  	}
22  	public void setType(String type) {
23  		this.type = type;
24  	}
25  	public String getAddressId() {
26  		return addressId;
27  	}
28  	public void setAddressId(String customerId) {
29  		this.addressId = customerId;
30  	}
31  	public boolean equals(Object other) {
32  		if ( !(other instanceof AddressId) ) return false;
33  		AddressId add = (AddressId) other;
34  		return type.equals(add.type) && addressId.equals(add.addressId);
35  	}
36  	public int hashCode() {
37  		return addressId.hashCode() + type.hashCode();
38  	}
39  	
40  	public String toString() {
41  		return type + '#' + addressId;
42  	}
43  
44  }
