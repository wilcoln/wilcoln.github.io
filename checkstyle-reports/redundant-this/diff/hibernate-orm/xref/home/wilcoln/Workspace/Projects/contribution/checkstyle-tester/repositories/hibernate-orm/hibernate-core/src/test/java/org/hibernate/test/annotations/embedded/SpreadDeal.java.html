1   package org.hibernate.test.annotations.embedded;
2   
3   import javax.persistence.AttributeOverride;
4   import javax.persistence.AttributeOverrides;
5   import javax.persistence.Column;
6   import javax.persistence.Embedded;
7   import javax.persistence.Entity;
8   
9   /**
10   * Represents a "curve spread" deal that consists of
11   * two Interest Rate Swaps with different tenors (short and long).
12   * For simplicity, tenors are not considered here.
13   */
14  @Entity
15  @AttributeOverrides(value = {
16  		@AttributeOverride(name = "swap.tenor", column = @Column(name = "MEDIUM_TENOR")),
17  		@AttributeOverride(name = "swap.fixedLeg.paymentFrequency", column = @Column(name = "MEDIUM_FIXED_FREQUENCY")),
18  		@AttributeOverride(name = "swap.fixedLeg.rate", column = @Column(name = "MEDIUM_FIXED_RATE")),
19  		@AttributeOverride(name = "swap.floatLeg.paymentFrequency", column = @Column(name = "MEDIUM_FLOAT_FREQUENCY")),
20  		@AttributeOverride(name = "swap.floatLeg.rateIndex", column = @Column(name = "MEDIUM_FLOAT_RATEINDEX")),
21  		@AttributeOverride(name = "swap.floatLeg.rateSpread", column = @Column(name = "MEDIUM_FLOAT_RATESPREAD"))
22  })
23  public class SpreadDeal extends NotonialDeal {
24  
25  	/**
26  	 * Swap with the tenor.
27  	 */
28  	private Swap longSwap;
29  
30  	@Embedded
31  	public Swap getLongSwap() {
32  		return longSwap;
33  	}
34  
35  	public void setLongSwap(Swap swap) {
36  		this.longSwap = swap;
37  	}
38  
39  
40  	/**
41  	 * Swap with the longer tenor.
42  	 */
43  	private Swap shortSwap;
44  
45  	@Embedded
46  	@AttributeOverrides(value = {
47  			@AttributeOverride(name = "tenor", column = @Column(name = "SHORT_TENOR")),
48  			@AttributeOverride(name = "fixedLeg.paymentFrequency", column = @Column(name = "SHORT_FIXED_FREQUENCY")),
49  			@AttributeOverride(name = "fixedLeg.rate", column = @Column(name = "SHORT_FIXED_RATE")),
50  			@AttributeOverride(name = "floatLeg.paymentFrequency", column = @Column(name = "SHORT_FLOAT_FREQUENCY")),
51  			@AttributeOverride(name = "floatLeg.rateIndex", column = @Column(name = "SHORT_FLOAT_RATEINDEX")),
52  			@AttributeOverride(name = "floatLeg.rateSpread", column = @Column(name = "SHORT_FLOAT_RATESPREAD"))
53  	})
54  	public Swap getShortSwap() {
55  		return shortSwap;
56  	}
57  
58  	public void setShortSwap(Swap shortSwap) {
59  		this.shortSwap = shortSwap;
60  	}
61  }
