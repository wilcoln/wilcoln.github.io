7   import java.util.Arrays;
8   import java.util.List;
9   import java.util.Objects;
10  
11  import org.apache.commons.lang3.StringUtils;
12  
13  import net.sourceforge.pmd.lang.ast.QualifiedName;

15  import apex.jorje.semantic.symbol.type.TypeInfo;

18  /**
19   * Qualified name of an apex class or method.
20   *
21   * @author Cl√©ment Fournier
22   */
23  public final class ApexQualifiedName implements QualifiedName {
24  
25  
26      private final String nameSpace;
27      private final String[] classes;
28      private final String operation;
29  
30  
31      private ApexQualifiedName(String nameSpace, String[] classes, String operation) {
32          this.nameSpace = nameSpace;
33          this.operation = operation;
34          this.classes = classes;
35      }
36  
37  
38      public String getOperation() {
39          return operation;
40      }
41  
42  
43      public String[] getClasses() {
44          return Arrays.copyOf(classes, classes.length);
45      }
46  
47  
48      /**
49       * Gets the namespace prefix of this resource.
50       *
51       * @return The namespace prefix
52       */
53      public String getNameSpace() {
54          return nameSpace;
55      }
56  
57  
58      @Override
59      public boolean isClass() {
60          return operation == null;
61      }
62  
63  
64      @Override
65      public boolean isOperation() {
66          return operation != null;
67      }
68  
69  
70      @Override
71      public String toString() {
72          StringBuilder sb = new StringBuilder();
73          sb.append(nameSpace).append("__");
74          sb.append(classes[0]);
75  
76          if (classes.length > 1) {
77              sb.append('.').append(classes[1]);
78          }
79  
80          if (isOperation()) {
81              sb.append("#").append(operation);
82          }
83  
84          return sb.toString();
85      }
86  
87  
88      @Override
89      public ApexQualifiedName getClassName() {
90          if (isClass()) {
91              return this;
92          }
93  
94          return new ApexQualifiedName(this.nameSpace, this.classes, null);
95      }
96  
97  
98      @Override
99      public int hashCode() {
100         int result = nameSpace.hashCode();
101         result = 31 * result + Arrays.hashCode(classes);
102         result = 31 * result + (operation != null ? operation.hashCode() : 0);
103         return result;
104     }
105 
106 
107     @Override
108     public boolean equals(Object obj) {
109         return obj instanceof ApexQualifiedName
110                && Objects.deepEquals(classes, ((ApexQualifiedName) obj).classes)
111                && Objects.equals(operation, ((ApexQualifiedName) obj).operation)
112                && Objects.equals(nameSpace, ((ApexQualifiedName) obj).nameSpace);
113 
114     }
115 
116 
117     /**
118      * Parses a string conforming to the format defined below and returns an ApexQualifiedName.
119      *
120      * <p>Here are some examples of the format:
121      * <ul>
122      * <li> {@code namespace__OuterClass.InnerClass}: name of an inner class
123      * <li> {@code namespace__Class#method(String, int)}: name of an operation
124      * </ul>
125      *
126      * @param toParse The string to parse
127      *
128      * @return An ApexQualifiedName, or null if the string couldn't be parsed
129      */
130     // private static final Pattern FORMAT = Pattern.compile("(\\w+)__(\\w+)(.(\\w+))?(#(\\w+))?"); // TODO
131     public static ApexQualifiedName ofString(String toParse) {
132         throw new UnsupportedOperationException();
133     }
134 
135 
136     static ApexQualifiedName ofOuterClass(ASTUserClassOrInterface astUserClass) {
137         String ns = astUserClass.getNode().getDefiningType().getNamespace().toString();
138         String[] classes = {astUserClass.getImage()};
139         return new ApexQualifiedName(StringUtils.isEmpty(ns) ? "c" : ns, classes, null);
140     }
141 
142 
143     static ApexQualifiedName ofNestedClass(ApexQualifiedName parent, ASTUserClassOrInterface astUserClass) {
144 
145         String[] classes = Arrays.copyOf(parent.classes, parent.classes.length + 1);
146         classes[classes.length - 1] = astUserClass.getImage();
147         return new ApexQualifiedName(parent.nameSpace, classes, null);
148     }
149 
150 
151     private static String getOperationString(ASTMethod node) {
152         StringBuilder sb = new StringBuilder();
153         sb.append(node.getImage()).append('(');
154 
155 
156         List<TypeInfo> paramTypes = node.getNode().getMethodInfo().getParameterTypes();
157 
158         if (!paramTypes.isEmpty()) {
159             sb.append(paramTypes.get(0).getApexName());
160 
161             for (int i = 1; i < paramTypes.size(); i++) {
162                 sb.append(",").append(paramTypes.get(i).getApexName());
163             }
164 
165         }
166 
167         sb.append(')');
168 
169         return sb.toString();
170     }
171 
172 
173     static ApexQualifiedName ofMethod(ASTMethod node) {
174         ASTUserClassOrInterface<?> parent = node.getFirstParentOfType(ASTUserClassOrInterface.class);
175         if (parent == null) {
176             ASTUserTrigger trigger = node.getFirstParentOfType(ASTUserTrigger.class);
177             String ns = trigger.getNode().getDefiningType().getNamespace().toString();
178             String targetObj = trigger.getNode().getTargetName().get(0).getValue();
179 
180             return new ApexQualifiedName(StringUtils.isEmpty(ns) ? "c" : ns, new String[]{"trigger", targetObj}, trigger.getImage()); // uses a reserved word as a class name to prevent clashes
181 
182         } else {
183             ApexQualifiedName baseName = parent.getQualifiedName();
184 
185             return new ApexQualifiedName(baseName.nameSpace, baseName.classes, getOperationString(node));
186         }
187     }
