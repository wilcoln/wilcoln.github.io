13  package spoon.javadoc.internal;
14  
15  import java.io.Serializable;
16  
17  /**
18  * A piece of text inside a Javadoc description.
19  *
20  * <p>For example in <code>A class totally unrelated to {@link String}, I swear!</code> we would
21  * have two snippets: one before and one after the inline tag (<code>{@link String}</code>).
22  */
23  public class JavadocSnippet implements JavadocDescriptionElement, Serializable {
24  	private String text;
25  
26  	public JavadocSnippet(String text) {
27  		if (text == null) {
28  			throw new NullPointerException();
29  		}
30  		if (text.startsWith("/**")) {
31  			text = text.substring(3);
32  		}
33  		this.text = text;
34  	}
35  
36  	@Override
37  	public String toText() {
38  		return this.text;
39  	}
40  
41  	@Override
42  	public boolean equals(Object o) {
43  		if (this == o) {
44  			return true;
45  		}
46  		if (o == null || getClass() != o.getClass()) {
47  			return false;
48  		}
49  
50  		JavadocSnippet that = (JavadocSnippet) o;
51  
52  		return text.equals(that.text);
53  	}
54  
55  	@Override
56  	public int hashCode() {
57  		return text.hashCode();
58  	}
59  
60  	@Override
61  	public String toString() {
62  		return "JavadocSnippet{" + "text='" + text + '\'' + '}';
63  	}
64  }
