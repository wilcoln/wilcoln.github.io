5   package net.sourceforge.pmd.lang.metrics;
6   
7   import net.sourceforge.pmd.lang.ast.Node;
8   
9   /**
10   * Abstract class for all metrics.
11   *
12   * @param <N> Type of nodes the metric can be computed on
13   *
14   * @author Cl√©ment Fournier
15   * @since 6.0.0
16   */
17  public abstract class AbstractMetric<N extends Node> implements Metric<N> {
18  
19      /**
20       * Metrics should be stateless, thus any instance of the same metric class should be equal.
21       *
22       * {@inheritDoc}
23       */
24      @Override
25      public final boolean equals(Object o) {
26          return o != null && o.getClass() == this.getClass();
27      }
28  
29  
30      @Override
31      public final int hashCode() {
32          return getClass().hashCode();
33      }
34  
35  }
