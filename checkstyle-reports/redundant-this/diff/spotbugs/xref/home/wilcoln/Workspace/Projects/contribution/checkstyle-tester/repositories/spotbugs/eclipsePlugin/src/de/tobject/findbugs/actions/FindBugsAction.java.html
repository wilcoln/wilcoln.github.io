64  /**
65   * Run FindBugs on the currently selected element(s) in the package explorer.
66   *
67   * @author Peter Friese
68   * @author Phil Crosby
69   * @author Andrei Loskutov
70   * @version 1.1
71   * @since 25.09.2003
72   */
73  public class FindBugsAction implements IObjectActionDelegate {
74  
75      /** The current selection. */
76      protected ISelection selection;
77  
78      /** true if this action is used from editor */
79      protected boolean usedInEditor;
80  
81      private IWorkbenchPart targetPart;
82  
83      private static boolean dialogAlreadyShown;
84  
85      @Override
86      public final void setActivePart(final IAction action, final IWorkbenchPart targetPart) {
87          this.targetPart = targetPart;
88      }
89  
90      public static boolean isFindBugsPerspectiveActive(IWorkbenchPart part) {
91          IPerspectiveDescriptor perspective = getWindow(part).getActivePage().getPerspective();
92          return perspective != null && FindBugsPerspectiveFactory.ID.equals(perspective.getId());
93      }
94  
95      @Override
96      public final void selectionChanged(final IAction action, final ISelection newSelection) {
97          if (!usedInEditor) {
98              this.selection = newSelection;
99          }
100     }
101 
102     @Override
103     public void run(final IAction action) {
104         if (selection == null || selection.isEmpty()) {
105             return;
106         }
107         IWorkspace workspace = ResourcesPlugin.getWorkspace();
108         IWorkspaceDescription description = workspace.getDescription();
109         if (!description.isAutoBuilding() && getClass().equals(FindBugsAction.class)) {
110             boolean confirm = MessageDialog.openConfirm(null, "Project -> 'Build Automatically' disabled",
111                     "You are going to run SpotBugs analysis on a not compiled or partially compiled project.\n\n"
112                             + "To get reliable analysis results, you should make sure that project is compiled first.\n\n"
113                             + "Continue with SpotBugs analysis?");
114             if (!confirm) {
115                 return;
116             }
117         }
118 
119         if (selection instanceof IStructuredSelection) {
120             IStructuredSelection sSelection = (IStructuredSelection) selection;
121 
122             dialogAlreadyShown = false;
123 
124             Map<IProject, List<WorkItem>> projectMap = ResourceUtils.getResourcesPerProject(sSelection);
125 
126             for (Map.Entry<IProject, List<WorkItem>> e : projectMap.entrySet()) {
127                 work(targetPart, e.getKey(), e.getValue());
128             }
129             targetPart = null;
130             selection = null;
131         }
132     }
133 
134     protected IDialogSettings getDialogSettings() {
135         IDialogSettings settings = FindbugsPlugin.getDefault().getDialogSettings();
136         String settingsId = getDialogSettingsId();
137         IDialogSettings section = settings.getSection(settingsId);
138         if (section == null) {
139             section = settings.addNewSection(settingsId);
140         }
141         return section;
142     }
143 
144     protected final IProject getProject(IStructuredSelection structuredSelection) {
145         Object element = structuredSelection.getFirstElement();
146         IResource resource = ResourceUtils.getResource(element);
147         if (resource == null) {
148             return null;
149         }
150         IProject project = resource.getProject();
151         return project;
152     }
153 
154     protected String getDialogSettingsId() {
155         return "findBugsAction";
156     }
157 
158     /**
159      * Run a FindBugs analysis on the given resource, displaying a progress
160      * monitor.
161      *
162      * @param part
163      *
164      * @param resources
165      *            The resource to run the analysis on.
166      */
167     protected void work(IWorkbenchPart part, final IResource resource, final List<WorkItem> resources) {
168         FindBugsJob runFindBugs = new StartedFromViewJob("Finding bugs in " + resource.getName() + "...", resource, resources, part);
169         runFindBugs.scheduleInteractive();
170     }
171 
172     protected static void refreshViewer(IWorkbenchPart targetPart, final List<WorkItem> resources) {
173         if (targetPart == null) {
174             return;
175         }
176         ISelectionProvider selProvider = (ISelectionProvider) targetPart.getAdapter(ISelectionProvider.class);
177         if (!(selProvider instanceof TreeViewer)) {
178             return;
179         }
180         final TreeViewer viewer = (TreeViewer) selProvider;
181         Display.getDefault().asyncExec(new Runnable() {
182             @Override
183             public void run() {
184                 if (viewer.getControl().isDisposed()) {
185                     return;
186                 }
187                 for (WorkItem workItem : resources) {
188                     if (workItem.getMarkerTarget() instanceof IProject) {
189                         // this element has to be refreshed manually, because
190                         // there is no one real
191                         // resource associated with it => no resource change
192                         // notification after
193                         // creating a marker...
194                         viewer.refresh(workItem.getCorespondingJavaElement(), true);
195                     }
196                 }
197             }
198         });
199     }
200 
201     protected static void askUserToSwitch(IWorkbenchPart part, int warningsNumber) {
202         final IPreferenceStore store = FindbugsPlugin.getDefault().getPreferenceStore();
203         String message = "SpotBugs analysis finished, " + warningsNumber
204                 + " warnings found.\n\nSwitch to the SpotBugs perspective?";
205 
206         MessageDialogWithToggle dialog = MessageDialogWithToggle.openYesNoCancelQuestion(null, "SpotBugs analysis finished",
207                 message, "Remember the choice and do not ask me in the future", false, store,
208                 FindBugsConstants.ASK_ABOUT_PERSPECTIVE_SWITCH);
209 
210         boolean remember = dialog.getToggleState();
211         int returnCode = dialog.getReturnCode();
212 
213         if (returnCode == IDialogConstants.YES_ID) {
214             if (remember) {
215                 store.setValue(FindBugsConstants.SWITCH_PERSPECTIVE_AFTER_ANALYSIS, true);
216             }
217             switchPerspective(part);
218         } else if (returnCode == IDialogConstants.NO_ID) {
219             if (remember) {
220                 store.setValue(FindBugsConstants.SWITCH_PERSPECTIVE_AFTER_ANALYSIS, false);
221             }
222         }
223     }
224 
225     protected static IWorkbenchWindow getWindow(IWorkbenchPart part) {
226         IWorkbenchWindow window;
227         IWorkbenchPartSite currentSite = part != null ? part.getSite() : null;
228         if (currentSite != null) {
229             window = currentSite.getWorkbenchWindow();
230         } else {
231             window = FindbugsPlugin.getActiveWorkbenchWindow();
232         }
233         return window;
234     }
235 
236     protected static void switchPerspective(IWorkbenchPart part) {
237         IWorkbenchWindow window = getWindow(part);
238         IWorkbenchPage page = window.getActivePage();
239         IAdaptable input;
240         if (page != null) {
241             input = page.getInput();
242         } else {
243             input = ResourcesPlugin.getWorkspace().getRoot();
244         }
245         try {
246             PlatformUI.getWorkbench().showPerspective(FindBugsPerspectiveFactory.ID, window, input);
247         } catch (WorkbenchException e) {
248             FindbugsPlugin.getDefault().logException(e, "Failed to open SpotBugs Perspective");
249         }
250     }

252     private final static class StartedFromViewJob extends FindBugsJob {
253         private final List<WorkItem> resources;
254 
255         private final IResource resource;
256 
257         private final IWorkbenchPart targetPart;
258 
259         private StartedFromViewJob(String name, IResource resource, List<WorkItem> resources, IWorkbenchPart targetPart) {
260             super(name, resource);
261             this.resources = resources;
262             this.resource = resource;
263             this.targetPart = targetPart;
264         }
265 
266         @Override
267         protected boolean supportsMulticore(){
268             return MutexSchedulingRule.MULTICORE;
269         }
270 
271         @Override
272         protected void runWithProgress(IProgressMonitor monitor) throws CoreException {
273             FindBugsWorker worker = new FindBugsWorker(resource, monitor);
274 
275             worker.work(resources);
276 
277             refreshViewer(targetPart, resources);
278 
279             checkPerspective();
280         }
281 
282         private void checkPerspective() {
283             if (isFindBugsPerspectiveActive(targetPart)) {
284                 return;
285             }
286             final IMarker[] allMarkers = MarkerUtil.getAllMarkers(resource);
287             if (allMarkers.length == 0) {
288                 return;
289             }
290 
291             Display.getDefault().asyncExec(new Runnable() {
292                 @Override
293                 public void run() {
294                     if (isFindBugsPerspectiveActive(targetPart)) {
295                         return;
296                     }
297                     final IPreferenceStore store = FindbugsPlugin.getDefault().getPreferenceStore();
298                     final boolean ask = store.getBoolean(FindBugsConstants.ASK_ABOUT_PERSPECTIVE_SWITCH);
299                     if (ask && !dialogAlreadyShown) {
300                         dialogAlreadyShown = true;
301                         askUserToSwitch(targetPart, allMarkers.length);
302                     } else if (store.getBoolean(FindBugsConstants.SWITCH_PERSPECTIVE_AFTER_ANALYSIS)) {
303                         switchPerspective(targetPart);
304                     }
305                 }
306             });
307         }
308 
309     }
