32  package com.sun.demo.scripting.jconsole;
33  
34  import java.awt.*;
35  import java.awt.event.*;
36  import java.util.concurrent.ExecutorService;
37  import java.util.concurrent.Executors;
38  import javax.swing.*;
39  import javax.swing.event.*;
40  import javax.swing.text.*;
41  
42  
43  /**
44   * A JPanel subclass containing a scrollable text area displaying the
45   * jconsole's script console.
46   */
47  
48  class ScriptShellPanel extends JPanel {
49  
50      private static final long serialVersionUID = 4116273141148726319L;
51  
52      // interface to evaluate script command and script prompt
53      interface CommandProcessor {
54          // execute given String as script and return the result
55          public String executeCommand(String cmd);
56          // get prompt used for interactive read-eval-loop
57          public String getPrompt();
58      }
59  
60      // my script command processor
61      private CommandProcessor commandProcessor;
62      // editor component for command editing
63      private JTextComponent editor;
64  
65      private final ExecutorService commandExecutor =
66              Executors.newSingleThreadExecutor();
67  
68      // document management
69      private boolean updating;
70  
71      public ScriptShellPanel(CommandProcessor cmdProc) {
72          setLayout(new BorderLayout());
73          this.commandProcessor = cmdProc;
74          this.editor = new JTextArea();
75          editor.setDocument(new EditableAtEndDocument());
76          JScrollPane scroller = new JScrollPane();
77          scroller.getViewport().add(editor);
78          add(scroller, BorderLayout.CENTER);
79  
80          editor.getDocument().addDocumentListener(new DocumentListener() {
81              @Override
82              public void changedUpdate(DocumentEvent e) {
83              }
84  
85              @Override
86              public void insertUpdate(DocumentEvent e) {
87                  if (updating) return;
88                  beginUpdate();
89                  editor.setCaretPosition(editor.getDocument().getLength());
90                  if (insertContains(e, '\n')) {
91                      String cmd = getMarkedText();
92                      // Handle multi-line input
93                      if ((cmd.length() == 0) ||
94                          (cmd.charAt(cmd.length() - 1) != '\\')) {
95                          // Trim "\\n" combinations
96                          final String cmd1 = trimContinuations(cmd);
97                          commandExecutor.execute(new Runnable() {
98                              @Override
99                              public void run() {
100                                 final String result = executeCommand(cmd1);
101 
102                                 SwingUtilities.invokeLater(new Runnable() {
103                                     @Override
104                                     public void run() {
105                                         if (result != null) {
106                                             print(result + "\n");
107                                         }
108                                         printPrompt();
109                                         setMark();
110                                         endUpdate();
111                                     }
112                                 });
113                             }
114                         });
115                     } else {
116                         endUpdate();
117                     }
118                 } else {
119                     endUpdate();
120                 }
121             }
122 
123             @Override
124             public void removeUpdate(DocumentEvent e) {
125             }
126         });
127 
128         // This is a bit of a hack but is probably better than relying on
129         // the JEditorPane to update the caret's position precisely the
130         // size of the insertion
131         editor.addCaretListener(new CaretListener() {
132             @Override
133             public void caretUpdate(CaretEvent e) {
134                 int len = editor.getDocument().getLength();
135                 if (e.getDot() > len) {
136                     editor.setCaretPosition(len);
137                 }
138             }
139         });
140 
141         Box hbox = Box.createHorizontalBox();
142         hbox.add(Box.createGlue());
143         JButton button = new JButton("Clear"); // FIXME: i18n?
144         button.addActionListener(new ActionListener() {
145             @Override
146             public void actionPerformed(ActionEvent e) {
147                 clear();
148             }
149         });
150         hbox.add(button);
151         hbox.add(Box.createGlue());
152         add(hbox, BorderLayout.SOUTH);
153 
154         clear();
155     }
156 
157     public void dispose() {
158         commandExecutor.shutdown();
159     }
160 
161     @Override
162     public void requestFocus() {
163         editor.requestFocus();
164     }
165 
166     public void clear() {
167         clear(true);
168     }
169 
170     public void clear(boolean prompt) {
171         EditableAtEndDocument d = (EditableAtEndDocument) editor.getDocument();
172         d.clear();
173         if (prompt) printPrompt();
174         setMark();
175         editor.requestFocus();
176     }
177 
178     public void setMark() {
179         ((EditableAtEndDocument) editor.getDocument()).setMark();
180     }
181 
182     public String getMarkedText() {
183         try {
184             String s = ((EditableAtEndDocument) editor.getDocument()).getMarkedText();
185             int i = s.length();
186             while ((i > 0) && (s.charAt(i - 1) == '\n')) {
187                 i--;
188             }
189             return s.substring(0, i);
190         } catch (BadLocationException e) {
191             e.printStackTrace();
192             return null;
193         }
194     }
195 
196     public void print(String s) {
197         Document d = editor.getDocument();
198         try {
199             d.insertString(d.getLength(), s, null);
200         } catch (BadLocationException e) {
201             e.printStackTrace();
202         }
203     }
204 
205 
206     //
207     // Internals only below this point
208     //
209 
210     private String executeCommand(String cmd) {
211         return commandProcessor.executeCommand(cmd);
212     }
213 
214     private String getPrompt() {
215         return commandProcessor.getPrompt();
216     }
217 
218     private void beginUpdate() {
219         editor.setEditable(false);
220         updating = true;
221     }
222 
223     private void endUpdate() {
224         editor.setEditable(true);
225         updating = false;
226     }
227 
228     private void printPrompt() {
229         print(getPrompt());
230     }
231 
232     private boolean insertContains(DocumentEvent e, char c) {
233         String s = null;
234         try {
235             s = editor.getText(e.getOffset(), e.getLength());
236             for (int i = 0; i < e.getLength(); i++) {
237                 if (s.charAt(i) == c) {
238                     return true;
239                 }
240             }
241         } catch (BadLocationException ex) {
242             ex.printStackTrace();
243         }
244         return false;
245     }
246 
247     private String trimContinuations(String text) {
248         int i;
249         while ((i = text.indexOf("\\\n")) >= 0) {
250             text = text.substring(0, i) + text.substring(i+1, text.length());
251         }
252         return text;
253     }
254 }
