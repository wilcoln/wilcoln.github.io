2   package org.hibernate.test.annotations.collectionelement;
3   import java.util.Map;
4   import java.util.SortedMap;
5   import java.util.TreeMap;
6   import javax.persistence.Column;
7   import javax.persistence.ElementCollection;
8   import javax.persistence.Entity;
9   import javax.persistence.GeneratedValue;
10  import javax.persistence.Id;
11  import javax.persistence.MapKeyColumn;
12  
13  import org.hibernate.annotations.MapKeyType;
14  import org.hibernate.annotations.Sort;
15  import org.hibernate.annotations.SortType;
16  import org.hibernate.annotations.Type;
17  
18  /**
19   * @author Emmanuel Bernard
20   */
21  @Entity(name="Mtx")
22  public class Matrix {
23  	@Id
24  	@GeneratedValue
25  	@Column(name="mId")
26  	private Integer id;
27  
28  	@MapKeyType( @Type(type="integer") )
29  	@ElementCollection
30  	@Sort(type = SortType.NATURAL) 
31  	@Type(type = "float")
32  	@MapKeyColumn(nullable = false)
33  	private SortedMap<Integer, Float> mvalues = new TreeMap<Integer, Float>();
34  
35  	public Integer getId() {
36  		return id;
37  	}
38  
39  	public void setId(Integer id) {
40  		this.id = id;
41  	}
42  
43  	public Map<Integer, Float> getMvalues() {
44  		return mvalues;
45  	}
46  
47  	public void setMvalues(SortedMap<Integer, Float> mValues) {
48  		this.mvalues = mValues;
49  	}
50  }
