29  /** Transformation class for geodetic systems.
30   *
31   * <p>The Helmert transformation is mainly used to convert between various
32   * realizations of geodetic frames, for example in the ITRF family.</p>
33   *
34   * <p>The original Helmert transformation is a 14 parameters transform that
35   * includes translation, velocity, rotation, rotation rate and scale factor.
36   * The scale factor is useful for coordinates near Earth surface, but it
37   * cannot be extended to outer space as it would correspond to a non-unitary
38   * transform. Therefore, the scale factor is <em>not</em> used here.
39   *
40   * <p>Instances of this class are guaranteed to be immutable.</p>
41   *
42   * @author Luc Maisonobe
43   * @since 5.1
44   */
45  public class HelmertTransformation implements TransformProvider {

178     /** Cartesian part of the transform. */
179     private final PVCoordinates cartesian;
180 
181     /** Global rotation vector (applying rotation is done by computing cross product). */
182     private final Vector3D rotationVector;
183 
184     /** First time derivative of the rotation (norm representing angular rate). */
185     private final Vector3D rotationRate;
186 
187     /** Reference epoch of the transform. */
188     private final AbsoluteDate epoch;
189 
190     /** Build a transform from its primitive operations.
191      * @param epoch reference epoch of the transform
192      * @param t1 translation parameter along X axis (BEWARE, this is in mm)
193      * @param t2 translation parameter along Y axis (BEWARE, this is in mm)
194      * @param t3 translation parameter along Z axis (BEWARE, this is in mm)
195      * @param r1 rotation parameter around X axis (BEWARE, this is in mas)
196      * @param r2 rotation parameter around Y axis (BEWARE, this is in mas)
197      * @param r3 rotation parameter around Z axis (BEWARE, this is in mas)
198      * @param t1Dot rate of translation parameter along X axis (BEWARE, this is in mm/y)
199      * @param t2Dot rate of translation parameter along Y axis (BEWARE, this is in mm/y)
200      * @param t3Dot rate of translation parameter along Z axis (BEWARE, this is in mm/y)
201      * @param r1Dot rate of rotation parameter around X axis (BEWARE, this is in mas/y)
202      * @param r2Dot rate of rotation parameter around Y axis (BEWARE, this is in mas/y)
203      * @param r3Dot rate of rotation parameter around Z axis (BEWARE, this is in mas/y)
204      */
205     public HelmertTransformation(final AbsoluteDate epoch,
206                                  final double t1, final double t2, final double t3,
207                                  final double r1, final double r2, final double r3,
208                                  final double t1Dot, final double t2Dot, final double t3Dot,
209                                  final double r1Dot, final double r2Dot, final double r3Dot) {
210 
211         // conversion parameters to SI units
212         final double mmToM    = 1.0e-3;
213         final double masToRad = 1.0e-3 * Constants.ARC_SECONDS_TO_RADIANS;
214 
215         this.epoch          = epoch;
216         this.cartesian = new PVCoordinates(new Vector3D(t1 * mmToM,
217                                                         t2 * mmToM,
218                                                         t3 * mmToM),
219                                            new Vector3D(t1Dot * mmToM / Constants.JULIAN_YEAR,
220                                                         t2Dot * mmToM / Constants.JULIAN_YEAR,
221                                                         t3Dot * mmToM / Constants.JULIAN_YEAR));
222         this.rotationVector = new Vector3D(r1 * masToRad,
223                                            r2 * masToRad,
224                                            r3 * masToRad);
225         this.rotationRate   = new Vector3D(r1Dot * masToRad / Constants.JULIAN_YEAR,
226                                            r2Dot * masToRad / Constants.JULIAN_YEAR,
227                                            r3Dot * masToRad / Constants.JULIAN_YEAR);
228 
229     }
230 
231     /** Get the reference epoch of the transform.
232      * @return reference epoch of the transform
233      */
234     public AbsoluteDate getEpoch() {
235         return epoch;
236     }
237 
238     /** Compute the transform at some date.
239      * @param date date at which the transform is desired
240      * @return computed transform at specified date
241      */
242     public Transform getTransform(final AbsoluteDate date) {
243 
244         // compute parameters evolution since reference epoch
245         final double dt = date.durationFrom(epoch);
246         final Vector3D dR = new Vector3D(1, rotationVector, dt, rotationRate);
247 
248         // build translation part
249         final Transform translationTransform = new Transform(date, cartesian.shiftedBy(dt));
250 
251         // build rotation part
252         final double angle = dR.getNorm();
253         final Transform rotationTransform =
254                 new Transform(date,
255                               (angle < Precision.SAFE_MIN) ?
256                               Rotation.IDENTITY :
257                               new Rotation(dR, angle, RotationConvention.VECTOR_OPERATOR),
258                               rotationRate);
259 
260         // combine both parts
261         return new Transform(date, translationTransform, rotationTransform);
262 
263     }
