6   package spoon.support.util;
7   
8   import java.util.Collection;
9   import java.util.Comparator;
10  import java.util.LinkedList;
11  import java.util.ListIterator;
12  
13  public class SortedList<E> extends LinkedList<E> {
14  
15  	private static final long serialVersionUID = 1L;
16  
17  	Comparator<? super E> comparator;
18  
19  	public SortedList(Comparator<? super E> comparator) {
20  		this.comparator = comparator;
21  	}
22  
23  	@Override
24  	public boolean add(E o) {
25  		for (ListIterator<E> iterator = this.listIterator(); iterator.hasNext();) {
26  			E e = iterator.next();
27  			if (comparator.compare(o, e) < 0) {
28  				iterator.previous();
29  				iterator.add(o);
30  				return true;
31  			}
32  		}
33  		return super.add(o);
34  	}
35  
36  	@Override
37  	public void add(int index, E element) {
38  		throw new IllegalArgumentException("cannot force a position with a sorted list that has its own ordering");
39  	}
40  
41  	@Override
42  	public boolean addAll(Collection<? extends E> c) {
43  		boolean ret = true;
44  		for (E e : c) {
45  			ret &= add(e);
46  		}
47  		return ret;
48  	}
49  
50  	public Comparator<? super E> getComparator() {
51  		return comparator;
52  	}
53  
54  	public void setComparator(Comparator<? super E> comparator) {
55  		this.comparator = comparator;
56  	}
57  
58  }
