7   import static org.junit.Assert.assertEquals;
8   
9   import java.io.IOException;
10  import java.nio.charset.StandardCharsets;
11  
12  import org.apache.commons.io.IOUtils;
13  import org.junit.Before;
14  import org.junit.Test;
15  
16  import net.sourceforge.pmd.PMD;
17  import net.sourceforge.pmd.testframework.AbstractTokenizerTest;
18  
19  public class PythonTokenizerTest extends AbstractTokenizerTest {
20  
21      private static final String FILENAME = "sample-python.py";
22  
23      @Before
24      @Override
25      public void buildTokenizer() throws IOException {
26          this.tokenizer = new PythonTokenizer();
27          this.sourceCode = new SourceCode(new SourceCode.StringCodeLoader(this.getSampleCode(), FILENAME));
28      }
29  
30      @Override
31      public String getSampleCode() throws IOException {
32          return IOUtils.toString(PythonTokenizer.class.getResourceAsStream(FILENAME), StandardCharsets.UTF_8);
33      }
34  
35      @Test
36      public void tokenizeTest() throws IOException {
37          this.expectedTokenCount = 1218;
38          super.tokenizeTest();
39      }
40  
41      @Test
42      public void testIgnoreBetweenSpecialComments() throws IOException {
43          SourceCode sourceCode = new SourceCode(new SourceCode.StringCodeLoader("import logging" + PMD.EOL
44                  + "# CPD-OFF" + PMD.EOL
45                  + "logger = logging.getLogger('django.request')" + PMD.EOL
46                  + "class BaseHandler(object):" + PMD.EOL
47                  + "    def __init__(self):" + PMD.EOL
48                  + "        self._request_middleware = None" + PMD.EOL
49                  + "        # CPD-ON" + PMD.EOL
50          ));
51          Tokens tokens = new Tokens();
52          tokenizer.tokenize(sourceCode, tokens);
53          TokenEntry.getEOF();
54          assertEquals(3, tokens.size()); // 3 tokens: "import" + "logging" + EOF
55      }
56  
57      @Test
58      public void testBackticks() throws IOException {
59          SourceCode sourceCode = new SourceCode(new SourceCode.StringCodeLoader("test = 'hello'" + PMD.EOL
60                  + "quoted = `test`" + PMD.EOL
61                  + "print quoted" + PMD.EOL
62          ));
63          Tokens tokens = new Tokens();
64          tokenizer.tokenize(sourceCode, tokens); // should not result in parse error
65          TokenEntry.getEOF();
66          assertEquals(3, tokens.getTokens().get(tokens.getTokens().size() - 2).getBeginLine());
67      }
68  }
