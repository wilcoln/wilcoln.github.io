22  package lombok.ast.grammar;
23  
24  import lombok.ast.Node;
25  
26  import org.parboiled.BaseParser;
27  import org.parboiled.Rule;
28  import org.parboiled.annotations.SuppressSubnodes;
29  
30  public class StructuresParser extends BaseParser<Node> {
31  	final ParserGroup group;
32  	final StructuresActions actions;
33  	
34  	public StructuresParser(ParserGroup group) {
35  		this.actions = new StructuresActions(group.getSource());
36  		this.group = group;
37  	}
38  	
39  	public Rule typeBody() {
40  		return Sequence(
41  				Ch('{'), group.basics.optWS(),
42  				typeBodyDeclarations(), set(),
43  				Ch('}'), group.basics.optWS(),
44  				set(actions.posify(value())));
45  	}
46  	
47  	public Rule typeBodyMember() {
48  		return FirstOf(
49  				anyTypeDeclaration(),
50  				fieldDeclaration(),
51  				methodDeclaration(),
52  				constructorDeclaration(),
53  				staticInitializer(),
54  				instanceInitializer(),
55  				emptyDeclaration());
56  	}
57  	
58  	Rule typeBodyDeclarations() {
59  		return Sequence(
60  				ZeroOrMore(typeBodyMember().label("member")).label("members"),
61  				set(actions.createNormalTypeBody(values("members/member"))));
62  	}
63  	
64  	Rule emptyDeclaration() {
65  		return Sequence(Ch(';'), group.basics.optWS(), set(actions.createEmptyDeclaration()));
66  	}
67  	
68  	public Rule methodArguments() {
69  		return Sequence(
70  				Ch('('),
71  				group.basics.optWS(),
72  				Optional(Sequence(
73  						group.expressions.anyExpression(),
74  						set(),
75  						ZeroOrMore(Sequence(
76  								Ch(','),
77  								group.basics.optWS(),
78  								group.expressions.anyExpression(), set())))),
79  				Ch(')'),
80  				group.basics.optWS(),
81  				set(actions.createMethodArguments(value("Optional/Sequence"), values("Optional/Sequence/ZeroOrMore/Sequence"))));
82  	}
83  	
84  	public Rule anyTypeDeclaration() {
85  		return Sequence(
86  				TestNot(FirstOf(Eoi(), Ch('}'))),
87  				FirstOf(
88  						classOrInterfaceDeclaration(),
89  						enumDeclaration(),
90  						annotationDeclaration(),
91  						emptyDeclaration()));
92  	}
93  	
94  	public Rule classOrInterfaceDeclaration() {
95  		return Sequence(
96  				typeDeclarationModifiers().label("modifiers"),
97  				FirstOf(String("class"), String("interface")).label("kind"),
98  				group.basics.testLexBreak(), group.basics.optWS(),
99  				group.basics.identifier().label("typeName"),
100 				group.types.typeVariables().label("typeParameters"),
101 				ZeroOrMore(FirstOf(
102 						extendsClause(),
103 						implementsClause()).label("addon")).label("addons"),
104 				typeBody().label("body"),
105 				set(actions.createTypeDeclaration(text("kind"), value("modifiers"), value("typeName"), value("typeParameters"), value("body"), values("addons/addon"))));
106 	}
107 	
108 	Rule extendsClause() {
109 		return Sequence(
110 				Sequence(String("extends"), group.basics.testLexBreak(), group.basics.optWS()),
111 				group.types.type().label("head"),
112 				ZeroOrMore(Sequence(
113 						Ch(','), group.basics.optWS(),
114 						group.types.type()).label("tail")),
115 				set(actions.createExtendsClause(value("head"), values("ZeroOrMore/tail"))));
116 	}
117 	
118 	Rule implementsClause() {
119 		return Sequence(
120 				Sequence(String("implements"), group.basics.testLexBreak(), group.basics.optWS()),
121 				group.types.type().label("head"),
122 				ZeroOrMore(Sequence(
123 						Ch(','), group.basics.optWS(),
124 						group.types.type()).label("tail")),
125 				set(actions.createImplementsClause(value("head"), values("ZeroOrMore/tail"))));
126 	}
127 	
128 	public Rule enumDeclaration() {
129 		return Sequence(
130 				typeDeclarationModifiers().label("modifiers"),
131 				String("enum"), group.basics.testLexBreak(), group.basics.optWS(),
132 				group.basics.identifier().label("typeName"),
133 				ZeroOrMore(FirstOf(
134 						extendsClause(),
135 						implementsClause()).label("addon")).label("addons"),
136 				enumBody().label("body"),
137 				set(actions.createEnumDeclaration(value("modifiers"), value("typeName"), value("body"), values("addons/addon"))));
138 	}
139 	
140 	public Rule annotationDeclaration() {
141 		return Sequence(
142 				typeDeclarationModifiers().label("modifiers"),
143 				Ch('@'), group.basics.optWS(),
144 				String("interface"), group.basics.testLexBreak(), group.basics.optWS(),
145 				group.basics.identifier().label("name"),
146 				Ch('{').label("typeOpen"), group.basics.optWS(),
147 				ZeroOrMore(annotationElementDeclaration().label("member")).label("members"),
148 				Ch('}').label("typeClose"), group.basics.optWS(),
149 				set(actions.createAnnotationDeclaration(value("modifiers"), value("name"), values("members/member"), node("typeOpen"), node("typeClose"))));
150 	}
151 	
152 	Rule annotationElementDeclaration() {
153 		return FirstOf(
154 				annotationMethodDeclaration(),
155 				fieldDeclaration(),
156 				classOrInterfaceDeclaration(),
157 				enumDeclaration(),
158 				annotationDeclaration(),
159 				Sequence(Ch(';'), group.basics.optWS())
160 				);
161 	}
162 	
163 	Rule enumBody() {
164 		return Sequence(
165 				Ch('{'), group.basics.optWS(),
166 				Optional(Sequence(
167 						enumConstant().label("head"),
168 						ZeroOrMore(Sequence(
169 								Ch(','), group.basics.optWS(),
170 								enumConstant()).label("tail")),
171 						Optional(Sequence(Ch(','), group.basics.optWS())))).label("constants"),
172 				Optional(Sequence(
173 						Ch(';'), group.basics.optWS(),
174 						typeBodyDeclarations())).label("typeBodyDeclarations"),
175 				Ch('}'), group.basics.optWS(),
176 				set(actions.createEnumBody(value("constants/Sequence/head"), values("constants/Sequence/ZeroOrMore/tail"), value("typeBodyDeclarations"))));
177 	}
178 	
179 	Rule enumConstant() {
180 		return Sequence(
181 				ZeroOrMore(annotation().label("annotation")).label("annotations"),
182 				group.basics.identifier().label("name"),
183 				Optional(methodArguments()).label("arguments"),
184 				Optional(typeBody()).label("body"),
185 				set(actions.createEnumConstant(values("annotations/annotation"), value("name"), value("arguments"), value("body"))));
186 	}
187 	
188 	public Rule constructorDeclaration() {
189 		return Sequence(
190 				methodDeclarationModifiers().label("modifiers"),
191 				group.types.typeVariables().label("typeParameters"),
192 				group.basics.identifier().label("typeName"),
193 				methodParameters().label("params"),
194 				Optional(Sequence(
195 						Sequence(String("throws"), group.basics.testLexBreak(), group.basics.optWS()),
196 						group.types.type().label("throwsHead"),
197 						ZeroOrMore(Sequence(Ch(','), group.basics.optWS(), group.types.type()).label("throwsTail"))
198 						)).label("throwsClause"),
199 				FirstOf(
200 						Sequence(Ch(';'), group.basics.optWS()),
201 						group.statements.blockStatement()).label("body"),
202 				set(actions.createConstructorDeclaration(value("modifiers"), value("typeParameters"), value("typeName"), value("params"), 
203 						value("throwsClause/Sequence/throwsHead"), values("throwsClause/Sequence/ZeroOrMore/throwsTail"),
204 						value("body"))));
205 	}
206 	
207 	public Rule annotationMethodDeclaration() {
208 		return Sequence(
209 				methodDeclarationModifiers().label("modifiers"),
210 				group.types.type().label("resultType"),
211 				group.basics.identifier().label("methodName"),
212 				Ch('('), group.basics.optWS(),
213 				Ch(')'), group.basics.optWS(),
214 				ZeroOrMore(Sequence(Ch('['), group.basics.optWS(), Ch(']'), group.basics.optWS()).label("dim")).label("dims"),
215 				Optional(Sequence(
216 						Sequence(String("default"), group.basics.testLexBreak(), group.basics.optWS()),
217 						annotationElementValue())).label("defaultValue"),
218 				Ch(';'), group.basics.optWS(),
219 				set(actions.createAnnotationMethodDeclaration(value("modifiers"), value("resultType"), value("methodName"), nodes("dims/dim"), value("defaultValue"))));
220 	}
221 	
222 	public Rule methodDeclaration() {
223 		return Sequence(
224 				methodDeclarationModifiers().label("modifiers"),
225 				group.types.typeVariables().label("typeParameters"),
226 				group.types.type().label("resultType"),
227 				group.basics.identifier().label("methodName"),
228 				methodParameters().label("params"),
229 				ZeroOrMore(Sequence(Ch('['), group.basics.optWS(), Ch(']'), group.basics.optWS()).label("dim")).label("dims"),
230 				Optional(Sequence(
231 						Sequence(String("throws"), group.basics.testLexBreak(), group.basics.optWS()),
232 						group.types.type().label("throwsHead"),
233 						ZeroOrMore(Sequence(Ch(','), group.basics.optWS(), group.types.type()).label("throwsTail"))
234 						)).label("throwsClause"),
235 				FirstOf(
236 						Sequence(Ch(';'), group.basics.optWS()),
237 						group.statements.blockStatement()).label("body"),
238 				set(actions.createMethodDeclaration(value("modifiers"), value("typeParameters"), value("resultType"), value("methodName"), value("params"), 
239 						nodes("dims/dim"), value("throwsClause/Sequence/throwsHead"), values("throwsClause/Sequence/ZeroOrMore/throwsTail"),
240 						value("body"))));
241 	}
242 	
243 	Rule methodParameters() {
244 		return Sequence(
245 				Ch('('), group.basics.optWS(),
246 				Optional(Sequence(
247 						methodParameter().label("head"),
248 						ZeroOrMore(Sequence(
249 								Ch(','), group.basics.optWS(),
250 								methodParameter().label("tail"))))),
251 				Ch(')'), group.basics.optWS(),
252 				set(actions.createMethodParameters(value("Optional/Sequence/head"), values("Optional/Sequence/ZeroOrMore/Sequence/tail"))));
253 	}
254 	
255 	Rule methodParameter() {
256 		return Sequence(
257 				variableDefinitionModifiers().label("modifiers"),
258 				group.types.type().label("type"),
259 				Optional(Sequence(String("..."), group.basics.optWS())).label("varargs"),
260 				group.basics.identifier().label("name"),
261 				ZeroOrMore(Sequence(Ch('[').label("open"), group.basics.optWS(), Ch(']').label("closed"), group.basics.optWS()).label("dim")).label("dims"),
262 				set(actions.createMethodParameter(value("modifiers"), value("type"), text("varargs"), value("name"), nodes("dims/dim/open"), nodes("dims/dim/closed"))));
263 	}
264 	
265 	public Rule instanceInitializer() {
266 		return Sequence(
267 				group.statements.blockStatement().label("initializer"),
268 				set(actions.createInstanceInitializer(value("initializer"))));
269 	}
270 	
271 	public Rule staticInitializer() {
272 		return Sequence(
273 				String("static"), group.basics.testLexBreak(), group.basics.optWS(),
274 				group.statements.blockStatement().label("initializer"),
275 				set(actions.createStaticInitializer(value("initializer"))));
276 	}
277 	
278 	public Rule fieldDeclaration() {
279 		return Sequence(
280 				fieldDeclarationModifiers().label("modifiers"),
281 				variableDefinition(), set(), set(actions.posify(value())),
282 				Ch(';'), group.basics.optWS(),
283 				set(actions.createFieldDeclaration(value(), value("modifiers"))));
284 	}

286 	/**
287 	 * Add your own modifiers!
288 	 */
289 	Rule variableDefinition() {
290 		return Sequence(
291 				group.types.type().label("type"),
292 				variableDefinitionPart().label("head"),
293 				ZeroOrMore(Sequence(
294 						Ch(','), group.basics.optWS(),
295 						variableDefinitionPart()).label("tail")),
296 				set(actions.createVariableDefinition(value("type"), value("head"), values("ZeroOrMore/tail"))));
297 	}
298 	
299 	Rule variableDefinitionPartNoAssign() {
300 		return Sequence(
301 				group.basics.identifier().label("varName"),
302 				ZeroOrMore(Sequence(Ch('['), group.basics.optWS(), Ch(']'), group.basics.optWS()).label("dim")).label("dims"),
303 				set(actions.createVariableDefinitionPart(value("varName"), texts("dims/dim"), null)));
304 	}
305 	
306 	Rule variableDefinitionPart() {
307 		return Sequence(
308 				group.basics.identifier().label("varName"),
309 				ZeroOrMore(Sequence(Ch('['), group.basics.optWS(), Ch(']'), group.basics.optWS()).label("dim")).label("dims"),
310 				Optional(Sequence(
311 						Ch('='), group.basics.optWS(),
312 						FirstOf(
313 								group.expressions.arrayInitializer(),
314 								group.expressions.anyExpression()))).label("initializer"),
315 				set(actions.createVariableDefinitionPart(value("varName"), texts("dims/dim"), value("initializer"))));
316 	}
317 	
318 	public Rule annotation() {
319 		return Sequence(
320 				Ch('@'), group.basics.optWS(),
321 				group.types.plainReferenceType().label("annotationType"),
322 				Optional(Sequence(
323 						Ch('('), group.basics.optWS(),
324 						Optional(FirstOf(
325 								annotationElements(),
326 								Sequence(annotationElementValue(),
327 										set(actions.createAnnotationFromElement(lastValue()))))),
328 						Ch(')'), group.basics.optWS())).label("content"),
329 				set(actions.createAnnotation(value("annotationType"), value("content"))));
330 	}
331 	
332 	Rule annotationElements() {
333 		return Sequence(
334 				annotationElement().label("head"),
335 				ZeroOrMore(Sequence(
336 						Ch(','), group.basics.optWS(),
337 						annotationElement()).label("tail")),
338 				set(actions.createAnnotationFromElements(value("head"), values("ZeroOrMore/tail"))));
339 	}
340 	
341 	Rule annotationElement() {
342 		return Sequence(
343 				group.basics.identifier().label("name"),
344 				Ch('='), group.basics.optWS(),
345 				annotationElementValue().label("value"),
346 				set(actions.createAnnotationElement(value("name"), value("value"))));
347 	}
348 	
349 	Rule annotationElementValue() {
350 		return FirstOf(
351 				annotation(),
352 				Sequence(
353 						Ch('{'), group.basics.optWS(),
354 						Optional(Sequence(
355 								annotationElementValue().label("head"),
356 								ZeroOrMore(Sequence(
357 										Ch(','), group.basics.optWS(),
358 										annotationElementValue()).label("tail")),
359 								Optional(Sequence(Ch(','), group.basics.optWS())))),
360 						Ch('}'), group.basics.optWS(),
361 						set(actions.createAnnotationElementValueArrayInitializer(value("Optional/Sequence/head"), values("Optional/Sequence/ZeroOrMore/tail")))),
362 				group.expressions.inlineIfExpressionChaining());
363 	}

365 	@SuppressSubnodes
366 	Rule anyKeyword() {
367 		return FirstOf("final", "strictfp", "abstract", "transient", "volatile",
368 				"public", "protected", "private", "synchronized", "static", "native");
369 	}

371 	public Rule keywordModifier() {
372 		return Sequence(
373 				anyKeyword().label("keyword"),
374 				group.basics.testLexBreak(),
375 				set(actions.createKeywordModifier(text("keyword"))),
376 				group.basics.optWS());
377 	}

382 				ZeroOrMore(anyModifier().label("modifier")),
383 				set(actions.createModifiers(values("ZeroOrMore/modifier"))));
384 	}

390 				set(actions.createModifiers(values("ZeroOrMore/modifier"))));
391 	}

396 				ZeroOrMore(anyModifier().label("modifier")),
397 				set(actions.createModifiers(values("ZeroOrMore/modifier"))));
398 	}

401 		return Sequence(
402 				TestNot(Ch('}')),
403 				ZeroOrMore(anyModifier().label("modifier")),
404 				set(actions.createModifiers(values("ZeroOrMore/modifier"))));
405 	}

407 	public Rule anyModifier() {
408 		return FirstOf(annotation(), keywordModifier());
409 	}
410 	
411 	public Rule packageDeclaration() {
412 		return Sequence(
413 				Sequence(
414 						ZeroOrMore(annotation().label("annotation")).label("annotations"),
415 						String("package"), group.basics.testLexBreak(), group.basics.optWS()),
416 				group.basics.identifier().label("head"),
417 				ZeroOrMore(group.basics.dotIdentifier().label("tail")),
418 				Ch(';'), group.basics.optWS(),
419 				set(actions.createPackageDeclaration(values("Sequence/annotations/annotation"), value("head"), values("ZeroOrMore/tail"))));
420 	}
421 	
422 	public Rule importDeclaration() {
423 		return Sequence(
424 				Sequence(String("import"), group.basics.testLexBreak(), group.basics.optWS()),
425 				Optional(Sequence(String("static"), group.basics.testLexBreak(), group.basics.optWS())).label("static"),
426 				group.basics.identifier().label("head"),
427 				ZeroOrMore(group.basics.dotIdentifier().label("tail")),
428 				Optional(Sequence(
429 						Ch('.'), group.basics.optWS(),
430 						Ch('*'), group.basics.optWS())).label("dotStar"),
431 				Ch(';'), group.basics.optWS(),
432 				set(actions.createImportDeclaration(text("static"), value("head"), values("ZeroOrMore/tail"), text("dotStar"))));
433 	}
434 	
435 	public Rule compilationUnitEoi() {
436 		return Sequence(compilationUnit(), Eoi());
437 	}
438 	
439 	public Rule compilationUnit() {
440 		return Sequence(
441 				group.basics.optWS(),
442 				Optional(packageDeclaration()).label("package"),
443 				ZeroOrMore(importDeclaration().label("import")).label("imports"),
444 				ZeroOrMore(anyTypeDeclaration().label("type")).label("types"),
445 				set(actions.createCompilationUnit(value("package"), values("imports/import"), values("types/type"))));
446 	}
