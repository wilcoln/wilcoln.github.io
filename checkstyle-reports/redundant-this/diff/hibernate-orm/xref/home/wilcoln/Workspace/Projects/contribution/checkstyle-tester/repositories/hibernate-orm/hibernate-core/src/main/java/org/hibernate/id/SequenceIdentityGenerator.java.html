24  package org.hibernate.id;
25  
26  import java.io.Serializable;
27  import java.sql.PreparedStatement;
28  import java.sql.SQLException;
29  import java.util.Properties;
30  
31  import org.jboss.logging.Logger;
32  
33  import org.hibernate.HibernateException;
34  import org.hibernate.MappingException;
35  import org.hibernate.dialect.Dialect;
36  import org.hibernate.engine.spi.SessionImplementor;
37  import org.hibernate.id.insert.AbstractReturningDelegate;
38  import org.hibernate.id.insert.IdentifierGeneratingInsert;
39  import org.hibernate.id.insert.InsertGeneratedIdentifierDelegate;
40  import org.hibernate.internal.CoreMessageLogger;
41  import org.hibernate.sql.Insert;
42  import org.hibernate.type.Type;
43  
44  /**
45   * A generator which combines sequence generation with immediate retrieval
46   * through JDBC3 {@link java.sql.Connection#prepareStatement(String, String[]) getGeneratedKeys}.
47   * In this respect it works much like ANSI-SQL IDENTITY generation.
48   * <p/>
49   * This generator only known to work with newer Oracle drivers compiled for
50   * JDK 1.4 (JDBC3).
51   * <p/>
52   * Note: Due to a bug in Oracle drivers, sql comments on these insert statements
53   * are completely disabled.
54   *
55   * @author Steve Ebersole
56   */
57  public class SequenceIdentityGenerator
58  		extends SequenceGenerator
59  		implements PostInsertIdentifierGenerator {
60  
61      private static final CoreMessageLogger LOG = Logger.getMessageLogger(
62  			CoreMessageLogger.class,
63  			SequenceIdentityGenerator.class.getName()
64  	);
65  
66  	@Override
67      public Serializable generate(SessionImplementor s, Object obj) {
68  		return IdentifierGeneratorHelper.POST_INSERT_INDICATOR;
69  	}
70  
71  	public InsertGeneratedIdentifierDelegate getInsertGeneratedIdentifierDelegate(
72  			PostInsertIdentityPersister persister,
73  	        Dialect dialect,
74  	        boolean isGetGeneratedKeysEnabled) throws HibernateException {
75  		return new Delegate( persister, dialect, getSequenceName() );
76  	}
77  
78  	@Override
79      public void configure(Type type, Properties params, Dialect dialect) throws MappingException {
80  		super.configure( type, params, dialect );
81  	}
82  
83  	public static class Delegate extends AbstractReturningDelegate {
84  		private final Dialect dialect;
85  		private final String sequenceNextValFragment;
86  		private final String[] keyColumns;
87  
88  		public Delegate(PostInsertIdentityPersister persister, Dialect dialect, String sequenceName) {
89  			super( persister );
90  			this.dialect = dialect;
91  			this.sequenceNextValFragment = dialect.getSelectSequenceNextValString( sequenceName );
92  			this.keyColumns = getPersister().getRootTableKeyColumnNames();
93  			if ( keyColumns.length > 1 ) {
94  				throw new HibernateException( "sequence-identity generator cannot be used with with multi-column keys" );
95  			}
96  		}
97  
98  		public IdentifierGeneratingInsert prepareIdentifierGeneratingInsert() {
99  			NoCommentsInsert insert = new NoCommentsInsert( dialect );
100 			insert.addColumn( getPersister().getRootTableKeyColumnNames()[0], sequenceNextValFragment );
101 			return insert;
102 		}
103 
104 		@Override
105         protected PreparedStatement prepare(String insertSQL, SessionImplementor session) throws SQLException {
106 			return session.getTransactionCoordinator().getJdbcCoordinator().getStatementPreparer().prepareStatement( insertSQL, keyColumns );
107 		}
108 
109 		@Override
110 		protected Serializable executeAndExtract(PreparedStatement insert, SessionImplementor session) throws SQLException {
111 						session.getTransactionCoordinator().getJdbcCoordinator().getResultSetReturn().executeUpdate( insert );
112 			return IdentifierGeneratorHelper.getGeneratedIdentity(
113 					insert.getGeneratedKeys(),
114 					getPersister().getRootTableKeyColumnNames()[0],
115 					getPersister().getIdentifierType()
116 			);
117 		}
118 	}
119 
120 	public static class NoCommentsInsert extends IdentifierGeneratingInsert {
121 		public NoCommentsInsert(Dialect dialect) {
122 			super( dialect );
123 		}
124 
125 		@Override
126         public Insert setComment(String comment) {
127 			// don't allow comments on these insert statements as comments totally
128 			// blow up the Oracle getGeneratedKeys "support" :(
129 			LOG.disallowingInsertStatementComment();
130 			return this;
131 		}
132 	}
133 }
