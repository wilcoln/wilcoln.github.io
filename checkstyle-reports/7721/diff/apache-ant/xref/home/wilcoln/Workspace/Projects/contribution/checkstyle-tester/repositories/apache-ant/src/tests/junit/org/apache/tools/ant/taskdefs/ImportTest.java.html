135     @Test
136     public void testSymlinkedImports() throws Exception {
137         String ln = "/usr/bin/ln";
138         if (!new File(ln).exists()) {
139             ln = "/bin/ln";
140         }
141         Assume.assumeTrue("Current system does not support Symlinks", new File(ln).exists());
142         String symlink = "src/etc/testcases/taskdefs/import/symlinks/d3b";
143         File symlinkFile = new File(System.getProperty("root"), symlink);
144         if (Runtime.getRuntime().exec(new String[] {ln, "-s", "d3a", symlinkFile.getAbsolutePath()}).waitFor() != 0) {
145             throw new IOException("'" + ln + " -s d3a " + symlink + "' failed");
146         }
147         try {
148             buildRule.configureProject(
149                 "src/etc/testcases/taskdefs/import/symlinks/d1/p1.xml");
150             assertEquals(
151                 buildRule.getProject().getProperty("ant.file.p2"),
152                 new File(System.getProperty("root"), "src/etc/testcases/taskdefs/import/symlinks/d2/p2.xml")
153                 .getAbsolutePath());
154             assertEquals(
155                 buildRule.getProject().getProperty("ant.file.p3"),
156                 new File(System.getProperty("root"), "src/etc/testcases/taskdefs/import/symlinks/d3b/p3.xml")
157                 .getAbsolutePath());
158         } finally {
159             symlinkFile.delete();
160         }
161     }
