125 	public void testAllMethodsSameSignatureFunction() {
126 		Factory factory = ModelUtils.build(new File("./src/test/java/spoon/test/refactoring/parameter/testclasses"));
127 		
128 		//each executable in test classes is marked with a annotation TestHierarchy,
129 		//which defines the name of the hierarchy where this executable belongs to. 
130 
131 		//collect all executables which are marked that they belong to hierarchy A_method1
132 		List<CtExecutable<?>> executablesOfHierarchyA = getExecutablesOfHierarchy(factory, "A_method1");
133 		//check executables of this hierarchy
134 		checkMethodHierarchies(executablesOfHierarchyA);
135 
136 		//collect all executables which are marked that they belong to hierarchy R_method1
137 		List<CtExecutable<?>> executablesOfHierarchyR = getExecutablesOfHierarchy(factory, "R_method1");
138 		//check executables of this hierarchy
139 		checkMethodHierarchies(executablesOfHierarchyR);
140 		
141 		//contract: CtConstructor has no other same signature
142 		CtConstructor<?> constructorTypeA = factory.Class().get(TypeA.class).getConstructors().iterator().next();
143 		CtExecutable<?> exec = constructorTypeA.map(new AllMethodsSameSignatureFunction()).first();
144 		assertNull("Unexpected executable found by Constructor of TypeA "+exec, exec);
145 		CtConstructor<?> constructorTypeB = factory.Class().get(TypeB.class).getConstructors().iterator().next();
146 		exec = constructorTypeA.map(new AllMethodsSameSignatureFunction()).first();
147 		assertNull("Unexpected executable found by Constructor of TypeA "+exec, exec);
148 		//contract: constructor is returned if includingSelf == true
149 		assertSame(constructorTypeA, constructorTypeA.map(new AllMethodsSameSignatureFunction().includingSelf(true)).first());
150 	}

152 	private void checkMethodHierarchies(List<CtExecutable<?>> expectedExecutables) {
153 		//contract: check that found methods does not depend on the starting point. 
154 		//The same set of executables has to be found if we start on any of them
155 		int countOfTestedLambdas = 0;
156 		int countOfTestedMethods = 0;
157 		for (CtExecutable<?> ctExecutable : expectedExecutables) {
158 			if (ctExecutable instanceof CtLambda) {
159 				countOfTestedLambdas++;
160 			} else {
161 				assertTrue(ctExecutable instanceof CtMethod);
162 				countOfTestedMethods++;
163 			}
164 			//start checking of method hierarchy from each expected executable. It must always return same results
165 			checkMethodHierarchy(expectedExecutables, ctExecutable);
166 		}
167 		assertTrue(countOfTestedLambdas>0);
168 		assertTrue(countOfTestedMethods>0);
169 	}

171 	private void checkMethodHierarchy(List<CtExecutable<?>> expectedExecutables, CtExecutable startExecutable) {
172 		//contract: check that by default it does not includes self
173 		//contract: check that by default it returns lambdas
174 		{
175 			final List<CtExecutable<?>> executables = startExecutable.map(new AllMethodsSameSignatureFunction()).list();
176 			assertFalse("Unexpected start executable "+startExecutable, containsSame(executables, startExecutable));
177 			//check that some method was found
178 			assertFalse(executables.isEmpty());
179 			//check that expected methods were found and remove them 
180 			expectedExecutables.forEach(m->{
181 				boolean found = removeSame(executables, m);
182 				if(startExecutable==m) {
183 					//it is start method. It should not be there
184 					assertFalse("The signature "+getQSignature(m)+" was returned too", found);
185 				} else {
186 					assertTrue("The signature "+getQSignature(m)+" not found", found);
187 				}
188 			});
189 			//check that there is no unexpected executable
190 			assertTrue("Unexpected executables: "+executables, executables.isEmpty());
191 		}
192 		
193 		//contract: check that includingSelf(true) returns startMethod too
194 		//contract: check that by default it still returns lambdas
195 		{
196 			final List<CtExecutable<?>> executables = startExecutable.map(new AllMethodsSameSignatureFunction().includingSelf(true)).list();
197 			assertTrue("Missing start executable "+startExecutable, containsSame(executables, startExecutable));
198 			//check that some method was found
199 			assertFalse(executables.isEmpty());
200 			//check that expected methods were found and remove them 
201 			expectedExecutables.forEach(m->{
202 				assertTrue("The signature "+getQSignature(m)+" not found", removeSame(executables, m));
203 			});
204 			//check that there is no unexpected executable
205 			assertTrue("Unexpected executables: "+executables, executables.isEmpty());
206 		}
207 		
208 		//contract: check that includingLambdas(false) returns no lambda expressions
209 		{
210 			final List<CtExecutable<?>> executables = startExecutable.map(new AllMethodsSameSignatureFunction().includingSelf(true).includingLambdas(false)).list();
211 			if (startExecutable instanceof CtLambda) {
212 				//lambda must not be returned even if it is first 
213 				assertFalse("Unexpected start executable "+startExecutable, containsSame(executables, startExecutable));
214 			} else {
215 				assertTrue("Missing start executable "+startExecutable, containsSame(executables, startExecutable));
216 			}
217 			
218 			//check that some method was found
219 			assertFalse(executables.isEmpty());
220 			//check that expected methods were found and remove them 
221 			expectedExecutables.forEach(m->{
222 				if(m instanceof CtLambda) {
223 					//the lambdas are not expected. Do not ask for them
224 					return;
225 				}
226 				assertTrue("The signature "+getQSignature(m)+" not found", removeSame(executables, m));
227 			});
228 			//check that there is no unexpected executable or lambda
229 			assertTrue("Unexepcted executables "+executables, executables.isEmpty());
230 		}
231 		//contract: check early termination
232 		//contract: check that first returned element is the startExecutable itself if includingSelf == true
233 		CtExecutable<?> exec = startExecutable.map(new AllMethodsSameSignatureFunction().includingSelf(true)).first();
234 		assertSame(startExecutable, exec);
235 		//contract: check that first returned element is not the startExecutable itself if includingSelf == false, but some other executable from the expected
236 		exec = startExecutable.map(new AllMethodsSameSignatureFunction().includingSelf(false)).first();
237 		assertNotSame(startExecutable, exec);
238 		assertTrue(containsSame(expectedExecutables, exec));
239 	}

241 	private String getQSignature(CtExecutable e) {
242 		if (e instanceof CtMethod<?>) {
243 			CtMethod<?> m = (CtMethod<?>) e;
244 			return m.getDeclaringType().getQualifiedName()+"#"+m.getSignature();
245 		}
246 		return e.getShortRepresentation();
247 	}

249 	private List<CtExecutable<?>> getExecutablesOfHierarchy(Factory factory, String hierarchyName) {
250 		return factory.getModel().filterChildren(new TypeFilter(CtExecutable.class)).select((CtExecutable<?> exec)->{
251 			//detect if found executable belongs to hierarchy 'hierarchyName'
252 			CtElement ele = exec;
253 			if (exec instanceof CtLambda) {
254 				//lambda is marked by annotation on the first statement of the lambda body.
255 				List<CtStatement> stats = exec.getBody().getStatements();
256 				if(!stats.isEmpty()) {
257 					ele = stats.get(0);
258 				}
259 			}
260 			TestHierarchy th = ele.getAnnotation(TestHierarchy.class);
261 			if (th!=null) {
262 				return Arrays.asList(th.value()).contains(hierarchyName);
263 			}
264 			return false;
265 		}).list();
266 	}

269 	public void testExecutableReferenceFilter() {
270 		Factory factory = ModelUtils.build(new File("./src/test/java/spoon/test/refactoring/parameter/testclasses"));
271 		
272 		List<CtExecutable<?>> executables = factory.getModel().filterChildren((CtExecutable<?> e)->true).list();
273 		int nrExecRefsTotal = 0;
274 		//contract check that ExecutableReferenceFilter found CtExecutableReferences of each executable individually 
275 		for (CtExecutable<?> executable : executables) {
276 			nrExecRefsTotal += checkExecutableReferenceFilter(factory, Collections.singletonList(executable));
277 		}
278 		//contract check that ExecutableReferenceFilter found CtExecutableReferences of all executables together 
279 		int nrExecRefsTotal2 = checkExecutableReferenceFilter(factory, executables);
280 		
281 		assertSame(nrExecRefsTotal, nrExecRefsTotal2);
282 
283 		//contract check that it found lambdas too
284 		CtLambda lambda = factory.getModel().filterChildren((CtLambda<?> e)->true).first();
285 		assertNotNull(lambda);
286 		//this test case is quite wild, because there is normally lambda reference in spoon model. So make one lambda reference here:
287 		CtExecutableReference<?> lambdaRef = lambda.getReference();
288 		List<CtExecutableReference<?>> refs = lambdaRef.filterChildren(null).select(new ExecutableReferenceFilter(lambda)).list();
289 		assertEquals(1, refs.size());
290 		assertSame(lambdaRef, refs.get(0));
291 	}

293 	private int checkExecutableReferenceFilter(Factory factory, List<CtExecutable<?>> executables) {
294 		assertFalse(executables.isEmpty());
295 		ExecutableReferenceFilter execRefFilter = new ExecutableReferenceFilter();
296 		executables.forEach((CtExecutable<?> e)->execRefFilter.addExecutable(e));
297 		final List<CtExecutableReference<?>> refs = new ArrayList<>(factory.getModel().filterChildren(execRefFilter).list());
298 		int nrExecRefs = refs.size();
299 		//use different (slower, but straight forward) algorithm to search for all executable references to check if ExecutableReferenceFilter returns correct results
300 		factory.getModel().filterChildren((CtExecutableReference er)->{
301 			return containsSame(executables, er.getDeclaration());
302 		}).forEach((CtExecutableReference er)->{
303 			//check that each expected reference was found by ExecutableReferenceFilter and remove it from that list
304 			assertTrue("Executable reference: "+er+" not found.", refs.remove(er));
305 		});
306 		//check that no other reference was found by ExecutableReferenceFilter
307 		assertSame(0, refs.size());
308 		return nrExecRefs;
309 	}

311 	private boolean containsSame(Collection list, Object item) {
312 		for (Object object : list) {
313 			if(object==item) {
314 				return true;
315 			}
316 		}
317 		return false;
318 	}
319 	private boolean removeSame(Collection list, Object item) {
320 		for (Iterator iter = list.iterator(); iter.hasNext();) {
321 			Object object = iter.next();
322 			if(object==item) {
323 				iter.remove();
324 				return true;
325 			}
326 		}
327 		return false;
328 	}

331 	public void testCtParameterRemoveRefactoring() throws FileNotFoundException {
332 		String testPackagePath = "spoon/test/refactoring/parameter/testclasses";
333 		final Launcher launcher = new Launcher();
334 		launcher.getEnvironment().setNoClasspath(true);
335 		SpoonModelBuilder comp = launcher.createCompiler();
336 		comp.addInputSource(SpoonResourceHelper.createResource(new File("./src/test/java/"+testPackagePath)));
337 		comp.build();
338 		Factory factory = comp.getFactory();
339 		
340 		CtType<?> typeA = factory.Class().get(TypeA.class);
341 		
342 		CtMethod<?> methodTypeA_method1 = typeA.getMethodsByName("method1").get(0);
343 		CtParameterRemoveRefactoring refactor = new CtParameterRemoveRefactoring();
344 		refactor.setTarget(methodTypeA_method1.getParameters().get(0));
345 		//check that expected methods are targets of refactoring
346 		List<CtExecutable<?>> execs = refactor.getTargetExecutables();
347 		execs.forEach(exec->{
348 			//check that each to be modified method has one parameter
349 			assertEquals(1, exec.getParameters().size());
350 		});
351 		refactor.refactor();
352 		execs.forEach(exec->{
353 			//check that each to be modified method has no parameter after refactoring
354 			assertEquals(0, exec.getParameters().size());
355 		});
356 		launcher.setSourceOutputDirectory(new File("./target/spooned/"));
357 		launcher.getModelBuilder().generateProcessedSourceFiles(OutputType.CLASSES);
358 		ModelUtils.canBeBuilt("./target/spooned/"+testPackagePath, 8);
359 	}
