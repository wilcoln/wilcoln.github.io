37  /** This class gathers the informations present in the Orbital Parameter Message (OPM), and contains
38   * methods to generate {@link CartesianOrbit}, {@link KeplerianOrbit} or {@link SpacecraftState}.
39   * @author sports
40   * @since 6.1
41   */
42  public class OPMFile extends OGMFile {
43  
44      /** Meta-data. */
45      private final ODMMetaData metaData;
46  
47      /** Position vector (m). */
48      private Vector3D position;
49  
50      /** Velocity vector (m/s. */
51      private Vector3D velocity;
52  
53      /** Maneuvers. */
54      private List<Maneuver> maneuvers;
55  
56      /** Create a new OPM file object. */
57      OPMFile() {
58          metaData  = new ODMMetaData(this);
59          maneuvers = new ArrayList<Maneuver>();
60      };
61  
62      /** Get the meta data.
63       * @return meta data
64       */
65      @Override
66      public ODMMetaData getMetaData() {
67          return metaData;
68      }
69  
70      /** Get position vector.
71       * @return the position vector
72       */
73      public Vector3D getPosition() {
74          return position;
75      }
76  
77      /** Set position vector.
78       * @param position the position vector to be set
79       */
80      void setPosition(final Vector3D position) {
81          this.position = position;
82      }
83  
84      /** Get velocity vector.
85       * @return the velocity vector
86       */
87      public Vector3D getVelocity() {
88          return velocity;
89      }
90  
91      /** Set velocity vector.
92       * @param velocity the velocity vector to be set
93       */
94      void setVelocity(final Vector3D velocity) {
95          this.velocity = velocity;
96      }
97  
98      /** Get the number of maneuvers present in the OPM.
99       * @return the number of maneuvers
100      */
101     public int getNbManeuvers() {
102         return maneuvers.size();
103     }
104 
105     /** Get a list of all maneuvers.
106      * @return unmodifiable list of all maneuvers.
107      */
108     public List<Maneuver> getManeuvers() {
109         return Collections.unmodifiableList(maneuvers);
110     }
111 
112     /** Get a maneuver.
113      * @param index maneuver index, counting from 0
114      * @return maneuver
115      */
116     public Maneuver getManeuver(final int index) {
117         return maneuvers.get(index);
118     }
119 
120     /** Add a maneuver.
121      * @param maneuver maneuver to be set
122      */
123     void addManeuver(final Maneuver maneuver) {
124         maneuvers.add(maneuver);
125     }
126 
127     /** Get boolean testing whether the OPM contains at least one maneuver.
128      * @return true if OPM contains at least one maneuver
129      *         false otherwise */
130     public boolean getHasManeuver() {
131         return !maneuvers.isEmpty();
132     }
133 
134     /** Get the comment for meta-data.
135      * @return comment for meta-data
136      */
137     public List<String> getMetaDataComment() {
138         return metaData.getComment();
139     }
140 
141     /** {@inheritDoc} */
142     @Override
143     public List<SatelliteTimeCoordinate> getSatelliteCoordinates(final String satId) {
144         if (getMetaData().getObjectID().equals(satId)) {
145             return Arrays.asList(new SatelliteTimeCoordinate(getEpoch(), getPVCoordinates()));
146         } else {
147             return Collections.emptyList();
148         }
149     }
150 
151     /** Get the {@link SatelliteTimeCoordinate} of the OPM.
152      * @return the {@link SatelliteTimeCoordinate}
153      */
154     public SatelliteTimeCoordinate getSatelliteCoordinatesOPM() {
155         return new SatelliteTimeCoordinate(getEpoch(), getPVCoordinates());
156     }
157 
158     /** Get the position/velocity coordinates contained in the OPM.
159      * @return the position/velocity coordinates contained in the OPM
160      */
161     public PVCoordinates getPVCoordinates() {
162         return new PVCoordinates(getPosition(), getVelocity());
163     }
164 
165     /** {@inheritDoc} */
166     @Override
167     public String getCoordinateSystem() {
168         return metaData.getFrame().toString();
169     }
170 
171     /** {@inheritDoc} */
172     @Override
173     public OrbitFile.TimeSystem getTimeSystem() {
174         return metaData.getTimeSystem();
175     }
176 
177     /**
178      * Generate a {@link CartesianOrbit} from the OPM state vector data. If the reference frame is not
179      * pseudo-inertial, an exception is raised.
180      * @return the {@link CartesianOrbit} generated from the OPM information
181      * @exception OrekitException if the reference frame is not pseudo-inertial or if the central body
182      * gravitational coefficient cannot be retrieved from the OPM
183      */
184     public CartesianOrbit generateCartesianOrbit()
185         throws OrekitException {
186         setMuUsed();
187         return new CartesianOrbit(getPVCoordinates(), metaData.getFrame(), getEpoch(), getMuUsed());
188     }
189 
190     /** Generate a {@link KeplerianOrbit} from the OPM keplerian elements if hasKeplerianElements is true,
191      * or from the state vector data otherwise.
192      * If the reference frame is not pseudo-inertial, an exception is raised.
193      * @return the {@link KeplerianOrbit} generated from the OPM information
194      * @exception OrekitException if the reference frame is not pseudo-inertial or if the central body
195      * gravitational coefficient cannot be retrieved from the OPM
196      */
197     public KeplerianOrbit generateKeplerianOrbit() throws OrekitException {
198         setMuUsed();
199         if (hasKeplerianElements()) {
200             return new KeplerianOrbit(getA(), getE(), getI(), getPa(), getRaan(), getAnomaly(),
201                                       getAnomalyType(), metaData.getFrame(), getEpoch(), getMuUsed());
202         } else {
203             return new KeplerianOrbit(getPVCoordinates(), metaData.getFrame(), getEpoch(), getMuUsed());
204         }
205     }
206 
207     /** Generate spacecraft state from the {@link CartesianOrbit} generated by generateCartesianOrbit.
208      *  Raises an exception if OPM doesn't contain spacecraft mass information.
209      * @return the spacecraft state of the OPM
210      * @exception OrekitException if there is no spacecraft mass associated with the OPM
211      */
212     public SpacecraftState generateSpacecraftState()
213         throws OrekitException {
214         return new SpacecraftState(generateCartesianOrbit(), getMass());
215     }

217     /** Maneuver in an OPM file.
218      */
219     public static class Maneuver {
220 
221         /** Epoch ignition. */
222         private AbsoluteDate epochIgnition;
223 
224         /** Coordinate system for velocity increment vector, for Local Orbital Frames. */
225         private LOFType refLofType;
226 
227         /** Coordinate system for velocity increment vector, for absolute frames. */
228         private Frame refFrame;
229 
230         /** Duration (value is 0 for impulsive maneuver). */
231         private double duration;
232 
233         /** Mass change during maneuver (value is < 0). */
234         private double deltaMass;
235 
236         /** Velocity increment. */
237         private Vector3D dV;
238 
239         /** Maneuvers data comment, each string in the list corresponds to one line of comment. */
240         private List<String> comment;
241 
242         /** Simple constructor.
243          */
244         public Maneuver() {
245             this.dV      = Vector3D.ZERO;
246             this.comment = Collections.emptyList();
247         }
248 
249         /** Get epoch ignition.
250          * @return epoch ignition
251          */
252         public AbsoluteDate getEpochIgnition() {
253             return epochIgnition;
254         }
255 
256         /** Set epoch ignition.
257          * @param epochIgnition epoch ignition
258          */
259         void setEpochIgnition(final AbsoluteDate epochIgnition) {
260             this.epochIgnition = epochIgnition;
261         }
262 
263         /** Get coordinate system for velocity increment vector, for Local Orbital Frames.
264          * @return coordinate system for velocity increment vector, for Local Orbital Frames
265          */
266         public LOFType getRefLofType() {
267             return refLofType;
268         }
269 
270         /** Set coordinate system for velocity increment vector, for Local Orbital Frames.
271          * @param refLofType coordinate system for velocity increment vector, for Local Orbital Frames
272          */
273         public void setRefLofType(final LOFType refLofType) {
274             this.refLofType = refLofType;
275             this.refFrame   = null;
276         }
277 
278         /** Get Coordinate system for velocity increment vector, for absolute frames.
279          * @return coordinate system for velocity increment vector, for absolute frames
280          */
281         public Frame getRefFrame() {
282             return refFrame;
283         }
284 
285         /** Set Coordinate system for velocity increment vector, for absolute frames.
286          * @param refFrame coordinate system for velocity increment vector, for absolute frames
287          */
288         public void setRefFrame(final Frame refFrame) {
289             this.refLofType = null;
290             this.refFrame   = refFrame;
291         }
292 
293         /** Get duration (value is 0 for impulsive maneuver).
294          * @return duration (value is 0 for impulsive maneuver)
295          */
296         public double getDuration() {
297             return duration;
298         }
299 
300         /** Set duration (value is 0 for impulsive maneuver).
301          * @param duration duration (value is 0 for impulsive maneuver)
302          */
303         public void setDuration(final double duration) {
304             this.duration = duration;
305         }
306 
307         /** Get mass change during maneuver (value is &lt; 0).
308          * @return mass change during maneuver (value is &lt; 0)
309          */
310         public double getDeltaMass() {
311             return deltaMass;
312         }
313 
314         /** Set mass change during maneuver (value is &lt; 0).
315          * @param deltaMass mass change during maneuver (value is &lt; 0)
316          */
317         public void setDeltaMass(final double deltaMass) {
318             this.deltaMass = deltaMass;
319         }
320 
321         /** Get velocity increment.
322          * @return velocity increment
323          */
324         public Vector3D getDV() {
325             return dV;
326         }
327 
328         /** Set velocity increment.
329          * @param dV velocity increment
330          */
331         public void setdV(final Vector3D dV) {
332             this.dV = dV;
333         }
334 
335         /** Get the maneuvers data comment, each string in the list corresponds to one line of comment.
336          * @return maneuvers data comment, each string in the list corresponds to one line of comment
337          */
338         public List<String> getComment() {
339             return Collections.unmodifiableList(comment);
340         }
341 
342         /** Set the maneuvers data comment, each string in the list corresponds to one line of comment.
343          * @param comment maneuvers data comment, each string in the list corresponds to one line of comment
344          */
345         public void setComment(final List<String> comment) {
346             this.comment = new ArrayList<String>(comment);
347         }
348 
349     }
