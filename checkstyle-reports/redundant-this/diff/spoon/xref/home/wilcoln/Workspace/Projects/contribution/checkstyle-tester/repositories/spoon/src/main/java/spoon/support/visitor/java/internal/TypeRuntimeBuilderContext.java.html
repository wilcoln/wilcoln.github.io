6   package spoon.support.visitor.java.internal;
7   
8   import spoon.reflect.declaration.CtAnnotation;
9   import spoon.reflect.declaration.CtField;
10  import spoon.reflect.declaration.CtMethod;
11  import spoon.reflect.declaration.CtPackage;
12  import spoon.reflect.declaration.CtType;
13  import spoon.reflect.declaration.CtTypeParameter;
14  import spoon.reflect.path.CtRole;
15  import spoon.reflect.reference.CtTypeReference;
16  
17  import java.lang.annotation.Annotation;
18  import java.lang.reflect.GenericDeclaration;
19  import java.lang.reflect.Type;
20  import java.util.HashMap;
21  import java.util.Map;
22  
23  public class TypeRuntimeBuilderContext extends AbstractRuntimeBuilderContext {
24  	protected CtType type;
25  	protected Type rtType;
26  	private Map<String, CtTypeParameter> mapTypeParameters;
27  
28  	public TypeRuntimeBuilderContext(Type rtType, CtType type) {
29  		super(type);
30  		this.type = type;
31  		this.rtType = rtType;
32  		this.mapTypeParameters = new HashMap<>();
33  	}
34  
35  	@Override
36  	public void addPackage(CtPackage ctPackage) {
37  		ctPackage.addType(type);
38  	}
39  
40  	@Override
41  	public void addType(CtType<?> aType) {
42  		type.addNestedType(aType);
43  	}
44  
45  	@Override
46  	public void addAnnotation(CtAnnotation<Annotation> ctAnnotation) {
47  		type.addAnnotation(ctAnnotation);
48  	}
49  
50  	@Override
51  	public void addMethod(CtMethod<?> ctMethod) {
52  		type.addMethod(ctMethod);
53  	}
54  
55  	@Override
56  	public void addField(CtField<?> ctField) {
57  		type.addField(ctField);
58  	}
59  
60  	@Override
61  	public void addFormalType(CtTypeParameter parameterRef) {
62  		this.type.addFormalCtTypeParameter(parameterRef);
63  		this.mapTypeParameters.put(parameterRef.getSimpleName(), parameterRef);
64  	}
65  
66  	@Override
67  	public void addTypeReference(CtRole role, CtTypeReference<?> typeReference) {
68  		switch (role) {
69  			case INTERFACE:
70  				type.addSuperInterface(typeReference);
71  				return;
72  			case SUPER_TYPE:
73  				if (type instanceof CtTypeParameter) {
74  					((CtTypeParameter) this.type).setSuperclass(typeReference);
75  					return;
76  				}
77  		}
78  		throw new UnsupportedOperationException();
79  	}
80  
81  	@Override
82  	public CtTypeParameter getTypeParameter(GenericDeclaration genericDeclaration, String string) {
83  		return rtType == genericDeclaration ? this.mapTypeParameters.get(string) : null;
84  	}
85  }
