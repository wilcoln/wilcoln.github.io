5   package net.sourceforge.pmd.lang.java.ast;
6   
7   import net.sourceforge.pmd.annotation.InternalApi;

9   /**
10   * Represents an import declaration in a Java file.
11   *
12   * <pre>
13   *
14   * ImportDeclaration ::= "import" "static"? {@linkplain ASTName Name} ( "." "*" )? ";"
15   *
16   * </pre>
17   *
18   * @see <a href="https://docs.oracle.com/javase/specs/jls/se9/html/jls-7.html#jls-7.5">JLS 7.5</a>
19   */
20  // TODO should this really be a type node?
21  // E.g. for on-demand imports, what's the type of this node? There's no type name, just a package name
22  // for on-demand static imports?
23  // for static imports of a field? the type of the field or the type of the enclosing type?
24  // for static imports of a method?
25  // I don't think we can work out a spec without surprising corner cases, and #1207 will abstract
26  // things away anyway, so I think we should make it a regular node
27  public class ASTImportDeclaration extends AbstractJavaTypeNode {
28  
29      private boolean isImportOnDemand;
30      private boolean isStatic;
31      private Package pkg;
32  
33      @InternalApi
34      @Deprecated
35      public ASTImportDeclaration(int id) {
36          super(id);
37      }
38  
39      @InternalApi
40      @Deprecated
41      public ASTImportDeclaration(JavaParser p, int id) {
42          super(p, id);
43      }
44  
45  
46      /**
47       * @deprecated Will be made private with 7.0.0
48       */
49      @InternalApi
50      @Deprecated
51      public void setImportOnDemand() {
52          isImportOnDemand = true;
53      }
54  
55  
56      // @formatter:off
57      /**
58       * Returns true if this is an import-on-demand declaration,
59       * aka "wildcard import".
60       *
61       * <ul>
62       *     <li>If this is a static import, then the imported names are those
63       *     of the accessible static members of the named type;
64       *     <li>Otherwise, the imported names are the names of the accessible types
65       *     of the named type or named package.
66       * </ul>
67       */
68      // @formatter:on
69      public boolean isImportOnDemand() {
70          return isImportOnDemand;
71      }
72  
73  
74      /**
75       * @deprecated Will be made private with 7.0.0
76       */
77      @InternalApi
78      @Deprecated
79      public void setStatic() {
80          isStatic = true;
81      }
82  
83  
84      /**
85       * Returns true if this is a static import. If this import is not on-demand,
86       * {@link #getImportedSimpleName()} returns the name of the imported member.
87       */
88      public boolean isStatic() {
89          return isStatic;
90      }
91  
92      /**
93       * @deprecated this will be removed with PMD 7.0.0
94       */
95      @Deprecated
96      public ASTName getImportedNameNode() {
97          return (ASTName) getChild(0);
98      }
99  
100 
101     /**
102      * Returns the full name of the import. For on-demand imports, this is the name without
103      * the final dot and asterisk.
104      */
105     public String getImportedName() {
106         return getChild(0).getImage();
107     }
108 
109 
110     /**
111      * Returns the simple name of the type or method imported by this declaration.
112      * For on-demand imports, returns {@code null}.
113      */
114     public String getImportedSimpleName() {
115         if (isImportOnDemand) {
116             return null;
117         }
118 
119         String importName = getImportedName();
120         return importName.substring(importName.lastIndexOf('.') + 1);
121     }
122 
123 
124     /**
125      * Returns the "package" prefix of the imported name. For type imports, including on-demand
126      * imports, this is really the package name of the imported type(s). For static imports,
127      * this is actually the qualified name of the enclosing type, including the type name.
128      */
129     public String getPackageName() {
130         String importName = getImportedName();
131         if (isImportOnDemand) {
132             return importName;
133         }
134         if (importName.indexOf('.') == -1) {
135             return "";
136         }
137         int lastDot = importName.lastIndexOf('.');
138         return importName.substring(0, lastDot);
139     }
140 
141     @Override
142     public Object jjtAccept(JavaParserVisitor visitor, Object data) {
143         return visitor.visit(this, data);
144     }
145 
146     @InternalApi
147     @Deprecated
148     public void setPackage(Package packge) {
149         this.pkg = packge;
150     }
151 
152 
153     /**
154      * Returns the {@link Package} instance representing the package of the
155      * type or method imported by this declaration. This may be null if the
156      * auxclasspath is not correctly set, as this method depends on correct
157      * type resolution.
158      *
159      * @deprecated this will be removed with PMD 7.0.0
160      */
161     @Deprecated
162     public Package getPackage() {
163         return this.pkg;
164     }
165 }
