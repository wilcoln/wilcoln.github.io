15  /**
16   * Enumerates the properties that can be built from the XML. Defining a property in
17   * the XML requires the {@code type} attribute, which acts as a mnemonic for the type
18   * of the property that should be built. The mapping between the values of this attribute
19   * and the concrete property that is built is encoded in the constants of this enum.
20   *
21   * <h1>Properties API changes</h1>This class' API is mainly provided to build GUIs for XPath rules
22   * like the rule designer, so that they have info about the available properties from XML. As such,
23   * the number of clients are probably low. Nevertheless, a bunch of members have been deprecated to
24   * warn about probable upcoming API changes with 7.0.0, but the amount of change may be greater.
25   * See {@link PropertyDescriptor} for more info about property framework changes with 7.0.0.
26   *
27   * @author Cl√©ment Fournier
28   * @see PropertyDescriptorExternalBuilder
29   * @since 6.0.0
30   */
31  public enum PropertyTypeId {
32      // These are exclusively used for XPath rules. It would make more sense to model the supported
33      // property types around XML Schema Datatypes (XSD) 1.0 or 1.1 instead of Java datatypes (save for
34      // e.g. the Class type), including the mnemonics (eg. xs:integer instead of Integer)
35  
36      BOOLEAN("Boolean", BooleanProperty.extractor(), ValueParserConstants.BOOLEAN_PARSER),
37      /** @deprecated see {@link BooleanMultiProperty} */
38      @Deprecated
39      BOOLEAN_LIST("List[Boolean]", BooleanMultiProperty.extractor(), ValueParserConstants.BOOLEAN_PARSER),
40  
41      STRING("String", StringProperty.extractor(), ValueParserConstants.STRING_PARSER),
42      STRING_LIST("List[String]", StringMultiProperty.extractor(), ValueParserConstants.STRING_PARSER),
43      CHARACTER("Character", CharacterProperty.extractor(), ValueParserConstants.CHARACTER_PARSER),
44      CHARACTER_LIST("List[Character]", CharacterMultiProperty.extractor(), ValueParserConstants.CHARACTER_PARSER),
45  
46      REGEX("Regex", RegexProperty.extractor(), ValueParserConstants.REGEX_PARSER),
47  
48      INTEGER("Integer", IntegerProperty.extractor(), ValueParserConstants.INTEGER_PARSER),
49      INTEGER_LIST("List[Integer]", IntegerMultiProperty.extractor(), ValueParserConstants.INTEGER_PARSER),
50      LONG("Long", LongProperty.extractor(), ValueParserConstants.LONG_PARSER),
51      LONG_LIST("List[Long]", LongMultiProperty.extractor(), ValueParserConstants.LONG_PARSER),
52      @Deprecated
53      FLOAT("Float", FloatProperty.extractor(), ValueParserConstants.FLOAT_PARSER),
54      @Deprecated
55      FLOAT_LIST("List[Float]", FloatMultiProperty.extractor(), ValueParserConstants.FLOAT_PARSER),
56      DOUBLE("Double", DoubleProperty.extractor(), ValueParserConstants.DOUBLE_PARSER),
57      DOUBLE_LIST("List[Double]", DoubleMultiProperty.extractor(), ValueParserConstants.DOUBLE_PARSER),
58      //    ENUM("Enum", EnumeratedProperty.FACTORY),                     // TODO:cf we need new syntax in the xml to support that
59      //    ENUM_LIST("List[Enum]", EnumeratedMultiProperty.FACTORY),
60      @Deprecated
61      CLASS("Class", TypeProperty.extractor(), ValueParserConstants.CLASS_PARSER),
62      @Deprecated
63      CLASS_LIST("List[Class]", TypeMultiProperty.extractor(), ValueParserConstants.CLASS_PARSER);
64  
65  
66      private static final Map<String, PropertyTypeId> CONSTANTS_BY_MNEMONIC;
67      private final String stringId;
68      private final PropertyDescriptorExternalBuilder<?> factory;
69      private final ValueParser<?> valueParser;
70  
71      static {
72          Map<String, PropertyTypeId> temp = new HashMap<>();
73          for (PropertyTypeId id : values()) {
74              temp.put(id.stringId, id);
75          }
76          CONSTANTS_BY_MNEMONIC = Collections.unmodifiableMap(temp);
77      }
78  
79  
80      PropertyTypeId(String id, PropertyDescriptorExternalBuilder<?> factory, ValueParser<?> valueParser) {
81          this.stringId = id;
82          this.factory = factory;
83          this.valueParser = valueParser;
84      }
85  
86  
87      /**
88       * Gets the value of the type attribute represented by this constant.
89       *
90       * @return The string id
91       */
92      public String getStringId() {
93          return stringId;
94      }
95  
96  
97      /**
98       * Gets the factory associated to the type id, that can build the
99       * property from strings extracted from the XML.
100      *
101      * @return The factory
102      */
103     @Deprecated
104     public PropertyDescriptorExternalBuilder<?> getFactory() {
105         return factory;
106     }
107 
108 
109     /**
110      * Returns true if the property corresponding to this factory is numeric,
111      * which means it can be safely cast to a {@link NumericPropertyDescriptor}.
112      *
113      * @return whether the property is numeric
114      */
115     @Deprecated
116     public boolean isPropertyNumeric() {
117         return factory instanceof PropertyDescriptorBuilderConversionWrapper.SingleValue.Numeric
118                || factory instanceof PropertyDescriptorBuilderConversionWrapper.MultiValue.Numeric;
119     }
120 
121 
122     /**
123      * Returns true if the property corresponding to this factory is packaged,
124      * which means it can be safely cast to a {@link PackagedPropertyDescriptor}.
125      *
126      * @return whether the property is packaged
127      */
128     @Deprecated
129     public boolean isPropertyPackaged() {
130         return factory instanceof PropertyDescriptorBuilderConversionWrapper.SingleValue.Packaged
131                || factory instanceof PropertyDescriptorBuilderConversionWrapper.MultiValue.Packaged;
132     }
133 
134 
135     /**
136      * Returns true if the property corresponding to this factory takes
137      * lists of values as its value.
138      *
139      * @return whether the property is multivalue
140      *
141      * @deprecated see {@link PropertyDescriptor#isMultiValue()}
142      */
143     @Deprecated
144     public boolean isPropertyMultivalue() {
145         return factory.isMultiValue();
146     }
147 
148 
149     /**
150      * Returns the value type of the property corresponding to this factory.
151      * This is the component type of the list if the property is multivalued.
152      *
153      * @return The value type of the property
154      *
155      * @deprecated see {@link PropertyDescriptor#type()}
156      */
157     @Deprecated
158     public Class<?> propertyValueType() {
159         return factory.valueType();
160     }
161 
162 
163     /**
164      * Gets the object used to parse the values of this property from a string.
165      * If the property is multivalued, the parser only parses individual components
166      * of the list. A list parser can be obtained with {@link ValueParserConstants#multi(ValueParser, char)}.
167      *
168      * @return The value parser
169      *
170      * @deprecated see {@link PropertyDescriptor#valueFrom(String)}
171      */
172     @Deprecated
173     public ValueParser<?> getValueParser() {
174         return valueParser;
175     }
176 
177 
178     /**
179      * Returns the full mappings from type ids to enum constants.
180      *
181      * @return The full mapping.
182      */
183     public static Map<String, PropertyTypeId> typeIdsToConstants() {
184         return CONSTANTS_BY_MNEMONIC;
185     }
186 
187 
188     /**
189      * Gets the factory for the descriptor identified by the string id.
190      *
191      * @param stringId The identifier of the type
192      *
193      * @return The factory used to build new instances of a descriptor
194      *
195      * @deprecated See {@link PropertyDescriptorExternalBuilder}
196      */
197     @Deprecated
198     public static PropertyDescriptorExternalBuilder<?> factoryFor(String stringId) {
199         PropertyTypeId cons = CONSTANTS_BY_MNEMONIC.get(stringId);
200         return cons == null ? null : cons.factory;
201     }
202 
203 
204     /**
205      * Gets the enum constant corresponding to the given mnemonic.
206      *
207      * @param stringId A mnemonic for the property type
208      *
209      * @return A PropertyTypeId
210      */
211     public static PropertyTypeId lookupMnemonic(String stringId) {
212         return CONSTANTS_BY_MNEMONIC.get(stringId);
213     }
214 
215 
216     /**
217      * Gets the string representation of this type, as it should be given
218      * when defining a descriptor in the xml.
219      *
220      * @param valueType  The type to look for
221      * @param multiValue Whether the descriptor is multivalued or not
222      *
223      * @return The string id
224      *
225      * @deprecated The signature will probably be altered in 7.0.0 but a similar functionality will be available
226      */
227     @Deprecated
228     public static String typeIdFor(Class<?> valueType, boolean multiValue) {
229         for (Map.Entry<String, PropertyTypeId> entry : CONSTANTS_BY_MNEMONIC.entrySet()) {
230             if (entry.getValue().propertyValueType() == valueType
231                 && entry.getValue().isPropertyMultivalue() == multiValue) {
232                 return entry.getKey();
233             }
234         }
235         return null;
236     }
