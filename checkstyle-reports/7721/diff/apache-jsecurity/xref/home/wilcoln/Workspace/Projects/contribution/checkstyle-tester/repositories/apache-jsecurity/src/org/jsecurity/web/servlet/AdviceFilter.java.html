19  package org.jsecurity.web.servlet;
20  
21  import org.apache.commons.logging.Log;
22  import org.apache.commons.logging.LogFactory;
23  
24  import javax.servlet.FilterChain;
25  import javax.servlet.ServletException;
26  import javax.servlet.ServletRequest;
27  import javax.servlet.ServletResponse;
28  import java.io.IOException;
29  
30  /**
31   * A Servlet Filter that enables AOP-style advice for a SerlvetRequest via
32   * {@link #preHandle(javax.servlet.ServletRequest, javax.servlet.ServletResponse) preHandle},
33   * {@link #postHandle(javax.servlet.ServletRequest, javax.servlet.ServletResponse) postHandle},
34   * and {@link #afterCompletion(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Exception) afterCompletion}
35   * hooks.
36   *
37   * @author Les Hazlewood
38   * @since 0.9
39   */
40  public abstract class AdviceFilter extends OncePerRequestFilter {
41  
42      /** The static logger available to this class only */
43      private static final Log log = LogFactory.getLog(AdviceFilter.class);
44  
45      /**
46       * Returns <code>true</code> if the filter chain should be allowed to continue, <code>false</code> otherwise.
47       * It is called before the chain is actually consulted/executed.
48       * <p/>
49       * The default implementation returns <code>true</code> always and exists as a template method for subclasses.
50       *
51       * @param request  the incoming ServletRequest
52       * @param response the outgoing ServletResponse
53       * @return <code>true</code> if the filter chain should be allowed to continue, <code>false</code> otherwise.
54       * @throws Exception if there is any error.
55       */
56      protected boolean preHandle(ServletRequest request, ServletResponse response) throws Exception {
57          return true;
58      }
59  
60      /**
61       * Allows 'post' advice logic to be called, but only if no exception occurs during filter chain execution.  That
62       * is, if {@link #executeChain executeChain} throws an exception, this method will never be called.  Be aware of
63       * this when implementing logic.  Most resource 'cleanup' behavior is often done in the
64       * {@link #afterCompletion(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Exception) afterCompletion(request,response,exception)}
65       * implementation, which is guaranteed to be called for every request, even when the chain processing throws
66       * an Exception.
67       * <p/>
68       * The default implementation does nothing (no-op) and exists as a template method for subclasses.
69       *
70       * @param request  the incoming ServletRequest
71       * @param response the outgoing ServletResponse
72       * @throws Exception if an error occurs.
73       */
74      protected void postHandle(ServletRequest request, ServletResponse response) throws Exception {
75      }
76  
77      /**
78       * Called in all cases in a <code>finally</code> block even if {@link #preHandle preHandle} returns
79       * <code>false</code> or if an exception is thrown during filter chain processing.  Can be used for resource
80       * cleanup if so desired.
81       * <p/>
82       * The default implementation does nothing (no-op) and exists as a template method for subclasses.
83       *
84       * @param request   the incoming ServletRequest
85       * @param response  the outgoing ServletResponse
86       * @param exception any exception thrown during {@link #preHandle preHandle}, {@link #executeChain executeChain},
87       *                  or {@link #postHandle postHandle} execution, or <code>null</code> if no exception was thrown
88       *                  (i.e. the chain processed successfully).
89       * @throws Exception if an error occurs.
90       */
91      public void afterCompletion(ServletRequest request, ServletResponse response, Exception exception) throws Exception {
92      }
93  
94      /**
95       * Actually executes the specified filter chain by calling <code>chain.doFilter(request,response);</code>.
96       * <p/>
97       * Can be overridden by subclasses for custom logic.
98       *
99       * @param request  the incoming ServletRequest
100      * @param response the outgoing ServletResponse
101      * @param chain    the filter chain to execute
102      * @throws Exception if there is any error executing the chain.
103      */
104     protected void executeChain(ServletRequest request, ServletResponse response, FilterChain chain) throws Exception {
105         chain.doFilter(request, response);
106     }
107 
108     /**
109      * Actually implements the chain execution logic, utilizing
110      * {@link #preHandle(javax.servlet.ServletRequest, javax.servlet.ServletResponse) pre},
111      * {@link #postHandle(javax.servlet.ServletRequest, javax.servlet.ServletResponse) post}, and
112      * {@link #afterCompletion(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Exception) after}
113      * advice hooks.
114      *
115      * @param request  the incoming ServletRequest
116      * @param response the outgoing ServletResponse
117      * @param chain    the filter chain to execute
118      * @throws ServletException if a servlet-related error occurs
119      * @throws IOException      if an IO error occurs
120      */
121     @SuppressWarnings({"ThrowFromFinallyBlock"})
122     public void doFilterInternal(ServletRequest request, ServletResponse response, FilterChain chain)
123             throws ServletException, IOException {
124 
125         Exception exception = null;
126 
127         try {
128 
129             boolean continueChain = preHandle(request, response);
130             if (log.isTraceEnabled()) {
131                 log.trace("Invked preHandle method.  Continuing chain?: [" + continueChain + "]");
132             }
133 
134             if (continueChain) {
135                 executeChain(request, response, chain);
136             }
137 
138             postHandle(request, response);
139             if (log.isTraceEnabled()) {
140                 log.trace("Successfully invoked postHandle method");
141             }
142 
143         } catch (Exception e) {
144             exception = e;
145         } finally {
146             cleanup( request, response, exception );
147         }
148     }
149 
150     /**
151      * Executes cleanup logic in the <code>finally</code> code block in the
152      * {@link #doFilterInternal(javax.servlet.ServletRequest, javax.servlet.ServletResponse, javax.servlet.FilterChain) doFilterInternal}
153      * implementation.
154      * <p/>
155      * This implementation specifically calls
156      * {@link #afterCompletion(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Exception) afterCompletion}
157      * as well as handles any exceptions properly.
158      *
159      * @param request the incoming <code>ServletRequest</code>
160      * @param response the outgoing <code>ServletResponse</code>
161      * @param existing any exception that might have occurred while executing the <code>FilterChain</code> or
162      * pre or post advice, or <code>null</code> if the pre/chain/post excution did not throw an <code>Exception</code>.
163      * @throws ServletException if any exception other than an <code>IOException</code> is thrown.
164      * @throws IOException if the pre/chain/post execution throw an <code>IOException</code>
165      */
166     protected void cleanup( ServletRequest request, ServletResponse response, Exception existing )
167         throws ServletException, IOException {
168         Exception exception = existing;
169         try {
170             afterCompletion(request, response, exception);
171             if (log.isTraceEnabled()) {
172                 log.trace("Successfully invoked afterCompletion method.");
173             }
174         } catch (Exception e) {
175             if (exception == null) {
176                 exception = e;
177             }
178         }
179         if (exception != null) {
180             if (exception instanceof ServletException) {
181                 throw (ServletException) exception;
182             } else if (exception instanceof IOException) {
183                 throw (IOException) exception;
184             } else {
185                 String msg = "Filter execution resulted in an unexpected Exception " +
186                         "(not IOException or ServletException as the Filter api recommends).  " +
187                         "Wrapping in ServletException and propagating.";
188                 throw new ServletException(msg, exception);
189             }
190         }
191     }
192 }
