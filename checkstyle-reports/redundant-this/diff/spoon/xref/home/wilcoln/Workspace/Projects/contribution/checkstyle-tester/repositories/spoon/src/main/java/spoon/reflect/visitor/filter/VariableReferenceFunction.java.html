6   package spoon.reflect.visitor.filter;
7   
8   
9   import spoon.SpoonException;
10  import spoon.reflect.code.CtCatchVariable;
11  import spoon.reflect.code.CtLocalVariable;
12  import spoon.reflect.declaration.CtElement;
13  import spoon.reflect.declaration.CtField;
14  import spoon.reflect.declaration.CtParameter;
15  import spoon.reflect.declaration.CtVariable;
16  import spoon.reflect.reference.CtVariableReference;
17  import spoon.reflect.visitor.CtScanner;
18  import spoon.reflect.visitor.chain.CtConsumableFunction;
19  import spoon.reflect.visitor.chain.CtConsumer;
20  
21  /**
22   * The mapping function, accepting {@link CtVariable}
23   * <ul>
24   * <li>CtLocalVariable - local variable declared in body
25   * <li>CtField - member field of an type
26   * <li>CtParameter - method parameter
27   * <li>CtCatchVariable - try - catch variable
28   * </ul>
29   * and returning all the {@link CtVariableReference}, which refers this variable
30   */
31  public class VariableReferenceFunction implements CtConsumableFunction<CtElement> {
32  
33  	protected final Visitor visitor = new Visitor();
34  	private final CtVariable<?> variable;
35  	protected CtConsumer<Object> outputConsumer;
36  	protected CtElement scope;
37  
38  	public VariableReferenceFunction() {
39  		this.variable = null;
40  	}
41  
42  	public VariableReferenceFunction(CtVariable<?> variable) {
43  		this.variable = variable;
44  	}
45  
46  	@Override
47  	public void apply(CtElement variableOrScope, CtConsumer<Object> outputConsumer) {
48  		scope = variableOrScope;
49  		CtVariable<?> var = this.variable;
50  		if (var == null) {
51  			if (variableOrScope instanceof CtVariable<?>) {
52  				var = (CtVariable<?>) variableOrScope;
53  			} else {
54  				throw new SpoonException("The input of VariableReferenceFunction must be a CtVariable but is a " + variableOrScope.getClass().getSimpleName());
55  			}
56  		}
57  		this.outputConsumer = outputConsumer;
58  		var.accept(visitor);
59  	}
60  
61  	protected class Visitor extends CtScanner {
62  		@Override
63  		protected void enter(CtElement e) {
64  			throw new SpoonException("Unsupported variable of type " + e.getClass().getName());
65  		}
66  		/**
67  		 * calls outputConsumer for each reference of the field
68  		 */
69  		@Override
70  		public <T> void visitCtField(CtField<T> field) {
71  			new FieldReferenceFunction((CtField<?>) variable).apply(scope, outputConsumer);
72  		}
73  
74  		/**
75  		 * calls outputConsumer for each reference of the local variable
76  		 */
77  		@Override
78  		public <T> void visitCtLocalVariable(CtLocalVariable<T> localVariable) {
79  			new LocalVariableReferenceFunction((CtLocalVariable<?>) variable).apply(scope, outputConsumer);
80  		}
81  
82  		/**
83  		 * calls outputConsumer for each reference of the parameter
84  		 */
85  		@Override
86  		public <T> void visitCtParameter(CtParameter<T> parameter) {
87  			new ParameterReferenceFunction((CtParameter<?>) variable).apply(scope, outputConsumer);
88  		}
89  
90  		/**
91  		 * calls outputConsumer for each reference of the catch variable
92  		 */
93  		@Override
94  		public <T> void visitCtCatchVariable(CtCatchVariable<T> catchVariable) {
95  			new CatchVariableReferenceFunction((CtCatchVariable<?>) variable).apply(scope, outputConsumer);
96  		}
97  	}
98  }
