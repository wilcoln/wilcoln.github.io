140 	private static ClassLoader lastClassLoader = null;

142 	/**
143 	 * Finds the class requested in {@link #getActualClass()}.
144 	 *
145 	 * Looks for the class in the standard Java classpath, but also in the sourceClassPath given as option.
146 	 */
147 	@SuppressWarnings("unchecked")
148 	protected Class<T> findClass() {
149 		String qualifiedName = getQualifiedName();
150 		ClassLoader classLoader = getFactory().getEnvironment().getInputClassLoader();
151 
152 		// an array class should not crash
153 		// see https://github.com/INRIA/spoon/pull/2882
154 		if (getSimpleName().contains("[]")) {
155 			// Class.forName does not work for primitive types and arrays :-(
156 			// we have to work-around
157 			// original idea from https://bugs.openjdk.java.net/browse/JDK-4031337
158 			return (Class<T>) RtHelper.getAllFields((Launcher.parseClass("public class Foo { public " + getQualifiedName() + " field; }").newInstance().getClass()))[0].getType();
159 		}
160 
161 		if (classLoader != lastClassLoader) {
162 			//clear cache because class loader changed
163 			classByQName.clear();
164 			lastClassLoader = classLoader;
165 		}
166 		return MapUtils.getOrCreate(classByQName, qualifiedName, () -> {
167 			try {
168 				// creating a classloader on the fly is not the most efficient
169 				// but it decreases the amount of state to maintain
170 				// since getActualClass is only used in rare cases, that's OK.
171 				return (Class<T>) classLoader.loadClass(qualifiedName);
172 			} catch (Throwable e) {
173 				throw new SpoonClassNotFoundException("cannot load class: " + getQualifiedName(), e);
174 			}
175 		});
176 	}

235 	@Override
236 	public boolean isSubtypeOf(CtTypeReference<?> type) {
237 		if (type instanceof CtTypeParameterReference) {
238 			return false;
239 		}
240 		if (NULL_TYPE_NAME.equals(getSimpleName()) || NULL_TYPE_NAME.equals(type.getSimpleName())) {
241 			return false;
242 		}
243 		if (isPrimitive() || type.isPrimitive()) {
244 			return equals(type);
245 		}
246 		if (this instanceof CtArrayTypeReference) {
247 			if (type instanceof CtArrayTypeReference) {
248 				return ((CtArrayTypeReference<?>) this).getComponentType().isSubtypeOf(((CtArrayTypeReference<?>) type).getComponentType());
249 			}
250 			if (Array.class.getName().equals(type.getQualifiedName())) {
251 				return true;
252 			}
253 		}
254 		if (Object.class.getName().equals(type.getQualifiedName())) {
255 			//everything is a sub type of Object
256 			return true;
257 		}
258 		return new ClassTypingContext(this).isSubtypeOf(type);
259 	}

261 	/**
262 	 * Detects if this type is an code responsible for implementing of that type.<br>
263 	 * In means it detects whether this type can access protected members of that type
264 	 * @return true if this type or any declaring type recursively is subtype of type or directly is the type.
265 	 */
266 	private boolean isImplementationOf(CtTypeReference<?> type) {
267 		CtTypeReference<?> impl = this;
268 		while (impl != null) {
269 			if (impl.isSubtypeOf(type)) {
270 				return true;
271 			}
272 			impl = impl.getDeclaringType();
273 		}
274 		return false;
275 	}

278 	public <C extends CtActualTypeContainer> C setActualTypeArguments(List<? extends CtTypeReference<?>> actualTypeArguments) {
279 		if (actualTypeArguments == null || actualTypeArguments.isEmpty()) {
280 			this.actualTypeArguments = CtElementImpl.emptyList();
281 			return (C) this;
282 		}
283 		if (this.actualTypeArguments == CtElementImpl.<CtTypeReference<?>>emptyList()) {
284 			this.actualTypeArguments = new ArrayList<>(TYPE_TYPE_PARAMETERS_CONTAINER_DEFAULT_CAPACITY);
285 		}
286 		getFactory().getEnvironment().getModelChangeListener().onListDeleteAll(this, TYPE_ARGUMENT, this.actualTypeArguments, new ArrayList<>(this.actualTypeArguments));
287 		this.actualTypeArguments.clear();
288 		for (CtTypeReference<?> actualTypeArgument : actualTypeArguments) {
289 			addActualTypeArgument(actualTypeArgument);
290 		}
291 		return (C) this;
292 	}

319 	@Override
320 	public CtTypeReference<?> unbox() {
321 		if (isPrimitive()) {
322 			return this;
323 		}
324 		Class<T> actualClass;
325 		try {
326 			actualClass = getActualClass();
327 		} catch (SpoonClassNotFoundException e) {
328 			return this;
329 		}
330 		if (actualClass == Integer.class) {
331 			return getFactory().Type().createReference(int.class);
332 		}
333 		if (actualClass == Float.class) {
334 			return getFactory().Type().createReference(float.class);
335 		}
336 		if (actualClass == Long.class) {
337 			return getFactory().Type().createReference(long.class);
338 		}
339 		if (actualClass == Character.class) {
340 			return getFactory().Type().createReference(char.class);
341 		}
342 		if (actualClass == Double.class) {
343 			return getFactory().Type().createReference(double.class);
344 		}
345 		if (actualClass == Boolean.class) {
346 			return getFactory().Type().createReference(boolean.class);
347 		}
348 		if (actualClass == Short.class) {
349 			return getFactory().Type().createReference(short.class);
350 		}
351 		if (actualClass == Byte.class) {
352 			return getFactory().Type().createReference(byte.class);
353 		}
354 		if (actualClass == Void.class) {
355 			return getFactory().Type().createReference(void.class);
356 		}
357 		return this;
358 	}

481 	public boolean isLocalType() {
482 		if (this.getDeclaration() != null) {
483 			return (this.getDeclaration().isLocalType());
484 		}
485 		// A local type doesn't have a fully qualified name but have an identifier
486 		// to know which is the local type member wanted by the developer.
487 		// Oracle documentation: https://docs.oracle.com/javase/specs/jls/se7/html/jls-6.html#jls-6.7
488 		// JDT documentation: http://help.eclipse.org/juno/topic/org.eclipse.jdt.doc.isv/reference/api/org/eclipse/jdt/core/dom/ITypeBinding.html#getQualifiedName()
489 		final Pattern pattern = Pattern.compile("^([0-9]+)([a-zA-Z]+)$");
490 		final Matcher m = pattern.matcher(getSimpleName());
491 		return m.find();
492 	}

574 	@Override
575 	public boolean canAccess(CtTypeReference<?> type) {
576 		try {
577 			if (type == null) {
578 				//noclasspath mode
579 				return true;
580 			}
581 			if (type.getTypeDeclaration() == null) {
582 				return true;
583 			}
584 			Set<ModifierKind> modifiers = type.getModifiers();
585 
586 			if (modifiers.contains(ModifierKind.PUBLIC)) {
587 				return true;
588 			}
589 			if (modifiers.contains(ModifierKind.PROTECTED)) {
590 				//the accessed type is protected in scope of declaring type.
591 				CtTypeReference<?> declaringType = type.getDeclaringType();
592 				if (declaringType == null) {
593 					//top level type cannot be protected. So this is a model inconsistency.
594 					throw new SpoonException("The protected class " + type.getQualifiedName() + " has no declaring class.");
595 				}
596 				if (isImplementationOf(declaringType)) {
597 					//type is visible in code which implements declaringType
598 					return true;
599 				} //else it is visible in same package, like package protected
600 				return isInSamePackage(type);
601 			}
602 			if (modifiers.contains(ModifierKind.PRIVATE)) {
603 				//it is visible in scope of the same class only
604 				return type.getTopLevelType().getQualifiedName().equals(this.getTopLevelType().getQualifiedName());
605 			}
606 			/*
607 			 * no modifier, we have to check if it is nested type and if yes, if parent is interface or class.
608 			 * In case of no parent then implicit access is package protected
609 			 * In case of parent is interface, then implicit access is PUBLIC
610 			 * In case of parent is class, then implicit access is package protected
611 			 */
612 			CtTypeReference<?> declaringTypeRef = type.getDeclaringType();
613 			if (declaringTypeRef != null && declaringTypeRef.isInterface()) {
614 				//the declaring type is interface, then implicit access is PUBLIC
615 				return true;
616 			}
617 			//package protected
618 			//visible only in scope of the same package
619 			return isInSamePackage(type);
620 		} catch (SpoonClassNotFoundException e) {
621 			handleParentNotFound(e);
622 			//if the modifiers cannot be resolved then we expect that it is visible
623 			return true;
624 		}
625 	}

627 	@Override
628 	public boolean canAccess(CtTypeMember typeMember) {
629 		CtType<?> declaringType = typeMember.getDeclaringType();
630 		if (declaringType == null) {
631 			//noclasspath mode
632 			return true;
633 		}
634 		CtTypeReference<?> declaringTypeRef = declaringType.getReference();
635 		if (!canAccess(declaringTypeRef)) {
636 			return false;
637 		}
638 		Set<ModifierKind> modifiers = typeMember.getModifiers();
639 
640 		if (modifiers.contains(ModifierKind.PUBLIC)) {
641 			return true;
642 		}
643 		if (modifiers.contains(ModifierKind.PROTECTED)) {
644 			if (isImplementationOf(declaringTypeRef)) {
645 				//type is visible in code which implements declaringType
646 				return true;
647 			} //else it is visible in same package, like package protected
648 			return isInSamePackage(declaringTypeRef);
649 		}
650 		if (modifiers.contains(ModifierKind.PRIVATE)) {
651 			//it is visible in scope of the same class only
652 			return declaringType.getTopLevelType().getQualifiedName().equals(this.getTopLevelType().getQualifiedName());
653 		}
654 		/*
655 		 * no modifier, we have to check if it is nested type and if yes, if parent is interface or class.
656 		 * In case of no parent then implicit access is package protected
657 		 * In case of parent is interface, then implicit access is PUBLIC
658 		 * In case of parent is class, then implicit access is package protected
659 		 */
660 		CtType<?> declaringTypeDeclaringType = declaringType.getDeclaringType();
661 		if (declaringTypeDeclaringType != null && declaringTypeDeclaringType.isInterface()) {
662 			//the declaring type is interface, then implicit access is PUBLIC
663 			return true;
664 		}
665 		//package protected
666 		//visible only in scope of the same package
667 		return isInSamePackage(declaringTypeRef);
668 	}

670 	private boolean isInSamePackage(CtTypeReference<?> type) {
671 		CtPackageReference thisPackage = this.getTopLevelType().getPackage();
672 		CtPackageReference otherPackage = type.getTopLevelType().getPackage();
673 		if (thisPackage == null || otherPackage == null) {
674 			return true;
675 		}
676 		return thisPackage.getQualifiedName().equals(otherPackage.getQualifiedName());
677 	}

679 	@Override
680 	public CtTypeReference<?> getTopLevelType() {
681 		CtTypeReference<?> type = this;
682 		while (true) {
683 			CtTypeReference<?> parentType = type.getDeclaringType();
684 			if (parentType == null) {
685 				return type;
686 			}
687 			type = parentType;
688 		}
689 	}

691 	@Override
692 	public CtTypeReference<?> getAccessType() {
693 		CtTypeReference<?> declType = this.getDeclaringType();
694 		if (declType == null) {
695 			throw new SpoonException("The declaring type is expected, but " + getQualifiedName() + " is top level type");
696 		}
697 		CtType<?> contextType = getParent(CtType.class);
698 		if (contextType == null) {
699 			return declType;
700 		}
701 		CtTypeReference<?> contextTypeRef = contextType.getReference();
702 		if (contextTypeRef != null && contextTypeRef.canAccess(declType) == false) {
703 			//search for visible declaring type
704 			CtTypeReference<?> visibleDeclType = null;
705 			CtTypeReference<?> type = contextTypeRef;
706 			//search which type or declaring type of startType extends from nestedType
707 			while (visibleDeclType == null && type != null) {
708 				visibleDeclType = getLastVisibleSuperClassExtendingFrom(type, declType);
709 				if (visibleDeclType != null) {
710 					//found one!
711 					applyActualTypeArguments(visibleDeclType, declType);
712 					break;
713 				}
714 				//try class hierarchy of declaring type
715 				type = type.getDeclaringType();
716 			}
717 			declType = visibleDeclType;
718 		}
719 		if (declType == null) {
720 			throw new SpoonException("Cannot compute access path to type: " + this.getQualifiedName() + " in context of type: " + contextType.getQualifiedName());
721 		}
722 		return declType;
723 	}

725 	/**
726 	 * adds the actualTypeArguments of sourceTypeRef to targetTypeRef. Type of targetTypeRef extends from type of sourceTypeRef
727 	 * @param targetTypeRef
728 	 * @param sourceTypeRef
729 	 */
730 	private static void applyActualTypeArguments(CtTypeReference<?> targetTypeRef, CtTypeReference<?> sourceTypeRef) {
731 		CtTypeReference<?> targetDeclType = targetTypeRef.getDeclaringType();
732 		CtTypeReference<?> sourceDeclType = sourceTypeRef.getDeclaringType();
733 		if (targetDeclType != null && sourceDeclType != null && targetDeclType.isSubtypeOf(sourceDeclType)) {
734 			applyActualTypeArguments(targetDeclType, sourceDeclType);
735 		}
736 		if (targetTypeRef.isSubtypeOf(sourceTypeRef) == false) {
737 			throw new SpoonException("Invalid arguments. targetTypeRef " + targetTypeRef.getQualifiedName() + " must be a sub type of sourceTypeRef " + sourceTypeRef.getQualifiedName());
738 		}
739 		List<CtTypeReference<?>> newTypeArgs = new ArrayList<>();
740 		/*
741 		 * for now simply copy the type arguments, to have it fixed fast. But it is not correct!
742 		 *
743 		 * For example in this case
744 		 *
745 		 * class A<T,K>{}
746 		 *
747 		 * class B<U,T> extends A<T,Integer>
748 		 *
749 		 * The sourceTypeRef: A<T,K>
750 		 * has to be applied to
751 		 * targetTypeRef: B<?,T>
752 		 */
753 		for (CtTypeReference<?> l_tr : sourceTypeRef.getActualTypeArguments()) {
754 			newTypeArgs.add(l_tr.clone());
755 		}
756 		targetTypeRef.setActualTypeArguments(newTypeArgs);
757 	}

759 	/**
760 	 *
761 	 * @param sourceType
762 	 * @param targetType
763 	 * @return sourceType or last super class of sourceType, which extends from targetType and which is visible from sourceType or null if sourceType does not extends from targetType
764 	 */
765 	private static CtTypeReference<?> getLastVisibleSuperClassExtendingFrom(CtTypeReference<?> sourceType, CtTypeReference<?> targetType) {
766 		String targetQN = targetType.getQualifiedName();
767 		CtTypeReference<?> adept = sourceType;
768 		CtTypeReference<?> type = sourceType;
769 		while (true) {
770 			if (targetQN.equals(type.getQualifiedName())) {
771 				return adept;
772 			}
773 			type = type.getSuperclass();
774 			if (type == null) {
775 				//there is no super type which extends from targetType
776 				return null;
777 			}
778 			if (sourceType.canAccess(type)) {
779 				//this super type is still visible. It is adept for returning
780 				adept = type;
781 			}
782 		}
783 	}

805 	@Override
806 	public CtTypeParameter getTypeParameterDeclaration() {
807 
808 		CtElement parent = this.getParent();
809 
810 		// case 1: this is an actual type argument of a type reference eg List<E>
811 		if (parent instanceof CtTypeReference) {
812 			CtType t = ((CtTypeReference) parent).getTypeDeclaration();
813 			return findTypeParamDeclarationByPosition(t, ((CtTypeReference) parent).getActualTypeArguments().indexOf(this));
814 		}
815 
816 		// case 2: this is an actual type argument of a method/constructor reference
817 		if (parent instanceof CtExecutableReference) {
818 			CtExecutable<?> exec = ((CtExecutableReference<?>) parent).getExecutableDeclaration();
819 			if (exec instanceof CtMethod || exec instanceof CtConstructor) {
820 				int idx = ((CtExecutableReference) parent).getActualTypeArguments().indexOf(this);
821 				return idx >= 0 ? findTypeParamDeclarationByPosition((CtFormalTypeDeclarer) exec, idx) : null;
822 			}
823 		}
824 
825 		if (parent instanceof CtFormalTypeDeclarer) {
826 			CtFormalTypeDeclarer exec = (CtFormalTypeDeclarer) parent;
827 			if (exec instanceof CtMethod || exec instanceof CtConstructor) {
828 				for (CtTypeParameter typeParam : exec.getFormalCtTypeParameters()) {
829 					if (typeParam.getSimpleName().equals(getSimpleName())) {
830 						return typeParam;
831 					}
832 				}
833 				return null;
834 			}
835 		}
836 		return null;
837 	}

840 	public boolean isGenerics() {
841 		if (getDeclaration() instanceof CtTypeParameter) {
842 			return true;
843 		}
844 		for (CtTypeReference ref : getActualTypeArguments()) {
845 			if (ref.isGenerics()) {
846 				return true;
847 			}
848 		}
849 		return false;
850 	}
