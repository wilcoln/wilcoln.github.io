89      /** Simple constructor.
90       * @param propagatorBuilder builder to user for propagation
91       * @param optimizer solver for least squares problem
92       * @exception OrekitException if some propagator parameter cannot be retrieved
93       */
94      public BatchLSEstimator(final NumericalPropagatorBuilder propagatorBuilder,
95                              final LeastSquaresOptimizer optimizer)
96          throws OrekitException {
97  
98          this.propagatorBuilder              = propagatorBuilder;
99          this.measurements                   = new ArrayList<ObservedMeasurement<?>>();
100         this.optimizer                      = optimizer;
101         this.parametersConvergenceThreshold = Double.NaN;
102         this.lsBuilder                      = new LeastSquaresBuilder();
103         this.estimations                    = null;
104         this.observer                       = null;
105 
106         // our model computes value and Jacobian in one call,
107         // so we don't use the lazy evaluation feature
108         lsBuilder.lazyEvaluation(false);
109 
110         // we manage weight by ourselves, as we change them during
111         // iterations (setting to 0 the identified outliers measurements)
112         // so the least squares problem should not see our weights
113         lsBuilder.weight(null);
114 
115     }

171     /** Get the orbital parameters supported by this estimator.
172      * @param estimatedOnly if true, only estimated parameters are returned
173      * @return orbital parameters supported by this estimator
174      * @exception OrekitException if different parameters have the same name
175      */
176     public ParameterDriversList getOrbitalParametersDrivers(final boolean estimatedOnly)
177         throws OrekitException {
178 
179         if (estimatedOnly) {
180             final ParameterDriversList estimated = new ParameterDriversList();
181             for (final DelegatingDriver delegating : propagatorBuilder.getOrbitalParametersDrivers().getDrivers()) {
182                 if (delegating.isSelected()) {
183                     for (final ParameterDriver driver : delegating.getRawDrivers()) {
184                         estimated.add(driver);
185                     }
186                 }
187             }
188             return estimated;
189         } else {
190             return propagatorBuilder.getOrbitalParametersDrivers();
191         }
192 
193     }

219     /** Get the measurements parameters supported by this estimator (including measurements and modifiers).
220      * @param estimatedOnly if true, only estimated parameters are returned
221      * @return measurements parameters supported by this estimator
222      * @exception OrekitException if different parameters have the same name
223      */
224     public ParameterDriversList getMeasurementsParametersDrivers(final boolean estimatedOnly)
225         throws OrekitException {
226 
227         final ParameterDriversList parameters =  new ParameterDriversList();
228         for (final  ObservedMeasurement<?> measurement : measurements) {
229             for (final ParameterDriver driver : measurement.getParametersDrivers()) {
230                 if ((!estimatedOnly) || driver.isSelected()) {
231                     parameters.add(driver);
232                 }
233             }
234         }
235 
236         parameters.sort();
237 
238         return parameters;
239 
240     }

242     /**
243      * Set convergence threshold.
244      * <p>
245      * The convergence used for estimation is based on the estimated
246      * parameters {@link ParameterDriver#getNormalizedValue() normalized values}.
247      * Convergence is considered to have been reached when the difference
248      * between previous and current normalized value is less than the
249      * convergence threshold for all parameters. The same value is used
250      * for all parameters since they are normalized and hence dimensionless.
251      * </p>
252      * <p>
253      * Normalized values are computed as {@code (current - reference)/scale},
254      * so convergence is reached when the following condition holds for
255      * all estimated parameters:
256      * {@code |current[i] - previous[i]| <= threshold * scale[i]}
257      * </p>
258      * <p>
259      * So the convergence threshold specified here can be considered as
260      * a multiplication factor applied to scale. Since for all parameters
261      * the scale is often small (typically about 1 m for orbital positions
262      * for example), then the threshold should not be too small. A value
263      * of 10⁻³ is often quite accurate.
264      *
265      * @param parametersConvergenceThreshold convergence threshold on
266      * normalized parameters (dimensionless, related to parameters scales)
267      * @see EvaluationRmsChecker
268      */
269     public void setParametersConvergenceThreshold(final double parametersConvergenceThreshold) {
270         this.parametersConvergenceThreshold = parametersConvergenceThreshold;
271     }

273     /** Estimate the orbital, propagation and measurements parameters.
274      * <p>
275      * The initial guess for all parameters must have been set before calling this method
276      * using {@link #getOrbitalParametersDrivers(boolean)}, {@link #getPropagatorParametersDrivers(boolean)},
277      * and {@link #getMeasurementsParametersDrivers(boolean)} and then {@link ParameterDriver#setValue(double)
278      * setting the values} of the parameters.
279      * </p>
280      * <p>
281      * After this method returns, the estimated parameters can be retrieved using
282      * {@link #getOrbitalParametersDrivers(boolean)}, {@link #getPropagatorParametersDrivers(boolean)},
283      * and {@link #getMeasurementsParametersDrivers(boolean)} and then {@link ParameterDriver#getValue()
284      * getting the values} of the parameters.
285      * </p>
286      * <p>
287      * As a convenience, the method also returns a fully configured and ready to use
288      * propagator set up with all the estimated values.
289      * </p>
290      * @return propagator configured with estimated orbit as initial state, and all
291      * propagator estimated parameters also set
292      * @exception OrekitException if there is a conflict in parameters names
293      * or if orbit cannot be determined
294      */
295     public NumericalPropagator estimate() throws OrekitException {
296 
297         // get all estimated parameters
298         final ParameterDriversList estimatedOrbitalParameters      = getOrbitalParametersDrivers(true);
299         final ParameterDriversList estimatedPropagatorParameters   = getPropagatorParametersDrivers(true);
300         final ParameterDriversList estimatedMeasurementsParameters = getMeasurementsParametersDrivers(true);
301 
302         // create start point
303         final double[] start = new double[estimatedOrbitalParameters.getNbParams() +
304                                           estimatedPropagatorParameters.getNbParams() +
305                                           estimatedMeasurementsParameters.getNbParams()];
306         int iStart = 0;
307         for (final ParameterDriver driver : estimatedOrbitalParameters.getDrivers()) {
308             start[iStart++] = driver.getNormalizedValue();
309         }
310         for (final ParameterDriver driver : estimatedPropagatorParameters.getDrivers()) {
311             start[iStart++] = driver.getNormalizedValue();
312         }
313         for (final ParameterDriver driver : estimatedMeasurementsParameters.getDrivers()) {
314             start[iStart++] = driver.getNormalizedValue();
315         }
316         lsBuilder.start(start);
317 
318         // create target (which is an array set to 0, as we compute weighted residuals ourselves)
319         int p = 0;
320         for (final ObservedMeasurement<?> measurement : measurements) {
321             if (measurement.isEnabled()) {
322                 p += measurement.getDimension();
323             }
324         }
325         final double[] target = new double[p];
326         lsBuilder.target(target);
327 
328         // set up the model
329         final ModelObserver modelObserver = new ModelObserver() {
330             /** {@inheritDoc} */
331             @Override
332             public void modelCalled(final Orbit newOrbit,
333                                     final Map<ObservedMeasurement<?>, EstimatedMeasurement<?>> newEstimations) {
334                 BatchLSEstimator.this.orbit       = newOrbit;
335                 BatchLSEstimator.this.estimations = newEstimations;
336             }
337         };
338         final Model model = new Model(propagatorBuilder, measurements, estimatedMeasurementsParameters,
339                                       modelObserver);
340         lsBuilder.model(model);
341 
342         // add a validator for orbital parameters
343         lsBuilder.parameterValidator(new Validator(estimatedOrbitalParameters,
344                                                    estimatedPropagatorParameters,
345                                                    estimatedMeasurementsParameters));
346 
347         lsBuilder.checker(new ConvergenceChecker<LeastSquaresProblem.Evaluation>() {
348             /** {@inheritDoc} */
349             @Override
350             public boolean converged(final int iteration,
351                                      final LeastSquaresProblem.Evaluation previous,
352                                      final LeastSquaresProblem.Evaluation current) {
353                 final double lInf = current.getPoint().getLInfDistance(previous.getPoint());
354                 return lInf <= parametersConvergenceThreshold;
355             }
356         });
357 
358         // set up the problem to solve
359         final LeastSquaresProblem problem = new TappedLSProblem(lsBuilder.build(),
360                                                                 model,
361                                                                 estimatedOrbitalParameters,
362                                                                 estimatedPropagatorParameters,
363                                                                 estimatedMeasurementsParameters);
364 
365         try {
366 
367             // solve the problem
368             optimum = optimizer.optimize(problem);
369 
370             // create a new configured propagator with all estimated parameters
371             return model.createPropagator(optimum.getPoint());
372 
373         } catch (MathRuntimeException mrte) {
374             throw new OrekitException(mrte);
375         } catch (OrekitExceptionWrapper oew) {
376             throw oew.getException();
377         }
378 
379     }

411     /** Wrapper used to tap the various counters. */
412     private class TappedLSProblem implements LeastSquaresProblem {
413 
414         /** Underlying problem. */
415         private final LeastSquaresProblem problem;
416 
417         /** Multivariate function model. */
418         private final Model model;
419 
420         /** Estimated orbital parameters. */
421         private final ParameterDriversList estimatedOrbitalParameters;
422 
423         /** Estimated propagator parameters. */
424         private final ParameterDriversList estimatedPropagatorParameters;
425 
426         /** Estimated measurements parameters. */
427         private final ParameterDriversList estimatedMeasurementsParameters;
428 
429         /** Simple constructor.
430          * @param problem underlying problem
431          * @param model multivariate function model
432          * @param estimatedOrbitalParameters estimated orbital parameters
433          * @param estimatedPropagatorParameters estimated propagator parameters
434          * @param estimatedMeasurementsParameters estimated measurements parameters
435          */
436         TappedLSProblem(final LeastSquaresProblem problem,
437                         final Model model,
438                         final ParameterDriversList estimatedOrbitalParameters,
439                         final ParameterDriversList estimatedPropagatorParameters,
440                         final ParameterDriversList estimatedMeasurementsParameters) {
441             this.problem                         = problem;
442             this.model                           = model;
443             this.estimatedOrbitalParameters      = estimatedOrbitalParameters;
444             this.estimatedPropagatorParameters   = estimatedPropagatorParameters;
445             this.estimatedMeasurementsParameters = estimatedMeasurementsParameters;
446         }
447 
448         /** {@inheritDoc} */
449         @Override
450         public Incrementor getEvaluationCounter() {
451             // tap the evaluations counter
452             BatchLSEstimator.this.evaluationsCounter = problem.getEvaluationCounter();
453             model.setEvaluationsCounter(BatchLSEstimator.this.evaluationsCounter);
454             return BatchLSEstimator.this.evaluationsCounter;
455         }
456 
457         /** {@inheritDoc} */
458         @Override
459         public Incrementor getIterationCounter() {
460             // tap the iterations counter
461             BatchLSEstimator.this.iterationsCounter = problem.getIterationCounter();
462             model.setIterationsCounter(BatchLSEstimator.this.iterationsCounter);
463             return BatchLSEstimator.this.iterationsCounter;
464         }
465 
466         /** {@inheritDoc} */
467         @Override
468         public ConvergenceChecker<Evaluation> getConvergenceChecker() {
469             return problem.getConvergenceChecker();
470         }
471 
472         /** {@inheritDoc} */
473         @Override
474         public RealVector getStart() {
475             return problem.getStart();
476         }
477 
478         /** {@inheritDoc} */
479         @Override
480         public int getObservationSize() {
481             return problem.getObservationSize();
482         }
483 
484         /** {@inheritDoc} */
485         @Override
486         public int getParameterSize() {
487             return problem.getParameterSize();
488         }
489 
490         /** {@inheritDoc} */
491         @Override
492         public Evaluation evaluate(final RealVector point) {
493 
494             // perform the evaluation
495             final Evaluation evaluation = problem.evaluate(point);
496 
497             // notify the observer
498             if (observer != null) {
499                 try {
500                     observer.evaluationPerformed(iterationsCounter.getCount(),
501                                                  evaluationsCounter.getCount(),
502                                                  orbit,
503                                                  estimatedOrbitalParameters,
504                                                  estimatedPropagatorParameters,
505                                                  estimatedMeasurementsParameters,
506                                                  new Provider(),
507                                                  evaluation);
508                 } catch (OrekitException oe) {
509                     throw new OrekitExceptionWrapper(oe);
510                 }
511             }
512 
513             return evaluation;
514 
515         }
516 
517     }

519     /** Provider for evaluations. */
520     private class Provider implements EstimationsProvider {
521 
522         /** Sorted estimations. */
523         private EstimatedMeasurement<?>[] sortedEstimations;
524 
525         /** {@inheritDoc} */
526         @Override
527         public int getNumber() {
528             return estimations.size();
529         }
530 
531         /** {@inheritDoc} */
532         @Override
533         public EstimatedMeasurement<?> getEstimatedMeasurement(final int index)
534             throws OrekitException {
535 
536             // safety checks
537             if (index < 0 || index >= estimations.size()) {
538                 throw new OrekitException(LocalizedCoreFormats.OUT_OF_RANGE_SIMPLE,
539                                           index, 0, estimations.size());
540             }
541 
542             if (sortedEstimations == null) {
543 
544                 // lazy evaluation of the sorted array
545                 sortedEstimations = new EstimatedMeasurement<?>[estimations.size()];
546                 int i = 0;
547                 for (final Map.Entry<ObservedMeasurement<?>, EstimatedMeasurement<?>> entry : estimations.entrySet()) {
548                     sortedEstimations[i++] = entry.getValue();
549                 }
550 
551                 // sort the array chronologically
552                 Arrays.sort(sortedEstimations, 0, sortedEstimations.length,
553                             new ChronologicalComparator());
554 
555             }
556 
557             return sortedEstimations[index];
558 
559         }
560 
561     }

563     /** Validator for estimated parameters. */
564     private static class Validator implements ParameterValidator {
565 
566         /** Estimated orbital parameters. */
567         private final ParameterDriversList estimatedOrbitalParameters;
568 
569         /** Estimated propagator parameters. */
570         private final ParameterDriversList estimatedPropagatorParameters;
571 
572         /** Estimated measurements parameters. */
573         private final ParameterDriversList estimatedMeasurementsParameters;
574 
575         /** Simple constructor.
576          * @param estimatedOrbitalParameters estimated orbital parameters
577          * @param estimatedPropagatorParameters estimated propagator parameters
578          * @param estimatedMeasurementsParameters estimated measurements parameters
579          */
580         Validator(final ParameterDriversList estimatedOrbitalParameters,
581                   final ParameterDriversList estimatedPropagatorParameters,
582                   final ParameterDriversList estimatedMeasurementsParameters) {
583             this.estimatedOrbitalParameters      = estimatedOrbitalParameters;
584             this.estimatedPropagatorParameters   = estimatedPropagatorParameters;
585             this.estimatedMeasurementsParameters = estimatedMeasurementsParameters;
586         }
587 
588         /** {@inheritDoc} */
589         @Override
590         public RealVector validate(final RealVector params)
591             throws OrekitExceptionWrapper {
592 
593             try {
594                 int i = 0;
595                 for (final ParameterDriver driver : estimatedOrbitalParameters.getDrivers()) {
596                     // let the parameter handle min/max clipping
597                     driver.setNormalizedValue(params.getEntry(i));
598                     params.setEntry(i++, driver.getNormalizedValue());
599                 }
600                 for (final ParameterDriver driver : estimatedPropagatorParameters.getDrivers()) {
601                     // let the parameter handle min/max clipping
602                     driver.setNormalizedValue(params.getEntry(i));
603                     params.setEntry(i++, driver.getNormalizedValue());
604                 }
605                 for (final ParameterDriver driver : estimatedMeasurementsParameters.getDrivers()) {
606                     // let the parameter handle min/max clipping
607                     driver.setNormalizedValue(params.getEntry(i));
608                     params.setEntry(i++, driver.getNormalizedValue());
609                 }
610 
611                 return params;
612             } catch (OrekitException oe) {
613                 throw new OrekitExceptionWrapper(oe);
614             }
615         }
616     }
