130     /** Orbit got from Observed Three Lines of Sight (angles only).
131      *
132      * @param O1 Observer position 1
133      * @param O2 Observer position 2
134      * @param O3 Observer position 3
135      * @param lineOfSight1 line of sight 1
136      * @param dateObs1 date of observation 1
137      * @param lineOfSight2 line of sight 2
138      * @param dateObs2 date of observation 1
139      * @param lineOfSight3 line of sight 3
140      * @param dateObs3 date of observation 1
141      * @param rho1init initial guess of the range problem. range 1, in meters
142      * @param rho3init initial guess of the range problem. range 3, in meters
143      * @return an estimate of the Keplerian orbit
144      */
145     public KeplerianOrbit estimate(final Vector3D O1, final Vector3D O2, final Vector3D O3,
146                                    final Vector3D lineOfSight1, final AbsoluteDate dateObs1,
147                                    final Vector3D lineOfSight2, final AbsoluteDate dateObs2,
148                                    final Vector3D lineOfSight3, final AbsoluteDate dateObs3,
149                                    final double rho1init, final double rho3init) {
150 
151         this.date1 = dateObs1;
152 
153         // normalizing coefficients
154         R = FastMath.max(rho1init, rho3init);
155         V = FastMath.sqrt(mu / R);
156         T = R / V;
157 
158         // Initialize Lambert's problem solver for non-dimensional units.
159         lambert = new IodLambert(1.);
160 
161         this.vObserverPosition1 = O1.scalarMultiply(1. / R);
162         this.vObserverPosition2 = O2.scalarMultiply(1. / R);
163         this.vObserverPosition3 = O3.scalarMultiply(1. / R);
164 
165         final int maxiter = 100; // maximum iter
166 
167         // solve the range problem
168         solveRangeProblem(rho1init / R, rho3init / R,
169                           dateObs3.durationFrom(dateObs1) / T, dateObs2.durationFrom(dateObs1) / T,
170                           0,
171                           true,
172                           lineOfSight1, lineOfSight2, lineOfSight3,
173                           maxiter);
174 
175         // use the Gibbs problem to get the orbit now that we have three
176         // position vectors.
177         final IodGibbs gibbs = new IodGibbs(mu);
178         return gibbs.estimate(frame,
179                               (vObserverPosition1.add(lineOfSight1.scalarMultiply(rho1))).scalarMultiply(R), dateObs1,
180                               (vObserverPosition2.add(lineOfSight2.scalarMultiply(rho2))).scalarMultiply(R), dateObs2,
181                               (vObserverPosition3.add(lineOfSight3.scalarMultiply(rho3))).scalarMultiply(R), dateObs3);
182     }

184     /** Solve the range problem when three line of sight are given.
185      *
186      * @param rho1init   initial value for range R1, in meters
187      * @param rho3init   initial value for range R3, in meters
188      * @param T13   time of flight 1->3, in seconds
189      * @param T12   time of flight 1->2, in seconds
190      * @param nrev number of revolutions
191      * @param direction  posigrade (true) or retrograde
192      * @param lineOfSight1  line of sight 1
193      * @param lineOfSight2  line of sight 2
194      * @param lineOfSight3  line of sight 3
195      * @param maxIterations         max iter
196      * @return nothing
197      */
198     private boolean solveRangeProblem(final double rho1init, final double rho3init,
199                                       final double T13, final double T12,
200                                       final int nrev,
201                                       final boolean direction,
202                                       final Vector3D lineOfSight1,
203                                       final Vector3D lineOfSight2,
204                                       final Vector3D lineOfSight3,
205                                       final int maxIterations) {
206         final double ARBF = 1e-6;   // finite differences step
207         boolean withHalley = true;  // use Halley's method
208         final double cvtol = 1e-14; // convergence tolerance
209 
210         rho1 = rho1init;
211         rho3 = rho3init;
212 
213 
214         int iter = 0;
215         double stoppingCriterion = 10 * cvtol;
216         while ((iter < maxIterations) && (FastMath.abs(stoppingCriterion) > cvtol))  {
217             facFiniteDiff = ARBF;
218 
219             // proposed in the original algorithm by Gooding.
220             // We change the threshold to maxIterations / 2.
221             if (iter >= (maxIterations / 2)) {
222                 withHalley = true;
223             }
224 
225             // tentative position for R2
226             final Vector3D P2 = getPositionOnLoS2(lineOfSight1, rho1,
227                                           lineOfSight3, rho3,
228                                           T13, T12, nrev, direction);
229 
230             if (P2 == null) {
231                 modifyIterate(lineOfSight1, lineOfSight2, lineOfSight3);
232             } else {
233                 //
234                 R2 = P2.getNorm();
235 
236                 // compute current line of sight for (2) and the associated range
237                 final Vector3D C = P2.subtract(vObserverPosition2);
238                 rho2 = C.getNorm();
239 
240                 final double R10 = R1;
241                 final double R30 = R3;
242 
243                 // indicate how close we are to the direction of sight for measurement (2)
244                 // for convergence test
245                 final double CR = lineOfSight2.dotProduct(C);
246 
247                 // construct a basis and the function f and g.
248                 // Specifically, f is the P-coordinate and g the N-coordinate.
249                 // They should be zero when line of sight 2 and current direction for 2 from O2 are aligned.
250                 final Vector3D u = lineOfSight2.crossProduct(C);
251                 final Vector3D P = (u.crossProduct(lineOfSight2)).normalize();
252                 final Vector3D EN = (lineOfSight2.crossProduct(P)).normalize();
253 
254                 // if EN is zero we have a solution!
255                 final double ENR = EN.getNorm();
256                 if (ENR == 0.) {
257                     return true;
258                 }
259 
260                 // Coordinate along 'F function'
261                 final double Fc = P.dotProduct(C);
262                 //Gc = EN.dotProduct(C);
263 
264                 // Now get partials, by finite differences
265                 final double[] FD = new double[2];
266                 final double[] GD = new double[2];
267                 computeDerivatives(rho1, rho3,
268                                    R10, R30,
269                                    lineOfSight1, lineOfSight3,
270                                    P, EN,
271                                    Fc,
272                                    T13, T12,
273                                    withHalley,
274                                    nrev,
275                                    direction,
276                                    FD, GD);
277 
278                 // terms of the Jacobian
279                 final double fr1 = FD[0];
280                 final double fr3 = FD[1];
281                 final double gr1 = GD[0];
282                 final double gr3 = GD[1];
283                 // determinant of the Jacobian matrix
284                 final double detj = fr1 * gr3 - fr3 * gr1;
285 
286                 // compute the Newton step
287                 D3 = -gr3 * Fc / detj;
288                 D1 = gr1 * Fc / detj;
289 
290                 // update current values of ranges
291                 rho1 = rho1 + D3;
292                 rho3 = rho3 + D1;
293 
294                 // convergence tests
295                 final double den = FastMath.max(CR, R2);
296                 stoppingCriterion = Fc / den;
297             }
298 
299             ++iter;
300         } // end while
301 
302         return true;
303     }

305     /** Change the current iterate if Lambert's problem solver failed to find a solution.
306      *
307      * @param lineOfSight1 line of sight 1
308      * @param lineOfSight2 line of sight 2
309      * @param lineOfSight3 line of sight 3
310      */
311     private void modifyIterate(final Vector3D lineOfSight1,
312                                final Vector3D lineOfSight2,
313                                final Vector3D lineOfSight3) {
314         // This is a modifier proposed in the initial paper of Gooding.
315         // Try to avoid Lambert-fail, by common-perpendicular starters
316         final Vector3D R13 = vObserverPosition3.subtract(vObserverPosition1);
317         D1 = R13.dotProduct(lineOfSight1);
318         D3 = R13.dotProduct(lineOfSight3);
319         final double D2 = lineOfSight1.dotProduct(lineOfSight3);
320         final double D4 = 1. - D2 * D2;
321         rho1 = FastMath.max((D1 - D3 * D2) / D4, 0.);
322         rho3 = FastMath.max((D1 * D2 - D3) / D4, 0.);
323     }

325     /** Compute the derivatives by finite-differences for the range problem.
326      * Specifically, we are trying to solve the problem:
327      *      f(x,y) = 0
328      *      g(x,y) = 0
329      * So, in a Newton-Raphson process, we would need the derivatives:
330      *  fx,fy,gx,gy
331      * Enventually,
332      *    dx =-f*gy / D
333      *    dy = f*gx / D
334      * where D is the determinant of the Jacobian matrix.
335      *
336      *
337      * @param x    current range 1
338      * @param y    current range 3
339      * @param R10   current radius 1
340      * @param R30   current radius 3
341      * @param lineOfSight1  line of sight
342      * @param lineOfSight3  line of sight
343      * @param Pin   basis vector
344      * @param Ein   basis vector
345      * @param F     value of the f-function
346      * @param T13   time of flight 1->3, in seconds
347      * @param T12   time of flight 1->2, in seconds
348      * @param withHailley    use Halley iterative method
349      * @param nrev  number of revolutions
350      * @param direction direction of motion
351      * @param FD    derivatives of f wrt (rho1, rho3) by finite differences
352      * @param GD    derivatives of g wrt (rho1, rho3) by finite differences
353      */
354     private void computeDerivatives(final double x, final double y,
355                                     final double R10, final double R30,
356                                     final Vector3D lineOfSight1, final Vector3D lineOfSight3,
357                                     final Vector3D Pin,
358                                     final Vector3D Ein,
359                                     final double F,
360                                     final double T13, final double T12,
361                                     final boolean withHailley,
362                                     final int nrev,
363                                     final boolean direction,
364                                     final double[] FD, final double[] GD) {
365 
366         final Vector3D P = Pin.normalize();
367         final Vector3D EN = Ein.normalize();
368 
369         // Now get partials, by finite differences
370         // steps for the differentiation
371         final double dx = facFiniteDiff * x;
372         final double dy = facFiniteDiff * y;
373 
374         final Vector3D Cm1 = getPositionOnLoS2 (lineOfSight1, x - dx,
375                               lineOfSight3, y,
376                               T13, T12, nrev, direction).subtract(vObserverPosition2);
377 
378         final double Fm1 = P.dotProduct(Cm1);
379         final double Gm1 = EN.dotProduct(Cm1);
380 
381         final Vector3D Cp1 = getPositionOnLoS2 (lineOfSight1, x + dx,
382                               lineOfSight3, y,
383                               T13, T12, nrev, direction).subtract(vObserverPosition2);
384 
385         final double Fp1  = P.dotProduct(Cp1);
386         final double Gp1 = EN.dotProduct(Cp1);
387 
388         // derivatives df/drho1 and dg/drho1
389         final double Fx = (Fp1 - Fm1) / (2 * dx);
390         final double Gx = (Gp1 - Gm1) / (2 * dx);
391 
392         final Vector3D Cm3 = getPositionOnLoS2 (lineOfSight1, x,
393                               lineOfSight3, y - dy,
394                               T13, T12, nrev, direction).subtract(vObserverPosition2);
395 
396         final double Fm3 = P.dotProduct(Cm3);
397         final double Gm3 = EN.dotProduct(Cm3);
398 
399         final Vector3D Cp3 = getPositionOnLoS2 (lineOfSight1, x,
400                               lineOfSight3, y + dy,
401                               T13, T12, nrev, direction).subtract(vObserverPosition2);
402 
403         final double Fp3 = P.dotProduct(Cp3);
404         final double Gp3 = EN.dotProduct(Cp3);
405 
406         // derivatives df/drho3 and dg/drho3
407         final double Fy = (Fp3 - Fm3) / (2. * dy);
408         final double Gy = (Gp3 - Gm3) / (2. * dy);
409         final double detJac = Fx * Gy - Fy * Gx;
410 
411         // Coefficients for the classical Newton-Raphson iterative method
412         FD[0] = Fx / detJac;
413         FD[1] = Fy / detJac;
414         GD[0] = Gx / detJac;
415         GD[1] = Gy / detJac;
416 
417         // Modified Newton-Raphson process, with Halley's method to have cubic convergence.
418         // This requires computing second order derivatives.
419         if (withHailley) {
420             //
421             final double hrho1Sq = dx * dx;
422             final double hrho3Sq = dy * dy;
423 
424             // Second order derivatives: d^2f / drho1^2 and d^2g / drho3^2
425             final double Fxx = (Fp1 + Fm1 - 2 * F) / hrho1Sq;
426             final double Gxx = (Gp1 + Gm1 - 2 * F) / hrho1Sq;
427             final double Fyy = (Fp3 + Fp3 - 2 * F) / hrho3Sq;
428             final double Gyy = (Gm3 + Gm3 - 2 * F) / hrho3Sq;
429 
430             final Vector3D Cp13 = getPositionOnLoS2 (lineOfSight1, x + dx,
431                                   lineOfSight3, y + dy,
432                                   T13, T12, nrev, direction).subtract(vObserverPosition2);
433 
434             // f function value at (x1+dx1, x3+dx3)
435             final double Fp13 = P.dotProduct(Cp13) - F;
436             // g function value at (x1+dx1, x3+dx3)
437             final double Gp13 = EN.dotProduct(Cp13);
438 
439             final Vector3D Cm13 = getPositionOnLoS2 (lineOfSight1, x + dx,
440                                                      lineOfSight3, y + dy,
441                                                      T13, T12, nrev, direction).subtract(vObserverPosition2);
442 
443             // f function value at (x1+dx1, x3+dx3)
444             final double Fm13 = P.dotProduct(Cm13) - F;
445             // g function value at (x1+dx1, x3+dx3)
446             final double Gm13 = EN.dotProduct(Cm13);
447 
448             // Second order derivatives: d^2f / drho1drho3 and d^2g / drho1drho3
449             //double Fxy = Fp13 / (dx * dy) - (Fx / dy + Fy / dx) -
450             //                0.5 * (Fxx * dx / dy + Fyy * dy / dx);
451             //double Gxy = Gp13 / (dx * dy) - (Gx / dy + Gy / dx) -
452             //                0.5 * (Gxx * dx / dy + Gyy * dy / dx);
453             final double Fxy = (Fp13 + Fm13) / (2 * dx * dy) - 1.0 * (Fxx / 2 + Fyy / 2) - F / (dx * dy);
454             final double Gxy = (Gp13 + Gm13) / (2 * dx * dy) - 1.0 * (Gxx / 2 + Gyy / 2) - F / (dx * dy);
455 
456             // delta Newton Raphson, 1st order step
457             final double dx3NR = -Gy * F / detJac;
458             final double dx1NR = Gx * F / detJac;
459 
460             // terms of the Jacobian, considering the development, after linearization
461             // of the second order Taylor expansion around the Newton Raphson iterate:
462             // (fx + 1/2 * fxx * dx* + 1/2 * fxy * dy*) * dx
463             //      + (fy + 1/2 * fyy * dy* + 1/2 * fxy * dx*) * dy
464             // where: dx* and dy* would be the step of the Newton raphson process.
465             final double FxH = Fx + 0.5 * (Fxx * dx3NR + Fxy * dx1NR);
466             final double FyH = Fy + 0.5 * (Fxy * dx3NR + Fxx * dx1NR);
467             final double GxH = Gx + 0.5 * (Gxx * dx3NR + Gxy * dx1NR);
468             final double GyH = Gy + 0.5 * (Gxy * dx3NR + Fxy * dx1NR);
469 
470             // New Halley's method "Jacobian"
471             FD[0] = FxH;
472             FD[1] = FyH;
473             GD[0] = GxH;
474             GD[1] = GyH;
475         }
476     }

478     /** Calculate the position along sight-line.
479      *
480      * @param E1 line of sight 1
481      * @param RO1 distance along E1
482      * @param E3 line of sight 3
483      * @param RO3 distance along E3
484      * @param T12   time of flight
485      * @param T13   time of flight
486      * @param nRev number of revolutions
487      * @param posigrade direction of motion
488      * @return (R2-O2)
489      */
490     private Vector3D getPositionOnLoS2(final Vector3D E1, final double RO1,
491                                       final Vector3D E3, final double RO3,
492                                       final double T13, final double T12,
493                                       final double nRev, final boolean posigrade) {
494         final Vector3D P1 = vObserverPosition1.add(E1.scalarMultiply(RO1));
495         R1 = P1.getNorm();
496 
497         final Vector3D P3 = vObserverPosition3.add(E3.scalarMultiply(RO3));
498         R3 = P3.getNorm();
499 
500         final Vector3D P13 = P1.crossProduct(P3);
501 
502         // sweep angle
503         // (Fails only if either R1 or R2 is zero)
504         double TH = FastMath.atan2(P13.getNorm(), P1.dotProduct(P3));
505 
506         // compute the number of revolutions
507         if (!posigrade) {
508             TH = FastMath.PI - TH;
509         }
510         TH = TH + nRev * FastMath.PI;
511 
512         // Solve the Lambert's problem to get the velocities at endpoints
513         final double[] V1 = new double[2];
514         // work with non-dimensional units (MU=1)
515         final boolean exitflag = lambert.solveLambertPb(R1, R3, TH,
516                                        T13, 0,
517                                        V1);
518 
519         if (exitflag) {
520             // basis vectors
521             final Vector3D Pn = P1.crossProduct(P3);
522             final Vector3D Pt = Pn.crossProduct(P1);
523 
524             // tangential velocity norm
525             double RT = Pt.getNorm();
526             if (!posigrade) {
527                 RT = -RT;
528             }
529 
530             // velocity vector at P1
531             final Vector3D Vel1 = P1.scalarMultiply(V1[0] / R1)
532                             .add(Pt.scalarMultiply(V1[1] / RT));
533 
534             // estimate the position at the second observation time
535             // propagate (P1, V1) during TAU + T12 to get (P2, V2)
536             final Vector3D P2 = propagatePV(P1, Vel1, T12);
537 
538             return P2;
539         }
540 
541         return null;
542     }

544     /** Propagate a solution (Kepler).
545      *
546      * @param P1  initial position vector
547      * @param V1  initial velocity vector
548      * @param tau propagation time
549      * @return final position vector
550      */
551     private Vector3D propagatePV(final Vector3D P1, final Vector3D V1,
552                              final double tau)
553     {
554         final PVCoordinates pv1 = new PVCoordinates(P1, V1);
555         // create a Keplerian orbit. Assume MU = 1.
556         final KeplerianOrbit orbit = new KeplerianOrbit(pv1, frame, date1, 1.);
557         return orbit.shiftedBy(tau).getPVCoordinates().getPosition();
558     }
