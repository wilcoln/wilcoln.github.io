60      @Test
61      public void testDayNightSwitch() throws OrekitException {
62          //  Initial state definition : date, orbit
63          final AbsoluteDate initialDate = new AbsoluteDate(2004, 01, 01, 23, 30, 00.000, TimeScalesFactory.getUTC());
64          final Vector3D position  = new Vector3D(-6142438.668, 3492467.560, -25767.25680);
65          final Vector3D velocity  = new Vector3D(505.8479685, 942.7809215, 7435.922231);
66          final Orbit initialOrbit = new KeplerianOrbit(new PVCoordinates(position, velocity),
67                                                        FramesFactory.getEME2000(), initialDate,
68                                                        Constants.EIGEN5C_EARTH_MU);
69  
70          final
71  
72          // Attitudes sequence definition
73          EventsLogger logger = new EventsLogger();
74          final AttitudesSequence attitudesSequence = new AttitudesSequence();
75          final AttitudeProvider dayObservationLaw = new LofOffset(initialOrbit.getFrame(), LOFType.VVLH,
76                                                                   RotationOrder.XYZ, FastMath.toRadians(20), FastMath.toRadians(40), 0);
77          final AttitudeProvider nightRestingLaw   = new LofOffset(initialOrbit.getFrame(), LOFType.VVLH);
78          final PVCoordinatesProvider sun = CelestialBodyFactory.getSun();
79          final PVCoordinatesProvider earth = CelestialBodyFactory.getEarth();
80          final EclipseDetector ed =
81                  new EclipseDetector(sun, 696000000., earth, Constants.WGS84_EARTH_EQUATORIAL_RADIUS).
82                  withHandler(new ContinueOnEvent<EclipseDetector>() {
83                      private static final long serialVersionUID = 1L;
84                      public EventHandler.Action eventOccurred(final SpacecraftState s, final EclipseDetector d, final boolean increasing) {
85                          setInEclipse(s.getDate(), !increasing);
86                          return EventHandler.Action.RESET_STATE;
87                      }
88                  });
89          final EventDetector monitored = logger.monitorDetector(ed);
90          final Handler dayToNightHandler = new Handler(dayObservationLaw, nightRestingLaw);
91          final Handler nightToDayHandler = new Handler(nightRestingLaw, dayObservationLaw);
92          attitudesSequence.addSwitchingCondition(dayObservationLaw, nightRestingLaw,
93                                                  monitored, false, true, 300.0,
94                                                  AngularDerivativesFilter.USE_RRA, dayToNightHandler);
95          attitudesSequence.addSwitchingCondition(nightRestingLaw, dayObservationLaw,
96                                                  monitored, true, false, 300.0,
97                                                  AngularDerivativesFilter.USE_RRA, nightToDayHandler);
98          SpacecraftState initialState = new SpacecraftState(initialOrbit);
99          initialState = initialState.addAdditionalState("fortyTwo", 42.0);
100         if (ed.g(initialState) >= 0) {
101             // initial position is in daytime
102             setInEclipse(initialDate, false);
103             attitudesSequence.resetActiveProvider(dayObservationLaw);
104         } else {
105             // initial position is in nighttime
106             setInEclipse(initialDate, true);
107             attitudesSequence.resetActiveProvider(nightRestingLaw);
108         }
109 
110         // Propagator : consider the analytical Eckstein-Hechler model
111         final Propagator propagator = new EcksteinHechlerPropagator(initialOrbit, attitudesSequence,
112                                                                     Constants.EIGEN5C_EARTH_EQUATORIAL_RADIUS,
113                                                                     Constants.EIGEN5C_EARTH_MU,  Constants.EIGEN5C_EARTH_C20,
114                                                                     Constants.EIGEN5C_EARTH_C30, Constants.EIGEN5C_EARTH_C40,
115                                                                     Constants.EIGEN5C_EARTH_C50, Constants.EIGEN5C_EARTH_C60);
116 
117         // Register the switching events to the propagator
118         attitudesSequence.registerSwitchEvents(propagator);
119 
120         propagator.setMasterMode(60.0, new OrekitFixedStepHandler() {
121             public void init(final SpacecraftState s0, final AbsoluteDate t) {
122             }
123             public void handleStep(SpacecraftState currentState, boolean isLast) throws OrekitException {
124                 // the Earth position in spacecraft frame should be along spacecraft Z axis
125                 // during night time and away from it during day time due to roll and pitch offsets
126                 final Vector3D earth = currentState.toTransform().transformPosition(Vector3D.ZERO);
127                 final double pointingOffset = Vector3D.angle(earth, Vector3D.PLUS_K);
128 
129                 // the g function is the eclipse indicator, its an angle between Sun and Earth limb,
130                 // positive when Sun is outside of Earth limb, negative when Sun is hidden by Earth limb
131                 final double eclipseAngle = ed.g(currentState);
132 
133                 if (currentState.getDate().durationFrom(lastChange) > 300) {
134                     if (inEclipse) {
135                         Assert.assertTrue(eclipseAngle <= 0);
136                         Assert.assertEquals(0.0, pointingOffset, 1.0e-6);
137                     } else {
138                         Assert.assertTrue(eclipseAngle >= 0);
139                         Assert.assertEquals(0.767215, pointingOffset, 1.0e-6);
140                     }
141                 } else {
142                     // we are in transition
143                     Assert.assertTrue(pointingOffset + " " + (0.767215 - pointingOffset),
144                                       pointingOffset <= 0.7672155);
145                 }
146             }
147         });
148 
149         // Propagate from the initial date for the fixed duration
150         propagator.propagate(initialDate.shiftedBy(12600.));
151 
152         // as we have 2 switch events (even if they share the same underlying event detector),
153         // and these events are triggered at both eclipse entry and exit, we get 8
154         // raw events on 2 orbits
155         Assert.assertEquals(8, logger.getLoggedEvents().size());
156 
157         // we have 4 attitudes switch on 2 orbits, 2 of each type
158         Assert.assertEquals(2, dayToNightHandler.dates.size());
159         Assert.assertEquals(2, nightToDayHandler.dates.size());
160 
161     }

163     @Test
164     public void testBackwardPropagation() throws OrekitException {
165 
166         //  Initial state definition : date, orbit
167         final AbsoluteDate initialDate = new AbsoluteDate(2004, 01, 01, 23, 30, 00.000, TimeScalesFactory.getUTC());
168         final Vector3D position  = new Vector3D(-6142438.668, 3492467.560, -25767.25680);
169         final Vector3D velocity  = new Vector3D(505.8479685, 942.7809215, 7435.922231);
170         final Orbit initialOrbit = new KeplerianOrbit(new PVCoordinates(position, velocity),
171                                                       FramesFactory.getEME2000(), initialDate,
172                                                       Constants.EIGEN5C_EARTH_MU);
173 
174         final AttitudesSequence attitudesSequence = new AttitudesSequence();
175         final AttitudeProvider past    = new InertialProvider(Rotation.IDENTITY);
176         final AttitudeProvider current = new InertialProvider(Rotation.IDENTITY);
177         final AttitudeProvider future  = new InertialProvider(Rotation.IDENTITY);
178         final Handler handler = new Handler(current, past);
179         attitudesSequence.addSwitchingCondition(past, current,
180                                                 new DateDetector(initialDate.shiftedBy(-500.0)),
181                                                 true, false, 10.0, AngularDerivativesFilter.USE_R, handler);
182         attitudesSequence.addSwitchingCondition(current, future,
183                                                 new DateDetector(initialDate.shiftedBy(+500.0)),
184                                                 true, false, 10.0, AngularDerivativesFilter.USE_R, null);
185         attitudesSequence.resetActiveProvider(current);
186 
187         SpacecraftState initialState = new SpacecraftState(initialOrbit);
188         initialState = initialState.addAdditionalState("fortyTwo", 42.0);
189         final Propagator propagator = new EcksteinHechlerPropagator(initialOrbit, attitudesSequence,
190                                                                     Constants.EIGEN5C_EARTH_EQUATORIAL_RADIUS,
191                                                                     Constants.EIGEN5C_EARTH_MU,  Constants.EIGEN5C_EARTH_C20,
192                                                                     Constants.EIGEN5C_EARTH_C30, Constants.EIGEN5C_EARTH_C40,
193                                                                     Constants.EIGEN5C_EARTH_C50, Constants.EIGEN5C_EARTH_C60);
194         propagator.resetInitialState(initialState);
195         Assert.assertEquals(42.0, propagator.getInitialState().getAdditionalState("fortyTwo")[0], 1.0e-10);
196 
197         // Register the switching events to the propagator
198         attitudesSequence.registerSwitchEvents(propagator);
199 
200         SpacecraftState finalState = propagator.propagate(initialDate.shiftedBy(-10000.0));
201         Assert.assertEquals(42.0, finalState.getAdditionalState("fortyTwo")[0], 1.0e-10);
202         Assert.assertEquals(1, handler.dates.size());
203         Assert.assertEquals(-500.0, handler.dates.get(0).durationFrom(initialDate), 1.0e-3);
204         Assert.assertEquals(-490.0, finalState.getDate().durationFrom(initialDate), 1.0e-3);
205 
206     }

208     @Test
209     public void testTooShortTransition() {
210         double threshold      = 1.5;
211         double transitionTime = 0.5;
212         try {
213             new AttitudesSequence().addSwitchingCondition(new InertialProvider(Rotation.IDENTITY),
214                                                           new InertialProvider(Rotation.IDENTITY),
215                                                           new DateDetector(1000.0, threshold,
216                                                                            AbsoluteDate.J2000_EPOCH),
217                                                           true, false, transitionTime,
218                                                           AngularDerivativesFilter.USE_R, null);
219             Assert.fail("an exception should have been thrown");
220         } catch (OrekitException oe) {
221             Assert.assertEquals(OrekitMessages.TOO_SHORT_TRANSITION_TIME_FOR_ATTITUDES_SWITCH,
222                                 oe.getSpecifier());
223             Assert.assertEquals(transitionTime, ((Double) oe.getParts()[0]).doubleValue(), 1.0e-10);
224             Assert.assertEquals(threshold,      ((Double) oe.getParts()[1]).doubleValue(), 1.0e-10);
225         }
226     }

228     private static class Handler implements AttitudesSequence.SwitchHandler {
229 
230         private AttitudeProvider   expectedPrevious;
231         private AttitudeProvider   expectedNext;
232         private List<AbsoluteDate> dates;
233 
234         public Handler(final AttitudeProvider expectedPrevious, final AttitudeProvider expectedNext) {
235             this.expectedPrevious = expectedPrevious;
236             this.expectedNext     = expectedNext;
237             this.dates            = new ArrayList<AbsoluteDate>();
238         }
239 
240         @Override
241         public void switchOccurred(AttitudeProvider previous, AttitudeProvider next,
242                                    SpacecraftState state) {
243             Assert.assertTrue(previous == expectedPrevious);
244             Assert.assertTrue(next     == expectedNext);
245             dates.add(state.getDate());
246         }
247 
248     }

250     private void setInEclipse(AbsoluteDate lastChange, boolean inEclipse) {
251         this.lastChange = lastChange;
252         this.inEclipse = inEclipse;
253     }
