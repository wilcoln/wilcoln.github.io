47  /** This class propagates a {@link org.orekit.propagation.SpacecraftState}
48   *  using the analytical Eckstein-Hechler model.
49   * <p>The Eckstein-Hechler model is suited for near circular orbits
50   * (e &lt; 0.1, with poor accuracy between 0.005 and 0.1) and inclination
51   * neither equatorial (direct or retrograde) nor critical (direct or
52   * retrograde).</p>
53   * <p>
54   * Note that before version 7.0, there was a large inconsistency in the generated
55   * orbits, and it was fixed as of version 7.0 of Orekit, with a visible side effect.
56   * The problems is that if the circular parameters produced by the Eckstein-Hechler
57   * model are used to build an orbit considered to be osculating, the velocity deduced
58   * from this orbit was <em>inconsistent with the position evolution</em>! The reason is
59   * that the model includes non-Keplerian effects but it does not include a corresponding
60   * circular/Cartesian conversion. As a consequence, all subsequent computation involving
61   * velocity were wrong. This includes attitude modes like yaw compensation and Doppler
62   * effect. As this effect was considered serious enough and as accurate velocities were
63   * considered important, the propagator now generates {@link CartesianOrbit Cartesian
64   * orbits} which are built in a special way to ensure consistency throughout propagation.
65   * A side effect is that if circular parameters are rebuilt by user from these propagated
66   * Cartesian orbit, the circular parameters will generally <em>not</em> match the initial
67   * orbit (differences in semi-major axis can exceed 120 m). The position however <em>will</em>
68   * match to sub-micrometer level, and this position will be identical to the positions
69   * that were generated by previous versions (in other words, the internals of the models
70   * have not been changed, only the output parameters have been changed). The correctness
71   * of the initialization has been assessed and is good, as it allows the subsequent orbit
72   * to remain close to a numerical reference orbit.
73   * </p>
74   * <p>
75   * If users need a more definitive initialization of an Eckstein-Hechler propagator, they
76   * should consider using a {@link org.orekit.propagation.conversion.PropagatorConverter
77   * propagator converter} to initialize their Eckstein-Hechler propagator using a complete
78   * sample instead of just a single initial orbit.
79   * </p>
80   * @see Orbit
81   * @author Guylaine Prat
82   */
83  public class EcksteinHechlerPropagator extends AbstractAnalyticalPropagator implements Serializable {

346     /** Compute mean parameters according to the Eckstein-Hechler analytical model.
347      * @param osculating osculating orbit
348      * @param mass constant mass
349      * @return Eckstein-Hechler mean model
350      * @exception OrekitException if orbit goes outside of supported range
351      * (trajectory inside the Brillouin sphere, too eccentric, equatorial, critical
352      * inclination) or if convergence cannot be reached
353      */
354     private EHModel computeMeanParameters(final CircularOrbit osculating, final double mass)
355         throws OrekitException {
356 
357         // sanity check
358         if (osculating.getA() < referenceRadius) {
359             throw new OrekitException(OrekitMessages.TRAJECTORY_INSIDE_BRILLOUIN_SPHERE,
360                                            osculating.getA());
361         }
362 
363         // rough initialization of the mean parameters
364         EHModel current = new EHModel(osculating, mass, referenceRadius, mu, ck0);
365 
366         // threshold for each parameter
367         final double epsilon         = 1.0e-13;
368         final double thresholdA      = epsilon * (1 + FastMath.abs(current.mean.getA()));
369         final double thresholdE      = epsilon * (1 + current.mean.getE());
370         final double thresholdAngles = epsilon * FastMath.PI;
371 
372         int i = 0;
373         while (i++ < 100) {
374 
375             // recompute the osculating parameters from the current mean parameters
376             final DerivativeStructure[] parameters = current.propagateParameters(current.mean.getDate());
377 
378             // adapted parameters residuals
379             final double deltaA      = osculating.getA()          - parameters[0].getValue();
380             final double deltaEx     = osculating.getCircularEx() - parameters[1].getValue();
381             final double deltaEy     = osculating.getCircularEy() - parameters[2].getValue();
382             final double deltaI      = osculating.getI()          - parameters[3].getValue();
383             final double deltaRAAN   = MathUtils.normalizeAngle(osculating.getRightAscensionOfAscendingNode() -
384                                                                 parameters[4].getValue(),
385                                                                 0.0);
386             final double deltaAlphaM = MathUtils.normalizeAngle(osculating.getAlphaM() - parameters[5].getValue(), 0.0);
387 
388             // update mean parameters
389             current = new EHModel(new CircularOrbit(current.mean.getA()          + deltaA,
390                                                     current.mean.getCircularEx() + deltaEx,
391                                                     current.mean.getCircularEy() + deltaEy,
392                                                     current.mean.getI()          + deltaI,
393                                                     current.mean.getRightAscensionOfAscendingNode() + deltaRAAN,
394                                                     current.mean.getAlphaM()     + deltaAlphaM,
395                                                     PositionAngle.MEAN,
396                                                     current.mean.getFrame(),
397                                                     current.mean.getDate(), mu),
398                                   mass, referenceRadius, mu, ck0);
399 
400             // check convergence
401             if ((FastMath.abs(deltaA)      < thresholdA) &&
402                 (FastMath.abs(deltaEx)     < thresholdE) &&
403                 (FastMath.abs(deltaEy)     < thresholdE) &&
404                 (FastMath.abs(deltaI)      < thresholdAngles) &&
405                 (FastMath.abs(deltaRAAN)   < thresholdAngles) &&
406                 (FastMath.abs(deltaAlphaM) < thresholdAngles)) {
407                 return current;
408             }
409 
410         }
411 
412         throw new OrekitException(OrekitMessages.UNABLE_TO_COMPUTE_ECKSTEIN_HECHLER_MEAN_PARAMETERS, i);
413 
414     }

416     /** {@inheritDoc} */
417     public CartesianOrbit propagateOrbit(final AbsoluteDate date)
418         throws OrekitException {
419         // compute Cartesian parameters, taking derivatives into account
420         // to make sure velocity and acceleration are consistent
421         final EHModel current = models.get(date);
422         return new CartesianOrbit(toCartesian(date, current.propagateParameters(date)),
423                                   current.mean.getFrame(), mu);
424     }

496         /** Create a model for specified mean orbit.
497          * @param mean mean orbit
498          * @param mass constant mass
499          * @param referenceRadius reference radius of the central body attraction model (m)
500          * @param mu central attraction coefficient (m³/s²)
501          * @param ck0 un-normalized zonal coefficients
502          * @exception OrekitException if mean orbit is not within model supported domain
503          */
504         EHModel(final CircularOrbit mean, final double mass,
505                 final double referenceRadius, final double mu, final double[] ck0)
506             throws OrekitException {
507 
508             this.mean            = mean;
509             this.mass            = mass;
510 
511             // preliminary processing
512             double q = referenceRadius / mean.getA();
513             double ql = q * q;
514             final double g2 = ck0[2] * ql;
515             ql *= q;
516             final double g3 = ck0[3] * ql;
517             ql *= q;
518             final double g4 = ck0[4] * ql;
519             ql *= q;
520             final double g5 = ck0[5] * ql;
521             ql *= q;
522             final double g6 = ck0[6] * ql;
523 
524             final double cosI1 = FastMath.cos(mean.getI());
525             final double sinI1 = FastMath.sin(mean.getI());
526             final double sinI2 = sinI1 * sinI1;
527             final double sinI4 = sinI2 * sinI2;
528             final double sinI6 = sinI2 * sinI4;
529 
530             if (sinI2 < 1.0e-10) {
531                 throw new OrekitException(OrekitMessages.ALMOST_EQUATORIAL_ORBIT,
532                                           FastMath.toDegrees(mean.getI()));
533             }
534 
535             if (FastMath.abs(sinI2 - 4.0 / 5.0) < 1.0e-3) {
536                 throw new OrekitException(OrekitMessages.ALMOST_CRITICALLY_INCLINED_ORBIT,
537                                           FastMath.toDegrees(mean.getI()));
538             }
539 
540             if (mean.getE() > 0.1) {
541                 // if 0.005 < e < 0.1 no error is triggered, but accuracy is poor
542                 throw new OrekitException(OrekitMessages.TOO_LARGE_ECCENTRICITY_FOR_PROPAGATION_MODEL,
543                                           mean.getE());
544             }
545 
546             xnotDot = FastMath.sqrt(mu / mean.getA()) / mean.getA();
547 
548             rdpom = -0.75 * g2 * (4.0 - 5.0 * sinI2);
549             rdpomp = 7.5 * g4 * (1.0 - 31.0 / 8.0 * sinI2 + 49.0 / 16.0 * sinI4) -
550                     13.125 * g6 * (1.0 - 8.0 * sinI2 + 129.0 / 8.0 * sinI4 - 297.0 / 32.0 * sinI6);
551 
552             q = 3.0 / (32.0 * rdpom);
553             eps1 = q * g4 * sinI2 * (30.0 - 35.0 * sinI2) -
554                     175.0 * q * g6 * sinI2 * (1.0 - 3.0 * sinI2 + 2.0625 * sinI4);
555             q = 3.0 * sinI1 / (8.0 * rdpom);
556             eps2 = q * g3 * (4.0 - 5.0 * sinI2) - q * g5 * (10.0 - 35.0 * sinI2 + 26.25 * sinI4);
557 
558             xim = mean.getI();
559             ommD = cosI1 * (1.50    * g2 - 2.25 * g2 * g2 * (2.5 - 19.0 / 6.0 * sinI2) +
560                             0.9375  * g4 * (7.0 * sinI2 - 4.0) +
561                             3.28125 * g6 * (2.0 - 9.0 * sinI2 + 8.25 * sinI4));
562 
563             rdl = 1.0 - 1.50 * g2 * (3.0 - 4.0 * sinI2);
564             aMD = rdl +
565                     2.25 * g2 * g2 * (9.0 - 263.0 / 12.0 * sinI2 + 341.0 / 24.0 * sinI4) +
566                     15.0 / 16.0 * g4 * (8.0 - 31.0 * sinI2 + 24.5 * sinI4) +
567                     105.0 / 32.0 * g6 * (-10.0 / 3.0 + 25.0 * sinI2 - 48.75 * sinI4 + 27.5 * sinI6);
568 
569             final double qq = -1.5 * g2 / rdl;
570             final double qA   = 0.75 * g2 * g2 * sinI2;
571             final double qB   = 0.25 * g4 * sinI2;
572             final double qC   = 105.0 / 16.0 * g6 * sinI2;
573             final double qD   = -0.75 * g3 * sinI1;
574             final double qE   = 3.75 * g5 * sinI1;
575             kh = 0.375 / rdpom;
576             kl = kh / sinI1;
577 
578             ax1 = qq * (2.0 - 3.5 * sinI2);
579             ay1 = qq * (2.0 - 2.5 * sinI2);
580             as1 = qD * (4.0 - 5.0 * sinI2) +
581                   qE * (2.625 * sinI4 - 3.5 * sinI2 + 1.0);
582             ac2 = qq * sinI2 +
583                   qA * 7.0 * (2.0 - 3.0 * sinI2) +
584                   qB * (15.0 - 17.5 * sinI2) +
585                   qC * (3.0 * sinI2 - 1.0 - 33.0 / 16.0 * sinI4);
586             axy3 = qq * 3.5 * sinI2;
587             as3 = qD * 5.0 / 3.0 * sinI2 +
588                   qE * 7.0 / 6.0 * sinI2 * (1.0 - 1.125 * sinI2);
589             ac4 = qA * sinI2 +
590                   qB * 4.375 * sinI2 +
591                   qC * 0.75 * (1.1 * sinI4 - sinI2);
592 
593             as5 = qE * 21.0 / 80.0 * sinI4;
594 
595             ac6 = qC * -11.0 / 80.0 * sinI4;
596 
597             ex1 = qq * (1.0 - 1.25 * sinI2);
598             exx2 = qq * 0.5 * (3.0 - 5.0 * sinI2);
599             exy2 = qq * (2.0 - 1.5 * sinI2);
600             ex3 = qq * 7.0 / 12.0 * sinI2;
601             ex4 = qq * 17.0 / 8.0 * sinI2;
602 
603             ey1 = qq * (1.0 - 1.75 * sinI2);
604             eyx2 = qq * (1.0 - 3.0 * sinI2);
605             eyy2 = qq * (2.0 * sinI2 - 1.5);
606             ey3 = qq * 7.0 / 12.0 * sinI2;
607             ey4 = qq * 17.0 / 8.0 * sinI2;
608 
609             q  = -qq * cosI1;
610             rx1 =  3.5 * q;
611             ry1 = -2.5 * q;
612             r2 = -0.5 * q;
613             r3 =  7.0 / 6.0 * q;
614             rl = g3 * cosI1 * (4.0 - 15.0 * sinI2) -
615                  2.5 * g5 * cosI1 * (4.0 - 42.0 * sinI2 + 52.5 * sinI4);
616 
617             q = 0.5 * qq * sinI1 * cosI1;
618             iy1 =  q;
619             ix1 = -q;
620             i2 =  q;
621             i3 =  q * 7.0 / 3.0;
622             ih = -g3 * cosI1 * (4.0 - 5.0 * sinI2) +
623                  2.5 * g5 * cosI1 * (4.0 - 14.0 * sinI2 + 10.5 * sinI4);
624 
625             lx1 = qq * (7.0 - 77.0 / 8.0 * sinI2);
626             ly1 = qq * (55.0 / 8.0 * sinI2 - 7.50);
627             l2 = qq * (1.25 * sinI2 - 0.5);
628             l3 = qq * (77.0 / 24.0 * sinI2 - 7.0 / 6.0);
629             ll = g3 * (53.0 * sinI2 - 4.0 - 57.5 * sinI4) +
630                  2.5 * g5 * (4.0 - 96.0 * sinI2 + 269.5 * sinI4 - 183.75 * sinI6);
631 
632         }

634         /** Extrapolate an orbit up to a specific target date.
635          * @param date target date for the orbit
636          * @return propagated parameters
637          * @exception OrekitException if some parameters are out of bounds
638          */
639         public DerivativeStructure[] propagateParameters(final AbsoluteDate date)
640             throws OrekitException {
641 
642             // keplerian evolution
643             final DerivativeStructure dt =
644                     new DerivativeStructure(1, 2, 0, date.durationFrom(mean.getDate()));
645             final DerivativeStructure xnot = dt.multiply(xnotDot);
646 
647             // secular effects
648 
649             // eccentricity
650             final DerivativeStructure x   = xnot.multiply(rdpom + rdpomp);
651             final DerivativeStructure cx  = x.cos();
652             final DerivativeStructure sx  = x.sin();
653             final DerivativeStructure exm = cx.multiply(mean.getCircularEx()).
654                                             add(sx.multiply(eps2 - (1.0 - eps1) * mean.getCircularEy()));
655             final DerivativeStructure eym = sx.multiply((1.0 + eps1) * mean.getCircularEx()).
656                                             add(cx.multiply(mean.getCircularEy() - eps2)).
657                                             add(eps2);
658 
659             // no secular effect on inclination
660 
661             // right ascension of ascending node
662             final DerivativeStructure omm =
663                     new DerivativeStructure(1, 2,
664                                             MathUtils.normalizeAngle(mean.getRightAscensionOfAscendingNode() + ommD * xnot.getValue(),
665                                                                      FastMath.PI),
666                                             ommD * xnotDot,
667                                             0.0);
668 
669             // latitude argument
670             final DerivativeStructure xlm =
671                     new DerivativeStructure(1, 2,
672                                             MathUtils.normalizeAngle(mean.getAlphaM() + aMD * xnot.getValue(), FastMath.PI),
673                                             aMD * xnotDot,
674                                             0.0);
675 
676             // periodical terms
677             final DerivativeStructure cl1 = xlm.cos();
678             final DerivativeStructure sl1 = xlm.sin();
679             final DerivativeStructure cl2 = cl1.multiply(cl1).subtract(sl1.multiply(sl1));
680             final DerivativeStructure sl2 = cl1.multiply(sl1).add(sl1.multiply(cl1));
681             final DerivativeStructure cl3 = cl2.multiply(cl1).subtract(sl2.multiply(sl1));
682             final DerivativeStructure sl3 = cl2.multiply(sl1).add(sl2.multiply(cl1));
683             final DerivativeStructure cl4 = cl3.multiply(cl1).subtract(sl3.multiply(sl1));
684             final DerivativeStructure sl4 = cl3.multiply(sl1).add(sl3.multiply(cl1));
685             final DerivativeStructure cl5 = cl4.multiply(cl1).subtract(sl4.multiply(sl1));
686             final DerivativeStructure sl5 = cl4.multiply(sl1).add(sl4.multiply(cl1));
687             final DerivativeStructure cl6 = cl5.multiply(cl1).subtract(sl5.multiply(sl1));
688 
689             final DerivativeStructure qh  = eym.subtract(eps2).multiply(kh);
690             final DerivativeStructure ql  = exm.multiply(kl);
691 
692             final DerivativeStructure exmCl1 = exm.multiply(cl1);
693             final DerivativeStructure exmSl1 = exm.multiply(sl1);
694             final DerivativeStructure eymCl1 = eym.multiply(cl1);
695             final DerivativeStructure eymSl1 = eym.multiply(sl1);
696             final DerivativeStructure exmCl2 = exm.multiply(cl2);
697             final DerivativeStructure exmSl2 = exm.multiply(sl2);
698             final DerivativeStructure eymCl2 = eym.multiply(cl2);
699             final DerivativeStructure eymSl2 = eym.multiply(sl2);
700             final DerivativeStructure exmCl3 = exm.multiply(cl3);
701             final DerivativeStructure exmSl3 = exm.multiply(sl3);
702             final DerivativeStructure eymCl3 = eym.multiply(cl3);
703             final DerivativeStructure eymSl3 = eym.multiply(sl3);
704             final DerivativeStructure exmCl4 = exm.multiply(cl4);
705             final DerivativeStructure exmSl4 = exm.multiply(sl4);
706             final DerivativeStructure eymCl4 = eym.multiply(cl4);
707             final DerivativeStructure eymSl4 = eym.multiply(sl4);
708 
709             // semi major axis
710             final DerivativeStructure rda = exmCl1.multiply(ax1).
711                                             add(eymSl1.multiply(ay1)).
712                                             add(sl1.multiply(as1)).
713                                             add(cl2.multiply(ac2)).
714                                             add(exmCl3.add(eymSl3).multiply(axy3)).
715                                             add(sl3.multiply(as3)).
716                                             add(cl4.multiply(ac4)).
717                                             add(sl5.multiply(as5)).
718                                             add(cl6.multiply(ac6));
719 
720             // eccentricity
721             final DerivativeStructure rdex = cl1.multiply(ex1).
722                                              add(exmCl2.multiply(exx2)).
723                                              add(eymSl2.multiply(exy2)).
724                                              add(cl3.multiply(ex3)).
725                                              add(exmCl4.add(eymSl4).multiply(ex4));
726             final DerivativeStructure rdey = sl1.multiply(ey1).
727                                              add(exmSl2.multiply(eyx2)).
728                                              add(eymCl2.multiply(eyy2)).
729                                              add(sl3.multiply(ey3)).
730                                              add(exmSl4.subtract(eymCl4).multiply(ey4));
731 
732             // ascending node
733             final DerivativeStructure rdom = exmSl1.multiply(rx1).
734                                              add(eymCl1.multiply(ry1)).
735                                              add(sl2.multiply(r2)).
736                                              add(eymCl3.subtract(exmSl3).multiply(r3)).
737                                              add(ql.multiply(rl));
738 
739             // inclination
740             final DerivativeStructure rdxi = eymSl1.multiply(iy1).
741                                              add(exmCl1.multiply(ix1)).
742                                              add(cl2.multiply(i2)).
743                                              add(exmCl3.add(eymSl3).multiply(i3)).
744                                              add(qh.multiply(ih));
745 
746             // latitude argument
747             final DerivativeStructure rdxl = exmSl1.multiply(lx1).
748                                              add(eymCl1.multiply(ly1)).
749                                              add(sl2.multiply(l2)).
750                                              add(exmSl3.subtract(eymCl3).multiply(l3)).
751                                              add(ql.multiply(ll));
752 
753             // osculating parameters
754             return new DerivativeStructure[] {
755                 rda.add(1.0).multiply(mean.getA()),
756                 rdex.add(exm),
757                 rdey.add(eym),
758                 rdxi.add(xim),
759                 rdom.add(omm),
760                 rdxl.add(xlm)
761             };
762 
763         }
764

767     /** Convert circular parameters <em>with derivatives</em> to Cartesian coordinates.
768      * @param date date of the orbital parameters
769      * @param parameters circular parameters (a, ex, ey, i, raan, alphaM)
770      * @return Cartesian coordinates consistent with values and derivatives
771      */
772     private TimeStampedPVCoordinates toCartesian(final AbsoluteDate date, final DerivativeStructure[] parameters) {
773 
774         // evaluate coordinates in the orbit canonical reference frame
775         final DerivativeStructure cosOmega = parameters[4].cos();
776         final DerivativeStructure sinOmega = parameters[4].sin();
777         final DerivativeStructure cosI     = parameters[3].cos();
778         final DerivativeStructure sinI     = parameters[3].sin();
779         final DerivativeStructure alphaE   = meanToEccentric(parameters[5], parameters[1], parameters[2]);
780         final DerivativeStructure cosAE    = alphaE.cos();
781         final DerivativeStructure sinAE    = alphaE.sin();
782         final DerivativeStructure ex2      = parameters[1].multiply(parameters[1]);
783         final DerivativeStructure ey2      = parameters[2].multiply(parameters[2]);
784         final DerivativeStructure exy      = parameters[1].multiply(parameters[2]);
785         final DerivativeStructure q        = ex2.add(ey2).subtract(1).negate().sqrt();
786         final DerivativeStructure beta     = q.add(1).reciprocal();
787         final DerivativeStructure bx2      = beta.multiply(ex2);
788         final DerivativeStructure by2      = beta.multiply(ey2);
789         final DerivativeStructure bxy      = beta.multiply(exy);
790         final DerivativeStructure u        = bxy.multiply(sinAE).subtract(parameters[1].add(by2.subtract(1).multiply(cosAE)));
791         final DerivativeStructure v        = bxy.multiply(cosAE).subtract(parameters[2].add(bx2.subtract(1).multiply(sinAE)));
792         final DerivativeStructure x        = parameters[0].multiply(u);
793         final DerivativeStructure y        = parameters[0].multiply(v);
794 
795         // canonical orbit reference frame
796         final FieldVector3D<DerivativeStructure> p =
797                 new FieldVector3D<DerivativeStructure>(x.multiply(cosOmega).subtract(y.multiply(cosI.multiply(sinOmega))),
798                                                        x.multiply(sinOmega).add(y.multiply(cosI.multiply(cosOmega))),
799                                                        y.multiply(sinI));
800 
801         // dispatch derivatives
802         final Vector3D p0 = new Vector3D(p.getX().getValue(),
803                                          p.getY().getValue(),
804                                          p.getZ().getValue());
805         final Vector3D p1 = new Vector3D(p.getX().getPartialDerivative(1),
806                                          p.getY().getPartialDerivative(1),
807                                          p.getZ().getPartialDerivative(1));
808         final Vector3D p2 = new Vector3D(p.getX().getPartialDerivative(2),
809                                          p.getY().getPartialDerivative(2),
810                                          p.getZ().getPartialDerivative(2));
811         return new TimeStampedPVCoordinates(date, p0, p1, p2);
812 
813     }

815     /** Computes the eccentric latitude argument from the mean latitude argument.
816      * @param alphaM = M + Ω mean latitude argument (rad)
817      * @param ex e cos(Ω), first component of circular eccentricity vector
818      * @param ey e sin(Ω), second component of circular eccentricity vector
819      * @return the eccentric latitude argument.
820      */
821     private DerivativeStructure meanToEccentric(final DerivativeStructure alphaM,
822                                                 final DerivativeStructure ex,
823                                                 final DerivativeStructure ey) {
824         // Generalization of Kepler equation to circular parameters
825         // with alphaE = PA + E and
826         //      alphaM = PA + M = alphaE - ex.sin(alphaE) + ey.cos(alphaE)
827         DerivativeStructure alphaE        = alphaM;
828         DerivativeStructure shift         = alphaM.getField().getZero();
829         DerivativeStructure alphaEMalphaM = alphaM.getField().getZero();
830         DerivativeStructure cosAlphaE     = alphaE.cos();
831         DerivativeStructure sinAlphaE     = alphaE.sin();
832         int                 iter          = 0;
833         do {
834             final DerivativeStructure f2 = ex.multiply(sinAlphaE).subtract(ey.multiply(cosAlphaE));
835             final DerivativeStructure f1 = alphaM.getField().getOne().subtract(ex.multiply(cosAlphaE)).subtract(ey.multiply(sinAlphaE));
836             final DerivativeStructure f0 = alphaEMalphaM.subtract(f2);
837 
838             final DerivativeStructure f12 = f1.multiply(2);
839             shift = f0.multiply(f12).divide(f1.multiply(f12).subtract(f0.multiply(f2)));
840 
841             alphaEMalphaM  = alphaEMalphaM.subtract(shift);
842             alphaE         = alphaM.add(alphaEMalphaM);
843             cosAlphaE      = alphaE.cos();
844             sinAlphaE      = alphaE.sin();
845 
846         } while ((++iter < 50) && (FastMath.abs(shift.getValue()) > 1.0e-12));
847 
848         return alphaE;
849 
850     }

857     /** Replace the instance with a data transfer object for serialization.
858      * @return data transfer object that will be serialized
859      * @exception NotSerializableException if an additional state provider is not serializable
860      */
861     private Object writeReplace() throws NotSerializableException {
862         try {
863             // managed states providers
864             final List<AdditionalStateProvider> serializableProviders = new ArrayList<AdditionalStateProvider>();
865             for (final AdditionalStateProvider provider : getAdditionalStateProviders()) {
866                 if (provider instanceof Serializable) {
867                     serializableProviders.add(provider);
868                 } else {
869                     throw new NotSerializableException(provider.getClass().getName());
870                 }
871             }
872 
873             // states transitions
874             final AbsoluteDate[]  transitionDates;
875             final CircularOrbit[] allOrbits;
876             final double[]        allMasses;
877             final SortedSet<TimeSpanMap.Transition<EHModel>> transitions = models.getTransitions();
878             if (transitions.size() == 1  && transitions.first().getBefore() == transitions.first().getAfter()) {
879                 // the single entry is a dummy one, without a real transition
880                 // we ignore it completely
881                 transitionDates = null;
882                 allOrbits       = null;
883                 allMasses       = null;
884             } else {
885                 transitionDates = new AbsoluteDate[transitions.size()];
886                 allOrbits       = new CircularOrbit[transitions.size() + 1];
887                 allMasses       = new double[transitions.size() + 1];
888                 int i = 0;
889                 for (final TimeSpanMap.Transition<EHModel> transition : transitions) {
890                     if (i == 0) {
891                         // model before the first transition
892                         allOrbits[i] = transition.getBefore().mean;
893                         allMasses[i] = transition.getBefore().mass;
894                     }
895                     transitionDates[i] = transition.getDate();
896                     allOrbits[++i]     = transition.getAfter().mean;
897                     allMasses[i]       = transition.getAfter().mass;
898                 }
899             }
900 
901             return new DataTransferObject(getInitialState().getOrbit(), initialModel.mass,
902                                           referenceRadius, mu, ck0, getAttitudeProvider(),
903                                           transitionDates, allOrbits, allMasses,
904                                           serializableProviders.toArray(new AdditionalStateProvider[serializableProviders.size()]));
905         } catch (OrekitException orekitException) {
906             // this should never happen
907             throw new OrekitInternalError(null);
908         }
909 
910     }

971         /** Replace the deserialized data transfer object with a {@link EcksteinHechlerPropagator}.
972          * @return replacement {@link EcksteinHechlerPropagator}
973          */
974         private Object readResolve() {
975             try {
976                 final EcksteinHechlerPropagator propagator =
977                                 new EcksteinHechlerPropagator(orbit, attitudeProvider,
978                                                               g[0], g[1], g[2],              // mass, referenceRadius, mu
979                                                               g[3], g[4], g[5], g[6], g[7]); // c20, c30, c40, c50, c60
980                 for (final AdditionalStateProvider provider : providers) {
981                     propagator.addAdditionalStateProvider(provider);
982 
983                 }
984                 if (transitionDates != null) {
985                     // override the state transitions
986                     final double[] ck0 = new double[] {
987                         0, 0, g[3], g[4], g[5], g[6], g[7]
988                     };
989                     propagator.models = new TimeSpanMap<EHModel>(new EHModel(allOrbits[0], allMasses[0],
990                                                                              g[1], g[2], ck0));
991                     for (int i = 0; i < transitionDates.length; ++i) {
992                         propagator.models.addValidAfter(new EHModel(allOrbits[i + 1], allMasses[i + 1],
993                                                                     g[1], g[2], ck0),
994                                                         transitionDates[i]);
995                     }
996                 }
997 
998                 return propagator;
999             } catch (OrekitException oe) {
1000                 throw new OrekitInternalError(oe);
1001             }
1002         }
