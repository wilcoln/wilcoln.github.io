6   package spoon.reflect.visitor.filter;
7   
8   import spoon.reflect.declaration.CtMethod;
9   import spoon.reflect.visitor.Filter;
10  
11  /**
12   * Gets all overriding method from the method given.
13   */
14  public class OverridingMethodFilter implements Filter<CtMethod<?>> {
15  	private final CtMethod<?> method;
16  	private boolean includingSelf = false;
17  
18  	/**
19  	 * Creates a new overriding method filter.
20  	 *
21  	 * @param method
22  	 * 		the executable to be tested for being invoked
23  	 */
24  	public OverridingMethodFilter(CtMethod<?> method) {
25  		this.method = method;
26  	}
27  
28  	/**
29  	 * @param includingSelf if false then element which is equal to the #method is not matching.
30  	 * false is default behavior
31  	 */
32  	public OverridingMethodFilter includingSelf(boolean includingSelf) {
33  		this.includingSelf = includingSelf;
34  		return this;
35  	}
36  
37  	@Override
38  	public boolean matches(CtMethod<?> element) {
39  		if (method == element) {
40  			return this.includingSelf;
41  		}
42  		return element.isOverriding(method);
43  	}
44  }
