6   package spoon.support.reflect.code;
7   
8   import spoon.reflect.annotations.MetamodelPropertyField;
9   import spoon.reflect.code.CtJavaDocTag;
10  import spoon.reflect.visitor.CtVisitor;
11  import spoon.support.reflect.declaration.CtElementImpl;
12  
13  import static spoon.reflect.path.CtRole.COMMENT_CONTENT;
14  import static spoon.reflect.path.CtRole.JAVADOC_TAG_VALUE;
15  import static spoon.reflect.path.CtRole.DOCUMENTATION_TYPE;
16  
17  public class CtJavaDocTagImpl extends CtElementImpl implements CtJavaDocTag {
18  
19  	@MetamodelPropertyField(role = DOCUMENTATION_TYPE)
20  	private TagType type;
21  	@MetamodelPropertyField(role = COMMENT_CONTENT)
22  	private String content;
23  	@MetamodelPropertyField(role = JAVADOC_TAG_VALUE)
24  	private String param;
25  
26  	@Override
27  	public TagType getType() {
28  		return type;
29  	}
30  
31  	@Override
32  	public <E extends CtJavaDocTag> E setType(String type) {
33  		this.setType(TagType.tagFromName(type));
34  		return (E) this;
35  	}
36  
37  	@Override
38  	public <E extends CtJavaDocTag> E setType(TagType type) {
39  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, DOCUMENTATION_TYPE, type, this.type);
40  		this.type = type;
41  		return (E) this;
42  	}
43  
44  	@Override
45  	public String getContent() {
46  		return content;
47  	}
48  
49  	@Override
50  	public <E extends CtJavaDocTag> E setContent(String content) {
51  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, COMMENT_CONTENT, content, this.content);
52  		this.content = content;
53  		return (E) this;
54  	}
55  
56  	@Override
57  	public String getParam() {
58  		return param;
59  	}
60  
61  	@Override
62  	public <E extends CtJavaDocTag> E setParam(String param) {
63  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, JAVADOC_TAG_VALUE, param, this.param);
64  		this.param = param;
65  		return (E) this;
66  	}
67  
68  	@Override
69  	public void accept(CtVisitor visitor) {
70  		visitor.visitCtJavaDocTag(this);
71  	}
72  
73  	@Override
74  	public CtJavaDocTag clone() {
75  		return (CtJavaDocTag) super.clone();
76  	}
77  
78  	@Override
79  	public String toString()	{
80  		return this.getType().toString()	//Tag type, with @ sign included
81  				+ " "	//Space required between tag type and parameter
82  				+ this.param + System.lineSeparator()	//Tag parameter
83  				+ "\t\t" + this.content + System.lineSeparator();
84  	}
85  }
