48  /** Bridge between {@link ObservedMeasurement measurements} and {@link
49   * org.hipparchus.fitting.leastsquares.LeastSquaresProblem
50   * least squares problems}.
51   * @author Luc Maisonobe
52   * @since 8.0
53   */
54  class Model implements MultivariateJacobianFunction {
55  
56      /** Number of estimated orbital parameters. */
57      private final int estimatedOrbitalParameters;
58  
59      /** Estimated propagator parameters. */
60      private final ParameterDriversList estimatedPropagatorParameters;
61  
62      /** Builder for propagator. */
63      private final NumericalPropagatorBuilder propagatorBuilder;
64  
65      /** Measurements. */
66      private final List<ObservedMeasurement<?>> measurements;
67  
68      /** Estimated measurements parameters. */
69      private final ParameterDriversList estimatedMeasurementsParameters;
70  
71      /** Map for measurements parameters columns. */
72      private final Map<String, Integer> parameterColumns;
73  
74      /** Last evaluations. */
75      private final Map<ObservedMeasurement<?>, EstimatedMeasurement<?>> evaluations;
76  
77      /** Observer to be notified at orbit changes. */
78      private final ModelObserver observer;
79  
80      /** Counter for the evaluations. */
81      private Incrementor evaluationsCounter;
82  
83      /** Counter for the iterations. */
84      private Incrementor iterationsCounter;
85  
86      /** Date of the first enabled measurement. */
87      private AbsoluteDate firstDate;
88  
89      /** Date of the last enabled measurement. */
90      private AbsoluteDate lastDate;
91  
92      /** Mapper for Jacobians. */
93      private JacobiansMapper mapper;
94  
95      /** Model function value. */
96      private RealVector value;
97  
98      /** Model function Jacobian. */
99      private RealMatrix jacobian;

101     /** Simple constructor.
102      * @param propagatorBuilder builder to user for propagation
103      * @param measurements measurements
104      * @param estimatedMeasurementsParameters estimated measurements parameters
105      * @param observer observer to be notified at model calls
106      * @exception OrekitException if some propagator parameter cannot be set properly
107      */
108     Model(final NumericalPropagatorBuilder propagatorBuilder,
109           final List<ObservedMeasurement<?>> measurements, final ParameterDriversList estimatedMeasurementsParameters,
110           final ModelObserver observer)
111         throws OrekitException {
112 
113         this.propagatorBuilder               = propagatorBuilder;
114         this.measurements                    = measurements;
115         this.estimatedMeasurementsParameters = estimatedMeasurementsParameters;
116         this.parameterColumns                = new HashMap<String, Integer>(estimatedMeasurementsParameters.getDrivers().size());
117         this.evaluations                     = new IdentityHashMap<ObservedMeasurement<?>, EstimatedMeasurement<?>>(measurements.size());
118         this.observer                        = observer;
119 
120         // allocate vector and matrix
121         int rows = 0;
122         for (final ObservedMeasurement<?> measurement : measurements) {
123             rows += measurement.getDimension();
124         }
125 
126         int columns = 0;
127         for (final ParameterDriver driver : propagatorBuilder.getOrbitalParametersDrivers().getDrivers()) {
128             if (driver.isSelected()) {
129                 ++columns;
130             }
131         }
132         this.estimatedOrbitalParameters = columns;
133 
134         this.estimatedPropagatorParameters = new ParameterDriversList();
135         for (final ParameterDriver driver : propagatorBuilder.getPropagationParametersDrivers().getDrivers()) {
136             if (driver.isSelected()) {
137                 estimatedPropagatorParameters.add(driver);
138                 ++columns;
139             }
140         }
141 
142         for (final ParameterDriver parameter : estimatedMeasurementsParameters.getDrivers()) {
143             parameterColumns.put(parameter.getName(), columns);
144             ++columns;
145         }
146 
147         value    = new ArrayRealVector(rows);
148         jacobian = MatrixUtils.createRealMatrix(rows, columns);
149 
150     }
151

166     /** {@inheritDoc} */
167     @Override
168     public Pair<RealVector, RealMatrix> value(final RealVector point)
169         throws OrekitExceptionWrapper {
170         try {
171 
172             // set up the propagator
173             final NumericalPropagator propagator = createPropagator(point);
174             configureDerivatives(propagator);
175             configureMeasurements(propagator, point);
176             final Orbit orbit = propagator.getInitialState().getOrbit();
177 
178             // reset value and Jacobian
179             evaluations.clear();
180             value.set(0.0);
181             for (int i = 0; i < jacobian.getRowDimension(); ++i) {
182                 for (int j = 0; j < jacobian.getColumnDimension(); ++j) {
183                     jacobian.setEntry(i, j, 0.0);
184                 }
185             }
186 
187             // run the propagation, gathering residuals on the fly
188             propagator.propagate(firstDate.shiftedBy(-1.0), lastDate.shiftedBy(+1.0));
189 
190             observer.modelCalled(orbit, evaluations);
191 
192             return new Pair<RealVector, RealMatrix>(value, jacobian);
193 
194         } catch (OrekitException oe) {
195             throw new OrekitExceptionWrapper(oe);
196         }
197     }

213     /** Create the propagator and parameters corresponding to an evaluation point.
214      * @param point evaluation point
215      * @return a new propagator
216      * @exception OrekitException if orbit cannot be created with the current point
217      */
218     public NumericalPropagator createPropagator(final RealVector point)
219         throws OrekitException {
220 
221         // set up the propagator
222         final int dimension = estimatedOrbitalParameters + estimatedPropagatorParameters.getNbParams();
223         final double[] propagatorArray = new double[dimension];
224         for (int i = 0; i < propagatorArray.length; ++i) {
225             propagatorArray[i] = point.getEntry(i);
226         }
227         final NumericalPropagator propagator =
228                         propagatorBuilder.buildPropagator(propagatorArray);
229 
230         return propagator;
231 
232     }
233 
234     /** Configure the propagator to handle measurements.
235      * @param propagator {@link Propagator} to configure
236      * @param point evaluation point
237      * @exception OrekitException if measurements parameters cannot be set with the current point
238      */
239     private void configureMeasurements(final Propagator propagator, final RealVector point)
240         throws OrekitException {
241 
242         firstDate = AbsoluteDate.FUTURE_INFINITY;
243         lastDate  = AbsoluteDate.PAST_INFINITY;
244 
245         // set up the measurement parameters
246         int index = estimatedOrbitalParameters + estimatedPropagatorParameters.getNbParams();
247         for (final ParameterDriver parameter : estimatedMeasurementsParameters.getDrivers()) {
248             parameter.setNormalizedValue(point.getEntry(index++));
249         }
250 
251         // set up events to handle measurements
252         int p = 0;
253         for (final ObservedMeasurement<?> measurement : measurements) {
254             if (measurement.isEnabled()) {
255                 AbsoluteDate md = measurement.getDate();
256                 final EstimatedMeasurement<?> previousEvaluation = evaluations.get(measurement);
257                 if (previousEvaluation != null) {
258                     // pre-compensate signal transit time
259                     md = md.shiftedBy(-previousEvaluation.getTimeOffset());
260                 }
261                 if (md.compareTo(firstDate) < 0) {
262                     firstDate = md;
263                 }
264                 if (md.compareTo(lastDate) > 0) {
265                     lastDate = md;
266                 }
267                 final MeasurementHandler mh = new MeasurementHandler(this, measurement, p);
268                 propagator.addEventDetector(new DateDetector(md).withHandler(mh));
269                 p += measurement.getDimension();
270             }
271         }
272 
273     }

275     /** Configure the propagator to compute derivatives.
276      * @param propagator {@link Propagator} to configure
277      * @exception OrekitException if orbit cannot be created with the current point
278      */
279     private void configureDerivatives(final NumericalPropagator propagator)
280         throws OrekitException {
281 
282         final String equationName = Model.class.getName() + "-derivatives";
283         final PartialDerivativesEquations partials = new PartialDerivativesEquations(equationName, propagator);
284 
285         // add the derivatives to the initial state
286         final SpacecraftState rawState = propagator.getInitialState();
287         final SpacecraftState stateWithDerivatives = partials.setInitialJacobians(rawState, 6);
288         propagator.resetInitialState(stateWithDerivatives);
289 
290         mapper = partials.getMapper();
291 
292     }

294     /** Fetch a measurement that was evaluated during propagation.
295      * @param index index of the measurement first component
296      * @param evaluation measurement evaluation
297      * @exception OrekitException if Jacobians cannot be computed
298      */
299     void fetchEvaluatedMeasurement(final int index, final EstimatedMeasurement<?> evaluation)
300         throws OrekitException {
301 
302         // compute weighted residuals
303         evaluations.put(evaluation.getObservedMeasurement(), evaluation);
304         final double[] evaluated = evaluation.getEstimatedValue();
305         final double[] observed  = evaluation.getObservedMeasurement().getObservedValue();
306         final double[] sigma     = evaluation.getObservedMeasurement().getTheoreticalStandardDeviation();
307         final double[] weight    = evaluation.getCurrentWeight();
308         for (int i = 0; i < evaluated.length; ++i) {
309             value.setEntry(index + i, weight[i] * (evaluated[i] - observed[i]) / sigma[i]);
310         }
311 
312         // partial derivatives of the current Cartesian coordinates with respect to current orbital state
313         final double[][] aCY = new double[6][6];
314         final Orbit currentOrbit = evaluation.getState().getOrbit();
315         currentOrbit.getJacobianWrtParameters(propagatorBuilder.getPositionAngle(), aCY);
316         final RealMatrix dCdY = new Array2DRowRealMatrix(aCY, false);
317 
318         // Jacobian of the measurement with respect to current orbital state
319         final RealMatrix dMdC = new Array2DRowRealMatrix(evaluation.getStateDerivatives(), false);
320         final RealMatrix dMdY = dMdC.multiply(dCdY);
321 
322         // Jacobian of the measurement with respect to initial orbital state
323         final double[][] aYY0 = new double[6][6];
324         mapper.getStateJacobian(evaluation.getState(), aYY0);
325         final RealMatrix dYdY0 = new Array2DRowRealMatrix(aYY0, false);
326         final RealMatrix dMdY0 = dMdY.multiply(dYdY0);
327         for (int i = 0; i < dMdY0.getRowDimension(); ++i) {
328             int jOrb = 0;
329             for (int j = 0; j < dMdY0.getColumnDimension(); ++j) {
330                 final ParameterDriver driver = propagatorBuilder.getOrbitalParametersDrivers().getDrivers().get(j);
331                 if (driver.isSelected()) {
332                     jacobian.setEntry(index + i, jOrb++,
333                                       weight[i] * dMdY0.getEntry(i, j) / sigma[i] * driver.getScale());
334                 }
335             }
336         }
337 
338         if (estimatedPropagatorParameters.getNbParams() > 0) {
339             // Jacobian of the measurement with respect to propagator parameters
340             final double[][] aYPp  = new double[6][estimatedPropagatorParameters.getNbParams()];
341             mapper.getParametersJacobian(evaluation.getState(), aYPp);
342             final RealMatrix dYdPp = new Array2DRowRealMatrix(aYPp, false);
343             final RealMatrix dMdPp = dMdY.multiply(dYdPp);
344             for (int i = 0; i < dMdPp.getRowDimension(); ++i) {
345                 int jPar = estimatedOrbitalParameters;
346                 for (int j = 0; j < estimatedPropagatorParameters.getNbParams(); ++j) {
347                     final ParameterDriver driver = estimatedPropagatorParameters.getDrivers().get(j);
348                     jacobian.setEntry(index + i, jPar++,
349                                       weight[i] * dMdPp.getEntry(i, j) / sigma[i] * driver.getScale());
350                 }
351             }
352         }
353 
354         // Jacobian of the measurement with respect to measurements parameters
355         final ObservedMeasurement<?> measurement = evaluation.getObservedMeasurement();
356         for (final ParameterDriver driver : measurement.getParametersDrivers()) {
357             if (driver.isSelected()) {
358                 final double[] aMPm = evaluation.getParameterDerivatives(driver);
359                 for (int i = 0; i < aMPm.length; ++i) {
360                     jacobian.setEntry(index + i, parameterColumns.get(driver.getName()),
361                                       weight[i] * aMPm[i] / sigma[i] * driver.getScale());
362                 }
363             }
364         }
365 
366     }
