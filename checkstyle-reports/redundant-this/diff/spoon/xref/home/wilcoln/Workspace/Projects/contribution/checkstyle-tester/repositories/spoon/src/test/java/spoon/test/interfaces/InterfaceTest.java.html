17  package spoon.test.interfaces;

19  import org.junit.Before;
20  import org.junit.Test;

22  import spoon.SpoonModelBuilder;
23  import spoon.reflect.declaration.CtInterface;
24  import spoon.reflect.declaration.CtMethod;
25  import spoon.reflect.declaration.ModifierKind;
26  import spoon.reflect.factory.Factory;
27  import spoon.test.interfaces.testclasses.ExtendsDefaultMethodInterface;
28  import spoon.test.interfaces.testclasses.ExtendsStaticMethodInterface;
29  import spoon.test.interfaces.testclasses.InterfaceWithDefaultMethods;
30  import spoon.test.interfaces.testclasses.RedefinesDefaultMethodInterface;
31  import spoon.test.interfaces.testclasses.RedefinesStaticMethodInterface;
32  
33  import java.io.File;
34  import java.util.function.Consumer;
35  
36  import static org.junit.Assert.assertEquals;
37  import static org.junit.Assert.assertFalse;
38  import static org.junit.Assert.assertSame;
39  import static org.junit.Assert.assertTrue;
40  
41  public class InterfaceTest {
42  
43  	private Factory factory;
44  
45  	@Before
46  	public void setUp() {
47  		final File testDirectory = new File("./src/test/java/spoon/test/interfaces/testclasses/");
48  
49  		final Launcher launcher = new Launcher();
50  
51  		this.factory = launcher.createFactory();
52  		factory.getEnvironment().setComplianceLevel(8);
53  		SpoonModelBuilder compiler = launcher.createCompiler(this.factory);
54  
55  		compiler.addInputSource(testDirectory);
56  		compiler.build();
57  	}
58  
59  	@Test
60  	public void testDefaultMethodInInterface() {
61  		final CtInterface<?> ctInterface = (CtInterface<?>) factory.Type().get(InterfaceWithDefaultMethods.class);
62  
63  		final CtMethod<?> ctMethod = ctInterface.getMethodsByName("getZonedDateTime").get(0);
64  		assertTrue("The method in the interface must to be default", ctMethod.isDefaultMethod());
65  
66  		// contract: the debug toString also shows the implicit modifiers (here "public")
67  		final String expected =
68  				"public default java.time.ZonedDateTime getZonedDateTime(java.lang.String zoneString) {"
69  						+ System.lineSeparator()
70  						+ "    return java.time.ZonedDateTime.of(getLocalDateTime(), spoon.test.interfaces.testclasses.InterfaceWithDefaultMethods.getZoneId(zoneString));"
71  						+ System.lineSeparator() + "}";
72  		assertEquals("The default method must to be well printed", expected, ctMethod.toString());
73  	}
74  
75  	@Test
76  	public void testDefaultMethodInConsumer() {
77  		final CtInterface<?> ctInterface = (CtInterface<?>) factory.Type().get(Consumer.class);
78  
79  		final CtMethod<?> ctMethod = ctInterface.getMethodsByName("andThen").get(0);
80  		assertTrue("The method in the interface must to be default", ctMethod.isDefaultMethod());
81  	}
82  
83  	@Test
84  	public void testExtendsDefaultMethodInSubInterface() {
85  		final CtInterface<?> ctInterface = (CtInterface<?>) factory.Type().get(ExtendsDefaultMethodInterface.class);
86  
87  		assertEquals("Sub interface must have only one method in its interface", 1, ctInterface.getMethods().size());
88  		assertEquals("Sub interface must have 6 methods in its interface and its super interfaces", 6, ctInterface.getAllMethods().size());
89  
90  		final CtMethod<?> getZonedDateTimeMethod = ctInterface.getMethodsByName("getZonedDateTime").get(0);
91  		assertTrue("Method in the sub interface must be a default method", getZonedDateTimeMethod.isDefaultMethod());
92  		assertSame("Interface of the default method must be the sub interface", ExtendsDefaultMethodInterface.class, getZonedDateTimeMethod.getDeclaringType().getActualClass());
93  	}
94  
95  	@Test
96  	public void testRedefinesDefaultMethodInSubInterface() {
97  		final CtInterface<?> ctInterface = (CtInterface<?>) factory.Type().get(RedefinesDefaultMethodInterface.class);
98  
99  		assertEquals("Sub interface must have only one method in its interface", 1, ctInterface.getMethods().size());
100 		assertEquals("Sub interface must have 6 methods in its interface and its super interfaces", 6, ctInterface.getAllMethods().size());
101 
102 		final CtMethod<?> getZonedDateTimeMethod = ctInterface.getMethodsByName("getZonedDateTime").get(0);
103 		assertFalse("Method in the sub interface mustn't be a default method", getZonedDateTimeMethod.isDefaultMethod());
104 		assertSame("Interface of the default method must be the sub interface", RedefinesDefaultMethodInterface.class, getZonedDateTimeMethod.getDeclaringType().getActualClass());
105 	}
106 
107 	@Test
108 	public void testExtendsStaticMethodInSubInterface() {
109 		final CtInterface<?> ctInterface = (CtInterface<?>) factory.Type().get(ExtendsStaticMethodInterface.class);
110 
111 		assertEquals("Sub interface must have only one method in its interface", 1, ctInterface.getMethods().size());
112 		assertEquals("Sub interface must have 6 methods in its interface and its super interfaces", 6, ctInterface.getAllMethods().size());
113 
114 		final CtMethod<?> getZoneIdMethod = ctInterface.getMethodsByName("getZoneId").get(0);
115 		assertTrue("Method in the sub interface must be a static method", getZoneIdMethod.getModifiers().contains(ModifierKind.STATIC));
116 		assertSame("Interface of the static method must be the sub interface", ExtendsStaticMethodInterface.class, getZoneIdMethod.getDeclaringType().getActualClass());
117 	}
118 
119 	@Test
120 	public void testRedefinesStaticMethodInSubInterface() {
121 		final CtInterface<?> ctInterface = (CtInterface<?>) factory.Type().get(RedefinesStaticMethodInterface.class);
122 
123 		assertEquals("Sub interface must have only one method in its interface", 1, ctInterface.getMethods().size());
124 		assertEquals("Sub interface must have 6+12(from java.lang.Object) methods in its interface and its super interfaces", 6, ctInterface.getAllMethods().size());
125 
126 		final CtMethod<?> getZoneIdMethod = ctInterface.getMethodsByName("getZoneId").get(0);
127 		assertFalse("Method in the sub interface mustn't be a static method", getZoneIdMethod.getModifiers().contains(ModifierKind.STATIC));
128 		assertSame("Interface of the static method must be the sub interface", RedefinesStaticMethodInterface.class, getZoneIdMethod.getDeclaringType().getActualClass());
129 	}
