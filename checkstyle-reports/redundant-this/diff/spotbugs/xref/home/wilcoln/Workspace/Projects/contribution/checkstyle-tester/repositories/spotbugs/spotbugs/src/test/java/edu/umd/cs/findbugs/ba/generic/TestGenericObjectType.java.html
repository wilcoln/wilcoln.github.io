39  /**
40   * @author Nat Ayewah
41   */
42  public class TestGenericObjectType {
43  
44      GenericObjectType obj;
45  
46      String javaSignature;
47  
48      String underlyingClass;
49  
50      GenericUtilities.TypeCategory typeCategory;
51  
52      @Nullable String variable;
53  
54      @Nullable Type extension;
55  
56      List<ReferenceType> parameters;
57  
58      public void initTest(String bytecodeSignature, String javaSignature, String underlyingClass,
59              GenericUtilities.TypeCategory typeCategory,
60              @Nullable String variable, @Nullable Type extension, @Nullable List<ReferenceType> parameters) {
61          this.obj = (GenericObjectType) GenericUtilities.getType(bytecodeSignature);
62          this.javaSignature = javaSignature;
63          this.underlyingClass = underlyingClass;
64          this.typeCategory = typeCategory;
65          this.variable = variable;
66          this.extension = extension;
67          this.parameters = parameters;
68      }
69  
70      public void processTest() {
71          assertEquals(obj.toString(true), javaSignature);
72          assertEquals(obj.getClassName(), underlyingClass);
73          assertEquals(obj.getTypeCategory(), typeCategory);
74  
75          if (typeCategory == TypeCategory.PARAMETERIZED) {
76              assertTrue(obj.hasParameters());
77              assertTrue(obj.getNumParameters() == parameters.size());
78              for (int i = 0; i < obj.getNumParameters(); i++) {
79                  compareTypes(obj.getParameterAt(i), parameters.get(i));
80              }
81              assertNull(obj.getVariable());
82              assertNull(obj.getExtension());
83          } else if (typeCategory == TypeCategory.TYPE_VARIABLE) {
84              assertFalse(obj.hasParameters());
85              assertNull(obj.getExtension());
86              assertNotNull(obj.getVariable());
87              assertEquals(obj.getVariable(), variable);
88          } else if (typeCategory == TypeCategory.WILDCARD) {
89              assertFalse(obj.hasParameters());
90              assertNull(obj.getExtension());
91              assertNotNull(obj.getVariable());
92              assertEquals(obj.getVariable(), "*");
93          } else if (typeCategory == TypeCategory.WILDCARD_EXTENDS || typeCategory == TypeCategory.WILDCARD_SUPER) {
94              assertFalse(obj.hasParameters());
95              assertNotNull(obj.getExtension());
96              assertNotNull(obj.getVariable());
97              assertEquals(obj.getVariable(), variable);
98              compareTypes(obj.getExtension(), extension);
99          }
100     }
101 
102     private static void compareTypes(Type a, Type b) {
103         assertEquals(a, b);
104         if (a instanceof GenericObjectType || b instanceof GenericObjectType) {
105             assertTrue(a instanceof GenericObjectType && b instanceof GenericObjectType);
106             assertEquals(((GenericObjectType) a).toString(true), ((GenericObjectType) b).toString(true));
107         }
108     }
109 
110     @Test
111     public void testParameterizedList() {
112         initTest("Ljava/util/List<Ljava/lang/Comparable;>;", "java.util.List<java.lang.Comparable>", "java.util.List",
113                 GenericUtilities.TypeCategory.PARAMETERIZED, null, null,
114                 GenericUtilities.getTypeParameters("Ljava/lang/Comparable;"));
115         processTest();
116     }
117 
118     public void notestCreateTypes() {
119         initTest("LDummyClass<Ljava/lang/Comparable;TE;>;", "DummyClass<java.lang.Comparable,E>", "DummyClass",
120                 GenericUtilities.TypeCategory.PARAMETERIZED, null, null, Arrays.asList(
121                         (ReferenceType) GenericUtilities.getType("Ljava/lang/Comparable;"),
122                         (ReferenceType) GenericUtilities.getType("TE;")));
123         processTest();
124     }
125 
126     public void notestTypeVariables() {
127         initTest("TE;", "E", "java.lang.Object", GenericUtilities.TypeCategory.TYPE_VARIABLE, "E", null, null);
128         processTest();
129 
130         initTest("*", "?", "java.lang.Object", GenericUtilities.TypeCategory.WILDCARD, "*", null, null);
131         processTest();
132 
133         initTest("+TE;", "? extends E", "java.lang.Object", GenericUtilities.TypeCategory.WILDCARD_EXTENDS, "+",
134                 GenericUtilities.getType("TE;"), null);
135         processTest();
136 
137         initTest("-TE;", "? super E", "java.lang.Object", GenericUtilities.TypeCategory.WILDCARD_SUPER, "-",
138                 GenericUtilities.getType("TE;"), null);
139         processTest();
140 
141         initTest("-[TE;", "? super E[]", "java.lang.Object", GenericUtilities.TypeCategory.WILDCARD_SUPER, "-",
142                 GenericUtilities.getType("[TE;"), null);
143         processTest();
144     }
