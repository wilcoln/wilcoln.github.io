32  package j2dbench;
33  
34  import java.util.Hashtable;
35  import java.util.Properties;
36  import java.util.Enumeration;
37  import java.util.Vector;
38  import java.io.PrintWriter;
39  
40  public class ResultSet {
41      static Hashtable preferprops;
42      static Hashtable ignoreprops;
43  
44      // Preferred props - will be listed even if undefined
45      static String preferredkeys[] = {
46          "java.version",
47          "java.runtime.version",
48          "java.class.version",
49          "java.vm.version",
50          "java.vm.name",
51          "java.vm.info",
52          "java.home",
53          "java.compiler",
54          "os.arch",
55          "os.name",
56          "os.version",
57          "user.name",
58          "sun.cpu.endian",
59          "sun.cpu.isalist",
60      };
61  
62      // Ignored props - will not be copied to results file
63      static String ignoredkeys[] = {
64          "user.dir",
65          "user.home",
66          "user.timezone",
67          "path.separator",
68          "line.separator",
69          "file.separator",
70          "file.encoding",
71          "file.encoding.pkg",
72          "java.class.path",
73          "java.library.path",
74          "java.io.tmpdir",
75          "java.ext.dirs",
76          "java.endorsed.dirs",
77          "java.util.prefs.PreferencesFactory",
78          "sun.java2d.fontpath",
79          "sun.boot.library.path",
80          "sun.boot.class.path",
81      };
82  
83      /*
84       * Other props, as of 1.4.0, not classified as "preferred" or "ignored"
85       * Allowed to propagate to the results file if defined.
86       *
87       * java.runtime.name
88       * java.vendor
89       * java.vendor.url
90       * java.vendor.url.bug
91       * java.specification.name
92       * java.specification.vendor
93       * java.specification.version
94       * java.vm.specification.name
95       * java.vm.specification.vendor
96       * java.vm.specification.version
97       * java.vm.vendor
98       * java.awt.graphicsenv
99       * java.awt.printerjob
100      * user.language
101      * sun.os.patch.level
102      * sun.arch.data.model
103      * sun.io.unicode.encoding
104      */
105 
106     static String unknown = "unknown";
107 
108     static {
109         preferprops = new Hashtable();
110         for (int i = 0; i < preferredkeys.length; i++) {
111             preferprops.put(preferredkeys[i], unknown);
112         }
113         ignoreprops = new Hashtable();
114         for (int i = 0; i < ignoredkeys.length; i++) {
115             ignoreprops.put(ignoredkeys[i], unknown);
116         }
117     }
118 
119     Hashtable props;
120     Vector results;
121     long start;
122     long end;
123     String title;
124     String description;
125 
126     public ResultSet() {
127         Properties sysprops = System.getProperties();
128         props = (Hashtable) preferprops.clone();
129         Enumeration enum_ = sysprops.keys();
130         while (enum_.hasMoreElements()) {
131             Object key = enum_.nextElement();
132             if (!ignoreprops.containsKey(key)) {
133                 props.put(key, sysprops.get(key));
134             }
135         }
136         results = new Vector();
137         start = System.currentTimeMillis();
138     }
139 
140     public void setTitle(String title) {
141         this.title = title;
142     }
143 
144     public void setDescription(String desc) {
145         this.description = desc;
146     }
147 
148     public void record(Result result) {
149         results.addElement(result);
150     }
151 
152     public void summarize() {
153         end = System.currentTimeMillis();
154         for (int i = 0; i < results.size(); i++) {
155             ((Result) results.elementAt(i)).summarize();
156         }
157     }
158 
159     public void write(PrintWriter pw) {
160         pw.println("<result-set version=\"0.1\" name=\""+title+"\">");
161         pw.println("  <test-desc>"+description+"</test-desc>");
162         pw.println("  <test-date start=\""+start+"\" end=\""+end+"\"/>");
163         for (int i = 0; i < preferredkeys.length; i++) {
164             String key = preferredkeys[i];
165             pw.println("  <sys-prop key=\""+key+
166                        "\" value=\""+props.get(key)+"\"/>");
167         }
168         Enumeration enum_ = props.keys();
169         while (enum_.hasMoreElements()) {
170             Object key = enum_.nextElement();
171             if (!preferprops.containsKey(key)) {
172                 pw.println("  <sys-prop key=\""+key+
173                            "\" value=\""+props.get(key)+"\"/>");
174             }
175         }
176         for (int i = 0; i < results.size(); i++) {
177             ((Result) results.elementAt(i)).write(pw);
178         }
179         pw.println("</result-set>");
180     }
181 }
