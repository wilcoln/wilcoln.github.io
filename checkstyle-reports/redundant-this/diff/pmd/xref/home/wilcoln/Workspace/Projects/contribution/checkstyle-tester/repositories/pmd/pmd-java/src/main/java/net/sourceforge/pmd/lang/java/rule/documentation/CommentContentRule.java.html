7   import static net.sourceforge.pmd.properties.PropertyFactory.booleanProperty;
8   import static net.sourceforge.pmd.properties.PropertyFactory.stringListProperty;
9   
10  import java.util.ArrayList;
11  import java.util.Collections;
12  import java.util.HashSet;
13  import java.util.List;
14  import java.util.Locale;
15  import java.util.Set;
16  
17  import org.apache.commons.lang3.StringUtils;
18  
19  import net.sourceforge.pmd.Rule;
20  import net.sourceforge.pmd.RuleContext;
21  import net.sourceforge.pmd.lang.java.ast.ASTCompilationUnit;
22  import net.sourceforge.pmd.lang.java.ast.Comment;
23  import net.sourceforge.pmd.properties.PropertyDescriptor;
24  import net.sourceforge.pmd.properties.PropertySource;

26  /**
27   * A rule that checks for illegal words in the comment text.
28   *
29   * TODO implement regex option
30   *
31   * @author Brian Remedios
32   */
33  public class CommentContentRule extends AbstractCommentRule {
34  
35      private boolean caseSensitive;
36      private List<String> originalBadWords;
37      private List<String> currentBadWords;
38  
39      // ignored when property above == True
40      public static final PropertyDescriptor<Boolean> CASE_SENSITIVE_DESCRIPTOR = booleanProperty("caseSensitive").defaultValue(false).desc("Case sensitive").build();
41  
42      public static final PropertyDescriptor<List<String>> DISSALLOWED_TERMS_DESCRIPTOR =
43              stringListProperty("disallowedTerms")
44                      .desc("Illegal terms or phrases")
45                      .defaultValues("idiot", "jerk").build(); // TODO make blank property? or add more defaults?
46  
47      private static final Set<PropertyDescriptor<?>> NON_REGEX_PROPERTIES;
48  
49      static {
50          NON_REGEX_PROPERTIES = new HashSet<>(1);
51          NON_REGEX_PROPERTIES.add(CASE_SENSITIVE_DESCRIPTOR);
52      }
53  
54      public CommentContentRule() {
55          definePropertyDescriptor(CASE_SENSITIVE_DESCRIPTOR);
56          definePropertyDescriptor(DISSALLOWED_TERMS_DESCRIPTOR);
57      }
58  
59      /**
60       * Capture values and perform all the case-conversions once per run
61       */
62      @Override
63      public void start(RuleContext ctx) {
64          originalBadWords = getProperty(DISSALLOWED_TERMS_DESCRIPTOR);
65          caseSensitive = getProperty(CASE_SENSITIVE_DESCRIPTOR);
66          if (caseSensitive) {
67              currentBadWords = originalBadWords;
68          } else {
69              currentBadWords = new ArrayList<>();
70              for (String badWord : originalBadWords) {
71                  currentBadWords.add(badWord.toUpperCase(Locale.ROOT));
72              }
73          }
74      }
75  
76      /**
77       * .
78       * @see Rule#end(RuleContext)
79       */
80      @Override
81      public void end(RuleContext ctx) {
82          // Override as needed
83      }
84  
85      private List<String> illegalTermsIn(Comment comment) {
86  
87          if (currentBadWords.isEmpty()) {
88              return Collections.emptyList();
89          }
90  
91          String commentText = filteredCommentIn(comment);
92          if (StringUtils.isBlank(commentText)) {
93              return Collections.emptyList();
94          }
95  
96          if (!caseSensitive) {
97              commentText = commentText.toUpperCase(Locale.ROOT);
98          }
99  
100         List<String> foundWords = new ArrayList<>();
101 
102         for (int i = 0; i < currentBadWords.size(); i++) {
103             if (commentText.contains(currentBadWords.get(i))) {
104                 foundWords.add(originalBadWords.get(i));
105             }
106         }
107 
108         return foundWords;
109     }
110 
111     private String errorMsgFor(List<String> badWords) {
112         StringBuilder msg = new StringBuilder(this.getMessage()).append(": ");
113         if (badWords.size() == 1) {
114             msg.append("Invalid term: '").append(badWords.get(0)).append('\'');
115         } else {
116             msg.append("Invalid terms: '");
117             msg.append(badWords.get(0));
118             for (int i = 1; i < badWords.size(); i++) {
119                 msg.append("', '").append(badWords.get(i));
120             }
121             msg.append('\'');
122         }
123         return msg.toString();
124     }
125 
126     @Override
127     public Object visit(ASTCompilationUnit cUnit, Object data) {
128 
129         // NPE patch: Eclipse plugin doesn't call start() at onset?
130         if (currentBadWords == null) {
131             start(null);
132         }
133 
134         for (Comment comment : cUnit.getComments()) {
135             List<String> badWords = illegalTermsIn(comment);
136             if (badWords.isEmpty()) {
137                 continue;
138             }
139 
140             addViolationWithMessage(data, cUnit, errorMsgFor(badWords), comment.getBeginLine(), comment.getEndLine());
141         }
142 
143         return super.visit(cUnit, data);
144     }
145 
146     public boolean hasDissallowedTerms() {
147         List<String> terms = getProperty(DISSALLOWED_TERMS_DESCRIPTOR);
148         return !terms.isEmpty();
149     }
150 
151     /**
152      * @see PropertySource#dysfunctionReason()
153      */
154     @Override
155     public String dysfunctionReason() {
156         return hasDissallowedTerms() ? null : "No disallowed terms specified";
157     }
158 }
