9   public class Employment implements Serializable {

23  	}

25  		this.organizationName = organizationName;
26  	}

29  	}
30  	public void setPersonName(String personName) {
31  		this.personName = personName;
32  	}
33  	public Date getEndDate() {

35  	}
36  	public void setEndDate(Date endDate) {
37  		this.endDate = endDate;
38  	}
39  	public Date getStartDate() {
40  		return startDate;
41  	}
42  	public void setStartDate(Date startDate) {
43  		this.startDate = startDate;
44  	}
45  }
