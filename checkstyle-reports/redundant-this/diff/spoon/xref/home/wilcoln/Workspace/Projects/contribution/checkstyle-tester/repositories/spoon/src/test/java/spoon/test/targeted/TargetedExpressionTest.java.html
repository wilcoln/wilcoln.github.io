183 		assertEqualsFieldAccess(new ExpectedTargetedExpression().type(CtFieldWrite.class).declaringType(expectedType).target(expectedTypeAccess).result("spoon.test.targeted.testclasses.Foo.k"), elements.get(4));

471 	private void assertEqualsFieldAccess(ExpectedTargetedExpression expected, CtFieldAccess<?> fieldAccess) {
472 		if (expected.declaringType == null) {
473 			assertNull(fieldAccess.getVariable().getDeclaringType());
474 		} else {
475 			assertEquals(expected.isLocal, fieldAccess.getVariable().getDeclaringType().isLocalType());
476 			assertEquals(expected.declaringType.getQualifiedName(), fieldAccess.getVariable().getDeclaringType().getQualifiedName());
477 		}
478 		if (expected.targetClass != null) {
479 			assertSame(expected.targetClass, fieldAccess.getTarget().getClass());
480 		} else if (expected.targetString != null) {
481 			assertEquals(expected.targetString, fieldAccess.getTarget().toString());
482 		}
483 		assertEquals(expected.result, fieldAccess.toString());
484 		if (expected.type != null) {
485 			assertTrue(expected.type.isInstance(fieldAccess));
486 		}
487 	}

490 	private void assertEqualsInvocation(ExpectedTargetedExpression expected, CtInvocation<?> invocation) {
491 		// two required parts: toString and declaringType (type containing the method to be called)
492 		assertEquals(expected.result, invocation.toString());
493 		assertEquals(expected.declaringType, invocation.getExecutable().getDeclaringType());
494 
495 		// + two optional parts
496 		if (expected.targetClass != null) {
497 			assertSame(expected.targetClass, invocation.getTarget().getClass());
498 		} else if (expected.targetString != null) {
499 			assertEquals(expected.targetString, invocation.getTarget().toString());
500 		}
