68      public static void init() {
69          pixelroot = new Group("pixel", "Pixel Access Benchmarks");
70  
71          pixeloptroot = new Group(pixelroot, "opts", "Pixel Access Options");
72          doRenderTo = new Option.Toggle(pixeloptroot, "renderto",
73                                         "Render to Image before test",
74                                         Option.Toggle.Off);
75          doRenderFrom = new Option.Toggle(pixeloptroot, "renderfrom",
76                                           "Render from Image before test",
77                                           Option.Toggle.Off);
78  
79          // BufferedImage Sources
80          bufimgsrcroot = new Group.EnableSet(pixelroot, "src",
81                                              "BufferedImage Sources");
82          new BufImg(BufferedImage.TYPE_BYTE_BINARY, 1);
83          new BufImg(BufferedImage.TYPE_BYTE_BINARY, 2);
84          new BufImg(BufferedImage.TYPE_BYTE_BINARY, 4);
85          new BufImg(BufferedImage.TYPE_BYTE_INDEXED);
86          new BufImg(BufferedImage.TYPE_BYTE_GRAY);
87          new BufImg(BufferedImage.TYPE_USHORT_555_RGB);
88          new BufImg(BufferedImage.TYPE_USHORT_565_RGB);
89          new BufImg(BufferedImage.TYPE_USHORT_GRAY);
90          new BufImg(BufferedImage.TYPE_3BYTE_BGR);
91          new BufImg(BufferedImage.TYPE_4BYTE_ABGR);
92          new BufImg(BufferedImage.TYPE_INT_RGB);
93          new BufImg(BufferedImage.TYPE_INT_BGR);
94          new BufImg(BufferedImage.TYPE_INT_ARGB);
95  
96          // BufferedImage Tests
97          bufimgtestroot = new Group(pixelroot, "bimgtests",
98                                     "BufferedImage Tests");
99          new BufImgTest.GetRGB();
100         new BufImgTest.SetRGB();
101 
102         // Raster Tests
103         rastertestroot = new Group(pixelroot, "rastests",
104                                    "Raster Tests");
105         new RasTest.GetDataElements();
106         new RasTest.SetDataElements();
107         new RasTest.GetPixel();
108         new RasTest.SetPixel();
109 
110         // DataBuffer Tests
111         dbtestroot = new Group(pixelroot, "dbtests",
112                                "DataBuffer Tests");

114         new DataBufTest.SetElem();
115     }

117     public PixelTests(Group root, String nodeName, String description) {

123     public static class Context {
124         BufferedImage bimg;
125         WritableRaster ras;
126         DataBuffer db;
127         int pixeldata[];
128         Object elemdata;
129     }
130 
131     public Object initTest(TestEnvironment env, Result result) {
132         Context ctx = new Context();
133         ctx.bimg = ((BufImg) env.getModifier(bufimgsrcroot)).getImage();
134         if (env.isEnabled(doRenderTo)) {
135             Graphics2D g2d = ctx.bimg.createGraphics();
136             g2d.setColor(Color.white);
137             g2d.fillRect(3, 0, 1, 1);
138             g2d.dispose();
139         }
140         if (env.isEnabled(doRenderFrom)) {
141             GraphicsConfiguration cfg =
142                 GraphicsEnvironment
143                 .getLocalGraphicsEnvironment()
144                 .getDefaultScreenDevice()
145                 .getDefaultConfiguration();
146             VolatileImage vimg = cfg.createCompatibleVolatileImage(8, 1);
147             vimg.validate(cfg);
148             Graphics2D g2d = vimg.createGraphics();
149             for (int i = 0; i < 100; i++) {
150                 g2d.drawImage(ctx.bimg, 0, 0, null);
151             }
152             g2d.dispose();
153             vimg.flush();
154         }
155         result.setUnits(1);
156         result.setUnitName(getUnitName());
157         return ctx;
158     }

165     public static class BufImg extends Option.Enable {
166         public static int rgbvals[] = {
167             0x00000000,
168             0xff0000ff,
169             0x8000ff00,
170             0xffffffff
171         };
172 
173         static int cmap[] = {
174             0xff000000,  // 0: opaque black
175             0xffffffff,  // 1: opaque white
176 
177             0x00000000,  // 2: transparent black
178             0x80ffffff,  // 3: translucent white
179 
180             0x00ffffff,  // 4: transparent white
181             0x80000000,  // 5: translucent black
182             0xff333333,  // 6: opaque dark gray
183             0xff666666,  // 7: opaque medium gray
184             0xff999999,  // 8: opaque gray
185             0xffcccccc,  // 9: opaque light gray
186             0xff0000ff,  // A: opaque blue
187             0xff00ff00,  // B: opaque green
188             0xff00ffff,  // C: opaque cyan
189             0xffff0000,  // D: opaque red
190             0xffff00ff,  // E: opaque magenta
191             0xffffff00,  // F: opaque yellow
192         };
193 
194         int type;
195         int nbits;
196 
197         public BufImg(int type) {
198             super(bufimgsrcroot,
199                   Destinations.BufImg.ShortNames[type],
200                   Destinations.BufImg.Descriptions[type], false);
201             this.type = type;
202             this.nbits = 0;
203         }
204 
205         public BufImg(int type, int nbits) {
206             super(bufimgsrcroot,
207                   nbits+"BitBinary",
208                   nbits+"-bit Binary Image", false);
209             this.type = type;
210             this.nbits = nbits;
211         }
212 
213         public String getModifierValueName(Object val) {
214             return "BufImg("+getNodeName()+")";
215         }
216 
217         public BufferedImage getImage() {
218             BufferedImage bimg;
219             if (nbits == 0) {
220                 bimg = new BufferedImage(8, 1, type);
221             } else {
222                 IndexColorModel icm =
223                     new IndexColorModel(nbits, (1 << nbits),
224                                         cmap, 0, (nbits > 1), -1,
225                                         DataBuffer.TYPE_BYTE);
226                 // Note that this constructor has bugs pre 1.4...
227                 // bimg = new BufferedImage(64/nbits, 1, type, icm);
228                 WritableRaster wr =
229                     icm.createCompatibleWritableRaster(64/nbits, 1);
230                 bimg = new BufferedImage(icm, wr, false, null);
231             }
232             for (int i = 0; i < bimg.getWidth(); i++) {
233                 bimg.setRGB(i, 0, rgbvals[i&3]);
234             }
235             return bimg;
236         }
237     }

249             public GetRGB() {
250                 super("getrgb", "BufferedImage.getRGB(x, y)");

254                 BufferedImage bimg = ((Context) context).bimg;

284         public Object initTest(TestEnvironment env, Result result) {
285             Context ctx = (Context) super.initTest(env, result);
286             ctx.ras = ctx.bimg.getRaster();
287             ctx.pixeldata = ctx.ras.getPixel(0, 0, (int[]) null);
