23  /**
24   * Base class for generic property builders.
25   * Property builders are obtained from the {@link PropertyFactory},
26   * and are used to build {@link PropertyDescriptor}s.
27   *
28   * <p>All properties <i>must</i> specify the following attributes to build
29   * properly:
30   * <ul>
31   *   <li>A name: filled-in when obtaining the builder
32   *   <li>A description: see {@link #desc(String)}
33   *   <li>A default value: see {@link #defaultValue(Object)}
34   * </ul>
35   *
36   * <p>The {@link PropertyDescriptor} may be built after those required steps by
37   * calling {@link #build()}.
38   *
39   * <p>A property builder may throw {@link IllegalArgumentException} at any
40   * stage during the build process to indicate invalid input. It usually tries
41   * to do so as early as possible, rather than waiting for the call to {@link #build()}.
42   *
43   * <p>Note: from 7.0.0 on, all property builders will
44   * extend this class instead of {@link PropertyDescriptorBuilder}.
45   *
46   * @param <B> Concrete type of this builder instance
47   * @param <T> Type of values the property handles
48   *
49   * @author Clément Fournier
50   * @since 6.10.0
51   */
52  // @formatter:on
53  public abstract class PropertyBuilder<B extends PropertyBuilder<B, T>, T> {
54  
55      private static final Pattern NAME_PATTERN = Pattern.compile("[a-zA-Z][\\w-]*");
56      private final Set<PropertyConstraint<? super T>> validators = new LinkedHashSet<>();
57      protected boolean isDefinedExternally;
58      private String name;
59      private String description;
60      private T defaultValue;
61  
62  
63      PropertyBuilder(String name) {
64  
65          if (StringUtils.isBlank(name)) {
66              throw new IllegalArgumentException("Name must be provided");
67          } else if (!NAME_PATTERN.matcher(name).matches()) {
68              throw new IllegalArgumentException("Invalid name '" + name + "'");
69          }
70          this.name = name;
71      }
72  
73  
74      void setDefinedExternally(boolean bool) {
75          this.isDefinedExternally = bool;
76      }
77  
78  
79      Set<PropertyConstraint<? super T>> getConstraints() {
80          return validators;
81      }
82  
83  
84      String getDescription() {
85          if (StringUtils.isBlank(description)) {
86              throw new IllegalArgumentException("Description must be provided");
87          }
88          return description;
89      }
90  
91  
92      /** Returns the value, asserting it has been set. */
93      T getDefaultValue() {
94          if (!isDefaultValueSet()) {
95              throw new IllegalArgumentException("A default value must be provided");
96          }
97          return defaultValue;
98      }
99  
100 
101     boolean isDefaultValueSet() {
102         return defaultValue != null;
103     }
104 
105 
106     /**
107      * Specify the description of the property. This is used for documentation.
108      * Please describe precisely how the property may change the behaviour of the
109      * rule. Providing complete information should be preferred over being concise.
110      *
111      * <p>Calling this method is required for {@link #build()} to succeed.
112      *
113      * @param desc The description
114      *
115      * @return The same builder
116      *
117      * @throws IllegalArgumentException If the description is null or whitespace
118      */
119     @SuppressWarnings("unchecked")
120     public B desc(String desc) {
121         if (StringUtils.isBlank(desc)) {
122             throw new IllegalArgumentException("Description must be provided");
123         }
124         this.description = desc;
125         return (B) this;
126     }

129     //
130     //     * <p>Constraints should be independent from each other, and should
131     //     * perform no side effects. PMD doesn't specify how many times a
132     //     * constraint predicate will be executed, or in what order.
133     //
134     // This is superfluous right now bc users may not create their own constraints

137     /**
138      * Add a constraint on the values that this property may take.
139      * The validity of values will be checked when parsing the XML,
140      * and invalid values will be reported. A rule will never be run
141      * if some of its properties violate some constraints.
142      *
143      * @param constraint The constraint
144      *
145      * @return The same builder
146      *
147      * @see net.sourceforge.pmd.properties.constraints.NumericConstraints
148      */
149     @SuppressWarnings("unchecked")
150     public B require(PropertyConstraint<? super T> constraint) {
151         validators.add(constraint);
152         return (B) this;
153     }

156     /**
157      * Specify a default value. Some subclasses provide convenient
158      * related methods, see e.g. {@link GenericCollectionPropertyBuilder#defaultValues(Object, Object[])}.
159      * Using the null value is prohibited.
160      *
161      * <p>Calling this method is required for {@link #build()} to succeed.
162      *
163      * @param val Default value
164      *
165      * @return The same builder
166      *
167      * @throws IllegalArgumentException If the argument is null
168      */
169     @SuppressWarnings("unchecked")
170     public B defaultValue(T val) {
171         if (val == null) {
172             throw new IllegalArgumentException("Property values may not be null.");
173         }
174         this.defaultValue = val;
175         return (B) this;
176     }

179     /**
180      * Builds the descriptor and returns it.
181      *
182      * @return The built descriptor
183      *
184      * @throws IllegalArgumentException if the description or default value were not provided, or if the default value doesn't satisfy the given constraints
185      */
186     public abstract PropertyDescriptor<T> build();

189     /**
190      * Returns the name of the property to be built.
191      */
192     public String getName() {
193         return name;
194     }

197     // Technically this may very well be merged into PropertyBuilder
198     // We'd have all properties (even collection properties) enjoy a ValueParser,
199     // which means they could be parsed in a <value> tag (collections would use delimiters) if they opt in.
200     // The delimiters wouldn't be configurable (we'd take our current defaults). If they could ambiguous
201     // then the <seq> syntax should be the only one available.
202     // This would allow specifying eg lists of numbers as <value>1,2,3</value>, for which the <seq> syntax would look clumsy
203     abstract static class BaseSinglePropertyBuilder<B extends PropertyBuilder<B, T>, T> extends PropertyBuilder<B, T> {
204         private final ValueParser<T> parser;
205         private final Class<T> type;
206 
207 
208         // Class is not final but a package-private constructor restricts inheritance
209         BaseSinglePropertyBuilder(String name, ValueParser<T> parser, Class<T> type) {
210             super(name);
211             this.parser = parser;
212             this.type = type;
213         }
214 
215 
216         protected ValueParser<T> getParser() {
217             return parser;
218         }
219 
220 
221         protected Class<T> getType() {
222             return type;
223         }
224 
225 
226         /**
227          * Returns a new builder that can be used to build a property
228          * handling lists of Ts. The validators already added are
229          * converted to list validators. The default value cannot have
230          * previously been set.
231          *
232          * @return A new list property builder
233          *
234          * @throws IllegalStateException if the default value has already been set
235          */
236         /* package private */ GenericCollectionPropertyBuilder<T, List<T>> toList() {
237 
238             Supplier<List<T>> listSupplier = new Supplier<List<T>>() {
239                 @Override
240                 public List<T> get() {
241                     return new ArrayList<>();
242                 }
243             };
244 
245             return toCollection(listSupplier);
246         }
247 
248 
249         // TODO 7.0.0 this can be inlined
250         private <C extends Collection<T>> GenericCollectionPropertyBuilder<T, C> toCollection(Supplier<C> emptyCollSupplier) {
251             if (isDefaultValueSet()) {
252                 throw new IllegalStateException("The default value is already set!");
253             }
254 
255             GenericCollectionPropertyBuilder<T, C> result = new GenericCollectionPropertyBuilder<>(getName(),
256                                                                                                    getParser(),
257                                                                                                    emptyCollSupplier,
258                                                                                                    getType());
259 
260             for (PropertyConstraint<? super T> validator : getConstraints()) {
261                 result.require(validator.toCollectionConstraint());
262             }
263 
264             return result;
265 
266         }
267 
268 
269         @Override
270         public PropertyDescriptor<T> build() {
271             return new GenericPropertyDescriptor<>(
272                     getName(),
273                     getDescription(),
274                     0f,
275                     getDefaultValue(),
276                     getConstraints(),
277                     parser,
278                     isDefinedExternally,
279                     type
280             );
281         }
282     }

302      * Specialized builder for regex properties. Allows specifying the pattern as a
303      * string, with {@link #defaultValue(String)}.

306      * @since 6.10.0

311             super(name, ValueParserConstants.REGEX_PARSER, Pattern.class);

315         /**
316          * Specify a default value using a string pattern. The argument is
317          * compiled to a pattern using {@link Pattern#compile(String)}.
318          *
319          * @param pattern String pattern
320          *
321          * @return The same builder
322          *
323          * @throws java.util.regex.PatternSyntaxException If the argument is not a valid pattern
324          */
325         public RegexPropertyBuilder defaultValue(String pattern) {
326             super.defaultValue(Pattern.compile(pattern));
327             return this;
328         }

331         /**
332          * Specify a default value using a string pattern. The argument is
333          * compiled to a pattern using {@link Pattern#compile(String, int)}.
334          *
335          * @param pattern String pattern
336          * @param flags   Regex compilation flags, the same as for {@link Pattern#compile(String, int)}
337          *
338          * @return The same builder
339          *
340          * @throws java.util.regex.PatternSyntaxException If the argument is not a valid pattern
341          * @throws IllegalArgumentException               If bit values other than those corresponding to the defined

344         public RegexPropertyBuilder defaultValue(String pattern, int flags) {
345             super.defaultValue(Pattern.compile(pattern, flags));
346             return this;

351     /**
352      * Generic builder for a collection-valued property.
353      * This class adds methods related to {@link #defaultValue(Collection)}
354      * to make its use more flexible. See e.g. {@link #defaultValues(Object, Object[])}.
355      *
356      * <p>Note: this is designed to support arbitrary collections.
357      * Pre-7.0.0, the only collections available from the {@link PropertyFactory}
358      * are list types though.
359      *
360      * @param <V> Component type of the collection
361      * @param <C> Collection type for the property being built
362      *
363      * @author Clément Fournier
364      * @since 6.10.0
365      */
366     public static final class GenericCollectionPropertyBuilder<V, C extends Collection<V>> extends PropertyBuilder<GenericCollectionPropertyBuilder<V, C>, C> {
367         private final ValueParser<V> parser;
368         private final Supplier<C> emptyCollSupplier;
369         private final Class<V> type;
370         private char multiValueDelimiter = MultiValuePropertyDescriptor.DEFAULT_DELIMITER;
371 
372 
373         /**
374          * Builds a new builder for a collection type. Package-private.
375          */
376         GenericCollectionPropertyBuilder(String name,
377                                          ValueParser<V> parser,
378                                          Supplier<C> emptyCollSupplier,
379                                          Class<V> type) {
380             super(name);
381             this.parser = parser;
382             this.emptyCollSupplier = emptyCollSupplier;
383             this.type = type;
384         }
385 
386 
387         private C getDefaultValue(Collection<? extends V> list) {
388             C coll = emptyCollSupplier.get();
389             coll.addAll(list);
390             return coll;
391         }
392 
393 
394         /**
395          * Specify a default value.
396          *
397          * @param val List of values
398          *
399          * @return The same builder
400          */
401         @SuppressWarnings("unchecked")
402         public GenericCollectionPropertyBuilder<V, C> defaultValue(Collection<? extends V> val) {
403             super.defaultValue(getDefaultValue(val));
404             return this;
405         }
406 
407 
408         /**
409          * Specify default values. To specify an empty
410          * default value, use {@link #emptyDefaultValue()}.
411          *
412          * @param head First value
413          * @param tail Rest of the values
414          *
415          * @return The same builder
416          */
417         @SuppressWarnings("unchecked")
418         public GenericCollectionPropertyBuilder<V, C> defaultValues(V head, V... tail) {
419             List<V> tmp = new ArrayList<>(tail.length + 1);
420             tmp.add(head);
421             tmp.addAll(Arrays.asList(tail));
422             return super.defaultValue(getDefaultValue(tmp));
423         }
424 
425 
426         /**
427          * Specify that the default value is an empty collection.
428          *
429          * @return The same builder
430          */
431         public GenericCollectionPropertyBuilder<V, C> emptyDefaultValue() {
432             return super.defaultValue(getDefaultValue(Collections.<V>emptyList()));
433         }
434 
435 
436         /**
437          * Require that the given constraint be fulfilled on each item of the
438          * value of this property. This is a convenient shorthand for
439          * {@code require(constraint.toCollectionConstraint())}.
440          *
441          * @param constraint Constraint to impose on the items of the collection value
442          *
443          * @return The same builder
444          */
445         public GenericCollectionPropertyBuilder<V, C> requireEach(PropertyConstraint<? super V> constraint) {
446             return super.require(constraint.toCollectionConstraint());
447         }
448 
449 
450         /**
451          * Specify a delimiter character. By default it's {@link MultiValuePropertyDescriptor#DEFAULT_DELIMITER}, or {@link
452          * MultiValuePropertyDescriptor#DEFAULT_NUMERIC_DELIMITER} for numeric properties.
453          *
454          * @param delim Delimiter
455          *
456          * @return The same builder
457          *
458          * @deprecated PMD 7.0.0 will introduce a new XML syntax for multi-valued properties which will not rely on delimiters.
459          * This method is kept until this is implemented for compatibility reasons with the pre-7.0.0 framework, but
460          * it will be scrapped come 7.0.0.
461          */
462         @Deprecated
463         public GenericCollectionPropertyBuilder<V, C> delim(char delim) {
464             this.multiValueDelimiter = delim;
465             return this;
466         }
467 
468 
469         @SuppressWarnings("unchecked")
470         @Override
471         public PropertyDescriptor<C> build() {
472             // Note: the unchecked cast is safe because pre-7.0.0,
473             // we only allow building property descriptors for lists.
474             // C is thus always List<V>, and the cast doesn't fail
475 
476             // Post-7.0.0, the multi-value property classes will be removed
477             // and C will be the actual type parameter of the returned property
478             // descriptor
479 
480             return (PropertyDescriptor<C>) new GenericMultiValuePropertyDescriptor<>(
481                     getName(),
482                     getDescription(),
483                     0f,
484                     getDefaultValue(),
485                     getConstraints(),
486                     parser,
487                     multiValueDelimiter,
488                     type
489             );
490         }
