143     /**
144      * Dedicated method to support I10N and conversion errors
145      *
146      * @param key message which contains formatting string
147      * @param expr that should be formatted
148      * @return formatted expr with format specified by key
149      */
150     public String getFormatted(String key, String expr) {
151         Map<String, ConversionData> conversionErrors = ActionContext.getContext().getConversionErrors();
152         if (conversionErrors.containsKey(expr)) {
153             String[] vals = (String[]) conversionErrors.get(expr).getValue();
154             return vals[0];
155         } else {
156             final ValueStack valueStack = ActionContext.getContext().getValueStack();
157             final Object val = valueStack.findValue(expr);
158             return getText(key, Arrays.asList(val));
159         }
160     }

305     /**
306      * If called first time it will create {@link com.opensymphony.xwork2.TextProviderFactory},
307      * inject dependency (if {@link com.opensymphony.xwork2.inject.Container} is accesible) into in,
308      * then will create new {@link com.opensymphony.xwork2.TextProvider} and store it in a field
309      * for future references and at the returns reference to that field
310      *
311      * @return reference to field with TextProvider
312      */
313     protected TextProvider getTextProvider() {
314         if (textProvider == null) {
315             final TextProviderFactory tpf = getContainer().getInstance(TextProviderFactory.class);
316             textProvider = tpf.createInstance(getClass());
317         }
318         return textProvider;
319     }
320 
321     protected LocaleProvider getLocaleProvider() {
322         if (localeProvider == null) {
323             final LocaleProviderFactory localeProviderFactory = getContainer().getInstance(LocaleProviderFactory.class);
324             localeProvider = localeProviderFactory.createLocaleProvider();
325         }
326         return localeProvider;
327     }

329     /**
330      * TODO: This a temporary solution, maybe we should consider stop injecting container into beans
331      */
332     protected Container getContainer() {
333         if (container == null) {
334             container = ActionContext.getContext().getContainer();
335             if (container != null) {
336                 boolean devMode = Boolean.parseBoolean(container.getInstance(String.class, StrutsConstants.STRUTS_DEVMODE));
337                 if (devMode) {
338                     LOG.warn("Container is null, action was created manually? Fallback to ActionContext");
339                 } else {
340                     LOG.debug("Container is null, action was created manually? Fallback to ActionContext");
341                 }
342             } else {
343                 LOG.warn("Container is null, action was created out of ActionContext scope?!?");
344             }
345         }
346         return container;
347     }
