203     @Test
204     public void multiLocationGeoDistanceTest() throws Exception {
205         try {
206             client().admin().indices().prepareDelete("test").execute().actionGet();
207         } catch (Exception e) {
208             // ignore
209         }
210         String mapping = XContentFactory.jsonBuilder().startObject().startObject("type1")
211                 .startObject("properties").startObject("location").field("type", "geo_point").field("lat_lon", true).endObject().endObject()
212                 .endObject().endObject().string();
213         client().admin().indices().prepareCreate("test").addMapping("type1", mapping).execute().actionGet();
214         client().admin().cluster().prepareHealth().setWaitForEvents(Priority.LANGUID).setWaitForGreenStatus().execute().actionGet();
215 
216         client().prepareIndex("test", "type1", "1").setSource(jsonBuilder().startObject()
217                 .field("num", 1)
218                 .startArray("location")
219                         // to NY: 0
220                 .startObject().field("lat", 40.7143528).field("lon", -74.0059731).endObject()
221                         // to NY: 5.286 km
222                 .startObject().field("lat", 40.759011).field("lon", -73.9844722).endObject()
223                 .endArray()
224                 .endObject()).execute().actionGet();
225 
226         client().prepareIndex("test", "type1", "3").setSource(jsonBuilder().startObject()
227                 .field("num", 3)
228                 .startArray("location")
229                         // to NY: 0.4621 km
230                 .startObject().field("lat", 40.718266).field("lon", -74.007819).endObject()
231                         // to NY: 1.055 km
232                 .startObject().field("lat", 40.7051157).field("lon", -74.0088305).endObject()
233                 .endArray()
234                 .endObject()).execute().actionGet();
235 
236 
237         client().admin().indices().prepareRefresh().execute().actionGet();
238 
239         SearchResponse searchResponse = client().prepareSearch() // from NY
240                 .setQuery(matchAllQuery())
241                 .addFacet(geoDistanceFacet("geo1").field("location").point(40.7143528, -74.0059731).unit(DistanceUnit.KILOMETERS)
242                         .addRange(0, 2)
243                         .addRange(2, 10)
244                 )
245                 .execute().actionGet();
246 
247         assertNoFailures(searchResponse);
248 
249         assertThat(searchResponse.getHits().totalHits(), equalTo(2l));
250         GeoDistanceFacet facet = searchResponse.getFacets().facet("geo1");
251         assertThat(facet.getEntries().size(), equalTo(2));
252 
253         assertThat(facet.getEntries().get(0).getFrom(), closeTo(0, 0.000001));
254         assertThat(facet.getEntries().get(0).getTo(), closeTo(2, 0.000001));
255         assertThat(facet.getEntries().get(0).getCount(), equalTo(2l));
256 
257         assertThat(facet.getEntries().get(1).getFrom(), closeTo(2, 0.000001));
258         assertThat(facet.getEntries().get(1).getTo(), closeTo(10, 0.000001));
259         assertThat(facet.getEntries().get(1).getCount(), equalTo(1l));
260     }
