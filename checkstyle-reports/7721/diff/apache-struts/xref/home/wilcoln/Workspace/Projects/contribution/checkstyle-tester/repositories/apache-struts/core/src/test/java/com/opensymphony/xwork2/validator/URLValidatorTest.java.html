145 
146         assertFalse(pattern.matcher("myapp://test.com").matches());

189     public void testLongRunningValidations() throws Exception {
190         URLValidator validator = new URLValidator();
191 
192         Pattern pattern = Pattern.compile(validator.getUrlRegex(), Pattern.CASE_INSENSITIVE);
193 
194         long time = System.currentTimeMillis();
195         assertFalse(pattern.matcher("ftp://aaaaaaaaaaaaaaaaaaaaaaaa|").matches());
196         assertTrue("Validation did not complete in half a second", System.currentTimeMillis() - time < 500);
197 
198         time = System.currentTimeMillis();
199         assertFalse(pattern.matcher("ftp://bbbbbbbbbbbbbbbbbbbbbbbb}").matches());
200         assertTrue("Validation did not complete in half a second", System.currentTimeMillis() - time < 500);
201 
202         time = System.currentTimeMillis();
203         assertFalse(pattern.matcher("ftp://cccccccccccccccccccccccc{").matches());
204         assertTrue("Validation did not complete in half a second", System.currentTimeMillis() - time < 500);
205     }

207     public void testValidUrlCaseInsensitive() throws Exception {
208         // given
209         final Map<String, Object> fieldErrors = new HashMap<>();
210 
211         URLValidator validator = new URLValidator() {
212             @Override
213             public String getFieldName() {
214                 return "url";
215             }
216 
217             @Override
218             protected Object getFieldValue(String name, Object object) throws ValidationException {
219                 return object;
220             }
221 
222             @Override
223             protected void addFieldError(String propertyName, Object object) {
224                 fieldErrors.put(propertyName, object);
225             }
226         };
227 
228         // when
229         validator.validate("http://localhost:8080/myapp");
230 
231         // then
232         assertTrue(fieldErrors.isEmpty());
233 
234         // when
235         validator.validate("http://LOCALHOST:8080/MYAPP");
236 
237         // then
238         assertTrue(fieldErrors.isEmpty());
239 
240         // when
241         validator.validate("http://www.appache.org/TEST");
242 
243         // then
244         assertTrue(fieldErrors.isEmpty());
245     }
