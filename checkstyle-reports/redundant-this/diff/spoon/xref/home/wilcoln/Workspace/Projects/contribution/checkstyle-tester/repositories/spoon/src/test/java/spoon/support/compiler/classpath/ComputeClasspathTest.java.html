17  package spoon.support.compiler.classpath;
18  
19  import org.junit.Before;
20  import org.junit.Test;
21  import spoon.Launcher;
22  import spoon.SpoonModelBuilder;
23  import spoon.compiler.builder.ClasspathOptions;
24  import spoon.reflect.factory.Factory;
25  import spoon.support.compiler.jdt.JDTBasedSpoonCompiler;
26  
27  import java.io.File;
28  
29  import static org.junit.Assert.assertEquals;
30  
31  public class ComputeClasspathTest {
32  
33  	private static final String TEST_CLASSPATH =
34  			"./src/test/java/spoon/test/annotation/"
35  					+ File.pathSeparator
36  					+ "./src/test/java/spoon/test/api/"
37  					+ File.pathSeparator
38  					+ "./src/test/java/spoon/test/arrays/"
39  					+ File.pathSeparator
40  					+ "./src/test/java/spoon/test/casts/"
41  					+ File.pathSeparator;
42  
43  	private JDTBasedSpoonCompiler compiler;
44  	private Class<? extends JDTBasedSpoonCompiler> compilerClass;
45  
46  	private String[] systemClasspath;
47  
48  	@Before
49  	public void setUp() {
50  		Launcher launcher = new Launcher() {
51  
52  			public SpoonModelBuilder createCompiler(Factory factory) {
53  				return new JDTBasedSpoonCompiler(factory);
54  			}
55  
56  		};
57  		launcher.getEnvironment().setLevel("OFF");
58  
59  		this.compiler = (JDTBasedSpoonCompiler) launcher.createCompiler();
60  		this.compilerClass = compiler.getClass();
61  
62  		this.systemClasspath = TEST_CLASSPATH.split(File.pathSeparator);
63  	}
64  
65  	@Test
66  	public void testSourceClasspath() {
67  		final ClasspathOptions options = new ClasspathOptions().classpath(systemClasspath);
68  		assertEquals("-cp " + TEST_CLASSPATH, String.join(" ", options.build()));
69  	}
70  }
