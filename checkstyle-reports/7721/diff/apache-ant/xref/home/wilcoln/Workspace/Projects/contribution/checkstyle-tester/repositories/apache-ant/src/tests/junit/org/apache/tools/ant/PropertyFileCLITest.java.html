19  package org.apache.tools.ant;
20  
21  import java.io.File;
22  import java.io.FileReader;
23  import java.io.FileWriter;
24  import org.apache.tools.ant.util.FileUtils;
25  import org.junit.Test;
26  
27  import static org.apache.tools.ant.AntAssert.assertContains;
28  
29  public class PropertyFileCLITest {
30  
31      @Test
32      public void testPropertyResolution() throws Exception {
33          FileUtils fu = FileUtils.getFileUtils();
34          File props = fu.createTempFile("propertyfilecli", ".properties",
35                                         null, true, true);
36          File build = fu.createTempFile("propertyfilecli", ".xml", null, true,
37                                         true);
38          File log = fu.createTempFile("propertyfilecli", ".log", null, true,
39                                       true);
40          FileWriter fw = null;
41          FileReader fr = null;
42          try {
43              fw = new FileWriter(props);
44              fw.write("w=world\nmessage=Hello, ${w}\n");
45              fw.close();
46              fw = new FileWriter(build);
47              fw.write("<project><echo>${message}</echo></project>");
48              fw.close();
49              fw = null;
50              Main m = new NoExitMain();
51              m.startAnt(new String[] {
52                      "-propertyfile", props.getAbsolutePath(),
53                      "-f", build.getAbsolutePath(),
54                      "-l", log.getAbsolutePath()
55                  }, null, null);
56              String l = FileUtils.safeReadFully(fr = new FileReader(log));
57              assertContains("Hello, world", l);
58          } finally {
59              FileUtils.close(fw);
60              FileUtils.close(fr);
61          }
62      }
63  
64      private static class NoExitMain extends Main {
65          protected void exit(int exitCode) {
66          }
67      }
68  }
