98      @Inject
99      public TransportClientNodesService(Settings settings, ClusterName clusterName, TransportService transportService,
100                                        ThreadPool threadPool, Headers headers, Version version) {
101         super(settings);
102         this.clusterName = clusterName;
103         this.transportService = transportService;
104         this.threadPool = threadPool;
105         this.minCompatibilityVersion = version.minimumCompatibilityVersion();
106         this.headers = headers;
107 
108         this.nodesSamplerInterval = componentSettings.getAsTime("nodes_sampler_interval", timeValueSeconds(5));
109         this.pingTimeout = componentSettings.getAsTime("ping_timeout", timeValueSeconds(5)).millis();
110         this.ignoreClusterName = componentSettings.getAsBoolean("ignore_cluster_name", false);
111 
112         if (logger.isDebugEnabled()) {
113             logger.debug("node_sampler_interval[" + nodesSamplerInterval + "]");
114         }
115 
116         if (componentSettings.getAsBoolean("sniff", false)) {
117             this.nodesSampler = new SniffNodesSampler();
118         } else {
119             this.nodesSampler = new SimpleNodeSampler();
120         }
121         this.nodesSamplerFuture = threadPool.schedule(nodesSamplerInterval, ThreadPool.Names.GENERIC, new ScheduledNodeSampler());
122     }

144     public TransportClientNodesService addTransportAddresses(TransportAddress... transportAddresses) {
145         synchronized (mutex) {
146             if (closed) {
147                 throw new ElasticsearchIllegalStateException("transport client is closed, can't add an address");
148             }
149             List<TransportAddress> filtered = Lists.newArrayListWithExpectedSize(transportAddresses.length);
150             for (TransportAddress transportAddress : transportAddresses) {
151                 boolean found = false;
152                 for (DiscoveryNode otherNode : listedNodes) {
153                     if (otherNode.address().equals(transportAddress)) {
154                         found = true;
155                         logger.debug("address [{}] already exists with [{}], ignoring...", transportAddress, otherNode);
156                         break;
157                     }
158                 }
159                 if (!found) {
160                     filtered.add(transportAddress);
161                 }
162             }
163             if (filtered.isEmpty()) {
164                 return this;
165             }
166             ImmutableList.Builder<DiscoveryNode> builder = ImmutableList.builder();
167             builder.addAll(listedNodes());
168             for (TransportAddress transportAddress : filtered) {
169                 DiscoveryNode node = new DiscoveryNode("#transport#-" + tempNodeIdGenerator.incrementAndGet(), transportAddress, minCompatibilityVersion);
170                 logger.debug("adding address [{}]", node);
171                 builder.add(node);
172             }
173             listedNodes = builder.build();
174             nodesSampler.sample();
175         }
176         return this;
177     }
178 
179     public TransportClientNodesService removeTransportAddress(TransportAddress transportAddress) {
180         synchronized (mutex) {
181             if (closed) {
182                 throw new ElasticsearchIllegalStateException("transport client is closed, can't remove an address");
183             }
184             ImmutableList.Builder<DiscoveryNode> builder = ImmutableList.builder();
185             for (DiscoveryNode otherNode : listedNodes) {
186                 if (!otherNode.address().equals(transportAddress)) {
187                     builder.add(otherNode);
188                 } else {
189                     logger.debug("removing address [{}]", otherNode);
190                 }
191             }
192             listedNodes = builder.build();
193             nodesSampler.sample();
194         }
195         return this;
196     }

198     public <Response> void execute(NodeListenerCallback<Response> callback, ActionListener<Response> listener) throws ElasticsearchException {
199         ImmutableList<DiscoveryNode> nodes = this.nodes;
200         ensureNodesAreAvailable(nodes);
201         int index = getNodeNumber();
202         RetryListener<Response> retryListener = new RetryListener<>(callback, listener, nodes, index, threadPool, logger);
203         DiscoveryNode node = nodes.get((index) % nodes.size());
204         try {
205             callback.doWithNode(node, retryListener);
206         } catch (Throwable t) {
207             //this exception can't come from the TransportService as it doesn't throw exception at all
208             listener.onFailure(t);
209         }
210     }

212     public static class RetryListener<Response> implements ActionListener<Response> {
213         private final NodeListenerCallback<Response> callback;
214         private final ActionListener<Response> listener;
215         private final ImmutableList<DiscoveryNode> nodes;
216         private final ESLogger logger;
217         private final int index;
218         private ThreadPool threadPool;
219 
220         private volatile int i;
221 
222         public RetryListener(NodeListenerCallback<Response> callback, ActionListener<Response> listener, ImmutableList<DiscoveryNode> nodes,
223                              int index, ThreadPool threadPool, ESLogger logger) {
224             this.callback = callback;
225             this.listener = listener;
226             this.nodes = nodes;
227             this.index = index;
228             this.threadPool = threadPool;
229             this.logger = logger;
230         }
231 
232         @Override
233         public void onResponse(Response response) {
234             listener.onResponse(response);
235         }
236 
237         @Override
238         public void onFailure(Throwable e) {
239             if (ExceptionsHelper.unwrapCause(e) instanceof ConnectTransportException) {
240                 int i = ++this.i;
241                 if (i >= nodes.size()) {
242                     runFailureInListenerThreadPool(new NoNodeAvailableException("None of the configured nodes were available: " + nodes, e));
243                 } else {
244                     try {
245                         callback.doWithNode(nodes.get((index + i) % nodes.size()), this);
246                     } catch(final Throwable t) {
247                         // this exception can't come from the TransportService as it doesn't throw exceptions at all
248                         runFailureInListenerThreadPool(t);
249                     }
250                 }
251             } else {
252                 runFailureInListenerThreadPool(e);
253             }
254         }
255 
256         // need to ensure to not block the netty I/O thread, in case of retry due to the node sampling
257         private void runFailureInListenerThreadPool(final Throwable t) {
258             threadPool.executor(ThreadPool.Names.LISTENER).execute(new AbstractRunnable() {
259                 @Override
260                 protected void doRun() throws Exception {
261                     listener.onFailure(t);
262                 }
263 
264                 @Override
265                 public void onFailure(Throwable t) {
266                     if (logger.isDebugEnabled()) {
267                         logger.debug("Could not execute failure listener: [{}]", t, t.getMessage());
268                     } else {
269                         logger.error("Could not execute failure listener: [{}]", t.getMessage());
270                     }
271                 }
272             });
273         }
274     }
275 
276     public void close() {
277         synchronized (mutex) {
278             if (closed) {
279                 return;
280             }
281             closed = true;
282             FutureUtils.cancel(nodesSamplerFuture);
283             for (DiscoveryNode node : nodes) {
284                 transportService.disconnectFromNode(node);
285             }
286             for (DiscoveryNode listedNode : listedNodes) {
287                 transportService.disconnectFromNode(listedNode);
288             }
289             nodes = ImmutableList.of();
290         }
291     }
292 
293     private int getNodeNumber() {
294         int index = randomNodeGenerator.incrementAndGet();
295         if (index < 0) {
296             index = 0;
297             randomNodeGenerator.set(0);
298         }
299         return index;
300     }
301 
302     private void ensureNodesAreAvailable(ImmutableList<DiscoveryNode> nodes) {
303         if (nodes.isEmpty()) {
304             String message = String.format(Locale.ROOT, "None of the configured nodes are available: %s", nodes);
305             throw new NoNodeAvailableException(message);
306         }
307     }

309     abstract class NodeSampler {
310         public void sample() {
311             synchronized (mutex) {
312                 if (closed) {
313                     return;
314                 }
315                 doSample();
316             }
317         }
318 
319         protected abstract void doSample();
320 
321         /**
322          * validates a set of potentially newly discovered nodes and returns an immutable
323          * list of the nodes that has passed.
324          */
325         protected ImmutableList<DiscoveryNode> validateNewNodes(Set<DiscoveryNode> nodes) {
326             for (Iterator<DiscoveryNode> it = nodes.iterator(); it.hasNext(); ) {
327                 DiscoveryNode node = it.next();
328                 if (!transportService.nodeConnected(node)) {
329                     try {
330                         logger.trace("connecting to node [{}]", node);
331                         transportService.connectToNode(node);
332                     } catch (Throwable e) {
333                         it.remove();
334                         logger.debug("failed to connect to discovered node [" + node + "]", e);
335                     }
336                 }
337             }
338 
339             return new ImmutableList.Builder<DiscoveryNode>().addAll(nodes).build();
340         }
341 
342     }

344     class ScheduledNodeSampler implements Runnable {
345         @Override
346         public void run() {
347             try {
348                 nodesSampler.sample();
349                 if (!closed) {
350                     nodesSamplerFuture = threadPool.schedule(nodesSamplerInterval, ThreadPool.Names.GENERIC, this);
351                 }
352             } catch (Exception e) {
353                 logger.warn("failed to sample", e);
354             }
355         }
356     }

358     class SimpleNodeSampler extends NodeSampler {
359 
360         @Override
361         protected void doSample() {
362             HashSet<DiscoveryNode> newNodes = new HashSet<>();
363             HashSet<DiscoveryNode> newFilteredNodes = new HashSet<>();
364             for (DiscoveryNode listedNode : listedNodes) {
365                 if (!transportService.nodeConnected(listedNode)) {
366                     try {
367                         // its a listed node, light connect to it...
368                         logger.trace("connecting to listed node (light) [{}]", listedNode);
369                         transportService.connectToNodeLight(listedNode);
370                     } catch (Throwable e) {
371                         logger.debug("failed to connect to node [{}], removed from nodes list", e, listedNode);
372                         continue;
373                     }
374                 }
375                 try {
376                     NodesInfoResponse nodeInfo = transportService.submitRequest(listedNode, NodesInfoAction.NAME,
377                             headers.applyTo(Requests.nodesInfoRequest("_local").clear()),
378                             TransportRequestOptions.options().withType(TransportRequestOptions.Type.STATE).withTimeout(pingTimeout),
379                             new FutureTransportResponseHandler<NodesInfoResponse>() {
380                                 @Override
381                                 public NodesInfoResponse newInstance() {
382                                     return new NodesInfoResponse();
383                                 }
384                             }).txGet();
385                     if (!ignoreClusterName && !clusterName.equals(nodeInfo.getClusterName())) {
386                         logger.warn("node {} not part of the cluster {}, ignoring...", listedNode, clusterName);
387                         newFilteredNodes.add(listedNode);
388                     } else if (nodeInfo.getNodes().length != 0) {
389                         // use discovered information but do keep the original transport address, so people can control which address is exactly used.
390                         DiscoveryNode nodeWithInfo = nodeInfo.getNodes()[0].getNode();
391                         newNodes.add(new DiscoveryNode(nodeWithInfo.name(), nodeWithInfo.id(), nodeWithInfo.getHostName(), nodeWithInfo.getHostAddress(), listedNode.address(), nodeWithInfo.attributes(), nodeWithInfo.version()));
392                     } else {
393                         // although we asked for one node, our target may not have completed initialization yet and doesn't have cluster nodes
394                         logger.debug("node {} didn't return any discovery info, temporarily using transport discovery node", listedNode);
395                         newNodes.add(listedNode);
396                     }
397                 } catch (Throwable e) {
398                     logger.info("failed to get node info for {}, disconnecting...", e, listedNode);
399                     transportService.disconnectFromNode(listedNode);
400                 }
401             }
402 
403             nodes = validateNewNodes(newNodes);
404             filteredNodes = ImmutableList.copyOf(newFilteredNodes);
405         }
406     }

408     class SniffNodesSampler extends NodeSampler {
409 
410         @Override
411         protected void doSample() {
412             // the nodes we are going to ping include the core listed nodes that were added
413             // and the last round of discovered nodes
414             Set<DiscoveryNode> nodesToPing = Sets.newHashSet();
415             for (DiscoveryNode node : listedNodes) {
416                 nodesToPing.add(node);
417             }
418             for (DiscoveryNode node : nodes) {
419                 nodesToPing.add(node);
420             }
421 
422             final CountDownLatch latch = new CountDownLatch(nodesToPing.size());
423             final ConcurrentMap<DiscoveryNode, ClusterStateResponse> clusterStateResponses = ConcurrentCollections.newConcurrentMap();
424             for (final DiscoveryNode listedNode : nodesToPing) {
425                 threadPool.executor(ThreadPool.Names.MANAGEMENT).execute(new Runnable() {
426                     @Override
427                     public void run() {
428                         try {
429                             if (!transportService.nodeConnected(listedNode)) {
430                                 try {
431 
432                                     // if its one of the actual nodes we will talk to, not to listed nodes, fully connect
433                                     if (nodes.contains(listedNode)) {
434                                         logger.trace("connecting to cluster node [{}]", listedNode);
435                                         transportService.connectToNode(listedNode);
436                                     } else {
437                                         // its a listed node, light connect to it...
438                                         logger.trace("connecting to listed node (light) [{}]", listedNode);
439                                         transportService.connectToNodeLight(listedNode);
440                                     }
441                                 } catch (Exception e) {
442                                     logger.debug("failed to connect to node [{}], ignoring...", e, listedNode);
443                                     latch.countDown();
444                                     return;
445                                 }
446                             }
447                             transportService.sendRequest(listedNode, ClusterStateAction.NAME,
448                                     headers.applyTo(Requests.clusterStateRequest().clear().nodes(true).local(true)),
449                                     TransportRequestOptions.options().withType(TransportRequestOptions.Type.STATE).withTimeout(pingTimeout),
450                                     new BaseTransportResponseHandler<ClusterStateResponse>() {
451 
452                                         @Override
453                                         public ClusterStateResponse newInstance() {
454                                             return new ClusterStateResponse();
455                                         }
456 
457                                         @Override
458                                         public String executor() {
459                                             return ThreadPool.Names.SAME;
460                                         }
461 
462                                         @Override
463                                         public void handleResponse(ClusterStateResponse response) {
464                                             clusterStateResponses.put(listedNode, response);
465                                             latch.countDown();
466                                         }
467 
468                                         @Override
469                                         public void handleException(TransportException e) {
470                                             logger.info("failed to get local cluster state for {}, disconnecting...", e, listedNode);
471                                             transportService.disconnectFromNode(listedNode);
472                                             latch.countDown();
473                                         }
474                                     });
475                         } catch (Throwable e) {
476                             logger.info("failed to get local cluster state info for {}, disconnecting...", e, listedNode);
477                             transportService.disconnectFromNode(listedNode);
478                             latch.countDown();
479                         }
480                     }
481                 });
482             }
483 
484             try {
485                 latch.await();
486             } catch (InterruptedException e) {
487                 return;
488             }
489 
490             HashSet<DiscoveryNode> newNodes = new HashSet<>();
491             HashSet<DiscoveryNode> newFilteredNodes = new HashSet<>();
492             for (Map.Entry<DiscoveryNode, ClusterStateResponse> entry : clusterStateResponses.entrySet()) {
493                 if (!ignoreClusterName && !clusterName.equals(entry.getValue().getClusterName())) {
494                     logger.warn("node {} not part of the cluster {}, ignoring...", entry.getValue().getState().nodes().localNode(), clusterName);
495                     newFilteredNodes.add(entry.getKey());
496                     continue;
497                 }
498                 for (ObjectCursor<DiscoveryNode> cursor : entry.getValue().getState().nodes().dataNodes().values()) {
499                     newNodes.add(cursor.value);
500                 }
501             }
502 
503             nodes = validateNewNodes(newNodes);
504             filteredNodes = ImmutableList.copyOf(newFilteredNodes);
505         }
506     }
