39      private static final char SEP = File.separatorChar;
40      private static char[] tokenPwd;
41      private static final char[] nssPwd =
42              new char[]{'t', 'e', 's', 't', '1', '2'};
43      private static final char[] solarisPwd =
44              new char[]{'p', 'i', 'n'};
45      private static SecretKey sk1;
46      private static SecretKey sk2;
47      private static SecretKey softkey;
48      private static KeyStore ks;
49      private static final String KS_TYPE = "PKCS11";

56      public void main(Provider p) throws Exception {
57          this.provider = p;
58  
59          // create secret key
60          byte[] keyVal = new byte[16];
61          (new SecureRandom()).nextBytes(keyVal);
62          // NSS will throw CKR_HOST_MEMORY if calling C_DecryptInit w/
63          // (keyVal[0] == 0)
64          if (keyVal[0] == 0) {
65              keyVal[0] = 1;
66          }
67          softkey = new SecretKeySpec(keyVal, "AES");
68          dumpKey("softkey", softkey);
69  
70          KeyGenerator kg = KeyGenerator.getInstance("DESede", provider);
71          sk1 = kg.generateKey();
72          dumpKey("skey1", sk1);
73          sk2 = kg.generateKey();
74          dumpKey("skey2", sk2);
75  
76          String token = System.getProperty("TOKEN");
77  
78          if (token == null || token.length() == 0) {
79              System.out.println("Error: missing TOKEN system property");
80              throw new Exception("token arg required");
81          }
82  
83          if ("nss".equals(token)) {
84              tokenPwd = nssPwd;
85          } else if ("solaris".equals(token)) {
86              tokenPwd = solarisPwd;
87          }
88  
89          int testnum = 1;
90          doTest();
91      }
92  
93      private static boolean checkSecretKeyEntry(String alias,
94              SecretKey expected,
95              boolean saveBeforeCheck)
96              throws Exception {
97          if (saveBeforeCheck) {
98              ks.setKeyEntry(alias, expected, null, null);
99          }
100         SecretKey result = (SecretKey) (ks.getKey(alias, null));
101         String keyEncFormat = result.getFormat();
102         if (keyEncFormat == null) {
103             // sensitive or un-extractable keys - verify by encrypt/decrypt
104             byte[] data = new byte[64];
105             Cipher c =
106                     Cipher.getInstance(result.getAlgorithm() + "/CBC/NoPadding",
107                     provider);
108             c.init(Cipher.ENCRYPT_MODE, expected);
109             byte[] encOut = c.doFinal(data);
110             c.init(Cipher.DECRYPT_MODE, result, c.getParameters());
111             byte[] decOut = c.doFinal(encOut);
112             if (!Arrays.equals(data, decOut)) {
113                 return false;
114             }
115         } else if (keyEncFormat.toUpperCase().equals("RAW")) {
116             if (!Arrays.equals(result.getEncoded(), expected.getEncoded())) {
117                 dumpKey("\texpected:", expected);
118                 dumpKey("\treturns:", result);
119                 return false;
120             }
121         }
122         return true;
123     }
124 
125     private static void dumpKey(String info, SecretKey key) {
126         System.out.println(info + "> " + key);
127         System.out.println("\tALGO=" + key.getAlgorithm());
128         if (key.getFormat() != null) {
129             System.out.println("\t[" + key.getFormat() + "] VALUE=" +
130                     new BigInteger(key.getEncoded()));
131         } else {
132             System.out.println("\tVALUE=n/a");
133         }
134     }
135 
136     private static void doTest() throws Exception {
137         if (ks == null) {
138             ks = KeyStore.getInstance(KS_TYPE, provider);
139             ks.load(null, tokenPwd);
140         }
141 
142         System.out.println("Number of entries: " + ks.size());
143         if (ks.size() != 0) {
144             System.out.println("Deleting entries under aliases: ");
145             for (Enumeration<String> aliases = ks.aliases();
146                     aliases.hasMoreElements();) {
147                 String alias = aliases.nextElement();
148                 System.out.println("\t" + alias);
149                 ks.deleteEntry(alias);
150             }
151         }
152 
153         String alias = "testSKey";
154 
155         boolean testResult = checkSecretKeyEntry(alias, softkey, true);
156         if (!testResult) {
157             System.out.println("FAILURE: setKey() w/ softSecretKey failed");
158         }
159 
160         if (!checkSecretKeyEntry(alias, sk1, true)) {
161             testResult = false;
162             System.out.println("FAILURE: setKey() w/ skey1 failed");
163         }
164         if (!checkSecretKeyEntry(alias, sk2, true)) {
165             testResult = false;
166             System.out.println("FAILURE: setKey() w/ skey2 failed");
167         }
168 
169         ks.store(null);
170         System.out.println("Reloading keystore...");
171 
172         ks.load(null, "whatever".toCharArray());
173         if (ks.size() != 1) {
174             System.out.println("FAILURE: reload#1 ks.size() != 1");
175         }
176         if (!checkSecretKeyEntry(alias, sk2, false)) {
177             testResult = false;
178             System.out.println("FAILURE: reload#1 ks entry check failed");
179         }
180 
181         ks.deleteEntry(alias);
182         ks.store(null);
183 
184         System.out.println("Reloading keystore...");
185         ks.load(null, "whatever".toCharArray());
186         if (ks.size() != 0) {
187             testResult = false;
188             System.out.println("FAILURE: reload#2 ks.size() != 0");
189         }
190         if (!testResult) {
191             throw new Exception("One or more test failed!");
192         }
193     }
