17  package spoon.test.processing.processors;
18  
19  import spoon.processing.AbstractProcessor;
20  import spoon.reflect.declaration.CtElement;
21  import spoon.reflect.declaration.CtNamedElement;
22  import spoon.reflect.reference.CtReference;
23  
24  public class RenameProcessor extends AbstractProcessor<CtElement> {
25      private String oldName;
26      private String newName;
27  
28      public RenameProcessor(String oldName, String newName) {
29          this.oldName = oldName;
30          this.newName = newName;
31      }
32  
33      @Override
34      public boolean isToBeProcessed(CtElement candidate) {
35          if (candidate instanceof CtNamedElement) {
36              CtNamedElement namedElement = (CtNamedElement) candidate;
37              return namedElement.getSimpleName().equals(oldName);
38          } else if (candidate instanceof CtReference) {
39              CtReference reference = (CtReference) candidate;
40              return reference.getSimpleName().equals(oldName);
41          }
42          return false;
43      }
44  
45      @Override
46      public void process(CtElement element) {
47          if (element instanceof CtNamedElement) {
48              CtNamedElement namedElement = (CtNamedElement) element;
49              namedElement.setSimpleName(this.newName);
50          } else if (element instanceof CtReference) {
51              CtReference reference = (CtReference) element;
52              reference.setSimpleName(this.newName);
53          }
54      }
55  }
