44  /**
45   * This class reads and provides solar activity data needed by
46   * atmospheric models: F107 solar flux and Kp indexes.
47   * <p>
48   * The data are retrieved through the NASA Marshall
49   * Solar Activity Future Estimation (MSAFE) as estimates of monthly
50   * F10.7 Mean solar flux and Ap geomagnetic parameter.
51   * The data can be retrieved at the NASA <a
52   * href="http://sail.msfc.nasa.gov/archive_index.htm">
53   * Marshall Solar Activity website</a>.
54   * Here Kp indices are deduced from Ap indexes, which in turn are tabulated
55   * equivalent of retrieved Ap values.
56   * </p>
57   * <p>
58   * If several MSAFE files are available, some dates may appear in several
59   * files (for example August 2007 is in all files from the first one
60   * published in March 1999 to the February 2008 file). In this case, the
61   * data from the most recent file is used and the older ones are discarded.
62   * The date of the file is assumed to be 6 months after its first entry
63   * (which explains why the file having August 2007 as its first entry is the
64   * February 2008 file). This implies that MSAFE files must <em>not</em> be
65   * edited to change their time span, otherwise this would break the old
66   * entries overriding mechanism.
67   * </p>
68   * <p>
69   * With these data, the {@link #getInstantFlux(AbsoluteDate)} and {@link
70   * #getMeanFlux(AbsoluteDate)} methods return the same values and the {@link
71   * #get24HoursKp(AbsoluteDate)} and {@link #getThreeHourlyKP(AbsoluteDate)}
72   * methods return the same values.
73   * </p>
74   * <p>
75   * Conversion from Ap index values in the MSAFE file to Kp values used by the atmosphere
76   * model is done using Jacchia's equation in [1].
77   * </p>
78   *
79   * <h2>References</h2>
80   *
81   * <ol> <li> Jacchia, L. G. "CIRA 1972, recent atmospheric models, and improvements in
82   * progress." COSPAR, 21st Plenary Meeting. Vol. 1. 1978. </li> </ol>
83   *
84   * @author Bruno Revelin
85   * @author Luc Maisonobe
86   * @author Evan Ward
87   */
88  public class MarshallSolarActivityFutureEstimation implements DTM2000InputParameters, DataLoader {

131     /** Simple constructor.
132      * <p>
133      * The original file names used by NASA Marshall space center are of the
134      * form: Dec2010F10.txt or Oct1999F10.TXT. So a recommended regular
135      * expression for the supported name that work with all published files is:
136      * "(?:Jan|Feb|Mar|Apr|May|Jun|Jul|Aug|Sep|Oct|Nov|Dec)\\p{Digit}\\p{Digit}\\p{Digit}\\p{Digit}F10\\.(?:txt|TXT)"
137      * </p>
138      * @param supportedNames regular expression for supported files names
139      * @param strengthLevel selected strength level of activity
140      */
141     public MarshallSolarActivityFutureEstimation(final String supportedNames,
142                                                  final StrengthLevel strengthLevel) {
143 
144         firstDate           = null;
145         lastDate            = null;
146         data                = new TreeSet<TimeStamped>(new ChronologicalComparator());
147         this.supportedNames = supportedNames;
148         this.strengthLevel  = strengthLevel;
149 
150         // the data lines have the following form:
151         // 2010.5003   JUL    83.4      81.3      78.7       6.4       5.9       5.2
152         // 2010.5837   AUG    87.3      83.4      78.5       7.0       6.1       4.9
153         // 2010.6670   SEP    90.8      85.5      79.4       7.8       6.2       4.7
154         // 2010.7503   OCT    94.2      87.6      80.4       9.1       6.4       4.9
155         final StringBuilder builder = new StringBuilder("^");
156 
157         // first group: year
158         builder.append("\\p{Blank}*(\\p{Digit}\\p{Digit}\\p{Digit}\\p{Digit})");
159 
160         // month as fraction of year, not stored in a group
161         builder.append("\\.\\p{Digit}+");
162 
163         // second group: month as a three upper case letters abbreviation
164         builder.append("\\p{Blank}+(");
165         for (final Month month : Month.values()) {
166             builder.append(month.getUpperCaseAbbreviation());
167             builder.append('|');
168         }
169         builder.delete(builder.length() - 1, builder.length());
170         builder.append(")");
171 
172         // third to eighth group: data fields
173         for (int i = 0; i < 6; ++i) {
174             builder.append("\\p{Blank}+([-+]?[0-9]+\\.[0-9]+)");
175         }
176 
177         // end of line
178         builder.append("\\p{Blank}*$");
179 
180         // compile the pattern
181         dataPattern = Pattern.compile(builder.toString());
182 
183     }

192     /** Find the data bracketing a specified date.
193      * @param date date to bracket
194      * @throws OrekitException if specified date is out of range
195      */
196     private void bracketDate(final AbsoluteDate date) throws OrekitException {
197 
198         if ((date.durationFrom(firstDate) < 0) || (date.durationFrom(lastDate) > 0)) {
199             throw new OrekitException(OrekitMessages.OUT_OF_RANGE_EPHEMERIDES_DATE,
200                                       date, firstDate, lastDate);
201         }
202 
203         // don't search if the cached selection is fine
204         if ((previousParam != null) &&
205             (date.durationFrom(previousParam.getDate()) > 0) &&
206             (date.durationFrom(currentParam.getDate()) <= 0 )) {
207             return;
208         }
209 
210         if (date.equals(firstDate)) {
211             currentParam  = (LineParameters) data.tailSet(date.shiftedBy(1)).first();
212             previousParam = (LineParameters) data.first();
213         } else if (date.equals(lastDate)) {
214             currentParam  = (LineParameters) data.last();
215             previousParam = (LineParameters) data.headSet(date.shiftedBy(-1)).last();
216         } else {
217             currentParam  = (LineParameters) data.tailSet(date).first();
218             previousParam = (LineParameters) data.headSet(date).last();
219         }
220 
221     }

251     /** {@inheritDoc} */
252     public double getMeanFlux(final AbsoluteDate date) throws OrekitException {
253 
254         // get the neighboring dates
255         bracketDate(date);
256 
257         // perform a linear interpolation
258         final AbsoluteDate previousDate = previousParam.getDate();
259         final AbsoluteDate currentDate  = currentParam.getDate();
260         final double dt                 = currentDate.durationFrom(previousDate);
261         final double previousF107       = previousParam.getF107();
262         final double currentF107        = currentParam.getF107();
263         final double previousWeight     = currentDate.durationFrom(date)  / dt;
264         final double currentWeight      = date.durationFrom(previousDate) / dt;
265 
266         return previousF107 * previousWeight + currentF107 * currentWeight;
267 
268     }

275     /** Get the date of the file from which data at the specified date comes from.
276      * <p>
277      * If several MSAFE files are available, some dates may appear in several
278      * files (for example August 2007 is in all files from the first one
279      * published in March 1999 to the February 2008 file). In this case, the
280      * data from the most recent file is used and the older ones are discarded.
281      * The date of the file is assumed to be 6 months after its first entry
282      * (which explains why the file having August 2007 as its first entry is the
283      * February 2008 file). This implies that MSAFE files must <em>not</em> be
284      * edited to change their time span, otherwise this would break the old
285      * entries overriding mechanism.
286      * </p>
287      * @param date date of the solar activity data
288      * @return date of the file
289      * @exception OrekitException if specified date is out of range
290      */
291     public DateComponents getFileDate(final AbsoluteDate date) throws OrekitException {
292         bracketDate(date);
293         final double dtP = date.durationFrom(previousParam.getDate());
294         final double dtC = currentParam.getDate().durationFrom(date);
295         return (dtP < dtC) ? previousParam.getFileDate() : currentParam.getFileDate();
296     }

298     /** The Kp index is derived from the Ap index.
299      * <p>The method used is explained on <a
300      * href="http://www.ngdc.noaa.gov/stp/GEOMAG/kp_ap.html">
301      * NOAA website.</a> as follows:</p>
302      * <p>The scale is 0 to 9 expressed in thirds of a unit, e.g. 5- is 4 2/3,
303      * 5 is 5 and 5+ is 5 1/3. The ap (equivalent range) index is derived from
304      * the Kp index as follows:</p>
305      * <table border="1">
306      * <caption>Kp / Ap Conversion Table</caption>
307      * <tbody>
308      * <tr>
309      * <td>Kp</td><td>0o</td><td>0+</td><td>1-</td><td>1o</td><td>1+</td><td>2-</td><td>2o</td><td>2+</td><td>3-</td><td>3o</td><td>3+</td><td>4-</td><td>4o</td><td>4+</td>
310      * </tr>
311      * <tr>
312      * <td>ap</td><td>0</td><td>2</td><td>3</td><td>4</td><td>5</td><td>6</td><td>7</td><td>9</td><td>12</td><td>15</td><td>18</td><td>22</td><td>27</td><td>32</td>
313      * </tr>
314      * <tr>
315      * <td>Kp</td><td>5-</td><td>5o</td><td>5+</td><td>6-</td><td>6o</td><td>6+</td><td>7-</td><td>7o</td><td>7+</td><td>8-</td><td>8o</td><td>8+</td><td>9-</td><td>9o</td>
316      * </tr>
317      * <tr>
318      * <td>ap</td><td>39</td><td>48</td><td>56</td><td>67</td><td>80</td><td>94</td><td>111</td><td>132</td><td>154</td><td>179</td><td>207</td><td>236</td><td>300</td><td>400</td>
319      * </tr>
320      * </tbody>
321      * </table>
322      * @param date date of the Kp data
323      * @return the 24H geomagnetic index
324      * @exception OrekitException if the date is out of range of available data
325      */
326     public double get24HoursKp(final AbsoluteDate date) throws OrekitException {
327 
328         // get the neighboring dates
329         bracketDate(date);
330 
331         // perform a linear interpolation
332         final AbsoluteDate previousDate = previousParam.getDate();
333         final AbsoluteDate currentDate  = currentParam.getDate();
334         final double dt                 = currentDate.durationFrom(previousDate);
335         final double previousAp         = previousParam.getAp();
336         final double currentAp          = currentParam.getAp();
337         final double previousWeight     = currentDate.durationFrom(date)  / dt;
338         final double currentWeight      = date.durationFrom(previousDate) / dt;
339         final double ap                 = previousAp * previousWeight + currentAp * currentWeight;
340 
341         // calculating Ap index, then corresponding Kp index
342         // equation 4 in [1] for Ap to Kp conversion
343         return 1.89 * FastMath.asinh(0.154 * ap);
344     }

407     /** {@inheritDoc} */
408     public void loadData(final InputStream input, final String name)
409         throws IOException, ParseException, OrekitException {
410 
411         // select the groups we want to store
412         final int f107Group;
413         final int apGroup;
414         switch (strengthLevel) {
415             case STRONG :
416                 f107Group = 3;
417                 apGroup   = 6;
418                 break;
419             case AVERAGE :
420                 f107Group = 4;
421                 apGroup   = 7;
422                 break;
423             default :
424                 f107Group = 5;
425                 apGroup   = 8;
426                 break;
427         }
428 
429         // read the data
430         final BufferedReader reader = new BufferedReader(new InputStreamReader(input, "UTF-8"));
431         boolean inData = false;
432         final TimeScale utc = TimeScalesFactory.getUTC();
433         DateComponents fileDate = null;
434         for (String line = reader.readLine(); line != null; line = reader.readLine()) {
435             line = line.trim();
436             if (line.length() > 0) {
437                 final Matcher matcher = dataPattern.matcher(line);
438                 if (matcher.matches()) {
439 
440                     // we are in the data section
441                     inData = true;
442 
443                     // extract the data from the line
444                     final int year = Integer.parseInt(matcher.group(1));
445                     final Month month = Month.parseMonth(matcher.group(2));
446                     final AbsoluteDate date = new AbsoluteDate(year, month, 1, utc);
447                     if (fileDate == null) {
448                         // the first entry of each file correspond exactly to 6 months before file publication
449                         // so we compute the file date by adding 6 months to its first entry
450                         if (month.getNumber() > 6) {
451                             fileDate = new DateComponents(year + 1, month.getNumber() - 6, 1);
452                         } else {
453                             fileDate = new DateComponents(year, month.getNumber() + 6, 1);
454                         }
455                     }
456 
457                     // check if there is already an entry for this date or not
458                     boolean addEntry = false;
459                     final Iterator<TimeStamped> iterator = data.tailSet(date).iterator();
460                     if (iterator.hasNext()) {
461                         final LineParameters existingEntry = (LineParameters) iterator.next();
462                         if (existingEntry.getDate().equals(date)) {
463                             // there is an entry for this date
464                             if (existingEntry.getFileDate().compareTo(fileDate) < 0) {
465                                 // the entry was read from an earlier file
466                                 // we replace it with the new entry as it is fresher
467                                 iterator.remove();
468                                 addEntry = true;
469                             }
470                         } else {
471                             // it is the first entry we get for this date
472                             addEntry = true;
473                         }
474                     } else {
475                         // it is the first entry we get for this date
476                         addEntry = true;
477                     }
478                     if (addEntry) {
479                         // we must add the new entry
480                         data.add(new LineParameters(fileDate, date,
481                                                     Double.parseDouble(matcher.group(f107Group)),
482                                                     Double.parseDouble(matcher.group(apGroup))));
483                     }
484 
485                 } else {
486                     if (inData) {
487                         // we have already read some data, so we are not in the header anymore
488                         // however, we don't recognize this non-empty line,
489                         // we consider the file is corrupted
490                         throw new OrekitException(OrekitMessages.NOT_A_MARSHALL_SOLAR_ACTIVITY_FUTURE_ESTIMATION_FILE,
491                                                   name);
492                     }
493                 }
494             }
495         }
496 
497         if (data.isEmpty()) {
498             throw new OrekitException(OrekitMessages.NOT_A_MARSHALL_SOLAR_ACTIVITY_FUTURE_ESTIMATION_FILE,
499                                       name);
500         }
501         firstDate = data.first().getDate();
502         lastDate  = data.last().getDate();
503 
504     }
