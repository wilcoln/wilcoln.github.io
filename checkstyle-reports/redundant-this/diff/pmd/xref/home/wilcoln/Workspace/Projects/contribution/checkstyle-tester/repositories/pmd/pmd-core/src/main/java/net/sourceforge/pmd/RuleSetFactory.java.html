122     /**
123      * Returns an Iterator of RuleSet objects loaded from descriptions from the
124      * "categories.properties" resource for each Language with Rule support.
125      *
126      * @return An Iterator of RuleSet objects.
127      *
128      * @throws RuleSetNotFoundException if the ruleset file could not be found
129      */
130     public Iterator<RuleSet> getRegisteredRuleSets() throws RuleSetNotFoundException {
131         String rulesetsProperties = null;
132         try {
133             List<RuleSetReferenceId> ruleSetReferenceIds = new ArrayList<>();
134             for (Language language : LanguageRegistry.findWithRuleSupport()) {
135                 Properties props = new Properties();
136                 rulesetsProperties = "category/" + language.getTerseName() + "/categories.properties";
137                 try (InputStream inputStream = resourceLoader.loadClassPathResourceAsStreamOrThrow(rulesetsProperties)) {
138                     props.load(inputStream);
139                     String rulesetFilenames = props.getProperty("rulesets.filenames");
140                     if (rulesetFilenames != null) {
141                         ruleSetReferenceIds.addAll(RuleSetReferenceId.parse(rulesetFilenames));
142                     }
143                 } catch (RuleSetNotFoundException e) {
144                     LOG.warning("The language " + language.getTerseName() + " provides no " + rulesetsProperties + ".");
145                 }
146             }
147             return createRuleSets(ruleSetReferenceIds).getRuleSetsIterator();
148         } catch (IOException ioe) {
149             throw new RuntimeException("Couldn't find " + rulesetsProperties
150                     + "; please ensure that the directory is on the classpath. The current classpath is: "
151                     + System.getProperty("java.class.path"));
152         }
153     }

249     /**
250      * Creates a new ruleset with the given metadata such as name, description,
251      * fileName, exclude/include patterns are used. The rules are taken from the given
252      * collection.
253      *
254      * <p><strong>Note:</strong> The rule instances are shared between the collection
255      * and the new ruleset (copy-by-reference). This might lead to concurrency issues,
256      * if the rules of the collection are also referenced by other rulesets and used
257      * in different threads.
258      * </p>
259      *
260      * @param name the name of the ruleset
261      * @param description the description
262      * @param fileName the filename
263      * @param excludePatterns list of exclude patterns, if any is not a valid regular expression, it will be ignored
264      * @param includePatterns list of include patterns, if any is not a valid regular expression, it will be ignored
265      * @param rules the collection with the rules to add to the new ruleset
266      * @return the new ruleset
267      */
268     public RuleSet createNewRuleSet(String name,
269                                     String description,
270                                     String fileName,
271                                     Collection<String> excludePatterns,
272                                     Collection<String> includePatterns,
273                                     Collection<Rule> rules) {
274         RuleSetBuilder builder = new RuleSetBuilder(0L); // TODO: checksum missing
275         builder.withName(name)
276                .withDescription(description)
277                .withFileName(fileName)
278                .replaceFileExclusions(toPatterns(excludePatterns))
279                .replaceFileInclusions(toPatterns(includePatterns));
280         for (Rule rule : rules) {
281             builder.addRule(rule);
282         }
283         return builder.build();
284     }

287         List<Pattern> result = new ArrayList<>();
288         for (String s : sources) {
289             try {
290                 result.add(Pattern.compile(s));
291             } catch (PatternSyntaxException ignored) {
292 
293             }
294         }
295         return result;
296     }

316                 .withDescription("RuleSet for " + rule.getName());

323      * currently configured ResourceLoader is used.
324      * <p>
325      * Any Rules in the RuleSet other than the one being created, are _not_
326      * created. Deprecated rules are _not_ ignored, so that they can be

328      *
329      * @param ruleSetReferenceId

341             throw new IllegalArgumentException(
342                     "Cannot parse a single Rule from an all Rule RuleSet reference: <" + ruleSetReferenceId + ">.");
343         }
344         RuleSet ruleSet = createRuleSet(ruleSetReferenceId, withDeprecatedRuleReferences);
345         return ruleSet.getRuleByName(ruleSetReferenceId.getRuleName());

348     /**
349      * Parse a ruleset node to construct a RuleSet.
350      *
351      * @param ruleSetReferenceId
352      *            The RuleSetReferenceId of the RuleSet being parsed.
353      * @param withDeprecatedRuleReferences
354      *            whether rule references that are deprecated should be ignored
355      *            or not
356      * @return The new RuleSet.
357      */
358     private RuleSet parseRuleSetNode(RuleSetReferenceId ruleSetReferenceId, boolean withDeprecatedRuleReferences)
359             throws RuleSetNotFoundException {
360         try (CheckedInputStream inputStream = new CheckedInputStream(
361                 ruleSetReferenceId.getInputStream(resourceLoader), new Adler32());) {
362             if (!ruleSetReferenceId.isExternal()) {
363                 throw new IllegalArgumentException(
364                         "Cannot parse a RuleSet from a non-external reference: <" + ruleSetReferenceId + ">.");
365             }
366             DocumentBuilder builder = createDocumentBuilder();
367             InputSource inputSource;
368             if (compatibilityFilter != null) {
369                 inputSource = new InputSource(compatibilityFilter.filterRuleSetFile(inputStream));
370             } else {
371                 inputSource = new InputSource(inputStream);
372             }
373             Document document = builder.parse(inputSource);
374             Element ruleSetElement = document.getDocumentElement();
375 
376             RuleSetBuilder ruleSetBuilder = new RuleSetBuilder(inputStream.getChecksum().getValue())
377                     .withFileName(ruleSetReferenceId.getRuleSetFileName());
378 
379             if (ruleSetElement.hasAttribute("name")) {
380                 ruleSetBuilder.withName(ruleSetElement.getAttribute("name"));
381             } else {
382                 LOG.warning("RuleSet name is missing. Future versions of PMD will require it.");
383                 ruleSetBuilder.withName("Missing RuleSet Name");
384             }
385 
386             Set<String> rulesetReferences = new HashSet<>();
387 
388             NodeList nodeList = ruleSetElement.getChildNodes();
389             for (int i = 0; i < nodeList.getLength(); i++) {
390                 Node node = nodeList.item(i);
391                 if (node.getNodeType() == Node.ELEMENT_NODE) {
392                     String nodeName = node.getNodeName();
393                     String text = parseTextNode(node);
394                     if (DESCRIPTION.equals(nodeName)) {
395                         ruleSetBuilder.withDescription(text);
396                     } else if ("include-pattern".equals(nodeName)) {
397                         final Pattern pattern = parseRegex(text);
398                         if (pattern == null) {
399                             continue;
400                         }
401                         ruleSetBuilder.withFileInclusions(pattern);
402                     } else if ("exclude-pattern".equals(nodeName)) {
403                         final Pattern pattern = parseRegex(text);
404                         if (pattern == null) {
405                             continue;
406                         }
407                         ruleSetBuilder.withFileExclusions(pattern);
408                     } else if ("rule".equals(nodeName)) {
409                         parseRuleNode(ruleSetReferenceId, ruleSetBuilder, node, withDeprecatedRuleReferences, rulesetReferences);
410                     } else {
411                         throw new IllegalArgumentException(UNEXPECTED_ELEMENT + node.getNodeName()
412                                 + "> encountered as child of <ruleset> element.");
413                     }
414                 }
415             }
416 
417             if (!ruleSetBuilder.hasDescription()) {
418                 LOG.warning("RuleSet description is missing. Future versions of PMD will require it.");
419                 ruleSetBuilder.withDescription("Missing description");
420             }
421 
422             ruleSetBuilder.filterRulesByPriority(minimumPriority);
423 
424             return ruleSetBuilder.build();
425         } catch (ReflectiveOperationException ex) {
426             ex.printStackTrace();
427             throw new RuntimeException("Couldn't find the class " + ex.getMessage(), ex);
428         } catch (ParserConfigurationException | IOException | SAXException ex) {
429             ex.printStackTrace();
430             throw new RuntimeException("Couldn't read the ruleset " + ruleSetReferenceId + ": " + ex.getMessage(), ex);
431         }
432     }

434     private Pattern parseRegex(String text) {
435         final Pattern pattern;
436         try {
437             pattern = Pattern.compile(text);
438         } catch (PatternSyntaxException pse) {
439             LOG.warning(pse.getMessage());
440             return null;
441         }
442         return pattern;
443     }

446     private DocumentBuilder createDocumentBuilder() throws ParserConfigurationException {
447         final DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
448 
449         try {
450             /*
451              * parser hardening
452              * https://www.owasp.org/index.php/XML_External_Entity_(XXE)_Prevention_Cheat_Sheet#JAXP_DocumentBuilderFactory.2C_SAXParserFactory_and_DOM4J
453              */
454             // This is the PRIMARY defense. If DTDs (doctypes) are disallowed, almost all XML entity attacks are prevented
455             // Xerces 2 only - http://xerces.apache.org/xerces2-j/features.html#disallow-doctype-decl
456             dbf.setFeature("http://apache.org/xml/features/disallow-doctype-decl", true);
457             
458             // If you can't completely disable DTDs, then at least do the following:
459             // Xerces 1 - http://xerces.apache.org/xerces-j/features.html#external-general-entities
460             // Xerces 2 - http://xerces.apache.org/xerces2-j/features.html#external-general-entities
461             // JDK7+ - http://xml.org/sax/features/external-general-entities    
462             dbf.setFeature("http://xml.org/sax/features/external-general-entities", false);
463             
464             // Xerces 1 - http://xerces.apache.org/xerces-j/features.html#external-parameter-entities
465             // Xerces 2 - http://xerces.apache.org/xerces2-j/features.html#external-parameter-entities
466             // JDK7+ - http://xml.org/sax/features/external-parameter-entities    
467             dbf.setFeature("http://xml.org/sax/features/external-parameter-entities", false);
468             
469             // Disable external DTDs as well
470             dbf.setFeature("http://apache.org/xml/features/nonvalidating/load-external-dtd", false);
471             
472             // and these as well, per Timothy Morgan's 2014 paper: "XML Schema, DTD, and Entity Attacks"
473             dbf.setXIncludeAware(false);
474             dbf.setExpandEntityReferences(false);
475         } catch (final ParserConfigurationException e) {
476             // an unsupported feature... too bad, but won't fail execution due to this
477             LOG.log(Level.WARNING, "Ignored unsupported XML Parser Feature for parsing rulesets", e);
478         }
479         
480         return dbf.newDocumentBuilder();
481     }

483     /**
484      * Parse a rule node.
485      *
486      * @param ruleSetReferenceId
487      *            The RuleSetReferenceId of the RuleSet being parsed.
488      * @param ruleSetBuilder
489      *            The RuleSet being constructed.
490      * @param ruleNode
491      *            Must be a rule element node.
492      * @param withDeprecatedRuleReferences
493      *            whether rule references that are deprecated should be ignored
494      *            or not
495      * @param rulesetReferences keeps track of already processed complete ruleset references in order to log a warning
496      */
497     private void parseRuleNode(RuleSetReferenceId ruleSetReferenceId, RuleSetBuilder ruleSetBuilder, Node ruleNode,
498             boolean withDeprecatedRuleReferences, Set<String> rulesetReferences)
499             throws ClassNotFoundException, InstantiationException, IllegalAccessException, RuleSetNotFoundException {
500         Element ruleElement = (Element) ruleNode;
501         String ref = ruleElement.getAttribute("ref");
502         if (ref.endsWith("xml")) {
503             parseRuleSetReferenceNode(ruleSetBuilder, ruleElement, ref, rulesetReferences);
504         } else if (StringUtils.isBlank(ref)) {
505             parseSingleRuleNode(ruleSetReferenceId, ruleSetBuilder, ruleNode);
506         } else {
507             parseRuleReferenceNode(ruleSetReferenceId, ruleSetBuilder, ruleNode, ref, withDeprecatedRuleReferences);
508         }
509     }

511     /**
512      * Parse a rule node as an RuleSetReference for all Rules. Every Rule from
513      * the referred to RuleSet will be added as a RuleReference except for those
514      * explicitly excluded, below the minimum priority threshold for this
515      * RuleSetFactory, or which are deprecated.
516      *
517      * @param ruleSetBuilder
518      *            The RuleSet being constructed.
519      * @param ruleElement
520      *            Must be a rule element node.
521      * @param ref
522      *            The RuleSet reference.
523      * @param rulesetReferences keeps track of already processed complete ruleset references in order to log a warning
524      */
525     private void parseRuleSetReferenceNode(RuleSetBuilder ruleSetBuilder, Element ruleElement, String ref, Set<String> rulesetReferences)
526             throws RuleSetNotFoundException {
527         String priority = null;
528         NodeList childNodes = ruleElement.getChildNodes();
529         Set<String> excludedRulesCheck = new HashSet<>();
530         for (int i = 0; i < childNodes.getLength(); i++) {
531             Node child = childNodes.item(i);
532             if (isElementNode(child, "exclude")) {
533                 Element excludeElement = (Element) child;
534                 String excludedRuleName = excludeElement.getAttribute("name");
535                 excludedRulesCheck.add(excludedRuleName);
536             } else if (isElementNode(child, PRIORITY)) {
537                 priority = parseTextNode(child).trim();
538             }
539         }
540         final RuleSetReference ruleSetReference = new RuleSetReference(ref, true, excludedRulesCheck);
541 
542         // load the ruleset with minimum priority low, so that we get all rules, to be able to exclude any rule
543         // minimum priority will be applied again, before constructing the final ruleset
544         RuleSetFactory ruleSetFactory = new RuleSetFactory(resourceLoader, RulePriority.LOW, false, this.compatibilityFilter != null);
545         RuleSet otherRuleSet = ruleSetFactory.createRuleSet(RuleSetReferenceId.parse(ref).get(0));
546         List<RuleReference> potentialRules = new ArrayList<>();
547         int countDeprecated = 0;
548         for (Rule rule : otherRuleSet.getRules()) {
549             excludedRulesCheck.remove(rule.getName());
550             if (!ruleSetReference.getExcludes().contains(rule.getName())) {
551                 RuleReference ruleReference = new RuleReference(rule, ruleSetReference);
552                 // override the priority
553                 if (priority != null) {
554                     ruleReference.setPriority(RulePriority.valueOf(Integer.parseInt(priority)));
555                 }
556 
557                 if (rule.isDeprecated()) {
558                     countDeprecated++;
559                 }
560                 potentialRules.add(ruleReference);
561             }
562         }
563 
564         boolean rulesetDeprecated = false;
565         if (!potentialRules.isEmpty() && potentialRules.size() == countDeprecated) {
566             // all rules in the ruleset have been deprecated - the ruleset itself is considered to be deprecated
567             rulesetDeprecated = true;
568             LOG.warning("The RuleSet " + ref + " has been deprecated and will be removed in PMD " + PMDVersion.getNextMajorRelease());
569         }
570 
571         for (RuleReference r : potentialRules) {
572             if (rulesetDeprecated || !r.getRule().isDeprecated()) {
573                 // add the rule, if either the ruleset itself is deprecated (then we add all rules)
574                 // or if the rule is not deprecated (in that case, the ruleset might contain deprecated as well
575                 // as valid rules)
576                 ruleSetBuilder.addRuleIfNotExists(r);
577             }
578         }
579 
580         if (!excludedRulesCheck.isEmpty()) {
581             if (LOG.isLoggable(Level.WARNING)) {
582                 LOG.warning(
583                     "Unable to exclude rules " + excludedRulesCheck + " from ruleset reference " + ref
584                     + "; perhaps the rule name is mispelled or the rule doesn't exist anymore?");
585             }
586         }
587 
588         if (rulesetReferences.contains(ref)) {
589             LOG.warning("The ruleset " + ref + " is referenced multiple times in \""
590                     + ruleSetBuilder.getName() + "\".");
591         }
592         rulesetReferences.add(ref);
593     }

595     /**
596      * Parse a rule node as a single Rule. The Rule has been fully defined
597      * within the context of the current RuleSet.
598      *
599      * @param ruleSetReferenceId
600      *            The RuleSetReferenceId of the RuleSet being parsed.
601      * @param ruleSetBuilder
602      *            The RuleSet being constructed.
603      * @param ruleNode
604      *            Must be a rule element node.
605      */
606     private void parseSingleRuleNode(RuleSetReferenceId ruleSetReferenceId, RuleSetBuilder ruleSetBuilder,
607             Node ruleNode) throws ClassNotFoundException, InstantiationException, IllegalAccessException {
608         Element ruleElement = (Element) ruleNode;
609 
610         // Stop if we're looking for a particular Rule, and this element is not
611         // it.
612         if (StringUtils.isNotBlank(ruleSetReferenceId.getRuleName())
613                 && !isRuleName(ruleElement, ruleSetReferenceId.getRuleName())) {
614             return;
615         }
616         Rule rule = new RuleFactory(resourceLoader).buildRule(ruleElement);
617         rule.setRuleSetName(ruleSetBuilder.getName());
618 
619         ruleSetBuilder.addRule(rule);
620     }

623     /**
624      * Parse a rule node as a RuleReference. A RuleReference is a single Rule
625      * which comes from another RuleSet with some of it's attributes potentially
626      * overridden.
627      *
628      * @param ruleSetReferenceId
629      *            The RuleSetReferenceId of the RuleSet being parsed.
630      * @param ruleSetBuilder
631      *            The RuleSet being constructed.
632      * @param ruleNode
633      *            Must be a rule element node.
634      * @param ref
635      *            A reference to a Rule.
636      * @param withDeprecatedRuleReferences
637      *            whether rule references that are deprecated should be ignored
638      *            or not
639      */
640     private void parseRuleReferenceNode(RuleSetReferenceId ruleSetReferenceId, RuleSetBuilder ruleSetBuilder,
641             Node ruleNode, String ref, boolean withDeprecatedRuleReferences) throws RuleSetNotFoundException {
642         Element ruleElement = (Element) ruleNode;
643 
644         // Stop if we're looking for a particular Rule, and this element is not
645         // it.
646         if (StringUtils.isNotBlank(ruleSetReferenceId.getRuleName())
647                 && !isRuleName(ruleElement, ruleSetReferenceId.getRuleName())) {
648             return;
649         }
650 
651         // load the ruleset with minimum priority low, so that we get all rules, to be able to exclude any rule
652         // minimum priority will be applied again, before constructing the final ruleset
653         RuleSetFactory ruleSetFactory = new RuleSetFactory(resourceLoader, RulePriority.LOW, false, this.compatibilityFilter != null);
654 
655         boolean isSameRuleSet = false;
656         RuleSetReferenceId otherRuleSetReferenceId = RuleSetReferenceId.parse(ref).get(0);
657         if (!otherRuleSetReferenceId.isExternal()
658                 && containsRule(ruleSetReferenceId, otherRuleSetReferenceId.getRuleName())) {
659             otherRuleSetReferenceId = new RuleSetReferenceId(ref, ruleSetReferenceId);
660             isSameRuleSet = true;
661         } else if (otherRuleSetReferenceId.isExternal()
662                 && otherRuleSetReferenceId.getRuleSetFileName().equals(ruleSetReferenceId.getRuleSetFileName())) {
663             otherRuleSetReferenceId = new RuleSetReferenceId(otherRuleSetReferenceId.getRuleName(), ruleSetReferenceId);
664             isSameRuleSet = true;
665         }
666         // do not ignore deprecated rule references
667         Rule referencedRule = ruleSetFactory.createRule(otherRuleSetReferenceId, true);
668 
669         if (referencedRule == null) {
670             throw new IllegalArgumentException("Unable to find referenced rule " + otherRuleSetReferenceId.getRuleName()
671                     + "; perhaps the rule name is mispelled?");
672         }
673 
674         if (warnDeprecated && referencedRule.isDeprecated()) {
675             if (referencedRule instanceof RuleReference) {
676                 RuleReference ruleReference = (RuleReference) referencedRule;
677                 if (LOG.isLoggable(Level.WARNING)) {
678                     LOG.warning("Use Rule name " + ruleReference.getRuleSetReference().getRuleSetFileName() + '/'
679                             + ruleReference.getOriginalName() + " instead of the deprecated Rule name "
680                             + otherRuleSetReferenceId
681                             + ". PMD " + PMDVersion.getNextMajorRelease()
682                             + " will remove support for this deprecated Rule name usage.");
683                 }
684             } else if (referencedRule instanceof MockRule) {
685                 if (LOG.isLoggable(Level.WARNING)) {
686                     LOG.warning("Discontinue using Rule name " + otherRuleSetReferenceId
687                             + " as it has been removed from PMD and no longer functions."
688                             + " PMD " + PMDVersion.getNextMajorRelease()
689                             + " will remove support for this Rule.");
690                 }
691             } else {
692                 if (LOG.isLoggable(Level.WARNING)) {
693                     LOG.warning("Discontinue using Rule name " + otherRuleSetReferenceId
694                             + " as it is scheduled for removal from PMD."
695                             + " PMD " + PMDVersion.getNextMajorRelease()
696                             + " will remove support for this Rule.");
697                 }
698             }
699         }
700 
701         RuleSetReference ruleSetReference = new RuleSetReference(otherRuleSetReferenceId.getRuleSetFileName(), false);
702 
703         RuleReference ruleReference = new RuleFactory(resourceLoader).decorateRule(referencedRule, ruleSetReference, ruleElement);
704 
705         if (warnDeprecated && ruleReference.isDeprecated() && !isSameRuleSet) {
706             if (LOG.isLoggable(Level.WARNING)) {
707                 LOG.warning("Use Rule name " + ruleReference.getRuleSetReference().getRuleSetFileName() + '/'
708                         + ruleReference.getOriginalName() + " instead of the deprecated Rule name "
709                         + ruleSetReferenceId.getRuleSetFileName() + '/' + ruleReference.getName()
710                         + ". PMD " + PMDVersion.getNextMajorRelease()
711                         + " will remove support for this deprecated Rule name usage.");
712             }
713         }
714 
715         if (withDeprecatedRuleReferences || !isSameRuleSet || !ruleReference.isDeprecated()) {
716             Rule existingRule = ruleSetBuilder.getExistingRule(ruleReference);
717             if (existingRule instanceof RuleReference) {
718                 RuleReference existingRuleReference = (RuleReference) existingRule;
719                 // the only valid use case is: the existing rule does not override anything yet
720                 // which means, it is a plain reference. And the new reference overrides.
721                 // for all other cases, we should log a warning
722                 if (existingRuleReference.hasOverriddenAttributes() || !ruleReference.hasOverriddenAttributes()) {
723                     LOG.warning("The rule " + ruleReference.getName() + " is referenced multiple times in \""
724                             + ruleSetBuilder.getName() + "\". "
725                             + "Only the last rule configuration is used.");
726                 }
727             }
728 
729             ruleSetBuilder.addRuleReplaceIfExists(ruleReference);
730         }
731     }

734     /**
735      * Check whether the given ruleName is contained in the given ruleset.
736      *
737      * @param ruleSetReferenceId the ruleset to check
738      * @param ruleName           the rule name to search for
739      *
740      * @return {@code true} if the ruleName exists
741      */
742     private boolean containsRule(RuleSetReferenceId ruleSetReferenceId, String ruleName) {
743         boolean found = false;
744         try (InputStream ruleSet = ruleSetReferenceId.getInputStream(resourceLoader)) {
745             DocumentBuilder builder = createDocumentBuilder();
746             Document document = builder.parse(ruleSet);
747             Element ruleSetElement = document.getDocumentElement();
748 
749             NodeList rules = ruleSetElement.getElementsByTagName("rule");
750             for (int i = 0; i < rules.getLength(); i++) {
751                 Element rule = (Element) rules.item(i);
752                 if (rule.hasAttribute("name") && rule.getAttribute("name").equals(ruleName)) {
753                     found = true;
754                     break;
755                 }
756             }
757         } catch (Exception e) {
758             throw new RuntimeException(e);
759         }
760 
761         return found;
762     }

764     private static boolean isElementNode(Node node, String name) {
765         return node.getNodeType() == Node.ELEMENT_NODE && node.getNodeName().equals(name);
766     }

768     /**
769      * Parse a String from a textually type node.
770      *
771      * @param node
772      *            The node.
773      * @return The String.
774      */
775     private static String parseTextNode(Node node) {
776 
777         final int nodeCount = node.getChildNodes().getLength();
778         if (nodeCount == 0) {
779             return "";
780         }
781 
782         StringBuilder buffer = new StringBuilder();
783 
784         for (int i = 0; i < nodeCount; i++) {
785             Node childNode = node.getChildNodes().item(i);
786             if (childNode.getNodeType() == Node.CDATA_SECTION_NODE || childNode.getNodeType() == Node.TEXT_NODE) {
787                 buffer.append(childNode.getNodeValue());
788             }
789         }
790         return buffer.toString();
791     }

794     /**
795      * Determine if the specified rule element will represent a Rule with the
796      * given name.
797      *
798      * @param ruleElement The rule element.
799      * @param ruleName    The Rule name.
800      *
801      * @return {@code true} if the Rule would have the given name, {@code false} otherwise.
802      */
803     private boolean isRuleName(Element ruleElement, String ruleName) {
804         if (ruleElement.hasAttribute("name")) {
805             return ruleElement.getAttribute("name").equals(ruleName);
806         } else if (ruleElement.hasAttribute("ref")) {
807             RuleSetReferenceId ruleSetReferenceId = RuleSetReferenceId.parse(ruleElement.getAttribute("ref")).get(0);
808             return ruleSetReferenceId.getRuleName() != null && ruleSetReferenceId.getRuleName().equals(ruleName);
809         } else {
810             return false;
811         }
812     }
