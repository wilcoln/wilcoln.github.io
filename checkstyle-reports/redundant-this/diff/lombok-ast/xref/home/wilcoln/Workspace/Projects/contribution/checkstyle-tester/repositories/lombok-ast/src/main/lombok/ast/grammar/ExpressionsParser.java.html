32  public class ExpressionsParser extends BaseParser<Node> {
33  	final ParserGroup group;
34  	final ExpressionsActions actions;

36  	public ExpressionsParser(ParserGroup group) {
37  		this.actions = new ExpressionsActions(group.getSource());
38  		this.group = group;

41  	/**
42  	 * P0
43  	 */
44  	public Rule primaryExpression() {
45  		return FirstOf(
46  				parenGrouping(),
47  				group.literals.anyLiteral(),
48  				unqualifiedThisOrSuperLiteral(),
49  				arrayCreationExpression(),
50  				unqualifiedConstructorInvocation(),
51  				qualifiedClassOrThisOrSuperLiteral(),
52  				identifierExpression());
53  	}
54  	
55  	Rule parenGrouping() {
56  		return Sequence(
57  				Ch('('), group.basics.optWS(),
58  				anyExpression(), set(),
59  				Ch(')'), set(actions.addParens(value())),
60  				group.basics.optWS());
61  	}
62  	
63  	Rule unqualifiedThisOrSuperLiteral() {
64  		return Sequence(
65  				FirstOf(String("this"), String("super")).label("thisOrSuper"),
66  				group.basics.testLexBreak(),
67  				group.basics.optWS(),
68  				TestNot(Ch('(')),
69  				set(actions.createThisOrSuperOrClass(null, text("thisOrSuper"), null)));
70  	}

73  	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/expressions.html#15.8.2">JLS section 15.8.2</a>
74  	 */
75  	Rule qualifiedClassOrThisOrSuperLiteral() {
76  		return Sequence(
77  				group.types.type().label("type"),
78  				Ch('.').label("dot"), group.basics.optWS(),
79  				FirstOf(String("this"), String("super"), String("class")).label("thisOrSuperOrClass"),
80  				group.basics.testLexBreak(),
81  				group.basics.optWS(),
82  				set(actions.createThisOrSuperOrClass(node("dot"), text("thisOrSuperOrClass"), value("type"))));
83  	}

85  	Rule unqualifiedConstructorInvocation() {
86  		return Sequence(
87  				String("new"), group.basics.testLexBreak(), group.basics.optWS(),
88  				group.types.typeArguments().label("constructorTypeArgs"),
89  				group.types.type().label("type"),
90  				group.structures.methodArguments().label("args"),
91  				Optional(group.structures.typeBody()).label("classBody"),
92  				set(actions.createUnqualifiedConstructorInvocation(value("constructorTypeArgs"), value("type"), value("args"), value("classBody"))));
93  	}

95  	/**
96  	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/arrays.html#10.3">JLS section 10.3</a>
97  	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/expressions.html#15.10">JLS section 15.10</a>

99  	Rule arrayCreationExpression() {
100 		return Sequence(
101 				String("new"), group.basics.testLexBreak(), group.basics.optWS(),
102 				group.types.nonArrayType().label("type"),
103 				OneOrMore(Sequence(
104 						Ch('[').label("openArray"), group.basics.optWS(),
105 						Optional(anyExpression()).label("dimension"), Ch(']'), group.basics.optWS(),
106 						set(actions.createDimension(value("dimension"), node("openArray"))))),
107 				Optional(arrayInitializer()).label("initializer"),
108 				set(actions.createArrayCreationExpression(value("type"), values("OneOrMore/Sequence"), value("initializer"))));
109 	}
110 	
111 	public Rule arrayInitializer() {
112 		return Sequence(
113 				Ch('{'), group.basics.optWS(),
114 				Optional(Sequence(
115 						FirstOf(arrayInitializer(), anyExpression()).label("head"),
116 						ZeroOrMore(Sequence(
117 								Ch(','), group.basics.optWS(),
118 								FirstOf(arrayInitializer(), anyExpression()).label("tail"))),
119 						Optional(Ch(',')),
120 						group.basics.optWS())),
121 				Ch('}'), group.basics.optWS(),
122 				set(actions.createArrayInitializerExpression(value("Optional/Sequence/head"), values("Optional/Sequence/ZeroOrMore/Sequence/tail"))));
123 	}

126 		return Sequence(
127 				group.basics.identifier(),
128 				set(),
129 				Optional(Sequence(group.structures.methodArguments(), set()).label("methodArgs")),
130 				set(actions.createPrimary(value(), value("Optional/methodArgs"))));
131 	}

155 	Rule level1ExpressionChaining() {
156 		return Sequence(
157 				primaryExpression().label("head"), set(),
158 				ZeroOrMore(FirstOf(
159 						arrayAccessOperation().label("arrayAccess"),
160 						methodInvocationWithTypeArgsOperation().label("methodInvocation"),

163 	}

167 				Ch('['), group.basics.optWS(),

169 				set(actions.createArrayAccessOperation(value())));
170 	}

176 				group.basics.identifier().label("name"),
177 				group.structures.methodArguments().label("methodArguments"),
178 				set(actions.createMethodInvocationOperation(node("dot"), value("typeArguments"), value("name"), value("methodArguments"))));

188 	/**
189 	 * P2''
190 	 * 
191 	 * This is the relational new operator; not just 'new', but new with context, so: "a.new InnerClass(params)". It is grouped with P2, but for some reason has higher precedence
192 	 * in all java parsers, and so we give it its own little precedence group here.
193 	 */
194 	Rule dotNewExpressionChaining() {
195 		return Sequence(
196 				level1ExpressionChaining().label("head"), set(),
197 				ZeroOrMore(Sequence(
198 						Sequence(
199 								Ch('.'),
200 								group.basics.optWS(),
201 								String("new"),
202 								group.basics.testLexBreak(),
203 								group.basics.optWS()),
204 						group.types.typeArguments().label("constructorTypeArgs"),
205 						group.basics.identifier().label("innerClassName"),
206 						group.types.typeArguments().label("classTypeArgs"),
207 						group.structures.methodArguments().label("methodArguments"),
208 						Optional(group.structures.typeBody()).label("classBody"),
209 						set(actions.createQualifiedConstructorInvocation(value("constructorTypeArgs"), node("innerClassName"), node("classTypeArgs"), value("methodArguments"), value("classBody"))))),
210 				set(actions.createChainOfQualifiedConstructorInvocations(node("head"), nodes("ZeroOrMore/Sequence"))));
211 	}

213 	/**
214 	 * P2'
215 	 * Technically, postfix increment operations are in P2 along with all the unary operators like ~ and !, as well as typecasts.
216 	 * However, because ALL of the P2 expression are right-associative, the postfix operators can be considered as a higher level of precedence.
217 	 */
218 	Rule postfixIncrementExpressionChaining() {
219 		return Sequence(
220 				dotNewExpressionChaining(), set(),
221 				ZeroOrMore(Sequence(
222 						FirstOf(String("++"), String("--")).label("operator"),
223 						group.basics.optWS()).label("operatorCt")),
224 				set(actions.createUnaryPostfixExpression(value(), nodes("ZeroOrMore/operatorCt/operator"), texts("ZeroOrMore/operatorCt/operator"))));
225 	}
226 	
227 	Rule postfixIncrementExpression() {
228 		return Sequence(
229 				dotNewExpressionChaining(), set(),
230 				OneOrMore(Sequence(
231 						FirstOf(String("++"), String("--")).label("operator"),
232 						group.basics.optWS()).label("operatorCt")),
233 				set(actions.createUnaryPostfixExpression(value(), nodes("OneOrMore/operatorCt/operator"), texts("OneOrMore/operatorCt/operator"))));
234 	}
235 	
236 	Rule prefixIncrementExpression() {
237 		return Sequence(
238 				OneOrMore(Sequence(
239 						FirstOf(String("++"), String("--")).label("operator"),
240 						group.basics.optWS()).label("operatorCt")),
241 						postfixIncrementExpressionChaining().label("operand"), set(),
242 				set(actions.createUnaryPrefixExpressions(node("operand"), nodes("OneOrMore/operatorCt/operator"), texts("OneOrMore/operatorCt/operator"))));
243 	}

245 	/**
246 	 * P2
247 	 */
248 	Rule level2ExpressionChaining() {
249 		return FirstOf(
250 				Sequence(
251 						FirstOf(
252 								String("++"), String("--"),
253 								Ch('!'), Ch('~'),
254 								solitarySymbol('+'), solitarySymbol('-'),
255 								Sequence(
256 										Ch('('), group.basics.optWS(),
257 										group.types.type().label("type"),
258 										Ch(')'),
259 										TestNot(Sequence(
260 												actions.typeIsAlsoLegalAsExpression(UP(UP(value("type")))),
261 												group.basics.optWS(),
262 												FirstOf(solitarySymbol('+'), solitarySymbol('-'))))).label("cast")
263 								).label("operator"),
264 						group.basics.optWS(),
265 						level2ExpressionChaining().label("operand"), set(),
266 						set(actions.createUnaryPrefixExpression(value("operand"), node("operator"), text("operator")))),
267 					Sequence(postfixIncrementExpressionChaining(), set()));
268 	}

297 	/**
298 	 * P6
299 	 * 
300 	 * Technically 'instanceof' is on equal footing with the other operators, but practically speaking this doesn't hold;
301 	 * for starters, the RHS of instanceof is a Type and not an expression, and the inevitable type of an instanceof expression (boolean) is
302 	 * not compatible as LHS to *ANY* of the operators in this class, including instanceof itself. Therefore, pragmatically speaking, there can only
303 	 * be one instanceof, and it has to appear at the end of the chain.
304 	 * 
305 	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#15.20">JLS section 15.20</a>
306 	 */
307 	Rule relationalExpressionChaining() {
308 		return Sequence(
309 				forLeftAssociativeBinaryExpression("exprRelational", FirstOf(String("<="), String(">="), solitarySymbol('<'), solitarySymbol('>')), shiftExpressionChaining()),
310 				set(),
311 				Optional(Sequence(
312 						Sequence(String("instanceof"), group.basics.testLexBreak(), group.basics.optWS()),
313 						group.types.type().label("type")).label("typeCt")).label("instanceof"),
314 				set(actions.createInstanceOfExpression(value(), value("instanceof/typeCt/type"))));
315 	}

387 		return Sequence(

441 				solitarySymbol('='),

448 	/**
449 	 * @param operator Careful; operator has to match _ONLY_ the operator, not any whitespace around it (otherwise we'd have to remove comments from it, which isn't feasible).
450 	 */
451 	@Cached
452 	Rule forLeftAssociativeBinaryExpression(String labelName, Rule operator, Rule nextHigher) {
453 		return Sequence(
454 				nextHigher.label("head"), new Action<Node>() {
455 					@Override public boolean run(Context<Node> context) {
456 						setContext(context);
457 						return set();
458 					}
459 				},
460 				group.basics.optWS(),
461 				ZeroOrMore(Sequence(
462 						operator.label("operator"),
463 						group.basics.optWS(),
464 						nextHigher.label("tail"),
465 						group.basics.optWS())),
466 				new Action<Node>() {
467 					@Override public boolean run(Context<Node> context) {
468 						setContext(context);
469 						return set(actions.createLeftAssociativeBinaryExpression(
470 								node("head"),
471 								nodes("ZeroOrMore/Sequence/operator"), texts("ZeroOrMore/Sequence/operator"),
472 								nodes("ZeroOrMore/Sequence/tail")));
473 					}
474 				},
475 				group.basics.optWS()).label(labelName);
476 	}

478 	Rule solitarySymbol(char c) {
479 		return Sequence(Ch(c), TestNot(Ch(c)));
480 	}
