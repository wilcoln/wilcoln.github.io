28  /* default */ class JavaTypeDefinitionSimple extends JavaTypeDefinition {
29      private final Class<?> clazz;
30      private final JavaTypeDefinition[] genericArgs;
31      // cached because calling clazz.getTypeParameters().length create a new array every time
32      private final int typeParameterCount;
33      private final boolean isGeneric;
34      private final boolean isRawType;
35      private final JavaTypeDefinition enclosingClass;
36  
37      private static final Logger LOG = Logger.getLogger(JavaTypeDefinitionSimple.class.getName());
38      private static final JavaTypeDefinition[] NO_GENERICS = {};
39  
40      protected JavaTypeDefinitionSimple(Class<?> clazz, JavaTypeDefinition... boundGenerics) {
41          super(EXACT);
42          this.clazz = clazz;
43  
44          final TypeVariable<?>[] typeParameters;
45          // the anonymous class can't have generics, but we may be binding generics from super classes
46          if (clazz.isAnonymousClass()) {
47              // is this an anonymous class based on an interface or a class?
48              if (clazz.getInterfaces().length != 0) {
49                  typeParameters = clazz.getInterfaces()[0].getTypeParameters();
50              } else {
51                  typeParameters = clazz.getSuperclass().getTypeParameters();
52              }
53          } else {
54              typeParameters = clazz.getTypeParameters();
55          }
56  
57          typeParameterCount = typeParameters.length;
58          isGeneric = typeParameters.length != 0;
59          isRawType = isGeneric && boundGenerics.length == 0;
60  
61          if (isGeneric) {
62              // Generics will be lazily loaded if not already known
63              this.genericArgs = Arrays.copyOf(boundGenerics, typeParameterCount);
64          } else {
65              this.genericArgs = NO_GENERICS;
66          }
67  
68          Class<?> enclosing = null;
69          try {
70              enclosing = clazz.getEnclosingClass();
71          } catch (LinkageError e) {
72              if (LOG.isLoggable(Level.WARNING)) {
73                  LOG.log(Level.WARNING, "Could not load enclosing class of " + clazz.getName() + ", due to: " + e);
74              }
75          }
76          enclosingClass = forClass(enclosing);
77      }

94      private JavaTypeDefinition getGenericType(final String parameterName, Method method,
95                                                List<JavaTypeDefinition> methodTypeArguments) {
96          if (method != null && methodTypeArguments != null) {
97              int paramIndex = getGenericTypeIndex(method.getTypeParameters(), parameterName);
98              if (paramIndex != -1) {
99                  return methodTypeArguments.get(paramIndex);
100             }
101         }
102 
103         return getGenericType(parameterName);
104     }

106     @Override
107     public JavaTypeDefinition getGenericType(final String parameterName) {
108         for (JavaTypeDefinition currTypeDef = this; currTypeDef != null;
109                 currTypeDef = currTypeDef.getEnclosingClass()) {
110 
111             int paramIndex = getGenericTypeIndex(currTypeDef.getType().getTypeParameters(), parameterName);
112             if (paramIndex != -1) {
113                 return currTypeDef.getGenericType(paramIndex);
114             }
115         }
116 
117         // throw because we could not find parameterName
118         StringBuilder builder = new StringBuilder("No generic parameter by name ").append(parameterName);
119         for (JavaTypeDefinition currTypeDef = this; currTypeDef != null;
120                 currTypeDef = currTypeDef.getEnclosingClass()) {
121 
122             builder.append("\n on class ");
123             builder.append(currTypeDef.getType().getSimpleName());
124         }
125 
126         LOG.log(Level.FINE, builder.toString());
127         // TODO: throw eventually
128         //throw new IllegalArgumentException(builder.toString());
129         return forClass(Object.class);
130     }

132     @Override
133     public JavaTypeDefinition getGenericType(final int index) {
134         // Check if it has been lazily initialized first
135         final JavaTypeDefinition cachedDefinition = genericArgs[index];
136         if (cachedDefinition != null) {
137             return cachedDefinition;
138         }
139 
140         /*
141          * Set a default to circuit-break any recursions (ie: raw types with no generic info)
142          * Object.class is a right answer in those scenarios
143          */
144         genericArgs[index] = forClass(Object.class);
145 
146         final TypeVariable<?> typeVariable = clazz.getTypeParameters()[index];
147         final JavaTypeDefinition typeDefinition = resolveTypeDefinition(typeVariable.getBounds()[0]);
148 
149         // cache result
150         genericArgs[index] = typeDefinition;
151         return typeDefinition;
152     }

159     @Override
160     public JavaTypeDefinition resolveTypeDefinition(final Type type, Method method,
161                                                     List<JavaTypeDefinition> methodTypeArgs) {
162         if (type == null) {
163             // Without more info, this is all we can tell...
164             return forClass(Object.class);
165         }
166 
167         if (type instanceof Class) { // Raw types take this branch as well
168             return forClass((Class<?>) type);
169         } else if (type instanceof ParameterizedType) {
170             final ParameterizedType parameterizedType = (ParameterizedType) type;
171 
172             // recursively determine each type argument's type def.
173             final Type[] typeArguments = parameterizedType.getActualTypeArguments();
174             final JavaTypeDefinition[] genericBounds = new JavaTypeDefinition[typeArguments.length];
175             for (int i = 0; i < typeArguments.length; i++) {
176                 genericBounds[i] = resolveTypeDefinition(typeArguments[i], method, methodTypeArgs);
177             }
178 
179             // TODO : is this cast safe?
180             return forClass((Class<?>) parameterizedType.getRawType(), genericBounds);
181         } else if (type instanceof TypeVariable) {
182             return getGenericType(((TypeVariable<?>) type).getName(), method, methodTypeArgs);
183         } else if (type instanceof WildcardType) {
184             final Type[] wildcardLowerBounds = ((WildcardType) type).getLowerBounds();
185 
186             if (wildcardLowerBounds.length != 0) { // lower bound wildcard
187                 return forClass(LOWER_WILDCARD, resolveTypeDefinition(wildcardLowerBounds[0], method, methodTypeArgs));
188             } else { // upper bound wildcard
189                 final Type[] wildcardUpperBounds = ((WildcardType) type).getUpperBounds();
190                 return forClass(UPPER_WILDCARD, resolveTypeDefinition(wildcardUpperBounds[0], method, methodTypeArgs));
191             }
192         } else if (type instanceof GenericArrayType) {
193             JavaTypeDefinition component = resolveTypeDefinition(((GenericArrayType) type).getGenericComponentType(), method, methodTypeArgs);
194             // TODO: retain the generic types of the array component...
195             return forClass(Array.newInstance(component.getType(), 0).getClass());
196         }
197 
198         // TODO : Shall we throw here?
199         return forClass(Object.class);
200     }

234     public JavaTypeDefinition withDimensions(int numDimensions) {
235         if (numDimensions < 0) {
236             throw new IllegalArgumentException("Negative array dimension");
237         }
238         return numDimensions == 0
239                 ? this
240                 : forClass(Array.newInstance(getType(), (int[]) Array.newInstance(int.class, numDimensions)).getClass());

274     @Override
275     public String toString() {
276         final StringBuilder sb = new StringBuilder("JavaTypeDefinition [clazz=").append(clazz)
277                 .append(", definitionType=").append(getDefinitionType())
278                 .append(", genericArgs=[");
279 
280         // Forcefully resolve all generic types
281         for (int i = 0; i < genericArgs.length; i++) {
282             getGenericType(i);
283         }
284 
285         for (final JavaTypeDefinition jtd : genericArgs) {
286             sb.append(jtd.shallowString()).append(", ");
287         }
288 
289         if (genericArgs.length != 0) {
290             sb.replace(sb.length() - 3, sb.length() - 1, "");   // remove last comma
291         }
292 
293         return sb.append("], isGeneric=").append(isGeneric)
294             .append("]\n").toString();
295     }

305     @Override
306     public boolean equals(Object obj) {
307         if (!(obj instanceof JavaTypeDefinitionSimple)) {
308             return false;
309         }
310 
311         // raw vs raw
312         // we assume that this covers raw types, because they are cached
313         if (this == obj) {
314             return true;
315         }
316 
317         JavaTypeDefinitionSimple otherTypeDef = (JavaTypeDefinitionSimple) obj;
318 
319         // This should cover
320         // raw vs proper
321         // proper vs raw
322         // proper vs proper
323 
324         if (clazz != otherTypeDef.clazz) {
325             return false;
326         }
327 
328         for (int i = 0; i < getTypeParameterCount(); ++i) {
329             // Note: we assume that cycles can only exist because of raw types
330             if (!getGenericType(i).equals(otherTypeDef.getGenericType(i))) {
331                 return false;
332             }
333         }
334 
335         return true;
336     }

348     @Override
349     protected Set<JavaTypeDefinition> getSuperTypeSet(Set<JavaTypeDefinition> destinationSet) {
350         destinationSet.add(this);
351 
352         if (this.clazz != Object.class) {
353 
354             resolveTypeDefinition(clazz.getGenericSuperclass()).getSuperTypeSet(destinationSet);
355 
356             for (Type type : clazz.getGenericInterfaces()) {
357                 resolveTypeDefinition(type).getSuperTypeSet(destinationSet);
358             }
359         }
360 
361         return destinationSet;
362     }

364     @Override
365     public Set<Class<?>> getErasedSuperTypeSet() {
366         Set<Class<?>> result = new HashSet<>();
367         result.add(Object.class);
368         return getErasedSuperTypeSet(this.clazz, result);
369     }

371     private static Set<Class<?>> getErasedSuperTypeSet(Class<?> clazz, Set<Class<?>> destinationSet) {
372         if (clazz != null) {
373             destinationSet.add(clazz);
374             getErasedSuperTypeSet(clazz.getSuperclass(), destinationSet);
375 
376             for (Class<?> interfaceType : clazz.getInterfaces()) {
377                 getErasedSuperTypeSet(interfaceType, destinationSet);
378             }
379         }
380 
381         return destinationSet;
382     }

385     @Override
386     public JavaTypeDefinition getAsSuper(Class<?> superClazz) {
387         if (Objects.equals(clazz, superClazz)) { // optimize for same class calls
388             return this;
389         }
390 
391         for (JavaTypeDefinition superTypeDef : getSuperTypeSet()) {
392             if (superTypeDef.getType() == superClazz) {
393                 return superTypeDef;
394             }
395         }
396 
397         return null;
398     }
