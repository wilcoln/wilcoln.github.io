98  	@Override
99  	public CtExecutable<T> getExecutableDeclaration() {
100 		CtTypeReference<?> declaringType = getDeclaringType();
101 		if (declaringType == null) {
102 			return null;
103 		}
104 
105 		if (declaringType instanceof CtArrayTypeReference && this.isConstructor()) {
106 			CtConstructor constructor = this.getFactory().createInvisibleArrayConstructor();
107 			constructor.setType(declaringType);
108 			return constructor;
109 		}
110 
111 		return getCtExecutable(declaringType.getTypeDeclaration());
112 	}

114 	private CtExecutable<T> getCtExecutable(CtType<?> typeDecl) {
115 		if (typeDecl == null) {
116 			return null;
117 		}
118 		CtTypeReference<?>[] arrayParameters = parameters.toArray(new CtTypeReferenceImpl<?>[0]);
119 		CtExecutable<T> method = typeDecl.getMethod(getSimpleName(), arrayParameters);
120 		if ((method == null) && (typeDecl instanceof CtClass) && this.isConstructor()) {
121 			try {
122 				CtClass<?> zeClass = (CtClass) typeDecl;
123 				CtConstructor<?> constructor = zeClass.getConstructor(arrayParameters);
124 				return (CtExecutable<T>) constructor;
125 			} catch (ClassCastException e) {
126 				Launcher.LOGGER.error(e.getMessage(), e);
127 			}
128 		} else if (method == null && getSimpleName().startsWith(CtExecutableReference.LAMBDA_NAME_PREFIX)) {
129 			final List<CtLambda> elements = typeDecl.getElements(new NamedElementFilter<>(CtLambda.class, getSimpleName()));
130 			if (elements.isEmpty()) {
131 				return null;
132 			}
133 			return elements.get(0);
134 		}
135 		return method;
136 	}

153 	@Override
154 	public <C extends CtExecutableReference<T>> C setParameters(List<CtTypeReference<?>> parameters) {
155 		if (parameters == null || parameters.isEmpty()) {
156 			this.parameters = CtElementImpl.emptyList();
157 			return (C) this;
158 		}
159 		if (this.parameters == CtElementImpl.<CtTypeReference<?>>emptyList()) {
160 			this.parameters = new ArrayList<>();
161 		}
162 		getFactory().getEnvironment().getModelChangeListener().onListDeleteAll(this, ARGUMENT_TYPE, this.parameters, new ArrayList<>(this.parameters));
163 		this.parameters.clear();
164 		for (CtTypeReference<?> parameter : parameters) {
165 			addParameter(parameter);
166 		}
167 		return (C) this;
168 	}

170 	private boolean addParameter(CtTypeReference<?> parameter) {
171 		if (parameter == null) {
172 			return false;
173 		}
174 		checkMethodParameterTypeRef(parameter);
175 		parameter.setParent(this);
176 		getFactory().getEnvironment().getModelChangeListener().onListAdd(this, ARGUMENT_TYPE, this.parameters, parameter);
177 		return this.parameters.add(parameter);
178 	}

180 	private void checkMethodParameterTypeRef(CtTypeReference<?> parameterType) {
181 		if (parameterType instanceof CtTypeParameterReference && !(parameterType instanceof CtWildcardReference)) {
182 			throw new SpoonException("CtExecutableReference cannot use CtTypeParameterReference. Use boundingType of CtTypeParameterReference instead.");
183 		}
184 		if (parameterType instanceof CtArrayTypeReference) {
185 			checkMethodParameterTypeRef(((CtArrayTypeReference<?>) parameterType).getComponentType());
186 		}
187 	}

189 	@Override
190 	@SuppressWarnings("unchecked")
191 	public <S extends T> CtExecutableReference<S> getOverridingExecutable(CtTypeReference<?> subType) {
192 		if ((subType == null) || subType.equals(getDeclaringType())) {
193 			return null;
194 		}
195 		CtType<?> t = subType.getDeclaration();
196 		if (t == null) {
197 			return null;
198 		}
199 		if (!(t instanceof CtClass)) {
200 			return null;
201 		}
202 		CtClass<?> c = (CtClass<?>) t;
203 		for (CtMethod<?> m : c.getMethods()) {
204 			if (m.getReference().isOverriding(this)) {
205 				return (CtExecutableReference<S>) m.getReference();
206 			}
207 		}
208 		return getOverridingExecutable(c.getSuperclass());
209 	}

211 	@Override
212 	public boolean isOverriding(CtExecutableReference<?> executable) {
213 		CtExecutable<?> exec = executable.getExecutableDeclaration();
214 		CtExecutable<?> thisExec = getExecutableDeclaration();
215 		if (exec == null || thisExec == null) {
216 			//the declaration of this executable is not in spoon model
217 			//use light detection algorithm, which ignores generic types
218 			final boolean isSame = getSimpleName().equals(executable.getSimpleName()) && getParameters().equals(executable.getParameters()) && getActualTypeArguments().equals(executable.getActualTypeArguments());
219 			if (!isSame) {
220 				return false;
221 			}
222 			CtTypeReference<?> declaringType = getDeclaringType();
223 			return declaringType != null && declaringType.isSubtypeOf(executable.getDeclaringType());
224 		}
225 		if (exec instanceof CtMethod<?> && thisExec instanceof CtMethod<?>) {
226 			return new ClassTypingContext(((CtTypeMember) thisExec).getDeclaringType()).isOverriding((CtMethod<?>) thisExec, (CtMethod<?>) exec);
227 		}
228 		//it is not a method. So we can return true only if it is reference to the this executable
229 		return exec == getDeclaration();
230 	}

244 			addActualTypeArgument(actualTypeArgument);

246 		return (C) this;

278 	@Override
279 	public Method getActualMethod() {
280 		List<CtTypeReference<?>> parameters = this.getParameters();
281 
282 		CtTypeReference<?> declaringType = getDeclaringType();
283 		if (declaringType == null) {
284 			return null;
285 		}
286 		method_loop:
287 		for (Method m : declaringType.getActualClass().getDeclaredMethods()) {
288 			if (!m.getDeclaringClass().isSynthetic() && m.isSynthetic()) {
289 				continue;
290 			}
291 			if (!m.getName().equals(getSimpleName())) {
292 				continue;
293 			}
294 			if (m.getParameterTypes().length != parameters.size()) {
295 				continue;
296 			}
297 			for (int i = 0; i < parameters.size(); i++) {
298 				Class<?> methodParameterType = m.getParameterTypes()[i];
299 				Class<?> currentParameterType = parameters.get(i).getActualClass();
300 				if (methodParameterType != currentParameterType) {
301 					continue method_loop;
302 				}
303 			}
304 
305 			return m;
306 		}
307 		return null;
308 	}

310 	@Override
311 	public Constructor<?> getActualConstructor() {
312 		List<CtTypeReference<?>> parameters = this.getParameters();
313 
314 		CtTypeReference<?> declaringType = getDeclaringType();
315 		if (declaringType == null) {
316 			return null;
317 		}
318 		constructor_loop:
319 		for (Constructor<?> c : declaringType.getActualClass().getDeclaredConstructors()) {
320 			if (c.getParameterTypes().length != parameters.size()) {
321 				continue;
322 			}
323 			for (int i = 0; i < parameters.size(); i++) {
324 				if (c.getParameterTypes()[i] != parameters.get(i).getActualClass()) {
325 					continue constructor_loop;
326 				}
327 			}
328 			return c;
329 		}
330 		return null;
331 	}

345 	@Override
346 	public boolean isFinal() {
347 		CtExecutable<T> e = getDeclaration();
348 		if (e != null) {
349 			if (e instanceof CtMethod) {
350 				return ((CtMethod<T>) e).hasModifier(ModifierKind.FINAL);
351 			} else if (e instanceof CtConstructor) {
352 				return ((CtConstructor<T>) e).hasModifier(ModifierKind.FINAL);
353 			}
354 			return false;
355 		}
356 		Method m = getActualMethod();
357 		return m != null && Modifier.isFinal(m.getModifiers());
358 	}

360 	public Set<ModifierKind> getModifiers() {
361 		CtExecutable<T> e = getDeclaration();
362 		if (e != null) {
363 			if (e instanceof CtMethod) {
364 				return ((CtMethod<T>) e).getModifiers();
365 			} else if (e instanceof CtConstructor) {
366 				return ((CtConstructor<T>) e).getModifiers();
367 			}
368 			return CtElementImpl.emptySet();
369 		}
370 		Method m = getActualMethod();
371 		if (m != null) {
372 			return RtHelper.getModifiers(m.getModifiers());
373 		}
374 		Constructor<?> c = getActualConstructor();
375 		if (c != null) {
376 			return RtHelper.getModifiers(c.getModifiers());
377 		}
378 		return Collections.emptySet();
379 	}

381 	@Override
382 	public CtExecutableReference<?> getOverridingExecutable() {
383 		CtTypeReference<Object> objectType = getFactory().Type().OBJECT;
384 		CtTypeReference<?> declaringType = getDeclaringType();
385 		if (declaringType == null) {
386 			return getOverloadedExecutable(objectType, objectType);
387 		}
388 		CtTypeReference<?> st = declaringType.getSuperclass();
389 		if (st == null) {
390 			return getOverloadedExecutable(objectType, objectType);
391 		}
392 		return getOverloadedExecutable(st, objectType);
393 	}
394 
395 	private CtExecutableReference<?> getOverloadedExecutable(CtTypeReference<?> t, CtTypeReference<Object> objectType) {
396 		if (t == null) {
397 			return null;
398 		}
399 		for (CtExecutableReference<?> e : t.getDeclaredExecutables()) {
400 			if (this.isOverriding(e)) {
401 				return e;
402 			}
403 		}
404 		if (t.equals(objectType)) {
405 			return null;
406 		}
407 		CtTypeReference<?> st = t.getSuperclass();
408 		if (st == null) {
409 			return getOverloadedExecutable(objectType, objectType);
410 		}
411 		return getOverloadedExecutable(t.getSuperclass(), objectType);
412 	}

414 	@Override
415 	public <C extends CtActualTypeContainer> C addActualTypeArgument(CtTypeReference<?> actualTypeArgument) {
416 		if (actualTypeArgument == null) {
417 			return (C) this;
418 		}
419 		if (actualTypeArguments == CtElementImpl.<CtTypeReference<?>>emptyList()) {
420 			actualTypeArguments = new ArrayList<>(METHOD_TYPE_PARAMETERS_CONTAINER_DEFAULT_CAPACITY);
421 		}
422 		actualTypeArgument.setParent(this);
423 		getFactory().getEnvironment().getModelChangeListener().onListAdd(this, TYPE_ARGUMENT, this.actualTypeArguments, actualTypeArgument);
424 		actualTypeArguments.add(actualTypeArgument);
425 		return (C) this;
426 	}

428 	@Override
429 	public boolean removeActualTypeArgument(CtTypeReference<?> actualTypeArgument) {
430 		if (actualTypeArguments == CtElementImpl.<CtTypeReference<?>>emptyList()) {
431 			return false;
432 		}
433 		getFactory().getEnvironment().getModelChangeListener().onListDelete(this, TYPE_ARGUMENT, actualTypeArguments, actualTypeArguments.indexOf(actualTypeArgument), actualTypeArgument);
434 		return actualTypeArguments.remove(actualTypeArgument);
435 	}
