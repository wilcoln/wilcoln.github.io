5   package net.sourceforge.pmd.properties.modules;
6   
7   import java.util.Collections;
8   import java.util.Map;
9   
10  import net.sourceforge.pmd.util.CollectionUtil;
11  
12  
13  /**
14   * Factorises common functionality for enumerated properties.
15   *
16   * @author Cl√©ment Fournier
17   */
18  @Deprecated
19  public class EnumeratedPropertyModule<E> {
20  
21      private final Map<String, E> choicesByLabel;
22      private final Map<E, String> labelsByChoice;
23      private final Class<E> valueType;
24  
25  
26      public EnumeratedPropertyModule(Map<String, E> choicesByLabel, Class<E> valueType) {
27          this.valueType = valueType;
28          this.choicesByLabel = Collections.unmodifiableMap(choicesByLabel);
29          this.labelsByChoice = Collections.unmodifiableMap(CollectionUtil.invertedMapFrom(choicesByLabel));
30      }
31  
32  
33      public Class<E> getValueType() {
34          return valueType;
35      }
36  
37  
38      public Map<E, String> getLabelsByChoice() {
39          return labelsByChoice;
40      }
41  
42  
43      public Map<String, E> getChoicesByLabel() {
44          return choicesByLabel;
45      }
46  
47  
48      private String nonLegalValueMsgFor(E value) {
49          return value + " is not a legal value";
50      }
51  
52  
53      public String errorFor(E value) {
54          return labelsByChoice.containsKey(value) ? null : nonLegalValueMsgFor(value);
55      }
56  
57  
58      public E choiceFrom(String label) {
59          E result = choicesByLabel.get(label);
60          if (result != null) {
61              return result;
62          }
63          throw new IllegalArgumentException(label);
64      }
65  
66  
67      public void checkValue(E value) {
68          if (!choicesByLabel.containsValue(value)) {
69              throw new IllegalArgumentException("Invalid default value: no mapping to this value");
70          }
71      }
72  
73  }
