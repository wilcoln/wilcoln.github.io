343     private void prepareMockInvocation() throws Exception {
344         ActionContext.getContext().setValueStack(stack);
345 
346         ActionConfig config = new ActionConfig.Builder("", "name", "").build();
347         ActionInvocation invocation = EasyMock.createNiceMock(ActionInvocation.class);
348         ActionProxy proxy = EasyMock.createNiceMock(ActionProxy.class);
349 
350         EasyMock.expect(invocation.getProxy()).andReturn(proxy).anyTimes();
351         EasyMock.expect(invocation.getAction()).andReturn(null).anyTimes();
352         EasyMock.expect(invocation.invoke()).andReturn(Action.SUCCESS).anyTimes();
353         EasyMock.expect(proxy.getMethod()).andReturn("execute").anyTimes();
354         EasyMock.expect(proxy.getConfig()).andReturn(config).anyTimes();
355 
356         EasyMock.replay(invocation);
357         EasyMock.replay(proxy);
358 
359         ActionContext.getContext().setActionInvocation(invocation);
360     }
