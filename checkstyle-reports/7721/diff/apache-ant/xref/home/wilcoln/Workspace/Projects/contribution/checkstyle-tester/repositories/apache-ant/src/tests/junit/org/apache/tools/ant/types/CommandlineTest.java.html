34      @Test
35      public void testTokenizer() {
36          String[] s = Commandline.translateCommandline("1 2 3");
37          assertEquals("Simple case", 3, s.length);
38          for (int i=0; i<3; i++) {
39              assertEquals(""+(i+1), s[i]);
40          }
41  
42          s = Commandline.translateCommandline("");
43          assertEquals("empty string", 0, s.length);
44  
45          s = Commandline.translateCommandline(null);
46          assertEquals("null", 0, s.length);
47  
48          s = Commandline.translateCommandline("1 \'2\' 3");
49          assertEquals("Simple case with single quotes", 3, s.length);
50          assertEquals("Single quotes have been stripped", "2", s[1]);
51  
52          s = Commandline.translateCommandline("1 \"2\" 3");
53          assertEquals("Simple case with double quotes", 3, s.length);
54          assertEquals("Double quotes have been stripped", "2", s[1]);
55  
56          s = Commandline.translateCommandline("1 \"2 3\" 4");
57          assertEquals("Case with double quotes and whitespace", 3, s.length);
58          assertEquals("Double quotes stripped, space included", "2 3", s[1]);
59  
60          s = Commandline.translateCommandline("1 \"2\'3\" 4");
61          assertEquals("Case with double quotes around single quote", 3, s.length);
62          assertEquals("Double quotes stripped, single quote included", "2\'3",
63                       s[1]);
64  
65          s = Commandline.translateCommandline("1 \'2 3\' 4");
66          assertEquals("Case with single quotes and whitespace", 3, s.length);
67          assertEquals("Single quotes stripped, space included", "2 3", s[1]);
68  
69          s = Commandline.translateCommandline("1 \'2\"3\' 4");
70          assertEquals("Case with single quotes around double quote", 3, s.length);
71          assertEquals("Single quotes stripped, double quote included", "2\"3",
72                       s[1]);
73  
74          // \ doesn't have a special meaning anymore - this is different from
75          // what the Unix sh does but causes a lot of problems on DOS
76          // based platforms otherwise
77          s = Commandline.translateCommandline("1 2\\ 3 4");
78          assertEquals("case with quoted whitespace", 4, s.length);
79          assertEquals("backslash included", "2\\", s[1]);
80  
81          // "" should become a single empty argument, same for ''
82          // PR 5906
83          s = Commandline.translateCommandline("\"\" a");
84          assertEquals("Doublequoted null arg prepend", 2, s.length);
85          assertEquals("Doublequoted null arg prepend", "", s[0]);
86          assertEquals("Doublequoted null arg prepend", "a", s[1]);
87          s = Commandline.translateCommandline("a \"\"");
88          assertEquals("Doublequoted null arg append", 2, s.length);
89          assertEquals("Doublequoted null arg append", "a", s[0]);
90          assertEquals("Doublequoted null arg append", "", s[1]);
91          s = Commandline.translateCommandline("\"\"");
92          assertEquals("Doublequoted null arg", 1, s.length);
93          assertEquals("Doublequoted null arg", "", s[0]);
94  
95          s = Commandline.translateCommandline("\'\' a");
96          assertEquals("Singlequoted null arg prepend", 2, s.length);
97          assertEquals("Singlequoted null arg prepend", "", s[0]);
98          assertEquals("Singlequoted null arg prepend", "a", s[1]);
99          s = Commandline.translateCommandline("a \'\'");
100         assertEquals("Singlequoted null arg append", 2, s.length);
101         assertEquals("Singlequoted null arg append", "a", s[0]);
102         assertEquals("Singlequoted null arg append", "", s[1]);
103         s = Commandline.translateCommandline("\'\'");
104         assertEquals("Singlequoted null arg", 1, s.length);
105         assertEquals("Singlequoted null arg", "", s[0]);
106 
107         // now to the expected failures
108 
109         try {
110             Commandline.translateCommandline("a \'b c");
111             fail("unbalanced single quotes undetected");
112         } catch (BuildException be) {
113             assertEquals("unbalanced quotes in a \'b c", be.getMessage());
114         }
115 
116         try {
117             Commandline.translateCommandline("a \"b c");
118             fail("unbalanced double quotes undetected");
119         } catch (BuildException be) {
120             assertEquals("unbalanced quotes in a \"b c", be.getMessage());
121         }
122     }

124     @Test
125     public void testToString() {
126         assertEquals("", Commandline.toString(new String[0]));
127         assertEquals("", Commandline.toString(null));
128         assertEquals("1 2 3", Commandline.toString(new String[] {"1", "2", "3"}));
129         assertEquals("1 \"2 3\"", Commandline.toString(new String[] {"1", "2 3"}));
130         assertEquals("1 \"2\'3\"", Commandline.toString(new String[] {"1", "2\'3"}));
131         assertEquals("1 \'2\"3\'", Commandline.toString(new String[] {"1", "2\"3"}));
132     }

134     @Test
135     public void testAwkCommand() {
136         Commandline c = new Commandline();
137         c.setExecutable("awk");
138         c.createArgument().setValue("'NR == 2 { print $NF }'");
139         String[] s = c.getCommandline();
140         assertNotNull(s);
141         assertEquals(2, s.length);
142         assertEquals("awk", s[0]);
143         assertEquals("'NR == 2 { print $NF }'", s[1]);
144     }
145 
146     @Test
147     public void testPrefix() {
148         Commandline c = new Commandline();
149         Commandline.Argument a = c.createArgument();
150         a.setValue("foo");
151         a.setPrefix("-f=");
152         String[] s = c.getCommandline();
153         assertEquals(1, s.length);
154         assertEquals("-f=foo", s[0]);
155     }
156 
157     @Test
158     public void testSuffix() {
159         Commandline c = new Commandline();
160         Commandline.Argument a = c.createArgument();
161         a.setValue("foo");
162         a.setSuffix(",1");
163         String[] s = c.getCommandline();
164         assertEquals(1, s.length);
165         assertEquals("foo,1", s[0]);
166     }
167 
168     @Test
169     public void testPrefixSuffixLine() {
170         Commandline c = new Commandline();
171         Commandline.Argument a = c.createArgument();
172         a.setLine("one two");
173         a.setPrefix("number ");
174         a.setSuffix(".");
175         String[] s = c.getCommandline();
176         assertEquals(2, s.length);
177         assertEquals("number one.", s[0]);
178         assertEquals("number two.", s[1]);
179     }
