20  package org.sonar.plugins.checkstyle;
21  
22  import static org.fest.assertions.Assertions.assertThat;
23  
24  import java.lang.reflect.Constructor;
25  
26  import org.junit.Assert;
27  import org.junit.Test;
28  import org.sonar.api.rules.RulePriority;
29  
30  public class CheckstyleSeverityUtilsTest {
31  
32      @Test
33      public void testToSeverity() {
34          assertThat(CheckstyleSeverityUtils.toSeverity(RulePriority.BLOCKER)).isEqualTo("error");
35          assertThat(CheckstyleSeverityUtils.toSeverity(RulePriority.CRITICAL)).isEqualTo("error");
36          assertThat(CheckstyleSeverityUtils.toSeverity(RulePriority.MAJOR)).isEqualTo("warning");
37          assertThat(CheckstyleSeverityUtils.toSeverity(RulePriority.MINOR)).isEqualTo("info");
38          assertThat(CheckstyleSeverityUtils.toSeverity(RulePriority.INFO)).isEqualTo("info");
39      }
40  
41      @Test
42      public void testToSeverityString() {
43          assertThat(CheckstyleSeverityUtils.toSeverity("BLOCKER")).isEqualTo("error");
44          assertThat(CheckstyleSeverityUtils.toSeverity("CRITICAL")).isEqualTo("error");
45          assertThat(CheckstyleSeverityUtils.toSeverity("MAJOR")).isEqualTo("warning");
46          assertThat(CheckstyleSeverityUtils.toSeverity("MINOR")).isEqualTo("info");
47          assertThat(CheckstyleSeverityUtils.toSeverity("INFO")).isEqualTo("info");
48      }
49  
50      @Test
51      public void testToSeverityWrongString() {
52          try {
53              CheckstyleSeverityUtils.toSeverity("nothing");
54              Assert.fail("IOException while writing should not be ignored");
55          }
56          catch (IllegalArgumentException ex) {
57              assertThat(ex.getMessage()).isEqualTo("Priority not supported: nothing");
58          }
59      }
60  
61      @Test
62      public void testFromSeverity() {
63          assertThat(CheckstyleSeverityUtils.fromSeverity("error")).isEqualTo(RulePriority.BLOCKER);
64          assertThat(CheckstyleSeverityUtils.fromSeverity("warning")).isEqualTo(RulePriority.MAJOR);
65          assertThat(CheckstyleSeverityUtils.fromSeverity("info")).isEqualTo(RulePriority.INFO);
66          assertThat(CheckstyleSeverityUtils.fromSeverity("ignore")).isEqualTo(RulePriority.INFO);
67          assertThat(CheckstyleSeverityUtils.fromSeverity("")).isNull();
68      }
69  
70      @Test
71      public void privateConstructor() throws ReflectiveOperationException {
72          final Constructor<CheckstyleSeverityUtils> constructor = CheckstyleSeverityUtils.class
73                  .getDeclaredConstructor();
74  
75          assertThat(constructor.isAccessible()).isFalse();
76          constructor.setAccessible(true);
77          constructor.newInstance();
78      }
79  
80  }
