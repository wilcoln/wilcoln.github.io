19  package org.apache.struts2.views.gxp;
20  
21  import com.google.common.base.Function;
22  import com.google.common.collect.MapMaker;
23  import com.opensymphony.xwork2.ObjectFactory;
24  import com.opensymphony.xwork2.inject.Inject;
25  
26  import java.util.Map;
27  import java.util.logging.Level;
28  import java.util.logging.Logger;
29  
30  /**
31   * Struts 2 to GXP adapter that uses instances of GXP Interfaces, as created by
32   * the {@link ObjectFactory}. Can be used to write a GXP or create a
33   * HtmlClosure. Pulls non-constructor GXP parameters from Struts 2 value stack.
34   *
35   * @author David P. Baker
36   */
37  public class GxpInstance extends Gxp {
38  
39      private static final Logger logger = Logger.getLogger(GxpInstance.class.getCanonicalName());
40  
41      private Class gxpInterface;
42      private Class gxpInstance;
43      private ObjectFactory objectFactory;
44  
45      GxpInstance(Class gxpClass) {
46          super(gxpClass,
47                  lookupMethodByName(getNestedClass(gxpClass, "Interface"), "write"),
48                  lookupMethodByName(getNestedClass(gxpClass, "Interface"), "getGxpClosure"));
49          this.gxpInterface = getNestedClass(gxpClass, "Interface");
50          this.gxpInstance = getNestedClass(gxpClass, "Instance");
51      }
52  
53      private static Class getNestedClass(Class clazz, String nestedClassName) {
54          for (Class nested : clazz.getDeclaredClasses()) {
55              if (nestedClassName.equals(nested.getSimpleName())) {
56                  return nested;
57              }
58          }
59          throw new IllegalArgumentException(String.format("Cannot find class %s.%s", clazz.getCanonicalName(), nestedClassName));
60      }
61  
62      /**
63       * {@inheritDoc}
64       * <p> This implementation uses the {@link ObjectFactory} to try to create an
65       * instance of the {@code Interface} class that is nested within the GXP
66       * class. If that doesn't work, it falls back to trying to use the
67       * {@code ObjectFactory} to create an instance of the nested {@code Instance}
68       * class, in case there is no binding for the {@code Interface}.</p>
69       *
70       * @return Gxp instance
71       */
72      @Override
73      protected Object getGxpInstance() {
74          try {
75              return objectFactory.buildBean(gxpInterface, null);
76          } catch (Exception e) {
77              logger.log(
78                      Level.INFO, "Error instantiating {0}; trying {1}",
79                      new Object[]{gxpInterface.getCanonicalName(), gxpInstance.getCanonicalName(),});
80              try {
81                  return objectFactory.buildBean(gxpInstance, null);
82              } catch (Exception e1) {
83                  throw new RuntimeException(String.format("Error instantiating %s",
84                          gxpInterface.getCanonicalName(), gxpInstance.getCanonicalName()),
85                          e1);
86              }
87          }
88      }
89  
90      @Override
91      public Class<?> getGxpClass() {
92          return this.gxpInterface;
93      }
94  
95      private static final Map<Class, GxpInstance> classToGxpInstance = new MapMaker().weakKeys().softValues()
96              .makeComputingMap(new Function<Class, GxpInstance>() {
97                  public GxpInstance apply(Class from) {
98                      return classToGxpInstance.containsKey(from) ? classToGxpInstance.get(from) : new GxpInstance(from);
99                  }
100             });
101 
102     private static final Map<String, GxpInstance> pathToGxpInstance = new MapMaker().softValues()
103             .makeComputingMap(new Function<String, GxpInstance>() {
104                 public GxpInstance apply(String from) {
105                     return pathToGxpInstance.containsKey(from) ? pathToGxpInstance.get(from) : getInstance(getGxpClassForPath(from));
106                 }
107             });
108 
109     /**
110      * Looks up Gxp instance for GXP with given path.
111      *
112      * @param gxpPath Gxp path
113      *
114      * @return Gxp instance
115      */
116     public static GxpInstance getInstance(String gxpPath) {
117         try {
118             return pathToGxpInstance.get(gxpPath);
119         } catch (RuntimeException e) {
120             if (e.getCause() instanceof ClassNotFoundException) {
121                 // Couldn't find or load the GXP class.  Return null.
122                 // It would be simpler if classToGxp.create() could return null,
123                 // but the contract of ReferenceCache doesn't allow it to.
124                 return null;
125             }
126             throw e;
127         }
128     }
129 
130     /**
131      * Looks up {@code GxpInstance} instance for the given GXP class.
132      *
133      * @param gxpClass Gxp class
134      *
135      * @return Gxp instance
136      */
137     public static GxpInstance getInstance(Class gxpClass) {
138         return classToGxpInstance.get(gxpClass);
139     }
140 
141     @Inject
142     public void setObjectFactory(ObjectFactory objectFactory) {
143         this.objectFactory = objectFactory;
144     }
145 
146 }
