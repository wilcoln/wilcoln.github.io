20  package edu.umd.cs.findbugs.ba.type;
21  
22  import java.util.BitSet;
23  import java.util.Iterator;
24  import java.util.NoSuchElementException;
25  
26  import org.apache.bcel.generic.ObjectType;
27  import org.apache.bcel.generic.ReferenceType;
28  import org.apache.bcel.generic.Type;
29  
30  import edu.umd.cs.findbugs.ba.AnalysisContext;
31  import edu.umd.cs.findbugs.ba.Hierarchy;
32  import edu.umd.cs.findbugs.ba.ch.Subtypes2;

34  /**
35   * Class for keeping track of exceptions that can be thrown by an instruction.
36   * We distinguish <em>explicit</em> and <em>implicit</em> exceptions. Explicit
37   * exceptions are explicitly declared, thrown, or caught. Implicit exceptions
38   * are runtime faults (NPE, array out of bounds) not explicitly handled by the
39   * user code.
40   *
41   * @author David Hovemeyer
42   * @see TypeAnalysis
43   */
44  public class ExceptionSet {
45  
46      private final ExceptionSetFactory factory;
47  
48      private final BitSet exceptionSet;
49  
50      private final BitSet explicitSet;
51  
52      private int size;
53  
54      private boolean universalHandler;
55  
56      private Type commonSupertype;
57  
58      /**
59       * Object to iterate over the exception types in the set.
60       */
61      public class ThrownExceptionIterator implements Iterator<ObjectType> {
62          private int last = -1, next = -1;
63  
64          ThrownExceptionIterator() {
65              findNext();
66          }
67  
68          @Override
69          public boolean hasNext() {
70              if (last == next) {
71                  findNext();
72              }
73              return next < factory.getNumTypes();
74          }
75  
76          @Override
77          public ObjectType next() {
78              if (!hasNext()) {
79                  throw new NoSuchElementException();
80              }
81              ObjectType result = factory.getType(next);
82              last = next;
83              return result;
84          }
85  
86          public boolean isExplicit() {
87              return explicitSet.get(last);
88          }
89  
90          @Override
91          public void remove() {
92              exceptionSet.clear(last);
93              explicitSet.clear(last);
94              --size;
95              commonSupertype = null;
96          }
97  
98          private void findNext() {
99              ++next;
100             while (next < factory.getNumTypes()) {
101                 if (exceptionSet.get(next)) {
102                     break;
103                 }
104                 ++next;
105             }
106         }
107     }
108 
109     /**
110      * Constructor. Creates an empty set.
111      */
112     ExceptionSet(ExceptionSetFactory factory) {
113         this.factory = factory;
114         this.exceptionSet = new BitSet();
115         this.explicitSet = new BitSet();
116         this.size = 0;
117         this.universalHandler = false;
118     }
119 
120     /**
121      * Return an exact copy of this object.
122      */
123     public ExceptionSet duplicate() {
124         ExceptionSet dup = factory.createExceptionSet();
125         dup.exceptionSet.clear();
126         dup.exceptionSet.or(this.exceptionSet);
127         dup.explicitSet.clear();
128         dup.explicitSet.or(this.explicitSet);
129         dup.size = this.size;
130         dup.universalHandler = this.universalHandler;
131         dup.commonSupertype = this.commonSupertype;
132 
133         return dup;
134     }
135 
136     @Override
137     public int hashCode() {
138         return exceptionSet.hashCode() + explicitSet.hashCode();
139     }
140 
141     @Override
142     public boolean equals(Object o) {
143         if (o == null) {
144             return false;
145         }
146         if (o.getClass() != this.getClass()) {
147             return false;
148         }
149 
150         ExceptionSet other = (ExceptionSet) o;
151         return exceptionSet.equals(other.exceptionSet) && explicitSet.equals(other.explicitSet)
152                 && universalHandler == other.universalHandler;
153     }
154 
155     /**
156      * Get the least (lowest in the lattice) common supertype of the exceptions
157      * in the set. Returns the special TOP type if the set is empty.
158      */
159     public Type getCommonSupertype() throws ClassNotFoundException {
160         if (commonSupertype != null) {
161             return commonSupertype;
162         }
163 
164         if (isEmpty()) {
165             // This probably means that we're looking at an
166             // infeasible exception path.
167             return TypeFrame.getTopType();
168         }
169 
170         // Compute first common superclass
171         ThrownExceptionIterator i = iterator();
172         ReferenceType result = i.next();
173         while (i.hasNext()) {
174             if (Subtypes2.ENABLE_SUBTYPES2_FOR_COMMON_SUPERCLASS_QUERIES) {
175                 result = AnalysisContext.currentAnalysisContext().getSubtypes2().getFirstCommonSuperclass(result, i.next());
176             } else {
177                 result = result.getFirstCommonSuperclass(i.next());
178             }
179             if (result == null) {
180                 // This should only happen if the class hierarchy
181                 // is incomplete. We'll just be conservative.
182                 result = Type.THROWABLE;
183                 break;
184             }
185         }
186 
187         // Cache and return the result
188         commonSupertype = result;
189         return result;
190     }
191 
192     /**
193      * Return an iterator over thrown exceptions.
194      */
195     public ThrownExceptionIterator iterator() {
196         return new ThrownExceptionIterator();
197     }
198 
199     /**
200      * Return whether or not the set is empty.
201      */
202     public boolean isEmpty() {
203         return size == 0;
204     }
205 
206     /**
207      * Checks to see if the exception set is a singleton set containing just the
208      * named exception
209      *
210      * @param exceptionName
211      *            (in dotted format)
212      * @return true if it is
213      */
214     public boolean isSingleton(String exceptionName) {
215         if (size != 1) {
216             return false;
217         }
218         ObjectType e = iterator().next();
219         return e.toString().equals(exceptionName);
220 
221     }
222 
223     /**
224      * Add an explicit exception.
225      *
226      * @param type
227      *            type of the exception
228      */
229     public void addExplicit(ObjectType type) {
230         add(type, true);
231     }
232 
233     /**
234      * Add an implicit exception.
235      *
236      * @param type
237      *            type of the exception
238      */
239     public void addImplicit(ObjectType type) {
240         add(type, false);
241     }
242 
243     /**
244      * Add an exception.
245      *
246      * @param type
247      *            the exception type
248      * @param explicit
249      *            true if the exception is explicitly declared or thrown, false
250      *            if implicit
251      */
252     public void add(ObjectType type, boolean explicit) {
253         int index = factory.getIndexOfType(type);
254         if (!exceptionSet.get(index)) {
255             ++size;
256         }
257         exceptionSet.set(index);
258         if (explicit) {
259             explicitSet.set(index);
260         }
261 
262         commonSupertype = null;
263     }
264 
265     /**
266      * Add all exceptions in the given set.
267      *
268      * @param other
269      *            the set
270      */
271     public void addAll(ExceptionSet other) {
272         exceptionSet.or(other.exceptionSet);
273         explicitSet.or(other.explicitSet);
274         size = countBits(exceptionSet);
275 
276         commonSupertype = null;
277     }
278 
279     private int countBits(BitSet bitSet) {
280         int count = 0;
281         for (int i = 0; i < factory.getNumTypes(); ++i) {
282             if (bitSet.get(i)) {
283                 ++count;
284             }
285         }
286         return count;
287     }
288 
289     /**
290      * Remove all exceptions from the set.
291      */
292     public void clear() {
293         exceptionSet.clear();
294         explicitSet.clear();
295         universalHandler = false;
296         commonSupertype = null;
297         size = 0;
298     }
299 
300     /**
301      * Return whether or not a universal exception handler was reached by the
302      * set.
303      */
304     public void sawUniversal() {
305         clear();
306         universalHandler = true;
307     }
308 
309     /**
310      * Mark the set as having reached a universal exception handler.
311      */
312     public boolean sawUniversalHandler() {
313         return universalHandler;
314     }
315 
316     /**
317      * Return whether or not the set contains any checked exceptions.
318      */
319     public boolean containsCheckedExceptions() throws ClassNotFoundException {
320         for (ThrownExceptionIterator i = iterator(); i.hasNext();) {
321             ObjectType type = i.next();
322             if (!Hierarchy.isUncheckedException(type)) {
323                 return true;
324             }
325         }
326         return false;
327     }
328 
329     /**
330      * Return whether or not the set contains any explicit exceptions.
331      */
332     public boolean containsExplicitExceptions() {
333         for (ThrownExceptionIterator i = iterator(); i.hasNext();) {
334             i.next();
335             if (i.isExplicit()) {
336                 return true;
337             }
338         }
339         return false;
340     }
341 
342     @Override
343     public String toString() {
344         StringBuilder buf = new StringBuilder();
345         buf.append('{');
346         boolean first = true;
347         for (ThrownExceptionIterator i = iterator(); i.hasNext();) {
348             ObjectType type = i.next();
349             if (first) {
350                 first = false;
351             } else {
352                 buf.append(',');
353             }
354             boolean implicit = !i.isExplicit();
355             if (implicit) {
356                 buf.append('[');
357             }
358             buf.append(type.toString());
359             if (implicit) {
360                 buf.append(']');
361             }
362         }
363         buf.append('}');
364         return buf.toString();
365     }
366 
367     public int size() {
368         return size;
369     }
370 }
