57      // Make sure that unordered, reversed, disjoint and/or overlapping ranges are supported
58      // Duel with filters
59      public void testRandomRanges() throws Exception {
60          final int numDocs = scaledRandomIntBetween(500, 5000);
61          final double[][] docs = new double[numDocs][];
62          for (int i = 0; i < numDocs; ++i) {
63              final int numValues = randomInt(5);
64              docs[i] = new double[numValues];
65              for (int j = 0; j < numValues; ++j) {
66                  docs[i][j] = randomDouble() * 100;
67              }
68          }
69  
70          createIndex("idx");
71          for (int i = 0; i < docs.length; ++i) {
72              XContentBuilder source = jsonBuilder()
73                      .startObject()
74                      .startArray("values");
75              for (int j = 0; j < docs[i].length; ++j) {
76                  source = source.value(docs[i][j]);
77              }
78              source = source.endArray().endObject();
79              client().prepareIndex("idx", "type").setSource(source).execute().actionGet();
80          }
81          assertNoFailures(client().admin().indices().prepareRefresh("idx").setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute().get());
82  
83          final int numRanges = randomIntBetween(1, 20);
84          final double[][] ranges = new double[numRanges][];
85          for (int i = 0; i < ranges.length; ++i) {
86              switch (randomInt(2)) {
87              case 0:
88                  ranges[i] = new double[] { Double.NEGATIVE_INFINITY, randomInt(100) };
89                  break;
90              case 1:
91                  ranges[i] = new double[] { randomInt(100), Double.POSITIVE_INFINITY };
92                  break;
93              case 2:
94                  ranges[i] = new double[] { randomInt(100), randomInt(100) };
95                  break;
96              default:
97                  throw new AssertionError();
98              }
99          }
100 
101         RangeBuilder query = range("range").field("values");
102         for (int i = 0; i < ranges.length; ++i) {
103             String key = Integer.toString(i);
104             if (ranges[i][0] == Double.NEGATIVE_INFINITY) {
105                 query.addUnboundedTo(key, ranges[i][1]);
106             } else if (ranges[i][1] == Double.POSITIVE_INFINITY) {
107                 query.addUnboundedFrom(key, ranges[i][0]);
108             } else {
109                 query.addRange(key, ranges[i][0], ranges[i][1]);
110             }
111         }
112 
113         SearchRequestBuilder reqBuilder = client().prepareSearch("idx").addAggregation(query);
114         for (int i = 0; i < ranges.length; ++i) {
115             RangeFilterBuilder filter = FilterBuilders.rangeFilter("values");
116             if (ranges[i][0] != Double.NEGATIVE_INFINITY) {
117                 filter = filter.from(ranges[i][0]);
118             }
119             if (ranges[i][1] != Double.POSITIVE_INFINITY){
120                 filter = filter.to(ranges[i][1]);
121             }
122             reqBuilder = reqBuilder.addAggregation(filter("filter" + i).filter(filter));
123         }
124 
125         SearchResponse resp = reqBuilder.execute().actionGet();
126         Range range = resp.getAggregations().get("range");
127 
128         for (int i = 0; i < ranges.length; ++i) {
129 
130             long count = 0;
131             for (double[] values : docs) {
132                 for (double value : values) {
133                     if (value >= ranges[i][0] && value < ranges[i][1]) {
134                         ++count;
135                         break;
136                     }
137                 }
138             }
139 
140             final Range.Bucket bucket = range.getBucketByKey(Integer.toString(i));
141             assertEquals(bucket.getKey(), count, bucket.getDocCount());
142 
143             final Filter filter = resp.getAggregations().get("filter" + i);
144             assertThat(filter.getDocCount(), equalTo(count));
145         }
146     }

148     // test long/double/string terms aggs with high number of buckets that require array growth
149     public void testDuelTerms() throws Exception {
150         final int numDocs = scaledRandomIntBetween(1000, 2000);
151         final int maxNumTerms = randomIntBetween(10, 5000);
152 
153         final IntOpenHashSet valuesSet = new IntOpenHashSet();
154         cluster().wipeIndices("idx");
155         prepareCreate("idx")
156                 .addMapping("type", jsonBuilder().startObject()
157                         .startObject("type")
158                         .startObject("properties")
159                         .startObject("string_values")
160                         .field("type", "string")
161                         .field("index", "not_analyzed")
162                         .startObject("fields")
163                             .startObject("doc_values")
164                                 .field("type", "string")
165                                 .field("index", "no")
166                                 .startObject("fielddata")
167                                     .field("format", "doc_values")
168                                 .endObject()
169                             .endObject()
170                         .endObject()
171                         .endObject()
172                         .startObject("long_values")
173                         .field("type", "long")
174                         .endObject()
175                         .startObject("double_values")
176                         .field("type", "double")
177                         .endObject()
178                         .endObject()
179                         .endObject()).execute().actionGet();
180 
181         List<IndexRequestBuilder> indexingRequests = Lists.newArrayList();
182         for (int i = 0; i < numDocs; ++i) {
183             final int[] values = new int[randomInt(4)];
184             for (int j = 0; j < values.length; ++j) {
185                 values[j] = randomInt(maxNumTerms - 1) - 1000;
186                 valuesSet.add(values[j]);
187             }
188             XContentBuilder source = jsonBuilder()
189                     .startObject()
190                     .field("num", randomDouble())
191                     .startArray("long_values");
192             for (int j = 0; j < values.length; ++j) {
193                 source = source.value(values[j]);
194             }
195             source = source.endArray().startArray("double_values");
196             for (int j = 0; j < values.length; ++j) {
197                 source = source.value((double) values[j]);
198             }
199             source = source.endArray().startArray("string_values");
200             for (int j = 0; j < values.length; ++j) {
201                 source = source.value(Integer.toString(values[j]));
202             }
203             source = source.endArray().endObject();
204             indexingRequests.add(client().prepareIndex("idx", "type").setSource(source));
205         }
206         indexRandom(true, indexingRequests);
207 
208         assertNoFailures(client().admin().indices().prepareRefresh("idx").setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute().get());
209 
210         TermsAggregatorFactory.ExecutionMode[] globalOrdinalModes = new TermsAggregatorFactory.ExecutionMode[]{
211                 TermsAggregatorFactory.ExecutionMode.GLOBAL_ORDINALS_HASH,
212                 TermsAggregatorFactory.ExecutionMode.GLOBAL_ORDINALS
213         };
214 
215         SearchResponse resp = client().prepareSearch("idx")
216                 .addAggregation(terms("long").field("long_values").size(maxNumTerms).collectMode(randomFrom(SubAggCollectionMode.values())).subAggregation(min("min").field("num")))
217                 .addAggregation(terms("double").field("double_values").size(maxNumTerms).collectMode(randomFrom(SubAggCollectionMode.values())).subAggregation(max("max").field("num")))
218                 .addAggregation(terms("string_map").field("string_values").collectMode(randomFrom(SubAggCollectionMode.values())).executionHint(TermsAggregatorFactory.ExecutionMode.MAP.toString()).size(maxNumTerms).subAggregation(stats("stats").field("num")))
219                 .addAggregation(terms("string_global_ordinals").field("string_values").collectMode(randomFrom(SubAggCollectionMode.values())).executionHint(globalOrdinalModes[randomInt(globalOrdinalModes.length - 1)].toString()).size(maxNumTerms).subAggregation(extendedStats("stats").field("num")))
220                 .addAggregation(terms("string_global_ordinals_doc_values").field("string_values.doc_values").collectMode(randomFrom(SubAggCollectionMode.values())).executionHint(globalOrdinalModes[randomInt(globalOrdinalModes.length - 1)].toString()).size(maxNumTerms).subAggregation(extendedStats("stats").field("num")))
221                 .execute().actionGet();
222         assertAllSuccessful(resp);
223         assertEquals(numDocs, resp.getHits().getTotalHits());
224 
225         final Terms longTerms = resp.getAggregations().get("long");
226         final Terms doubleTerms = resp.getAggregations().get("double");
227         final Terms stringMapTerms = resp.getAggregations().get("string_map");
228         final Terms stringGlobalOrdinalsTerms = resp.getAggregations().get("string_global_ordinals");
229         final Terms stringGlobalOrdinalsDVTerms = resp.getAggregations().get("string_global_ordinals_doc_values");
230 
231         assertEquals(valuesSet.size(), longTerms.getBuckets().size());
232         assertEquals(valuesSet.size(), doubleTerms.getBuckets().size());
233         assertEquals(valuesSet.size(), stringMapTerms.getBuckets().size());
234         assertEquals(valuesSet.size(), stringGlobalOrdinalsTerms.getBuckets().size());
235         assertEquals(valuesSet.size(), stringGlobalOrdinalsDVTerms.getBuckets().size());
236         for (Terms.Bucket bucket : longTerms.getBuckets()) {
237             final Terms.Bucket doubleBucket = doubleTerms.getBucketByKey(Double.toString(Long.parseLong(bucket.getKeyAsText().string())));
238             final Terms.Bucket stringMapBucket = stringMapTerms.getBucketByKey(bucket.getKeyAsText().string());
239             final Terms.Bucket stringGlobalOrdinalsBucket = stringGlobalOrdinalsTerms.getBucketByKey(bucket.getKeyAsText().string());
240             final Terms.Bucket stringGlobalOrdinalsDVBucket = stringGlobalOrdinalsDVTerms.getBucketByKey(bucket.getKeyAsText().string());
241             assertNotNull(doubleBucket);
242             assertNotNull(stringMapBucket);
243             assertNotNull(stringGlobalOrdinalsBucket);
244             assertNotNull(stringGlobalOrdinalsDVBucket);
245             assertEquals(bucket.getDocCount(), doubleBucket.getDocCount());
246             assertEquals(bucket.getDocCount(), stringMapBucket.getDocCount());
247             assertEquals(bucket.getDocCount(), stringGlobalOrdinalsBucket.getDocCount());
248             assertEquals(bucket.getDocCount(), stringGlobalOrdinalsDVBucket.getDocCount());
249         }
250     }

252     // Duel between histograms and scripted terms
253     public void testDuelTermsHistogram() throws Exception {
254         createIndex("idx");
255 
256         final int numDocs = scaledRandomIntBetween(500, 5000);
257         final int maxNumTerms = randomIntBetween(10, 2000);
258         final int interval = randomIntBetween(1, 100);
259 
260         final Integer[] values = new Integer[maxNumTerms];
261         for (int i = 0; i < values.length; ++i) {
262             values[i] = randomInt(maxNumTerms * 3) - maxNumTerms;
263         }
264 
265         for (int i = 0; i < numDocs; ++i) {
266             XContentBuilder source = jsonBuilder()
267                     .startObject()
268                     .field("num", randomDouble())
269                     .startArray("values");
270             final int numValues = randomInt(4);
271             for (int j = 0; j < numValues; ++j) {
272                 source = source.value(randomFrom(values));
273             }
274             source = source.endArray().endObject();
275             client().prepareIndex("idx", "type").setSource(source).execute().actionGet();
276         }
277         assertNoFailures(client().admin().indices().prepareRefresh("idx").setIndicesOptions(IndicesOptions.lenientExpandOpen()).execute().get());
278 
279         SearchResponse resp = client().prepareSearch("idx")
280                 .addAggregation(terms("terms").field("values").collectMode(randomFrom(SubAggCollectionMode.values())).script("floor(_value / interval)").param("interval", interval).size(maxNumTerms))
281                 .addAggregation(histogram("histo").field("values").interval(interval))
282                 .execute().actionGet();
283 
284         assertSearchResponse(resp);
285 
286         Terms terms = resp.getAggregations().get("terms");
287         assertThat(terms, notNullValue());
288         Histogram histo = resp.getAggregations().get("histo");
289         assertThat(histo, notNullValue());
290         assertThat(terms.getBuckets().size(), equalTo(histo.getBuckets().size()));
291         for (Terms.Bucket bucket : terms.getBuckets()) {
292             final long key = bucket.getKeyAsNumber().longValue() * interval;
293             final Histogram.Bucket histoBucket = histo.getBucketByKey(key);
294             assertEquals(bucket.getDocCount(), histoBucket.getDocCount());
295         }
296     }

298     public void testLargeNumbersOfPercentileBuckets() throws Exception {
299         // test high numbers of percentile buckets to make sure paging and release work correctly
300         createIndex("idx");
301 
302         final int numDocs = scaledRandomIntBetween(2500, 5000);
303         logger.info("Indexing [" + numDocs +"] docs");
304         List<IndexRequestBuilder> indexingRequests = Lists.newArrayList();
305         for (int i = 0; i < numDocs; ++i) {
306             indexingRequests.add(client().prepareIndex("idx", "type", Integer.toString(i)).setSource("double_value", randomDouble()));
307         }
308         indexRandom(true, indexingRequests);
309 
310         SearchResponse response = client().prepareSearch("idx").addAggregation(terms("terms").field("double_value").collectMode(randomFrom(SubAggCollectionMode.values())).subAggregation(percentiles("pcts").field("double_value"))).execute().actionGet();
311         assertAllSuccessful(response);
312         assertEquals(numDocs, response.getHits().getTotalHits());
313     }

315     // https://github.com/elasticsearch/elasticsearch/issues/6435
316     public void testReduce() throws Exception {
317         createIndex("idx");
318         final int value = randomIntBetween(0, 10);
319         indexRandom(true, client().prepareIndex("idx", "type").setSource("f", value));
320         ensureYellow("idx"); // only one document let's make sure all shards have an active primary
321         SearchResponse response = client().prepareSearch("idx")
322                 .addAggregation(filter("filter").filter(FilterBuilders.matchAllFilter())
323                 .subAggregation(range("range")
324                         .field("f")
325                         .addUnboundedTo(6)
326                         .addUnboundedFrom(6)
327                 .subAggregation(sum("sum").field("f"))))
328                 .execute().actionGet();
329 
330         assertSearchResponse(response);
331 
332         Filter filter = response.getAggregations().get("filter");
333         assertNotNull(filter);
334         assertEquals(1, filter.getDocCount());
335 
336         Range range = filter.getAggregations().get("range");
337         assertThat(range, notNullValue());
338         assertThat(range.getName(), equalTo("range"));
339         assertThat(range.getBuckets().size(), equalTo(2));
340 
341         Range.Bucket bucket = range.getBucketByKey("*-6.0");
342         assertThat(bucket, notNullValue());
343         assertThat(bucket.getKey(), equalTo("*-6.0"));
344         assertThat(bucket.getFrom().doubleValue(), equalTo(Double.NEGATIVE_INFINITY));
345         assertThat(bucket.getTo().doubleValue(), equalTo(6.0));
346         assertThat(bucket.getDocCount(), equalTo(value < 6 ? 1L : 0L));
347         Sum sum = bucket.getAggregations().get("sum");
348         assertEquals(value < 6 ? value : 0, sum.getValue(), 0d);
349 
350         bucket = range.getBucketByKey("6.0-*");
351         assertThat(bucket, notNullValue());
352         assertThat(bucket.getKey(), equalTo("6.0-*"));
353         assertThat(bucket.getFrom().doubleValue(), equalTo(6.0));
354         assertThat(bucket.getTo().doubleValue(), equalTo(Double.POSITIVE_INFINITY));
355         assertThat(bucket.getDocCount(), equalTo(value >= 6 ? 1L : 0L));
356         sum = bucket.getAggregations().get("sum");
357         assertEquals(value >= 6 ? value : 0, sum.getValue(), 0d);
358     }
