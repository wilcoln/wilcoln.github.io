51      @Override
52      public void setupSuiteScopeCluster() throws Exception {
53          assertAcked(
54                  prepareCreate("test")
55                      .addMapping("article", "_id", "index=not_analyzed")
56                      .addMapping("comment", "_parent", "type=article", "_id", "index=not_analyzed")
57          );
58  
59          List<IndexRequestBuilder> requests = new ArrayList<>();
60          String[] uniqueCategories = new String[randomIntBetween(1, 25)];
61          for (int i = 0; i < uniqueCategories.length; i++) {
62              uniqueCategories[i] = Integer.toString(i);
63          }
64          int catIndex = 0;
65  
66          int numParentDocs = randomIntBetween(uniqueCategories.length, uniqueCategories.length * 5);
67          for (int i = 0; i < numParentDocs; i++) {
68              String id = Integer.toString(i);
69  
70              String[] categories = new String[randomIntBetween(1,1)];
71              for (int j = 0; j < categories.length; j++) {
72                  String category = categories[j] = uniqueCategories[catIndex++ % uniqueCategories.length];
73                  Control control = categoryToControl.get(category);
74                  if (control == null) {
75                      categoryToControl.put(category, control = new Control(category));
76                  }
77                  control.articleIds.add(id);
78              }
79  
80              requests.add(client().prepareIndex("test", "article", id).setCreate(true).setSource("category", categories, "randomized", true));
81          }
82  
83          String[] commenters = new String[randomIntBetween(5, 50)];
84          for (int i = 0; i < commenters.length; i++) {
85              commenters[i] = Integer.toString(i);
86          }
87  
88          int id = 0;
89          for (Control control : categoryToControl.values()) {
90              for (String articleId : control.articleIds) {
91                  int numChildDocsPerParent = randomIntBetween(0, 5);
92                  for (int i = 0; i < numChildDocsPerParent; i++) {
93                      String commenter = commenters[id % commenters.length];
94                      String idValue = Integer.toString(id++);
95                      control.commentIds.add(idValue);
96                      Set<String> ids = control.commenterToCommentId.get(commenter);
97                      if (ids == null) {
98                          control.commenterToCommentId.put(commenter, ids = new HashSet<>());
99                      }
100                     ids.add(idValue);
101                     requests.add(client().prepareIndex("test", "comment", idValue).setCreate(true).setParent(articleId).setSource("commenter", commenter));
102                 }
103             }
104         }
105 
106         requests.add(client().prepareIndex("test", "article", "a").setSource("category", new String[]{"a"}, "randomized", false));
107         requests.add(client().prepareIndex("test", "article", "b").setSource("category", new String[]{"a", "b"}, "randomized", false));
108         requests.add(client().prepareIndex("test", "article", "c").setSource("category", new String[]{"a", "b", "c"}, "randomized", false));
109         requests.add(client().prepareIndex("test", "article", "d").setSource("category", new String[]{"c"}, "randomized", false));
110         requests.add(client().prepareIndex("test", "comment", "a").setParent("a").setSource("{}"));
111         requests.add(client().prepareIndex("test", "comment", "c").setParent("c").setSource("{}"));
112 
113         indexRandom(true, requests);
114         ensureSearchable("test");
115     }

117     @Test
118     public void testChildrenAggs() throws Exception {
119         SearchResponse searchResponse = client().prepareSearch("test")
120                 .setQuery(matchQuery("randomized", true))
121                 .addAggregation(
122                         terms("category").field("category").size(0).subAggregation(
123                                 children("to_comment").childType("comment").subAggregation(
124                                         terms("commenters").field("commenter").size(0).subAggregation(
125                                                 topHits("top_comments")
126                                         ))
127                         )
128                 ).get();
129         assertSearchResponse(searchResponse);
130 
131         Terms categoryTerms = searchResponse.getAggregations().get("category");
132         assertThat(categoryTerms.getBuckets().size(), equalTo(categoryToControl.size()));
133         for (Map.Entry<String, Control> entry1 : categoryToControl.entrySet()) {
134             Terms.Bucket categoryBucket = categoryTerms.getBucketByKey(entry1.getKey());
135             assertThat(categoryBucket.getKey(), equalTo(entry1.getKey()));
136             assertThat(categoryBucket.getDocCount(), equalTo((long) entry1.getValue().articleIds.size()));
137 
138             Children childrenBucket = categoryBucket.getAggregations().get("to_comment");
139             assertThat(childrenBucket.getName(), equalTo("to_comment"));
140             assertThat(childrenBucket.getDocCount(), equalTo((long) entry1.getValue().commentIds.size()));
141 
142             Terms commentersTerms = childrenBucket.getAggregations().get("commenters");
143             assertThat(commentersTerms.getBuckets().size(), equalTo(entry1.getValue().commenterToCommentId.size()));
144             for (Map.Entry<String, Set<String>> entry2 : entry1.getValue().commenterToCommentId.entrySet()) {
145                 Terms.Bucket commentBucket = commentersTerms.getBucketByKey(entry2.getKey());
146                 assertThat(commentBucket.getKey(), equalTo(entry2.getKey()));
147                 assertThat(commentBucket.getDocCount(), equalTo((long) entry2.getValue().size()));
148 
149                 TopHits topHits = commentBucket.getAggregations().get("top_comments");
150                 for (SearchHit searchHit : topHits.getHits().getHits()) {
151                     assertThat(entry2.getValue().contains(searchHit.getId()), is(true));
152                 }
153             }
154         }
155     }

157     @Test
158     public void testParentWithMultipleBuckets() throws Exception {
159         SearchResponse searchResponse = client().prepareSearch("test")
160                 .setQuery(matchQuery("randomized", false))
161                 .addAggregation(
162                         terms("category").field("category").size(0).subAggregation(
163                                 children("to_comment").childType("comment").subAggregation(topHits("top_comments").addSort("_id", SortOrder.ASC))
164                         )
165                 ).get();
166         assertSearchResponse(searchResponse);
167 
168         Terms categoryTerms = searchResponse.getAggregations().get("category");
169         assertThat(categoryTerms.getBuckets().size(), equalTo(3));
170 
171         for (Terms.Bucket bucket : categoryTerms.getBuckets()) {
172             logger.info("bucket=" + bucket.getKey());
173             Children childrenBucket = bucket.getAggregations().get("to_comment");
174             TopHits topHits = childrenBucket.getAggregations().get("top_comments");
175             logger.info("total_hits={}", topHits.getHits().getTotalHits());
176             for (SearchHit searchHit : topHits.getHits()) {
177                 logger.info("hit= {} {} {}", searchHit.sortValues()[0], searchHit.getType(), searchHit.getId());
178             }
179         }
180 
181         Terms.Bucket categoryBucket = categoryTerms.getBucketByKey("a");
182         assertThat(categoryBucket.getKey(), equalTo("a"));
183         assertThat(categoryBucket.getDocCount(), equalTo(3l));
184 
185         Children childrenBucket = categoryBucket.getAggregations().get("to_comment");
186         assertThat(childrenBucket.getName(), equalTo("to_comment"));
187         assertThat(childrenBucket.getDocCount(), equalTo(2l));
188         TopHits topHits = childrenBucket.getAggregations().get("top_comments");
189         assertThat(topHits.getHits().totalHits(), equalTo(2l));
190         assertThat(topHits.getHits().getAt(0).sortValues()[0].toString(), equalTo("a"));
191         assertThat(topHits.getHits().getAt(0).getId(), equalTo("a"));
192         assertThat(topHits.getHits().getAt(0).getType(), equalTo("comment"));
193         assertThat(topHits.getHits().getAt(1).sortValues()[0].toString(), equalTo("c"));
194         assertThat(topHits.getHits().getAt(1).getId(), equalTo("c"));
195         assertThat(topHits.getHits().getAt(1).getType(), equalTo("comment"));
196 
197         categoryBucket = categoryTerms.getBucketByKey("b");
198         assertThat(categoryBucket.getKey(), equalTo("b"));
199         assertThat(categoryBucket.getDocCount(), equalTo(2l));
200 
201         childrenBucket = categoryBucket.getAggregations().get("to_comment");
202         assertThat(childrenBucket.getName(), equalTo("to_comment"));
203         assertThat(childrenBucket.getDocCount(), equalTo(1l));
204         topHits = childrenBucket.getAggregations().get("top_comments");
205         assertThat(topHits.getHits().totalHits(), equalTo(1l));
206         assertThat(topHits.getHits().getAt(0).getId(), equalTo("c"));
207         assertThat(topHits.getHits().getAt(0).getType(), equalTo("comment"));
208 
209         categoryBucket = categoryTerms.getBucketByKey("c");
210         assertThat(categoryBucket.getKey(), equalTo("c"));
211         assertThat(categoryBucket.getDocCount(), equalTo(2l));
212 
213         childrenBucket = categoryBucket.getAggregations().get("to_comment");

222     @Test
223     public void testWithDeletes() throws Exception {
224         String indexName = "xyz";
225         assertAcked(
226                 prepareCreate(indexName)
227                         .addMapping("parent")
228                         .addMapping("child", "_parent", "type=parent", "count", "type=long")
229         );
230 
231         List<IndexRequestBuilder> requests = new ArrayList<>();
232         requests.add(client().prepareIndex(indexName, "parent", "1").setSource("{}"));
233         requests.add(client().prepareIndex(indexName, "child", "0").setParent("1").setSource("count", 1));
234         requests.add(client().prepareIndex(indexName, "child", "1").setParent("1").setSource("count", 1));
235         requests.add(client().prepareIndex(indexName, "child", "2").setParent("1").setSource("count", 1));
236         requests.add(client().prepareIndex(indexName, "child", "3").setParent("1").setSource("count", 1));
237         indexRandom(true, requests);
238 
239         for (int i = 0; i < 10; i++) {
240             SearchResponse searchResponse = client().prepareSearch(indexName)
241                     .addAggregation(children("children").childType("child").subAggregation(sum("counts").field("count")))
242                     .get();
243 
244             assertNoFailures(searchResponse);
245             Children children = searchResponse.getAggregations().get("children");
246             assertThat(children.getDocCount(), equalTo(4l));
247 
248             Sum count = children.getAggregations().get("counts");
249             assertThat(count.getValue(), equalTo(4.));
250 
251             String idToUpdate = Integer.toString(randomInt(3));
252             UpdateResponse updateResponse = client().prepareUpdate(indexName, "child", idToUpdate)
253                     .setParent("1")
254                     .setDoc("count", 1)
255                     .get();
256             assertThat(updateResponse.getVersion(), greaterThan(1l));
257             refresh();
258         }
259     }

334     @Test
335     public void testHierarchicalChildrenAggs() {
336         String indexName = "geo";
337         String grandParentType = "continent";
338         String parentType = "country";
339         String childType = "city";
340         assertAcked(
341                 prepareCreate(indexName)
342                         .setSettings(ImmutableSettings.builder()
343                                 .put(IndexMetaData.SETTING_NUMBER_OF_SHARDS, 1)
344                                 .put(IndexMetaData.SETTING_NUMBER_OF_REPLICAS, 0)
345                         )
346                         .addMapping(grandParentType)
347                         .addMapping(parentType, "_parent", "type=" + grandParentType)
348                         .addMapping(childType, "_parent", "type=" + parentType)
349         );
350 
351         client().prepareIndex(indexName, grandParentType, "1").setSource("name", "europe").get();
352         client().prepareIndex(indexName, parentType, "2").setParent("1").setSource("name", "belgium").get();
353         client().prepareIndex(indexName, childType, "3").setParent("2").setRouting("1").setSource("name", "brussels").get();
354         refresh();
355 
356         SearchResponse response = client().prepareSearch(indexName)
357                 .setQuery(matchQuery("name", "europe"))
358                 .addAggregation(
359                         children(parentType).childType(parentType).subAggregation(
360                                 children(childType).childType(childType).subAggregation(
361                                         terms("name").field("name")
362                                 )
363                         )
364                 )
365                 .get();
366         assertNoFailures(response);
367         assertHitCount(response, 1);
368 
369         Children children = response.getAggregations().get(parentType);
370         assertThat(children.getName(), equalTo(parentType));
371         assertThat(children.getDocCount(), equalTo(1l));
372         children = children.getAggregations().get(childType);
373         assertThat(children.getName(), equalTo(childType));
374         assertThat(children.getDocCount(), equalTo(1l));
375         Terms terms = children.getAggregations().get("name");
376         assertThat(terms.getBuckets().size(), equalTo(1));
377         assertThat(terms.getBuckets().get(0).getKey().toString(), equalTo("brussels"));
378         assertThat(terms.getBuckets().get(0).getDocCount(), equalTo(1l));
379     }
