37  /**
38   * This is a Jaxen based XPathRule query.
39   */
40  public class JaxenXPathRuleQuery extends AbstractXPathRuleQuery {
41  
42      private static final Logger LOG = Logger.getLogger(JaxenXPathRuleQuery.class.getName());
43  
44      private enum InitializationStatus {
45          NONE, PARTIAL, FULL
46      }
47  
48      // Mapping from Node name to applicable XPath queries
49      private InitializationStatus initializationStatus = InitializationStatus.NONE;
50      private Map<String, List<XPath>> nodeNameToXPaths;
51  
52      private static final String AST_ROOT = "_AST_ROOT_";
53  
54      @Override
55      public boolean isSupportedVersion(String version) {
56          return XPATH_1_0.equals(version);
57      }
58  
59      @Override
60      public List<Node> evaluate(final Node node, final RuleContext data) {
61          final List<Node> results = new ArrayList<>();
62  
63          try {
64              initializeExpressionIfStatusIsNoneOrPartial(data.getLanguageVersion().getLanguageVersionHandler().getXPathHandler().getNavigator());
65  
66              List<XPath> xPaths = getXPathsForNodeOrDefault(node.getXPathNodeName());
67              for (XPath xpath : xPaths) {
68                  @SuppressWarnings("unchecked")
69                  final List<Node> matchedNodes = xpath.selectNodes(node);
70                  results.addAll(matchedNodes);
71              }
72          } catch (final JaxenException e) {
73              throw new RuntimeException(e);
74          }
75          return results;
76      }
77  
78      /**
79       * Get the XPath queries associated with the node name. If there are none, the XPath queries for the {@link #AST_ROOT}
80       * are obtained.
81       *
82       * @param nodeName the id of the node
83       * @return the list of XPath queries that match the node name
84       */
85      private List<XPath> getXPathsForNodeOrDefault(final String nodeName) {
86          List<XPath> xPaths = nodeNameToXPaths.get(nodeName);
87          if (xPaths == null) {
88              xPaths = nodeNameToXPaths.get(AST_ROOT);
89          }
90          return xPaths;
91      }
92  
93      @Override
94      public List<String> getRuleChainVisits() {
95          try {
96              // No Navigator available in this context
97              initializeExpressionIfStatusIsNoneOrPartial(null);
98              return super.getRuleChainVisits();
99          } catch (final JaxenException ex) {
100             throw new RuntimeException(ex);
101         }
102     }
103 
104     /**
105      *
106      * @param navigator the navigator which is required to be non-null if the {@link #initializationStatus} is PARTIAL.
107      * @throws JaxenException
108      */
109     @SuppressWarnings("unchecked")
110     private void initializeExpressionIfStatusIsNoneOrPartial(final Navigator navigator) throws JaxenException {
111         if (initializationStatus == InitializationStatus.FULL) {
112             return;
113         }
114         if (initializationStatus == InitializationStatus.PARTIAL && navigator == null) {
115             LOG.severe("XPathRule is not initialized because no navigator was provided. "
116                     + "Please make sure to implement getXPathHandler in the handler of the language. "
117                     + "See also AbstractLanguageVersionHandler.");
118             return;
119         }
120         initializeXPathExpression(navigator);
121     }
122 
123     private void initializeXPathExpression(final Navigator navigator) throws JaxenException {
124         /*
125         Attempt to use the RuleChain with this XPath query.
126 
127         To do so, the queries should generally look like //TypeA or //TypeA | //TypeB. We will look at the parsed XPath
128         AST using the Jaxen APIs to make this determination.
129 
130         If the query is not exactly what we are looking for, do not use the
131         RuleChain.
132         */
133         nodeNameToXPaths = new HashMap<>();
134 
135         final BaseXPath originalXPath = createXPath(xpath, navigator);
136         addQueryToNode(originalXPath, AST_ROOT);
137 
138         boolean useRuleChain = true;
139         final Deque<Expr> pending = new ArrayDeque<>();
140         pending.push(originalXPath.getRootExpr());
141         while (!pending.isEmpty()) {
142             final Expr node = pending.pop();
143 
144             // Need to prove we can handle this part of the query
145             boolean valid = false;
146 
147             // Must be a LocationPath... that is something like //Type
148             if (node instanceof LocationPath) {
149                 final LocationPath locationPath = (LocationPath) node;
150                 if (locationPath.isAbsolute()) {
151                     // Should be at least two steps
152                     @SuppressWarnings("unchecked")
153                     final List<Step> steps = locationPath.getSteps();
154 
155                     if (steps.size() >= 2) {
156                         final Step step1 = steps.get(0);
157                         final Step step2 = steps.get(1);
158                         // First step should be an AllNodeStep using the
159                         // descendant or self axis
160                         if (step1 instanceof AllNodeStep
161                                 && step1.getAxis() == Axis.DESCENDANT_OR_SELF) {
162                             // Second step should be a NameStep using the child
163                             // axis.
164                             if (step2 instanceof NameStep && step2.getAxis() == Axis.CHILD) {
165                                 // Construct a new expression that is
166                                 // appropriate for RuleChain use
167                                 final XPathFactory xpathFactory = new DefaultXPathFactory();
168 
169                                 // Instead of an absolute location path, we'll
170                                 // be using a relative path
171                                 final LocationPath relativeLocationPath = xpathFactory.createRelativeLocationPath();
172                                 // The first step will be along the self axis
173                                 final Step allNodeStep = xpathFactory.createAllNodeStep(Axis.SELF);
174                                 // Retain all predicates from the original name
175                                 // step
176                                 @SuppressWarnings("unchecked")
177                                 final List<Predicate> predicates = step2.getPredicates();
178 
179                                 for (Predicate predicate : predicates) {
180                                     allNodeStep.addPredicate(predicate);
181                                 }
182                                 relativeLocationPath.addStep(allNodeStep);
183 
184                                 // Retain the remaining steps from the original
185                                 // location path
186                                 for (int i = 2; i < steps.size(); i++) {
187                                     relativeLocationPath.addStep(steps.get(i));
188                                 }
189 
190                                 final BaseXPath xpath = createXPath(relativeLocationPath.getText(), navigator);
191                                 addQueryToNode(xpath, ((NameStep) step2).getLocalName());
192                                 valid = true;
193                             }
194                         }
195                     }
196                 }
197             } else if (node instanceof UnionExpr) { // Or a UnionExpr, that is
198                 // something like //TypeA |
199                 // //TypeB
200                 UnionExpr unionExpr = (UnionExpr) node;
201                 pending.push(unionExpr.getLHS());
202                 pending.push(unionExpr.getRHS());
203                 valid = true;
204             }
205             if (!valid) {
206                 useRuleChain = false;
207                 break;
208             }
209         }
210 
211         if (useRuleChain) {
212             // Use the RuleChain for all the nodes extracted from the xpath
213             // queries
214             super.ruleChainVisits.addAll(nodeNameToXPaths.keySet());
215         } else {
216             // Use original XPath if we cannot use the RuleChain
217             nodeNameToXPaths.clear();
218             addQueryToNode(originalXPath, AST_ROOT);
219             if (LOG.isLoggable(Level.FINE)) {
220                 LOG.log(Level.FINE, "Unable to use RuleChain for XPath: " + xpath);
221             }
222         }
223 
224         if (navigator == null) {
225             this.initializationStatus = InitializationStatus.PARTIAL;
226             // Clear the node data, because we did not have a Navigator
227             nodeNameToXPaths = null;
228         } else {
229             this.initializationStatus = InitializationStatus.FULL;
230         }
231     }
232 
233     /**
234      * Relates an XPath query to a node by adding the query to the {@link #nodeNameToXPaths}.
235      *
236      * @param xPath    the query to do over a node
237      * @param nodeName the node on which to do the query
238      */
239     private void addQueryToNode(final XPath xPath, final String nodeName) {
240         List<XPath> xPathsForNode = nodeNameToXPaths.get(nodeName);
241         if (xPathsForNode == null) {
242             xPathsForNode = new ArrayList<>();
243             nodeNameToXPaths.put(nodeName, xPathsForNode);
244         }
245         xPathsForNode.add(xPath);
246     }
247 
248     private BaseXPath createXPath(final String xpathQueryString, final Navigator navigator) throws JaxenException {
249         final BaseXPath xpath = new BaseXPath(xpathQueryString, navigator);
250 
251         if (properties.size() > 1) {
252             final SimpleVariableContext vc = new SimpleVariableContext();
253             for (Entry<PropertyDescriptor<?>, Object> e : properties.entrySet()) {
254                 final String propName = e.getKey().name();
255                 if (!"xpath".equals(propName)) {
256                     final Object value = e.getValue();
257                     vc.setVariableValue(propName, value != null ? value.toString() : null);
258                 }
259             }
260             xpath.setVariableContext(vc);
261         }
262         return xpath;
263     }
