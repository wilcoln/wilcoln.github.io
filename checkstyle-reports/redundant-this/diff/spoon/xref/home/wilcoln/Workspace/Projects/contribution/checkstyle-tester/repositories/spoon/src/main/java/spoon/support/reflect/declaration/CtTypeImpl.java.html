89  	/** Adds a type member.
90  	 * If it has a position, adds it at the right place according to the position (sourceStart).
91  	 * If it is implicit, adds it at the beginning.
92  	 * Otherwise, adds it at the end.
93  	 */
94  	@Override
95  	public <C extends CtType<T>> C addTypeMember(CtTypeMember member) {
96  		if (member == null) {
97  			return (C) this;
98  		}
99  		Comparator c = new CtLineElementComparator();
100 
101 		if (member.isImplicit()) {
102 			return addTypeMemberAt(0, member);
103 		}
104 
105 		// by default, inserting at the end
106 		int insertionPosition = typeMembers.size();
107 
108 		// we search for an insertion position only if this one has one position
109 		if (member.getPosition().isValidPosition()) {
110 			for (int i = typeMembers.size() - 1; i >= 0; i--) {
111 				CtTypeMember m = this.typeMembers.get(i);
112 
113 				if (m.isImplicit() || (m.getPosition().isValidPosition() && c.compare(member, m) > 0)) {
114 					break;
115 				}
116 				insertionPosition--;
117 			}
118 		}
119 		return addTypeMemberAt(insertionPosition, member);
120 	}

122 	@Override
123 	public <C extends CtType<T>> C addTypeMemberAt(int position, CtTypeMember member) {
124 		if (member == null) {
125 			return (C) this;
126 		}
127 		if (this.typeMembers == CtElementImpl.<CtTypeMember>emptyList()) {
128 			this.typeMembers = new ArrayList<>();
129 		}
130 		if (!this.typeMembers.stream().anyMatch(m -> m == member)) {
131 			member.setParent(this);
132 			CtRole role = CtRole.TYPE_MEMBER.getMatchingSubRoleFor(member);
133 			getFactory().getEnvironment().getModelChangeListener().onListAdd(this, role, this.typeMembers, position, member);
134 			if (position < typeMembers.size()) {
135 				this.typeMembers.add(position, member);
136 			} else {
137 				this.typeMembers.add(member);
138 			}
139 		}
140 		return (C) this;
141 	}

143 	@Override
144 	public boolean removeTypeMember(CtTypeMember member) {
145 		CtRole role = CtRole.TYPE_MEMBER.getMatchingSubRoleFor(member);
146 		if (typeMembers.size() == 1) {
147 			if (typeMembers.contains(member)) {
148 				getFactory().getEnvironment().getModelChangeListener().onListDelete(this, role, this.typeMembers, this.typeMembers.indexOf(member), member);
149 				typeMembers = emptyList();
150 				return true;
151 			} else {
152 				return false;
153 			}
154 		}
155 		if (typeMembers.contains(member)) {
156 			getFactory().getEnvironment().getModelChangeListener().onListDelete(this, role, this.typeMembers, this.typeMembers.indexOf(member), member);
157 			return typeMembers.remove(member);
158 		}
159 		return false;
160 	}

164 		for (CtTypeMember typeMember : new ArrayList<>(typeMembers)) {
165 			removeTypeMember(typeMember);
166 		}
167 		if (members == null || members.isEmpty()) {
168 			this.typeMembers = emptyList();
169 			return (C) this;
170 		}
171 		typeMembers.clear();
172 		for (CtTypeMember typeMember : members) {
173 			addTypeMember(typeMember);
174 		}
175 		return (C) this;
176 	}

295 	/**
296 	 * Return the package reference for the corresponding type reference. For
297 	 * inner type, return the package reference of the top-most enclosing type.
298 	 * This helper method is meant to deal with package references that are
299 	 * <code>null</code> for inner types.
300 	 *
301 	 * @param tref the type reference
302 	 * @return the corresponding package reference
303 	 * @see CtTypeReference#getPackage()
304 	 * @since 4.0
305 	 */
306 	private static CtPackageReference getPackageReference(CtTypeReference<?> tref) {
307 		CtPackageReference pref = tref.getPackage();
308 		while (pref == null) {
309 			tref = tref.getDeclaringType();
310 			pref = tref.getPackage();
311 		}
312 		return pref;
313 	}

332 		CtType<?> top = this;

345 	public <N extends CtType<?>> N getNestedType(final String name) {
346 		class NestedTypeScanner extends EarlyTerminatingScanner<CtType<?>> {
347 
348 			private boolean checkType(CtType<?> type) {
349 				if (type.getSimpleName().equals(name) && CtTypeImpl.this.equals(type.getDeclaringType())) {
350 					setResult(type);
351 					terminate();
352 					return true;
353 				}
354 				return false;
355 			}
356 
357 			@Override
358 			public <U> void visitCtClass(spoon.reflect.declaration.CtClass<U> ctClass) {
359 				if (!checkType(ctClass)) {
360 					final List<CtTypeMember> typeMembers = new ArrayList<>();
361 					for (CtTypeMember typeMember : ctClass.getTypeMembers()) {
362 						if (typeMember instanceof CtType || typeMember instanceof CtConstructor || typeMember instanceof CtMethod) {
363 							typeMembers.add(typeMember);
364 						}
365 					}
366 					scan(typeMembers);
367 				}
368 			}
369 
370 			@Override
371 			public <U> void visitCtInterface(spoon.reflect.declaration.CtInterface<U> intrface) {
372 				if (!checkType(intrface)) {
373 					final List<CtTypeMember> typeMembers = new ArrayList<>();
374 					for (CtTypeMember typeMember : intrface.getTypeMembers()) {
375 						if (typeMember instanceof CtType || typeMember instanceof CtMethod) {
376 							typeMembers.add(typeMember);
377 						}
378 					}
379 					scan(typeMembers);
380 				}
381 			}
382 
383 			@Override
384 			public <U extends java.lang.Enum<?>> void visitCtEnum(spoon.reflect.declaration.CtEnum<U> ctEnum) {
385 				if (!checkType(ctEnum)) {
386 					final List<CtTypeMember> typeMembers = new ArrayList<>();
387 					for (CtTypeMember typeMember : ctEnum.getTypeMembers()) {
388 						if (typeMember instanceof CtType || typeMember instanceof CtConstructor || typeMember instanceof CtMethod) {
389 							typeMembers.add(typeMember);
390 						}
391 					}
392 					scan(typeMembers);
393 				}
394 			}
395 
396 			@Override
397 			public <A extends Annotation> void visitCtAnnotationType(CtAnnotationType<A> annotationType) {
398 				if (!checkType(annotationType)) {
399 					scan(annotationType.getNestedTypes());
400 				}
401 			}
402 		}
403 		NestedTypeScanner scanner = new NestedTypeScanner();
404 		scanner.scan(this);
405 		return (N) scanner.getResult();
406 	}

579 	public <M, C extends CtType<T>> C addMethod(CtMethod<M> method) {
580 		if (method != null) {
581 			for (CtTypeMember typeMember : new ArrayList<>(typeMembers)) {
582 				if (!(typeMember instanceof CtMethod)) {
583 					continue;
584 				}
585 				CtMethod<?> m = (CtMethod<?>) typeMember;
586 				if (m.getSignature().equals(method.getSignature())) {
587 					// replace old method by new one (based on signature and not equality)
588 					// we have to do it by hand
589 					removeTypeMember(m);
590 				}
591 			}
592 		}
593 		return addTypeMember(method);
594 	}

601 	@Override
602 	public <S, C extends CtType<T>> C addSuperInterface(CtTypeReference<S> interfac) {
603 		if (interfac == null) {
604 			return (C) this;
605 		}
606 		if (interfaces == CtElementImpl.<CtTypeReference<?>>emptySet()) {
607 			interfaces = new QualifiedNameBasedSortedSet<>();
608 		}
609 		interfac.setParent(this);
610 		getFactory().getEnvironment().getModelChangeListener().onSetAdd(this, CtRole.INTERFACE, this.interfaces, interfac);
611 		interfaces.add(interfac);
612 		return (C) this;
613 	}

615 	@Override
616 	public <S> boolean removeSuperInterface(CtTypeReference<S> interfac) {
617 		getFactory().getEnvironment().getModelChangeListener().onSetDelete(this, CtRole.INTERFACE, interfaces, interfac);
618 		if (interfaces == CtElementImpl.<CtTypeReference<?>>emptySet()) {
619 			return false;
620 		} else if (interfaces.size() == 1) {
621 			if (interfaces.contains(interfac)) {
622 				interfaces = CtElementImpl.<CtTypeReference<?>>emptySet();
623 				return true;
624 			} else {
625 				return false;
626 			}
627 		} else {
628 			// contains() not needed. see comment in removeMethod()
629 			return interfaces.contains(interfac) && interfaces.remove(interfac);
630 		}
631 	}

679 	@SuppressWarnings("unchecked")
680 	public <R> CtMethod<R> getMethod(CtTypeReference<R> returnType, String name, CtTypeReference<?>... parameterTypes) {
681 		for (CtTypeMember typeMember : typeMembers) {
682 			if (!(typeMember instanceof CtMethod)) {
683 				continue;
684 			}
685 			CtMethod<R> m = (CtMethod<R>) typeMember;
686 			if (m.getSimpleName().equals(name)) {
687 				if (!m.getType().equals(returnType)) {
688 					continue;
689 				}
690 				boolean cont = m.getParameters().size() == parameterTypes.length;
691 				for (int i = 0; cont && (i < m.getParameters().size()) && (i < parameterTypes.length); i++) {
692 					if (!m.getParameters().get(i).getType().getQualifiedName().equals(parameterTypes[i].getQualifiedName())) {
693 						cont = false;
694 					}
695 				}
696 				if (cont) {
697 					return m;
698 				}
699 			}
700 		}
701 		return null;
702 	}

704 	@Override
705 	@SuppressWarnings("unchecked")
706 	public <R> CtMethod<R> getMethod(String name, CtTypeReference<?>... parameterTypes) {
707 		if (name == null) {
708 			return null;
709 		}
710 
711 		for (CtMethod<?> candidate : getMethodsByName(name)) {
712 			if (hasSameParameters(candidate, parameterTypes)) {
713 				return (CtMethod<R>) candidate;
714 			}
715 		}
716 		return null;
717 	}

719 	protected boolean hasSameParameters(CtExecutable<?> candidate, CtTypeReference<?>... parameterTypes) {
720 		if (candidate.getParameters().size() != parameterTypes.length) {
721 			return false;
722 		}
723 		for (int i = 0; (i < candidate.getParameters().size()) && (i < parameterTypes.length); i++) {
724 			final CtTypeReference<?> ctParameterType = candidate.getParameters().get(i).getType();
725 			final CtTypeReference<?> parameterType = parameterTypes[i];
726 			if (parameterType instanceof CtArrayTypeReference) {
727 				if (ctParameterType instanceof CtArrayTypeReference) {
728 					if (!isSameParameter(candidate, ((CtArrayTypeReference) ctParameterType).getComponentType(), ((CtArrayTypeReference) parameterType).getComponentType())) {
729 						return false;
730 					} else {
731 						if (((CtArrayTypeReference) ctParameterType).getDimensionCount() != ((CtArrayTypeReference) parameterType).getDimensionCount()) {
732 							return false;
733 						}
734 					}
735 				} else {
736 					return false;
737 				}
738 			} else if (!isSameParameter(candidate, ctParameterType, parameterType)) {
739 				return false;
740 			}
741 		}
742 		return true;
743 	}

746 		return ctParameterType.getTypeErasure().getQualifiedName().equals(expectedType.getTypeErasure().getQualifiedName());
747 	}

792 	@Override
793 	public boolean hasMethod(CtMethod<?> method) {
794 		if (method == null) {
795 			return false;
796 		}
797 
798 		final String over = method.getSignature();
799 		for (CtMethod<?> m : getMethods()) {
800 			if (m.getSignature().equals(over)) {
801 				return true;
802 			}
803 		}
804 
805 		// Checking whether a super class has the method.
806 		final CtTypeReference<?> superCl = getSuperclass();
807 		try {
808 			if (superCl != null && superCl.getTypeDeclaration().hasMethod(method)) {
809 				return true;
810 			}
811 		} catch (SpoonException ex) {
812 			// No matter, trying something else.
813 		}
814 
815 		// Finally, checking whether an interface has the method.
816 		for (CtTypeReference<?> interf : getSuperInterfaces()) {
817 			try {
818 				if (interf.getTypeDeclaration().hasMethod(method)) {
819 					return true;
820 				}
821 			} catch (SpoonException ex) {
822 				// No matter, trying something else.
823 			}
824 		}
825 
826 		return false;
827 	}

831 		if (isTopLevel()) {
832 			if (getPackage() != null && !getPackage().isUnnamedPackage()) {
833 				return getPackage().getQualifiedName() + "." + getSimpleName();
834 			} else {
835 				return getSimpleName();
836 			}
837 		} else if (getDeclaringType() != null) {
838 			return getDeclaringType().getQualifiedName() + INNERTTYPE_SEPARATOR + getSimpleName();
839 		} else {
840 			return getSimpleName();
841 		}

908 	@Override
909 	public Set<CtMethod<?>> getAllMethods() {
910 		final Set<CtMethod<?>> l = new HashSet<>();
911 		final ClassTypingContext ctc = new ClassTypingContext(this);
912 		map(new AllTypeMembersFunction(CtMethod.class)).forEach(new CtConsumer<CtMethod<?>>() {
913 			@Override
914 			public void accept(CtMethod<?> currentMethod) {
915 				for (CtMethod<?> alreadyVisitedMethod : l) {
916 					if (ctc.isSameSignature(currentMethod, alreadyVisitedMethod)) {
917 						return;
918 					}
919 				}
920 
921 				l.add(currentMethod);
922 			}
923 		});
924 		return Collections.unmodifiableSet(l);
925 	}
