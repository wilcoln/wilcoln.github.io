12  /**
13   * Used to set all data in the cloned element.

17  class CloneBuilderTemplate extends CtInheritanceScanner {
18  	public void copy(spoon.reflect.declaration.CtElement element, spoon.reflect.declaration.CtElement other) {
19  		this.setOther(other);

21  	}
22  
23  	public static <T extends CtElement> T build(CloneBuilderTemplate builder, CtElement element, CtElement other) {
24  		builder.setOther(other);
25  		builder.scan(element);
26  		return (T) builder.other;
27  	}
28  
29  	private CtElement other;
30  
31  	public void setOther(CtElement other) {
