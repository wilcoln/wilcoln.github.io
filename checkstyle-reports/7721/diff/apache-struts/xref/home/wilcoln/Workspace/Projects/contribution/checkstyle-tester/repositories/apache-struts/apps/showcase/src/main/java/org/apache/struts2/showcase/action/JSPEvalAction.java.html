19  package org.apache.struts2.showcase.action;
20  
21  import com.opensymphony.xwork2.Action;
22  import com.opensymphony.xwork2.interceptor.annotations.After;
23  import org.apache.struts2.ServletActionContext;
24  
25  import java.io.*;
26  import java.net.URL;
27  
28  /**
29   * Will only work on containers that unzip war files
30   */
31  public class JSPEvalAction extends ExampleAction {
32  	private String jsp;
33  	private final static String FILE = "/interactive/demo.jsp";
34  
35  	public String execute() throws IOException {
36  		if (jsp != null) {
37  			//write it to file
38  			URL url = ServletActionContext.getServletContext().getResource(FILE);
39  			BufferedWriter writer = new BufferedWriter(new FileWriter(new File(url
40  					.getFile())));
41  			try {
42  				//directive
43  				writer.write("<%@ taglib prefix=\"s\" uri=\"/struts-tags\" %>");
44  				writer.write(jsp);
45  			} finally {
46  				if (writer != null)
47  					writer.close();
48  			}
49  		}
50  		return Action.SUCCESS;
51  	}
52  
53  	@After
54  	public void cleanUp() throws IOException {
55  		URL url = ServletActionContext.getServletContext().getResource(FILE);
56  		FileOutputStream out = new FileOutputStream(new File(url.getFile()));
57  		try {
58  			out.getChannel().truncate(0);
59  		} finally {
60  			if (out != null)
61  				out.close();
62  		}
63  	}
64  
65  	public void setJsp(String jsp) {
66  		this.jsp = jsp;
67  	}
68  
69  }
