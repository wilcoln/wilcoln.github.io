6   package spoon.compiler.builder;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  public abstract class Options<T extends Options<T>> {
12  	static final String COMMA_DELIMITER = ",";
13  	final List<String> args = new ArrayList<>();
14  	final T myself;
15  
16  	public Options(Class<?> type) {
17  		this.myself = (T) type.cast(this);
18  	}
19  
20  	public String[] build() {
21  		return args.toArray(new String[0]);
22  	}
23  
24  	protected String join(String delimiter, String[] classpath) {
25  		if (classpath == null || classpath.length == 0) {
26  			return "";
27  		}
28  		final StringBuilder builder = new StringBuilder();
29  		for (String entry : classpath) {
30  			builder.append(entry);
31  			builder.append(delimiter);
32  		}
33  		return builder.toString();
34  	}
35  }
