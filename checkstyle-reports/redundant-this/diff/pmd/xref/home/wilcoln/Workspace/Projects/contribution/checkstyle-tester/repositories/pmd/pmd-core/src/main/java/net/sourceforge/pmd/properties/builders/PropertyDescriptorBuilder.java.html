5   package net.sourceforge.pmd.properties.builders;
6   
7   import org.apache.commons.lang3.StringUtils;
8   
9   import net.sourceforge.pmd.properties.PropertyBuilder;
10  import net.sourceforge.pmd.properties.PropertyDescriptor;
11  import net.sourceforge.pmd.properties.PropertyFactory;
12  
13  
14  /**
15   * Base class for property builders.
16   *
17   * @param <E> Value type of the built descriptor
18   * @param <T> Concrete type of this builder instance. Removes code duplication at the expense of a few unchecked casts.
19   *            Everything goes well if this parameter's value is correctly set.
20   *
21   * @deprecated From 7.0.0 on, the only supported way to build properties will be through {@link PropertyFactory}.
22   *             This class hierarchy is replaced by the newer {@link PropertyBuilder}.
23   * @author Cl√©ment Fournier
24   * @since 6.0.0
25   */
26  @Deprecated
27  public abstract class PropertyDescriptorBuilder<E, T extends PropertyDescriptorBuilder<E, T>> {
28  
29      protected String name;
30      protected String description;
31      protected float uiOrder = 0f;
32      protected boolean isDefinedInXML = false;
33  
34  
35      protected PropertyDescriptorBuilder(String name) {
36          if (StringUtils.isBlank(name)) {
37              throw new IllegalArgumentException("Name must be provided");
38          }
39          this.name = name;
40      }
41      
42  
43      /**
44       * Specify the description of the property.
45       *
46       * @param desc The description
47       *
48       * @return The same builder
49       */
50      @SuppressWarnings("unchecked")
51      public T desc(String desc) {
52          if (StringUtils.isBlank(desc)) {
53              throw new IllegalArgumentException("Description must be provided");
54          }
55          this.description = desc;
56          return (T) this;
57      }
58  
59  
60      /**
61       * Specify the UI order of the property.
62       *
63       * @param f The UI order
64       *
65       * @return The same builder
66       * @deprecated See {@link PropertyDescriptor#uiOrder()}
67       */
68      @SuppressWarnings("unchecked")
69      @Deprecated
70      public T uiOrder(float f) {
71          this.uiOrder = f;
72          return (T) this;
73      }
74  
75  
76      /**
77       * Builds the descriptor and returns it.
78       *
79       * @return The built descriptor
80       * @throws IllegalArgumentException if parameters are incorrect
81       */
82      public abstract PropertyDescriptor<E> build();
83  
84  
85      /**
86       * Returns the name of the property to be built.
87       */
88      public String getName() {
89          return name;
90      }
91  
92  }
