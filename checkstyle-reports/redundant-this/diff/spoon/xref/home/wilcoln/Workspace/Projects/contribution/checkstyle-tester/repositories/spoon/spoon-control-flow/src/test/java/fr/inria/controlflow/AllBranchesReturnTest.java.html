130 	@Test
131 	public void nestedIfSomeNotReturning() throws Exception {
132 		testSegment(new AbstractProcessor<CtIf>() {
133 			@Override
134 			public void process(CtIf element) {
135 				CtMethod m = element.getParent().getParent(CtMethod.class);
136 				if (m != null && m.getSimpleName().equals("nestedIfSomeNotReturning"))
137 					if (element.getCondition().toString().contains("b < 1")) {
138 						AllBranchesReturn alg = new AllBranchesReturn();
139 						assertFalse(alg.execute(element));
140 					}
141 			}
142 		});
143 	}
