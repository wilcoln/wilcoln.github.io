129 	/**
130 	 * Builds tree of {@link SourcePosition}s of `element` and all it's children
131 	 * @param element the root element of the tree
132 	 */
133 	public void addTreeOfSourceFragmentsOfElement(CtElement element) {
134 		Deque<ElementSourceFragment> parents = new ArrayDeque<>();
135 		parents.push(this);
136 		/*
137 		 * scan all children of `element` and build tree of SourceFragments
138 		 * Note: we cannot skip implicit elements,
139 		 * because CtBlock can be implicit but contains non implicit elements, which has to be processed.
140 		 */
141 		new EarlyTerminatingScanner<Void>() {
142 			public <T> void visitCtFieldReference(final CtFieldReference<T> reference) {
143 				// bug 3133: we must must not visit the type of a field reference
144 				enter(reference);
145 				scan(CtRole.DECLARING_TYPE, reference.getDeclaringType());
146 				scan(CtRole.ANNOTATION, reference.getAnnotations());
147 				exit(reference);
148 			}
149 
150 			@Override
151 			protected void enter(CtElement e) {
152 				if (e instanceof CtCompilationUnit) {
153 					return;
154 				}
155 				ElementSourceFragment newFragment = addChild(parents.peek(), scannedRole, e);
156 				if (newFragment != null) {
157 					parents.push(newFragment);
158 					if (e instanceof CtModifiable) {
159 						CtModifiable modifiable = (CtModifiable) e;
160 						Set<CtExtendedModifier> modifiers = modifiable.getExtendedModifiers();
161 						for (CtExtendedModifier ctExtendedModifier : modifiers) {
162 							addChild(newFragment, CtRole.MODIFIER, ctExtendedModifier);
163 						}
164 					}
165 				}
166 			}
167 			@Override
168 			protected void exit(CtElement e) {
169 				if (e instanceof CtCompilationUnit) {
170 					return;
171 				}
172 				ElementSourceFragment topFragment = parents.peek();
173 				if (topFragment != null && topFragment.getElement() == e) {
174 					parents.pop();
175 				}
176 			}
177 		}
178 		.setVisitCompilationUnitContent(true)
179 		.scan(element.getRoleInParent(), element);
180 	}
181 	/**
182 	 * @param parentFragment the parent {@link ElementSourceFragment}, which will receive {@link ElementSourceFragment} made for `otherElement`
183 	 * @param roleInParent the {@link CtRole} of `otherElement` in scope of element of `parentFragment`
184 	 * @param otherElement {@link SourcePositionHolder} whose {@link ElementSourceFragment} has to be added to `parentFragment`
185 	 * @return new {@link ElementSourceFragment} created for `otherElement` or null if `otherElement` has no source position or doesn't belong to the same compilation unit
186 	 */
187 	private ElementSourceFragment addChild(ElementSourceFragment parentFragment, CtRole roleInParent, SourcePositionHolder otherElement) {
188 		SourcePosition otherSourcePosition = otherElement.getPosition();
189 		if (otherSourcePosition instanceof SourcePositionImpl && !(otherSourcePosition.getCompilationUnit() instanceof NoSourcePosition.NullCompilationUnit)) {
190 			if (parentFragment.isFromSameSource(otherSourcePosition)) {
191 				ElementSourceFragment otherFragment = new ElementSourceFragment(otherElement, parentFragment.getRoleHandler(roleInParent, otherElement));
192 				//parent and child are from the same file. So we can connect their positions into one tree
193 				CMP cmp = parentFragment.compare(otherFragment);
194 				if (cmp == CMP.OTHER_IS_CHILD) {
195 					//child belongs under parent - OK
196 					parentFragment.addChild(otherFragment);
197 					return otherFragment;
198 				} else {
199 					if (cmp == CMP.OTHER_IS_AFTER || cmp == CMP.OTHER_IS_BEFORE) {
200 						if (otherElement instanceof CtComment) {
201 							/*
202 							 * comments of elements are sometime not included in source position of element.
203 							 * because comments are ignored tokens for java compiler, which computes start/end of elements
204 							 * Example:
205 							 *
206 							 * 		//a comment
207 							 * 		aStatement();
208 							 *
209 							 */
210 							if (otherFragment.getStart() == 0) {
211 								//it is CompilationUnit comment, which is before package and imports, so it doesn't belong to class
212 								//No problem. Simply add comment at correct position into SourceFragment tree, starting from ROOT
213 								addChild(otherFragment);
214 								return otherFragment;
215 							}
216 							//add this child into parent's source fragment and extend that parent source fragment
217 							parentFragment.addChild(otherFragment);
218 							return otherFragment;
219 						}
220 					}
221 					//the source position of child element is not included in source position of parent element
222 					//I (Pavel) am not sure how to handle it, so let's wait until it happens...
223 //						if (otherElement instanceof CtAnnotation<?>) {
224 //							/*
225 //							 * it can happen for annotations of type TYPE_USE and FIELD
226 //							 * In such case the annotation belongs to 2 elements
227 //							 * And one of them cannot have matching source position - OK
228 //							 */
229 //							return null;
230 //						}
231 					//something is wrong ...
232 					throw new SpoonException("The SourcePosition of elements are not consistent\nparentFragment: " + parentFragment + "\notherFragment: " + otherFragment);
233 				}
234 			} else {
235 				throw new SpoonException("SourcePosition from unexpected compilation unit: " + otherSourcePosition + " expected is: " + parentFragment.getSourcePosition());
236 			}
237 		}
238 		//do not connect that undefined source position
239 		return null;
240 	}

242 	private RoleHandler getRoleHandler(CtRole roleInParent, SourcePositionHolder otherElement) {
243 		SourcePositionHolder parent = element;
244 		if (parent == null) {
245 			if (otherElement instanceof CtElement) {
246 				parent = ((CtElement) otherElement).getParent();
247 			}
248 		}
249 		if (parent instanceof CtElement) {
250 			CtElement ele = (CtElement) parent;
251 			return RoleHandlerHelper.getRoleHandler(ele.getClass(), roleInParent);
252 		}
253 		return null;
254 	}

256 	/**
257 	 * adds `other` {@link ElementSourceFragment} into tree of {@link ElementSourceFragment}s represented by this root element
258 	 *
259 	 * @param other to be added {@link ElementSourceFragment}
260 	 * @return new root of the tree of the {@link ElementSourceFragment}s. It can be be this or `other`
261 	 */
262 	public ElementSourceFragment add(ElementSourceFragment other) {
263 		if (this == other) {
264 			throw new SpoonException("SourceFragment#add must not be called twice for the same SourceFragment");
265 			//optionally we might accept that and simply return this
266 		}
267 		CMP cmp = this.compare(other);
268 		switch (cmp) {
269 		case OTHER_IS_AFTER:
270 			//other is after this
271 			addNextSibling(other);
272 			return this;
273 		case OTHER_IS_BEFORE:
274 			//other is before this
275 			other.addNextSibling(this);
276 			return other;
277 		case OTHER_IS_CHILD:
278 			//other is child of this
279 			addChild(other);
280 			return this;
281 		case OTHER_IS_PARENT:
282 			//other is parent of this, merge this and all siblings of `this` as children and siblings of `other`
283 			other.merge(this);
284 			return other;
285 		}
286 		throw new SpoonException("Unexpected compare result: " + cmp);
287 	}

289 	private void merge(ElementSourceFragment tobeMerged) {
290 		while (tobeMerged != null) {
291 			ElementSourceFragment nextTobeMerged = tobeMerged.getNextSibling();
292 			//disconnect tobeMerged from nextSiblings before we add it. So it is added individually and not with wrong siblings too
293 			tobeMerged.nextSibling = null;
294 			add(tobeMerged);
295 			tobeMerged = nextTobeMerged;
296 		}
297 	}

299 	/**
300 	 * adds `fragment` as child fragment of this fragment. If child is located before or after this fragment,
301 	 * then start/end of this fragment is moved
302 	 * @param fragment to be add
303 	 */
304 	public void addChild(ElementSourceFragment fragment) {
305 		if (firstChild == null) {
306 			firstChild = fragment;
307 		} else {
308 			firstChild = firstChild.add(fragment);
309 		}
310 		if (fragment.getElement() instanceof CtElement) {
311 			CtElement fragmentEleParent = ((CtElement) fragment.getElement()).getParent();
312 			if (element != fragmentEleParent && !(element instanceof CtCompilationUnit) && fragmentEleParent.getPosition().isValidPosition()) {
313 				throw new SpoonException("Inconsistent child fragment " + fragment.getElement().getClass() + " has unexpected parent " + element.getClass());
314 			}
315 		}
316 	}

318 	private void addNextSibling(ElementSourceFragment sibling) {
319 		if (nextSibling == null) {
320 			nextSibling = sibling;
321 		} else {
322 			nextSibling = nextSibling.add(sibling);
323 		}
324 	}

326 	private ElementSourceFragment getLastSibling() {
327 		ElementSourceFragment lastSibling = this;
328 		while (lastSibling.nextSibling != null) {
329 			lastSibling = lastSibling.nextSibling;
330 		}
331 		return lastSibling;
332 	}

334 	private enum CMP {
335 		OTHER_IS_BEFORE,
336 		OTHER_IS_AFTER,
337 		OTHER_IS_CHILD,
338 		OTHER_IS_PARENT
339 	}
340 
341 	/**
342 	 * compares this and other
343 	 * @param other other {@link SourcePosition}
344 	 * @return CMP
345 	 * throws {@link SpoonException} if intervals overlap or start/end is negative
346 	 */
347 	private CMP compare(ElementSourceFragment other) {
348 		if (other == this) {
349 			throw new SpoonException("SourcePositionImpl#addNextSibling must not be called twice for the same SourcePosition");
350 		}
351 		if (getEnd() <= other.getStart()) {
352 			//other is after this
353 			return CMP.OTHER_IS_AFTER;
354 		}
355 		if (other.getEnd() <= getStart()) {
356 			//other is before this
357 			return CMP.OTHER_IS_BEFORE;
358 		}
359 		if (getStart() <= other.getStart() && getEnd() >= other.getEnd()) {
360 			//other is child of this
361 			return CMP.OTHER_IS_CHILD;
362 		}
363 		if (getStart() >= other.getStart() && getEnd() <= other.getEnd()) {
364 			//other is parent of this
365 			return CMP.OTHER_IS_PARENT;
366 		}
367 		//the fragments overlap - it is not allowed
368 		throw new SpoonException("Cannot compare this: [" + getStart() + ", " + getEnd() + "] with other: [\"" + other.getStart() + "\", \"" + other.getEnd() + "\"]");
369 	}

385 	/**
386 	 * Searches the tree of fragments for the {@link ElementSourceFragment} with expected `element`,
387 	 * which contains `start` and `end` source interval.
388 	 * It searches in siblings and children of this {@link ElementSourceFragment} recursively.
389 	 * @param element the {@link SourcePositionHolder} of fragment it is looking for or null for any element
390 	 * @param start the start offset of searched fragment
391 	 * @param end the offset of next character after the end of searched fragment
392 	 *
393 	 * @return {@link ElementSourceFragment} which represents the root of the CtElement whose sources are in interval [start, end]
394 	 */
395 	public ElementSourceFragment getSourceFragmentOf(SourcePositionHolder element, int start, int end) {
396 		int myEnd = getEnd();
397 		if (myEnd <= start) {
398 			//search in next sibling
399 			if (nextSibling == null) {
400 				return null;
401 			}
402 			return getRootFragmentOfElement(nextSibling.getSourceFragmentOf(element, start, end));
403 		}
404 		int myStart = getStart();
405 		if (myStart <= start) {
406 			if (myEnd >= end) {
407 				if (myStart == start && myEnd == end) {
408 					//we have found exact match
409 					if (element != null && getElement() != element) {
410 						if (firstChild == null) {
411 							throw new SpoonException("There is no source fragment for element " + element.getClass() + ". There is one for class " + getElement().getClass());
412 						}
413 						return firstChild.getSourceFragmentOf(element, start, end);
414 					}
415 					return this;
416 				}
417 				//it is the child
418 				if (firstChild == null) {
419 					if (element != null && getElement() != element) {
420 						throw new SpoonException("There is no source fragment for element " + element.getClass() + ". There is one for class " + getElement().getClass());
421 					}
422 					return this;
423 				}
424 				ElementSourceFragment child = getRootFragmentOfElement(firstChild.getSourceFragmentOf(element, start, end));
425 				if (child != null) {
426 					//all children are smaller then this element
427 					return child;
428 				}
429 				//so this fragment is last one which wraps whole element
430 				if (element != null && getElement() != element) {
431 					throw new SpoonException("There is no source fragment for element " + element.getClass() + ". There is one for class " + getElement().getClass());
432 				}
433 				return this;
434 			}
435 			//start - end overlaps over multiple fragments
436 			throw new SpoonException("Invalid start/end interval. It overlaps multiple fragments.");
437 		}
438 		return null;
439 	}

441 	private ElementSourceFragment getRootFragmentOfElement(ElementSourceFragment childFragment) {
442 		if (childFragment != null && getElement() != null && childFragment.getElement() == getElement()) {
443 			//child fragment and this fragment have same element. Return this fragment,
444 			//because we have to return root fragment of CtElement
445 			return this;
446 		}
447 		return childFragment;
448 	}

457 	/**
458 	 * Note: the List of children is flat. The child fragments of collections (parameters, type members, ...) are next to each other.
459 	 * @return list of child fragments of this {@link ElementSourceFragment}.
460 	 */
461 	public List<SourceFragment> getChildrenFragments() {
462 		if (element instanceof CtLiteral) {
463 			return Collections.singletonList(new TokenSourceFragment(getSourceCode(), TokenType.LITERAL));
464 		}
465 		List<SourceFragment> children = new ArrayList<>();
466 		int off = getStart();
467 		ElementSourceFragment child = getFirstChild();
468 		while (child != null) {
469 			forEachConstantFragment(off, child.getStart(), cf -> children.add(cf));
470 			children.add(child);
471 			off = child.getEnd();
472 			child = child.getNextSibling();
473 		}
474 		forEachConstantFragment(off, getEnd(), cf -> children.add(cf));
475 		return children;
476 	}

478 	/**
479 	 * Detects all child fragments of this {@link ElementSourceFragment}.
480 	 * Note: the List of children contains one {@link CollectionSourceFragment} for each collection of fragments (parameters, type members, ...).
481 	 * Note: the {@link CollectionSourceFragment} may contain a mix of fragments of different roles, when they overlap.
482 	 * For example this code contains mix of annotations and modifiers
483 	 * <code>public @Deprecated static @Ignored void method()</code>
484 	 * @return list of child fragments of this {@link ElementSourceFragment} where fragments,
485 	 * which belongs to the same collection are grouped into {@link CollectionSourceFragment}
486 	 */
487 	public List<SourceFragment> getGroupedChildrenFragments() {
488 		List<SourceFragment> flatChildren = getChildrenFragments();
489 		List<SourceFragment> result = new ArrayList<>();
490 		int i = 0;
491 		while (i < flatChildren.size()) {
492 			SourceFragment child = flatChildren.get(i);
493 			if (child instanceof TokenSourceFragment) {
494 				result.add(child);
495 				i++;
496 				continue;
497 			} else if (child instanceof ElementSourceFragment) {
498 				ElementSourceFragment esf = (ElementSourceFragment) child;
499 				ContainerKind kind = esf.getContainerKindInParent();
500 				if (kind == ContainerKind.SINGLE) {
501 					//it is root element or there is always only one child instance in parent
502 					result.add(child);
503 					i++;
504 					continue;
505 				}
506 				//there can be 0, 1 or more items of children of the same role
507 				//search for another element of the same role
508 				Set<CtRole> foundRoles = new HashSet<>();
509 				foundRoles.add(checkNotNull(esf.getRoleInParent()));
510 				List<SourceFragment> childrenInSameCollection = new ArrayList<>();
511 				//but first include prefix whitespace
512 				SourceFragment spaceChild = removeSuffixSpace(result);
513 				if (spaceChild != null) {
514 					childrenInSameCollection.add(spaceChild);
515 				}
516 				childrenInSameCollection.add(esf);
517 				int lastOfSameRole = findIndexOfLastChildTokenOfRoleHandler(flatChildren, i, esf.getRoleInParent());
518 				//search for other roles in that interval
519 				i++;
520 				while (i <= lastOfSameRole) {
521 					child = flatChildren.get(i);
522 					childrenInSameCollection.add(child);
523 					CtRole role = null;
524 					if (child instanceof ElementSourceFragment) {
525 						ElementSourceFragment esf2 = (ElementSourceFragment) child;
526 						role = esf2.getRoleInParent();
527 					}
528 					if (role != null && role != CtRole.COMMENT && foundRoles.add(role)) {
529 						//there is another role in same block, search for last one
530 						lastOfSameRole = Math.max(lastOfSameRole, findIndexOfLastChildTokenOfRoleHandler(flatChildren, i + 1, role));
531 					}
532 					i++;
533 				}
534 				//add suffix space
535 				/*
536 				if (i < flatChildren.size()) {
537 					SourceFragment nextChild = flatChildren.get(i);
538 					if (isSpaceFragment(nextChild)) {
539 						childrenInSameCollection.add(nextChild);
540 						i++;
541 					}
542 				}
543 				*/
544 				result.add(new CollectionSourceFragment(childrenInSameCollection));
545 			} else {
546 				throw new SpoonException("Unexpected SourceFragment of type " + child.getClass());
547 			}
548 		}
549 		return result;
550 	}

552 	private SourceFragment removeSuffixSpace(List<SourceFragment> list) {
553 		if (list.size() > 0) {
554 			SourceFragment lastChild = list.get(list.size() - 1);
555 			if (isSpaceFragment(lastChild)) {
556 				list.remove(list.size() - 1);
557 				return lastChild;
558 			}
559 		}
560 		return null;
561 	}

570 	private static int findIndexOfLastChildTokenOfRoleHandler(List<SourceFragment> childFragments, int start, CtRole role) {
571 		return findIndexOfPreviousFragment(childFragments, start,
572 				filter(ElementSourceFragment.class, fragment -> fragment.getRoleInParent() == role));
573 	}

659 	/**
660 	 * Calls `consumer` once for each constant {@link SourceFragment} found in source code between `start` and `end`
661 	 */
662 	private void forEachConstantFragment(int start, int end, Consumer<SourceFragment> consumer) {
663 		if (start == end) {
664 			return;
665 		}
666 		if (start > end) {
667 			throw new SpoonException("Inconsistent start/end. Start=" + start + " is greater then End=" + end);
668 		}
669 		String sourceCode = getOriginalSourceCode();
670 		StringBuilder buff = new StringBuilder();
671 		CharType lastType = null;
672 		int off = start;
673 		while (off < end) {
674 			char c = sourceCode.charAt(off);
675 			CharType type = CharType.fromChar(c);
676 			if (type != lastType) {
677 				if (lastType != null) {
678 					onCharSequence(lastType, buff, consumer);
679 					buff.setLength(0);
680 				}
681 				lastType = type;
682 			}
683 			buff.append(c);
684 			off++;
685 		}
686 		onCharSequence(lastType, buff, consumer);
687 	}
688 
689 	private void onCharSequence(CharType type, StringBuilder buff, Consumer<SourceFragment> consumer) {
690 		if (type == CharType.SPACE) {
691 			consumer.accept(new TokenSourceFragment(buff.toString(), TokenType.SPACE));
692 			return;
693 		}
694 		char[] str = new char[buff.length()];
695 		buff.getChars(0, buff.length(), str, 0);
696 		int off = 0;
697 		while (off < str.length) {
698 			//detect java identifier or keyword
699 			int lenOfIdentifier = detectJavaIdentifier(str, off);
700 			if (lenOfIdentifier > 0) {
701 				String identifier = new String(str, off, lenOfIdentifier);
702 				if (javaKeywords.contains(identifier)) {
703 					//it is a java keyword
704 					consumer.accept(new TokenSourceFragment(identifier, TokenType.KEYWORD));
705 				} else {
706 					//it is a java identifier
707 					consumer.accept(new TokenSourceFragment(identifier, TokenType.IDENTIFIER));
708 				}
709 				off += lenOfIdentifier;
710 				continue;
711 			}
712 			//detect longest match in matchers
713 			StringMatcher longestMatcher = null;
714 			for (StringMatcher strMatcher : matchers) {
715 				if (strMatcher.isMatch(str, off)) {
716 					longestMatcher = strMatcher.getLonger(longestMatcher);
717 				}
718 			}
719 			if (longestMatcher == null) {
720 				throw new SpoonException("Unexpected source text: " + buff.toString());
721 			}
722 			consumer.accept(new TokenSourceFragment(longestMatcher.toString(), longestMatcher.getType()));
723 			off += longestMatcher.getLength();
724 		}
725 	}

727 	/**
728 	 * @return number of characters in buff starting from start which are java identifier
729 	 */
730 	private int detectJavaIdentifier(char[] buff, int start) {
731 		int len = buff.length;
732 		int o = start;
733 		if (start <= len) {
734 			char c = buff[o];
735 			if (Character.isJavaIdentifierStart(c)) {
736 				o++;
737 				while (o < len) {
738 					c = buff[o];
739 					if (Character.isJavaIdentifierPart(c) == false) {
740 						break;
741 					}
742 					o++;
743 				}
744 			}
745 		}
746 		return o - start;
747 	}

754 		return null;

757 	private static final class StringMatcher {
758 		private final TokenType type;
759 		private final char[] chars;
760 
761 		private StringMatcher(final String str, TokenType type) {
762 			this.type = type;
763 			chars = str.toCharArray();
764 		}
765 
766 		public boolean isMatch(final char[] buffer, int pos) {
767 			final int len = chars.length;
768 			if (pos + len > buffer.length) {
769 				return false;
770 			}
771 			for (int i = 0; i < chars.length; i++, pos++) {
772 				if (chars[i] != buffer[pos]) {
773 					return false;
774 				}
775 			}
776 			return true;
777 		}
778 
779 		@Override
780 		public String toString() {
781 			return new String(chars);
782 		}
783 
784 		public int getLength() {
785 			return chars.length;
786 		}
787 
788 		public StringMatcher getLonger(StringMatcher m) {
789 			if (m != null && m.getLength() > getLength()) {
790 				return m;
791 			}
792 			return this;
793 		}
794 
795 		public TokenType getType() {
796 			return type;
797 		}
798 	}

818 	/**
819 	 * looks for next fragment whose {@link Predicate} `test` returns true
820 	 * @param start - the index of first to be checked fragment
821 	 * @return index of found fragment, or -1 if not found
822 	 */
823 	static int findIndexOfNextFragment(List<SourceFragment> fragments, int start, Predicate<SourceFragment> test) {
824 		while (start < fragments.size()) {
825 			SourceFragment fragment = fragments.get(start);
826 			if (test.test(fragment)) {
827 				return start;
828 			}
829 			start++;
830 		}
831 		return -1;
832 	}

834 	/**
835 	 * @param start the index of element with lower index which is checked and may be returned
836 	 * @param test a {@link Predicate}, which is evaluated for each item of `fragments` starting from last one and ending with item in index `start`
837 	 * @return index of found fragment, or -1 if not found
838 	 */
839 	static int findIndexOfPreviousFragment(List<SourceFragment> fragments, int start, Predicate<SourceFragment> test) {
840 		int i = fragments.size() - 1;
841 		while (i >= start) {
842 			if (test.test(fragments.get(i))) {
843 				return i;
844 			}
845 			i--;
846 		}
847 		return -1;
848 	}

850 	/**
851 	 * @param predicate the {@link Predicate}, which has to be checkd for each item of {@link CollectionSourceFragment}
852 	 * @return {@link Predicate} which calls `predicate` for each item of {@link CollectionSourceFragment}
853 	 * Returned {@link Predicate} returns true only if `predicate` returns true on at least one item
854 	 */
855 	static Predicate<SourceFragment> checkCollectionItems(Predicate<SourceFragment> predicate) {
856 		return (SourceFragment fragment) -> {
857 			if (fragment instanceof CollectionSourceFragment) {
858 				CollectionSourceFragment collectionFragment = (CollectionSourceFragment) fragment;
859 				for (SourceFragment itemFragment : collectionFragment.getItems()) {
860 					if (predicate.test(itemFragment)) {
861 						return true;
862 					}
863 				}
864 				return false;
865 			} else {
866 				return predicate.test(fragment);
867 			}
868 		};
869 	}

871 	/**
872 	 * @param predicate to be called {@link Predicate}
873 	 * @return {@link Predicate} which calls `predicate` only for {@link SourceFragment}s of of type `clazz` and returns false for others
874 	 */
875 	static <T extends SourceFragment> Predicate<SourceFragment> filter(Class<T> clazz, Predicate<T> predicate) {
876 		return fragment -> {
877 			if (clazz.isInstance(fragment)) {
878 				return predicate.test((T) fragment);
879 			}
880 			return false;
881 		};
