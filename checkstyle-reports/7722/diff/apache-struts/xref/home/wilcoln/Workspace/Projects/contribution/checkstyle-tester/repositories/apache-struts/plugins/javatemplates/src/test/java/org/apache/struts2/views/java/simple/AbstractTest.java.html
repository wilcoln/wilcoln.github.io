168     protected void assertScriptingAttrs(String str) {
169         for (Map.Entry<String, String> entry : scriptingAttrs.entrySet()) {
170             String substr = entry.getKey() + "=\"" + entry.getValue() + "\"";
171             assertTrue("String [" + substr + "] was not found in [" + str + "]", str.indexOf(substr) >= 0);
172         }
173     }
174 
175     protected void assertCommonAttrs(String str) {
176         for (Map.Entry<String, String> entry : commonAttrs.entrySet()) {
177             String substr = entry.getKey() + "=\"" + entry.getValue() + "\"";
178             assertTrue("String [" + substr + "] was not found in [" + str + "]", str.indexOf(substr) >= 0);
179         }
180     }
181 
182     protected void assertDynamicAttrs(String str) {
183         for (Map.Entry<String, Object> entry : dynamicAttrs.entrySet()) {
184             String substr = entry.getKey() + "=\"" + entry.getValue() + "\"";
185             assertTrue("String [" + substr + "] was not found in [" + str + "]", str.indexOf(substr) >= 0);
186         }
187     }

189     protected Object doFindValue(String expr, Class toType) {
190         Object val = stack.findValue(expr);
191 
192         if (toType == String.class)
193             return val == null ? expr : val;
194         else
195             return val == null ? null : val;
196     }
