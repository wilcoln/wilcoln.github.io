192     /** {@inheritDoc} */
193     @Override
194     public boolean isAdditionalStateManaged(final String name) {
195 
196         // first look at already integrated states
197         if (super.isAdditionalStateManaged(name)) {
198             return true;
199         }
200 
201         // then look at states we integrate ourselves
202         for (final AdditionalEquations equation : additionalEquations) {
203             if (equation.getName().equals(name)) {
204                 return true;
205             }
206         }
207 
208         return false;
209     }

211     /** {@inheritDoc} */
212     @Override
213     public String[] getManagedAdditionalStates() {
214         final String[] alreadyIntegrated = super.getManagedAdditionalStates();
215         final String[] managed = new String[alreadyIntegrated.length + additionalEquations.size()];
216         System.arraycopy(alreadyIntegrated, 0, managed, 0, alreadyIntegrated.length);
217         for (int i = 0; i < additionalEquations.size(); ++i) {
218             managed[i + alreadyIntegrated.length] = additionalEquations.get(i).getName();
219         }
220         return managed;

369     /** {@inheritDoc} */
370     public SpacecraftState propagate(final AbsoluteDate target) throws OrekitException {
371         if (getStartDate() == null) {
372             if (getInitialState() == null) {
373                 throw new OrekitException(OrekitMessages.INITIAL_STATE_NOT_SPECIFIED_FOR_ORBIT_PROPAGATION);
374             }
375             setStartDate(getInitialState().getDate());
376         }
377         return propagate(getStartDate(), target);

380     /** {@inheritDoc} */
381     public SpacecraftState propagate(final AbsoluteDate tStart, final AbsoluteDate tEnd)
382         throws OrekitException {
383 
384         if (getInitialState() == null) {
385             throw new OrekitException(OrekitMessages.INITIAL_STATE_NOT_SPECIFIED_FOR_ORBIT_PROPAGATION);
386         }
387 
388         if (!tStart.equals(getInitialState().getDate())) {
389             // if propagation start date is not initial date,
390             // propagate from initial to start date without event detection
391             propagate(tStart, false);
392         }
393 
394         // propagate from start date to end date with event detection
395         return propagate(tEnd, true);
396 
397     }

399     /** Propagation with or without event detection.
400      * @param tEnd target date to which orbit should be propagated
401      * @param activateHandlers if true, step and event handlers should be activated
402      * @return state at end of propagation
403      * @exception OrekitException if orbit cannot be propagated
404      */
405     protected SpacecraftState propagate(final AbsoluteDate tEnd, final boolean activateHandlers)
406         throws OrekitException {
407         try {
408 
409             if (getInitialState().getDate().equals(tEnd)) {
410                 // don't extrapolate
411                 return getInitialState();
412             }
413 
414             // space dynamics view
415             stateMapper = createMapper(getInitialState().getDate(), stateMapper.getMu(),
416                                        stateMapper.getOrbitType(), stateMapper.getPositionAngleType(),
417                                        stateMapper.getAttitudeProvider(), getInitialState().getFrame());
418 
419 
420             // set propagation orbit type
421             final Orbit initialOrbit = stateMapper.getOrbitType().convertType(getInitialState().getOrbit());
422             if (Double.isNaN(getMu())) {
423                 setMu(initialOrbit.getMu());
424             }
425 
426             if (getInitialState().getMass() <= 0.0) {
427                 throw new OrekitException(OrekitMessages.SPACECRAFT_MASS_BECOMES_NEGATIVE,
428                                           getInitialState().getMass());
429             }
430 
431             integrator.clearEventHandlers();
432 
433             // set up events added by user
434             setUpUserEventDetectors();
435 
436             // convert space flight dynamics API to math API
437             final ODEState mathInitialState = createInitialState(getInitialIntegrationState());
438             final ExpandableODE mathODE = createODE(integrator, mathInitialState);
439             equationsMapper = mathODE.getMapper();
440             mathInterpolator = null;
441 
442             // initialize mode handler
443             if (modeHandler != null) {
444                 modeHandler.initialize(activateHandlers, tEnd);
445             }
446 
447             // mathematical integration
448             final ODEStateAndDerivative mathFinalState;
449             try {
450                 beforeIntegration(getInitialState(), tEnd);
451                 mathFinalState = integrator.integrate(mathODE, mathInitialState,
452                                                       tEnd.durationFrom(getInitialState().getDate()));
453                 afterIntegration();
454             } catch (OrekitExceptionWrapper oew) {
455                 throw oew.getException();
456             }
457 
458             // get final state
459             SpacecraftState finalState =
460                             stateMapper.mapArrayToState(stateMapper.mapDoubleToDate(mathFinalState.getTime(),
461                                                                                     tEnd),
462                                                         mathFinalState.getPrimaryState(),
463                                                         meanOrbit);
464             finalState = updateAdditionalStates(finalState);
465             for (int i = 0; i < additionalEquations.size(); ++i) {
466                 final double[] secondary = mathFinalState.getSecondaryState(i + 1);
467                 finalState = finalState.addAdditionalState(additionalEquations.get(i).getName(),
468                                                            secondary);
469             }
470             resetInitialState(finalState);
471             setStartDate(finalState.getDate());
472 
473             return finalState;
474 
475         } catch (MathRuntimeException mre) {
476             throw OrekitException.unwrap(mre);
477         }
478     }

488     /** Create an initial state.
489      * @param initialState initial state in flight dynamics world
490      * @return initial state in mathematics world
491      * @exception OrekitException if initial state cannot be mapped
492      */
493     private ODEState createInitialState(final SpacecraftState initialState)
494         throws OrekitException {
495 
496         // retrieve initial state
497         final double[] primary  = new double[getBasicDimension()];
498         stateMapper.mapStateToArray(initialState, primary);
499 
500         // secondary part of the ODE
501         final double[][] secondary = new double[additionalEquations.size()][];
502         for (int i = 0; i < additionalEquations.size(); ++i) {
503             final AdditionalEquations additional = additionalEquations.get(i);
504             secondary[i] = getInitialState().getAdditionalState(additional.getName());
505         }
506 
507         return new ODEState(0.0, primary, secondary);
508 
509     }

511     /** Create an ODE with all equations.
512      * @param integ numerical integrator to use for propagation.
513      * @param mathInitialState initial state
514      * @return a new ode
515      * @exception OrekitException if initial state cannot be mapped
516      */
517     private ExpandableODE createODE(final ODEIntegrator integ,
518                                     final ODEState mathInitialState)
519         throws OrekitException {
520 
521         final ExpandableODE ode =
522                 new ExpandableODE(new ConvertedMainStateEquations(getMainStateEquations(integ)));
523 
524         // secondary part of the ODE
525         for (int i = 0; i < additionalEquations.size(); ++i) {
526             final AdditionalEquations additional = additionalEquations.get(i);
527             final SecondaryODE secondary =
528                     new ConvertedSecondaryStateEquations(additional,
529                                                          mathInitialState.getSecondaryStateDimension(i + 1));
530             ode.addSecondaryEquations(secondary);
531         }
532 
533         return ode;
534 
535     }

577     /** Get a complete state with all additional equations.
578      * @param t current value of the independent <I>time</I> variable
579      * @param y array containing the current value of the state vector
580      * @return complete state
581      * @exception OrekitException if state cannot be mapped
582      */
583     private SpacecraftState getCompleteState(final double t, final double[] y)
584         throws OrekitException {
585 
586         // main state
587         SpacecraftState state = stateMapper.mapArrayToState(t, y, true);  //not sure of the mean orbit, should be true
588 
589         // pre-integrated additional states
590         state = updateAdditionalStates(state);
591 
592         // additional states integrated here
593         if (!additionalEquations.isEmpty()) {
594 
595             for (int i = 0; i < additionalEquations.size(); ++i) {
596                 state = state.addAdditionalState(additionalEquations.get(i).getName(),
597                                                  equationsMapper.extractEquationData(i + 1, y));
598             }
599 
600         }
601 
602         return state;
603 
604     }

634     /** Differential equations for the main state (orbit, attitude and mass), with converted API. */
635     private class ConvertedMainStateEquations implements OrdinaryDifferentialEquation {
636 
637         /** Main state equations. */
638         private final MainStateEquations main;
639 
640         /** Simple constructor.
641          * @param main main state equations
642          */
643         ConvertedMainStateEquations(final MainStateEquations main) {
644             this.main = main;
645             calls = 0;
646         }
647 
648         /** {@inheritDoc} */
649         public int getDimension() {
650             return getBasicDimension();
651         }
652 
653         @Override
654         public void init(final double t0, final double[] y0, final double finalTime) {
655             try {
656                 // update space dynamics view
657                 // use only ODE elements
658                 SpacecraftState initialState = stateMapper.mapArrayToState(t0, y0, true);
659                 initialState = updateAdditionalStates(initialState);
660                 final AbsoluteDate target = stateMapper.mapDoubleToDate(finalTime);
661                 main.init(initialState, target);
662             } catch (OrekitException oe) {
663                 throw new OrekitExceptionWrapper(oe);
664             }
665         }
666 
667         /** {@inheritDoc} */
668         public double[] computeDerivatives(final double t, final double[] y)
669             throws OrekitExceptionWrapper {
670             try {
671 
672                 // increment calls counter
673                 ++calls;
674 
675                 // update space dynamics view
676                 // use only ODE elements
677                 SpacecraftState currentState = stateMapper.mapArrayToState(t, y, true);
678                 currentState = updateAdditionalStates(currentState);
679 
680                 // compute main state differentials
681                 return main.computeDerivatives(currentState);
682 
683             } catch (OrekitException oe) {
684                 throw new OrekitExceptionWrapper(oe);
685             }
686         }
687 
688     }

697         private final int dimension;
698 
699         /** Simple constructor.
700          * @param equations additional equations
701          * @param dimension dimension of the additional state
702          */
703         ConvertedSecondaryStateEquations(final AdditionalEquations equations,
704                                          final int dimension) {
705             this.equations = equations;
706             this.dimension = dimension;
707         }
708 
709         /** {@inheritDoc} */
710         public int getDimension() {
711             return dimension;
712         }
713 
714         /** {@inheritDoc} */
715         public double[] computeDerivatives(final double t, final double[] primary,
716                                            final double[] primaryDot, final double[] secondary)
717             throws OrekitExceptionWrapper {
718 
719             try {
720 
721                 // update space dynamics view
722                 // the state contains only the ODE elements
723                 SpacecraftState currentState = stateMapper.mapArrayToState(t, primary, true);
724                 currentState = updateAdditionalStates(currentState);
725                 currentState = currentState.addAdditionalState(equations.getName(), secondary);
726 
727                 // compute additional derivatives
728                 final double[] secondaryDot = new double[secondary.length];
729                 final double[] additionalMainDot =
730                         equations.computeDerivatives(currentState, secondaryDot);
731                 if (additionalMainDot != null) {
732                     // the additional equations have an effect on main equations
733                     for (int i = 0; i < additionalMainDot.length; ++i) {
734                         primaryDot[i] += additionalMainDot[i];
735                     }
736                 }
737                 return secondaryDot;
738 
739             } catch (OrekitException oe) {
740                 throw new OrekitExceptionWrapper(oe);
741             }
742 
743         }
744

747     /** Adapt an {@link org.orekit.propagation.events.EventDetector}
748      * to Hipparchus {@link org.hipparchus.ode.events.ODEEventHandler} interface.
749      * @param <T> class type for the generic version
750      * @author Fabien Maussion
751      */
752     private class AdaptedEventDetector implements ODEEventHandler {
753 
754         /** Underlying event detector. */
755         private final EventDetector detector;
756 
757         /** Time of the previous call to g. */
758         private double lastT;
759 
760         /** Value from the previous call to g. */
761         private double lastG;
762 
763         /** Build a wrapped event detector.
764          * @param detector event detector to wrap
765         */
766         AdaptedEventDetector(final EventDetector detector) {
767             this.detector = detector;
768             this.lastT    = Double.NaN;
769             this.lastG    = Double.NaN;
770         }
771 
772         /** {@inheritDoc} */
773         public void init(final ODEStateAndDerivative s0, final double t) {
774             try {
775 
776                 detector.init(getCompleteState(s0.getTime(), s0.getCompleteState()),
777                               stateMapper.mapDoubleToDate(t));
778                 this.lastT = Double.NaN;
779                 this.lastG = Double.NaN;
780 
781             } catch (OrekitException oe) {
782                 throw new OrekitExceptionWrapper(oe);
783             }
784         }
785 
786         /** {@inheritDoc} */
787         public double g(final ODEStateAndDerivative s) {
788             try {
789                 if (!Precision.equals(lastT, s.getTime(), 0)) {
790                     lastT = s.getTime();
791                     lastG = detector.g(getCompleteState(s.getTime(), s.getCompleteState()));
792                 }
793                 return lastG;
794             } catch (OrekitException oe) {
795                 throw new OrekitExceptionWrapper(oe);
796             }
797         }
798 
799         /** {@inheritDoc} */
800         public Action eventOccurred(final ODEStateAndDerivative s, final boolean increasing) {
801             try {
802 
803                 final EventHandler.Action whatNext = detector.eventOccurred(getCompleteState(s.getTime(),
804                                                                                              s.getCompleteState()),
805                                                                             increasing);
806 
807                 switch (whatNext) {
808                     case STOP :
809                         return Action.STOP;
810                     case RESET_STATE :
811                         return Action.RESET_STATE;
812                     case RESET_DERIVATIVES :
813                         return Action.RESET_DERIVATIVES;
814                     default :
815                         return Action.CONTINUE;
816                 }
817             } catch (OrekitException oe) {
818                 throw new OrekitExceptionWrapper(oe);
819             }
820         }
821 
822         /** {@inheritDoc} */
823         public ODEState resetState(final ODEStateAndDerivative s) {
824             try {
825 
826                 final SpacecraftState oldState = getCompleteState(s.getTime(), s.getCompleteState());
827                 final SpacecraftState newState = detector.resetState(oldState);
828 
829                 // main part
830                 final double[] primary    = new double[s.getPrimaryStateDimension()];
831                 stateMapper.mapStateToArray(newState, primary);
832 
833                 // secondary part
834                 final double[][] secondary    = new double[additionalEquations.size()][];
835                 for (int i = 0; i < additionalEquations.size(); ++i) {
836                     secondary[i] = newState.getAdditionalState(additionalEquations.get(i).getName());
837                 }
838 
839                 return new ODEState(newState.getDate().durationFrom(getStartDate()),
840                                     primary, secondary);
841 
842             } catch (OrekitException oe) {
843                 throw new OrekitExceptionWrapper(oe);
844             }
845         }

965     private class EphemerisModeHandler implements ModeHandler, ODEStepHandler {
966 
967         /** Underlying raw mathematical model. */
968         private DenseOutputModel model;
969 
970         /** Generated ephemeris. */
971         private BoundedPropagator ephemeris;
972 
973         /** Flag for handler . */
974         private boolean activate;
975 
976         /** the user supplied end date. Propagation may not end on this date. */
977         private AbsoluteDate endDate;
978 
979         /** User's integration step handler. May be null. */
980         private final AdaptedStepHandler handler;
981 
982         /** Creates a new instance of EphemerisModeHandler which must be
983          *  filled by the propagator.
984          */
985         EphemerisModeHandler() {
986             this.handler = null;
987         }
988 
989         /** Creates a new instance of EphemerisModeHandler which must be
990          *  filled by the propagator.
991          *  @param handler the handler to notify of every integrator step.
992          */
993         EphemerisModeHandler(final OrekitStepHandler handler) {
994             this.handler = new AdaptedStepHandler(handler);
995         }
996 
997         /** {@inheritDoc} */
998         public void initialize(final boolean activateHandlers,
999                                final AbsoluteDate targetDate) {
1000             this.activate = activateHandlers;
1001             this.model    = new DenseOutputModel();
1002             this.endDate  = targetDate;
1003 
1004             // ephemeris will be generated when last step is processed
1005             this.ephemeris = null;
1006             if (this.handler != null) {
1007                 this.handler.initialize(activateHandlers, targetDate);
1008             }
1009         }
1010 
1011         /** Get the generated ephemeris.
1012          * @return a new instance of the generated ephemeris
1013          */
1014         public BoundedPropagator getEphemeris() {
1015             return ephemeris;
1016         }
1017 
1018         /** {@inheritDoc} */
1019         public void handleStep(final ODEStateInterpolator interpolator, final boolean isLast)
1020             throws OrekitExceptionWrapper {
1021             try {
1022                 if (activate) {
1023                     if (this.handler != null) {
1024                         this.handler.handleStep(interpolator, isLast);
1025                     }
1026 
1027                     model.handleStep(interpolator, isLast);
1028                     if (isLast) {
1029 
1030                         // set up the boundary dates
1031                         final double tI = model.getInitialTime();
1032                         final double tF = model.getFinalTime();
1033                         // tI is almost? always zero
1034                         final AbsoluteDate startDate =
1035                                 stateMapper.mapDoubleToDate(tI);
1036                         final AbsoluteDate finalDate =
1037                                 stateMapper.mapDoubleToDate(tF, this.endDate);
1038                         final AbsoluteDate minDate;
1039                         final AbsoluteDate maxDate;
1040                         if (tF < tI) {
1041                             minDate = finalDate;
1042                             maxDate = startDate;
1043                         } else {
1044                             minDate = startDate;
1045                             maxDate = finalDate;
1046                         }
1047 
1048                         // get the initial additional states that are not managed
1049                         final Map<String, double[]> unmanaged = new HashMap<String, double[]>();
1050                         for (final Map.Entry<String, double[]> initial : getInitialState().getAdditionalStates().entrySet()) {
1051                             if (!isAdditionalStateManaged(initial.getKey())) {
1052                                 // this additional state was in the initial state, but is unknown to the propagator
1053                                 // we simply copy its initial value as is
1054                                 unmanaged.put(initial.getKey(), initial.getValue());
1055                             }
1056                         }
1057 
1058                         // get the names of additional states managed by differential equations
1059                         final String[] names = new String[additionalEquations.size()];
1060                         for (int i = 0; i < names.length; ++i) {
1061                             names[i] = additionalEquations.get(i).getName();
1062                         }
1063 
1064                         // create the ephemeris
1065                         ephemeris = new IntegratedEphemeris(startDate, minDate, maxDate,
1066                                                             stateMapper, meanOrbit, model, unmanaged,
1067                                                             getAdditionalStateProviders(), names);
1068 
1069                     }
1070                 }
1071             } catch (OrekitException oe) {
1072                 throw new OrekitExceptionWrapper(oe);
1073             }
1074         }
1075 
1076         /** {@inheritDoc} */
1077         public void init(final ODEStateAndDerivative s0, final double t) {
1078             model.init(s0, t);
1079             if (this.handler != null) {
1080                 this.handler.init(s0, t);
1081             }
1082         }
