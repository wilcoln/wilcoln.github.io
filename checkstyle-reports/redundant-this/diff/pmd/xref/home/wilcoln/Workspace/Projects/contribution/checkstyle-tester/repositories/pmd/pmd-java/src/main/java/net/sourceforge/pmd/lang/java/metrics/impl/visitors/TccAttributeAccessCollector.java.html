29  /**
30   * Returns the map of method names to the set of local attributes accessed when visiting a class.
31   *
32   * @author Cl√©ment Fournier
33   * @since 6.0.0
34   * @deprecated Is internal API, will be moved in 7.0.0
35   */
36  @Deprecated
37  @InternalApi
38  public class TccAttributeAccessCollector extends JavaParserVisitorReducedAdapter {
39  
40      private final ASTAnyTypeDeclaration exploredClass;
41  
42  
43      /** The name of the current method. */
44      private String currentMethodName;
45  
46      private Map<String, Set<String>> methodAttributeAccess;
47  
48  
49      public TccAttributeAccessCollector(ASTAnyTypeDeclaration exploredClass) {
50          this.exploredClass = exploredClass;
51      }
52  
53  
54      /**
55       * Collects the attribute accesses by method into a map.
56       */
57      @SuppressWarnings("unchecked")
58      public Map<String, Set<String>> start() {
59          return (Map<String, Set<String>>) this.visit(exploredClass, new HashMap<String, Set<String>>());
60      }
61  
62  
63      @Override
64      public Object visit(ASTAnyTypeDeclaration node, Object data) {
65          if (Objects.equals(node, exploredClass)) {
66              methodAttributeAccess = new HashMap<>();
67              super.visit(node, data);
68          } else if (node instanceof ASTClassOrInterfaceDeclaration
69                  && ((ASTClassOrInterfaceDeclaration) node).isLocal()) {
70              super.visit(node, data);
71          }
72          return methodAttributeAccess;
73      }
74  
75  
76      @Override
77      public Object visit(ASTMethodDeclaration node, Object data) {
78  
79          if (!node.isAbstract()) {
80              if (node.getFirstParentOfType(ASTAnyTypeDeclaration.class) == exploredClass) {
81                  currentMethodName = node.getQualifiedName().getOperation();
82                  methodAttributeAccess.put(currentMethodName, new HashSet<String>());
83  
84                  super.visit(node, data);
85                  currentMethodName = null;
86              } else {
87                  super.visit(node, data);
88              }
89          }
90  
91          return null;
92      }
93  
94  
95      @Override
96      public Object visit(ASTConstructorDeclaration node, Object data) {
97          return data; // we're only looking for method pairs
98      }
99  
100 
101     /**
102      * The primary expression node is used to detect access
103      * to attributes and method calls. If the access is not for a
104      * foreign class, then the {@link #methodAttributeAccess}
105      * map is updated for the current method.
106      */
107     @Override
108     public Object visit(ASTPrimaryExpression node, Object data) {
109         if (currentMethodName != null) {
110             Set<String> methodAccess = methodAttributeAccess.get(currentMethodName);
111             String variableName = getVariableName(node);
112             if (isLocalAttributeAccess(variableName, node.getScope())) {
113                 methodAccess.add(variableName);
114             }
115         }
116 
117         return super.visit(node, data);
118     }
119 
120 
121     private String getVariableName(ASTPrimaryExpression node) {
122         ASTPrimaryPrefix prefix = node.getFirstDescendantOfType(ASTPrimaryPrefix.class);
123 
124         if (prefix.usesThisModifier()) {
125             List<ASTPrimarySuffix> suffixes = node.findChildrenOfType(ASTPrimarySuffix.class);
126             if (suffixes.size() > 1) {
127                 if (!suffixes.get(1).isArguments()) { // not a method call
128                     return suffixes.get(0).getImage();
129                 }
130             }
131         }
132 
133         ASTName name = prefix.getFirstDescendantOfType(ASTName.class);
134 
135         String variableName = null;
136 
137         if (name != null) {
138             int dotIndex = name.getImage().indexOf(".");
139             if (dotIndex == -1) {
140                 variableName = name.getImage();
141             } else {
142                 variableName = name.getImage().substring(0, dotIndex);
143             }
144         }
145 
146         return variableName;
147     }
148 
149 
150     private boolean isLocalAttributeAccess(String varName, Scope scope) {
151         Scope currentScope = scope;
152 
153         while (currentScope != null) {
154             for (VariableNameDeclaration decl : currentScope.getDeclarations(VariableNameDeclaration.class).keySet()) {
155                 if (decl.getImage().equals(varName)) {
156                     if (currentScope instanceof ClassScope
157                             && ((ClassScope) currentScope).getClassDeclaration().getNode() == exploredClass) {
158                         return true;
159                     }
160                 }
161             }
162             currentScope = currentScope.getParent(); // WARNING doesn't consider inherited fields or static imports
163         }
164 
165         return false;
166     }
