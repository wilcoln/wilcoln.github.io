6   package spoon.support.reflect.declaration;
7   
8   import spoon.reflect.annotations.MetamodelPropertyField;
9   import spoon.reflect.code.CtBlock;
10  import spoon.reflect.code.CtBodyHolder;
11  import spoon.reflect.code.CtExpression;
12  import spoon.reflect.code.CtStatement;
13  import spoon.reflect.declaration.CtAnnotationMethod;
14  import spoon.reflect.declaration.CtExecutable;
15  import spoon.reflect.declaration.CtFormalTypeDeclarer;
16  import spoon.reflect.declaration.CtParameter;
17  import spoon.reflect.declaration.CtTypeParameter;
18  import spoon.reflect.reference.CtTypeReference;
19  import spoon.reflect.visitor.CtVisitor;
20  import spoon.support.DerivedProperty;
21  import spoon.support.UnsettableProperty;
22  
23  import static spoon.reflect.path.CtRole.DEFAULT_EXPRESSION;
24  
25  import java.util.List;
26  import java.util.Set;
27  
28  /**
29   * The implementation for {@link spoon.reflect.declaration.CtAnnotationMethod}.
30   */
31  public class CtAnnotationMethodImpl<T> extends CtMethodImpl<T> implements CtAnnotationMethod<T> {
32  	@MetamodelPropertyField(role = DEFAULT_EXPRESSION)
33  	CtExpression<T> defaultExpression;
34  
35  	@Override
36  	public void accept(CtVisitor v) {
37  		v.visitCtAnnotationMethod(this);
38  	}
39  
40  	@Override
41  	public CtExpression<T> getDefaultExpression() {
42  		return defaultExpression;
43  	}
44  
45  	@Override
46  	public <C extends CtAnnotationMethod<T>> C setDefaultExpression(CtExpression<T> assignedExpression) {
47  		if (assignedExpression != null) {
48  			assignedExpression.setParent(this);
49  		}
50  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, DEFAULT_EXPRESSION, assignedExpression, this.defaultExpression);
51  		this.defaultExpression = assignedExpression;
52  		return (C) this;
53  	}
54  
55  	@Override
56  	@DerivedProperty
57  	public CtBlock<T> getBody() {
58  		return null;
59  	}
60  
61  	@Override
62  	@UnsettableProperty
63  	public <T extends CtBodyHolder> T setBody(CtStatement statement) {
64  		return (T) this;
65  	}
66  
67  	@Override
68  	@DerivedProperty
69  	public Set<CtTypeReference<? extends Throwable>> getThrownTypes() {
70  		return emptySet();
71  	}
72  
73  	@Override
74  	@UnsettableProperty
75  	public <U extends CtExecutable<T>> U setThrownTypes(Set<CtTypeReference<? extends Throwable>> thrownTypes) {
76  		return (U) this;
77  	}
78  
79  	@Override
80  	@DerivedProperty
81  	public List<CtTypeParameter> getFormalCtTypeParameters() {
82  		return emptyList();
83  	}
84  
85  	@Override
86  	@UnsettableProperty
87  	public <C extends CtFormalTypeDeclarer> C setFormalCtTypeParameters(List<CtTypeParameter> formalTypeParameters) {
88  		return (C) this;
89  	}
90  
91  	@Override
92  	@DerivedProperty
93  	public List<CtParameter<?>> getParameters() {
94  		return emptyList();
95  	}
96  
97  	@Override
98  	@UnsettableProperty
99  	public <U extends CtExecutable<T>> U setParameters(List<CtParameter<?>> parameters) {
100 		return (U) this;
101 	}
102 
103 	@Override
104 	public CtAnnotationMethod<T> clone() {
105 		return (CtAnnotationMethod<T>) super.clone();
106 	}
107 }
