21  package it.org.apache.struts2.showcase;
22  
23  import org.junit.Assert;
24  import org.junit.Test;
25  
26  import com.gargoylesoftware.htmlunit.WebClient;
27  import com.gargoylesoftware.htmlunit.html.HtmlCheckBoxInput;
28  import com.gargoylesoftware.htmlunit.html.HtmlForm;
29  import com.gargoylesoftware.htmlunit.html.HtmlPage;
30  import com.gargoylesoftware.htmlunit.html.HtmlSelect;
31  import com.gargoylesoftware.htmlunit.html.HtmlSubmitInput;
32  import com.gargoylesoftware.htmlunit.html.HtmlTextArea;
33  import com.gargoylesoftware.htmlunit.html.HtmlTextInput;
34  
35  public class UITagExampleTest {
36      @Test
37      public void testInputForm() throws Exception {
38          try (final WebClient webClient = new WebClient()) {
39              final HtmlPage page = webClient.getPage(ParameterUtils.getBaseUrl() + "/tags/ui/example!input.action");
40  
41              final HtmlForm form = page.getFormByName("exampleSubmit");
42              Assert.assertNotNull(form);
43  
44              final HtmlTextInput textField = form.getInputByName("name");
45              final HtmlTextArea textField2 = form.getTextAreaByName("bio");
46              final HtmlSelect textField3 = form.getSelectByName("favouriteColor");
47              final HtmlCheckBoxInput textField4 = form.getInputByValue("Patrick");
48              final HtmlCheckBoxInput textField41 = form.getInputByValue("Jason");
49              final HtmlCheckBoxInput textField5 = form.getInputByName("legalAge");
50  
51              Assert.assertNotNull(textField);
52              Assert.assertNotNull(textField2);
53              Assert.assertNotNull(textField3);
54              Assert.assertNotNull(textField4);
55              Assert.assertNotNull(textField41);
56              Assert.assertNotNull(textField5);
57  
58              textField.type("name");
59              textField2.type("bio");
60              textField3.setSelectedAttribute("Red", true);
61              textField4.setChecked(true);
62              textField41.setChecked(true);
63              textField5.setChecked(true);
64  
65              final HtmlSubmitInput button = form.getInputByValue("Submit");
66              final HtmlPage page2 = button.click();
67  
68              Assert.assertEquals("name", page2.getElementById("name").asText());
69              Assert.assertEquals("bio", page2.getElementById("bio").asText());
70              Assert.assertEquals("Red", page2.getElementById("favouriteColor").asText());
71              Assert.assertEquals("[Patrick, Jason]", page2.getElementById("friends").asText());
72              Assert.assertEquals("true", page2.getElementById("legalAge").asText());
73          }
74      }
75  }
