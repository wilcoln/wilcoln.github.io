6   package spoon.support.compiler;
7   
8   import java.util.ArrayList;
9   import java.util.List;
10  
11  import spoon.SpoonException;
12  
13  public class SnippetCompilationError extends SpoonException {
14  
15  	private static final long serialVersionUID = 7805276558728052328L;
16  
17  	public List<String> problems;
18  
19  	public SnippetCompilationError(List<String> problems) {
20  		this.problems = problems;
21  	}
22  
23  	public SnippetCompilationError(String string) {
24  		super(string);
25  		this.problems = new ArrayList<>();
26  		this.problems.add(string);
27  
28  	}
29  
30  }
