24  package org.hibernate.test.annotations.various.readwriteexpression;
25  import javax.persistence.Column;
26  import javax.persistence.Entity;
27  import javax.persistence.Id;
28  import javax.persistence.Table;
29  
30  import org.hibernate.annotations.ColumnTransformer;
31  
32  /**
33   * @author Emmanuel Bernard
34   */
35  @Entity
36  @Table(name="t_staff")
37  public class Staff {
38  
39  	public Staff(double sizeInInches, double radius, double diameter, Integer id) {
40  		this.sizeInInches = sizeInInches;
41  		this.radiusS = radius;
42  		this.diameter = diameter;
43  		this.id = id;
44  	}
45  
46  	@Id
47  	public Integer getId() { return id; }
48  	public void setId(Integer id) {  this.id = id; }
49  	private Integer id;
50  
51  	@Column(name="size_in_cm")
52  	@ColumnTransformer(
53  			forColumn = "size_in_cm",
54  			read = "size_in_cm / 2.54E0",
55  			write = "? * 2.54E0" )
56  	public double getSizeInInches() { return sizeInInches; }
57  	public void setSizeInInches(double sizeInInches) {  this.sizeInInches = sizeInInches; }
58  	private double sizeInInches;
59  
60  	//Weird extra S to avoid potential SQL keywords
61  	@ColumnTransformer(
62  			read = "radiusS / 2.54E0",
63  			write = "? * 2.54E0" )
64  	public double getRadiusS() { return radiusS; }
65  	public void setRadiusS(double radiusS) {  this.radiusS = radiusS; }
66  	private double radiusS;
67  
68  	@Column(name="diamet")
69  	@ColumnTransformer(
70  			read = "diamet / 2.54E0",
71  			write = "? * 2.54E0" )
72  	public double getDiameter() { return diameter; }
73  	public void setDiameter(double diameter) {  this.diameter = diameter; }
74  	private double diameter;
75  }
