79  	/**
80  	 * Creates a JDTCommentBuilder that will insert all comment of the declarationUnit into the Spoon AST
81  	 * @param declarationUnit the declaration unit
82  	 * @param factory the Spoon AST
83  	 */
84  	JDTCommentBuilder(CompilationUnitDeclaration declarationUnit,  Factory factory) {
85  		this.declarationUnit = declarationUnit;
86  		if (declarationUnit.comments == null) {
87  			return;
88  		}
89  		this.factory = factory;
90  		this.sourceUnit = declarationUnit.compilationResult.compilationUnit;
91  		this.contents = sourceUnit.getContents();
92  		this.filePath = CharOperation.charToString(sourceUnit.getFileName());
93  		this.spoonUnit = JDTTreeBuilder.getOrCreateCompilationUnit(declarationUnit, factory);
94  	}

96  	/**
97  	 * Start the build process
98  	 */
99  	public void build() {
100 		if (declarationUnit.comments == null) {
101 			return;
102 		}
103 		for (int j = 0; j < declarationUnit.comments.length; j++) {
104 			int[] positions = declarationUnit.comments[j];
105 			buildComment(positions);
106 		}
107 	}

109 	/**
110 	 * Inserts the comment at the position positions in the AST
111 	 * @param positions the position of the comment
112 	 */
113 	private void buildComment(int[] positions) {
114 		int start = positions[0];
115 		int end = -positions[1];
116 
117 		CtComment comment;
118 
119 		// Javadoc comments have negative end position
120 		if (end <= 0) {
121 			comment = factory.Core().createJavaDoc();
122 			end = -end;
123 		} else {
124 			comment = factory.Core().createComment();
125 			comment.setCommentType(CtComment.CommentType.BLOCK);
126 
127 			// the inline comments have negative start
128 			if (start < 0) {
129 				comment.setCommentType(CtComment.CommentType.INLINE);
130 				start = -start;
131 			}
132 		}
133 
134 		comment.setContent(getCommentContent(start, end));
135 
136 		// set the position
137 		int[] lineSeparatorPositions = declarationUnit.compilationResult.lineSeparatorPositions;
138 		SourcePosition sourcePosition = factory.Core().createSourcePosition(spoonUnit, start, end - 1, lineSeparatorPositions);
139 		comment.setPosition(sourcePosition);
140 
141 		insertCommentInAST(comment);
142 	}

144 	/**
145 	 * Insert the element to nearer element in the elements collections
146 	 * @param comment the comment to insert
147 	 * @param elements the collection that content the ast elements
148 	 * @return
149 	 */
150 	private CtElement addCommentToNear(final CtComment comment, final Collection<CtElement> elements) {
151 		CtElement best = null;
152 		int smallDistance = Integer.MAX_VALUE;
153 
154 		for (CtElement element : elements) {
155 			if (element.getPosition().isValidPosition() == false) {
156 				continue;
157 			}
158 			if (element.isImplicit()) {
159 				continue;
160 			}
161 			if (element instanceof CtComment) {
162 				continue;
163 			}
164 			final boolean isAfter = element.getPosition().getSourceEnd() < comment.getPosition().getSourceStart();
165 			int distance = Math.abs(element.getPosition().getSourceStart() - comment.getPosition().getSourceEnd());
166 			if (isAfter) {
167 				distance = Math.abs(element.getPosition().getSourceEnd() - comment.getPosition().getSourceStart());
168 			}
169 
170 			int elementEndLine = element.getPosition().getEndLine();
171 			int commentLine = comment.getPosition().getLine();
172 
173 			if (distance < smallDistance && (!isAfter || elementEndLine == commentLine || element instanceof CtType)) {
174 				best = element;
175 				smallDistance = distance;
176 			}
177 		}
178 		// adds the comment to the nearest element
179 		if (best != null) {
180 			best.addComment(comment);
181 		}
182 		return best;
183 	}

185 	/**
186 	 * Inserts the comment into the AST.
187 	 * @param comment the comment to insert
188 	 */
189 	private void insertCommentInAST(final CtComment comment) {
190 		CtElement commentParent = findCommentParent(comment);
191 		if (commentParent instanceof CtPackageDeclaration || commentParent instanceof CtCompilationUnit) {
192 			File file = spoonUnit.getFile();
193 			if (file == null) {
194 				//it is a virtual compilation unit - e.g. Snipet compilation unit
195 				//all such comments belongs to declared type
196 				List<CtType<?>> types = spoonUnit.getDeclaredTypes();
197 				if (types.size() > 0) {
198 					types.get(0).addComment(comment);
199 					return;
200 				}
201 			} else if (file.getName().equals(DefaultJavaPrettyPrinter.JAVA_PACKAGE_DECLARATION)) {
202 				//all compilation unit comments and package declaration comments are in package
203 				//other comments can belong to imports or types declared in package-info.java file
204 				spoonUnit.getDeclaredPackage().addComment(comment);
205 				return;
206 			} else if (file.getName().equals(DefaultJavaPrettyPrinter.JAVA_MODULE_DECLARATION)) {
207 				spoonUnit.getDeclaredModule().addComment(comment);
208 				return;
209 			}
210 		}
211 		// visitor that inserts the comment in the element
212 		CtInheritanceScanner insertionVisitor = new CtInheritanceScanner() {
213 			private boolean isScanned = false;
214 
215 			@Override
216 			public void scan(CtElement e) {
217 				if (e == null) {
218 					return;
219 				}
220 				// Do not visit the AST, only the first element
221 				if (!isScanned) {
222 					isScanned = true;
223 					SourcePosition sp = e.getPosition();
224 					if (sp.getSourceStart() == comment.getPosition().getSourceStart()) {
225 						e.addComment(comment);
226 						return;
227 					}
228 					if (sp instanceof DeclarationSourcePosition) {
229 						DeclarationSourcePosition dsp = (DeclarationSourcePosition) sp;
230 						if (comment.getPosition().getSourceEnd() < dsp.getModifierSourceEnd()) {
231 							e.addComment(comment);
232 							return;
233 						}
234 					}
235 					super.scan(e);
236 				}
237 			}
238 
239 			@Override
240 			public void scanCtReference(CtReference reference) {
241 				reference.addComment(comment);
242 				super.scanCtReference(reference);
243 			}
244 
245 			@Override
246 			public <R> void visitCtStatementList(CtStatementList e) {
247 				addCommentToNear(comment, new ArrayList<>(e.getStatements()));
248 				try {
249 					comment.getParent();
250 				} catch (ParentNotInitializedException ex) {
251 					e.addStatement(comment);
252 				}
253 			}

257 				e.addComment(comment);
258 			}

263 			}

266 			public <T> void visitCtConditional(CtConditional<T> e) {
267 				List<CtElement> elements = new ArrayList<>();
268 				elements.add(e.getElseExpression());
269 				elements.add(e.getThenExpression());
270 				elements.add(e.getCondition());
271 				addCommentToNear(comment, elements);
272 			}
273 
274 			@Override
275 			public <T> void visitCtBinaryOperator(CtBinaryOperator<T> e) {
276 				List<CtElement> elements = new ArrayList<>();
277 				elements.add(e.getLeftHandOperand());
278 				elements.add(e.getRightHandOperand());
279 				addCommentToNear(comment, elements);
280 			}
281 
282 			@Override
283 			public <T> void visitCtClass(CtClass<T> e) {
284 				//all the comments located before the body brackets belongs to class
285 				if (comment.getPosition().getSourceEnd() < ((BodyHolderSourcePosition) e.getPosition()).getBodyStart()) {
286 					e.addComment(comment);
287 					return;
288 				}
289 				final List<CtElement> elements = new ArrayList<>();
290 				for (CtTypeMember typeMember : e.getTypeMembers()) {
291 					if (typeMember instanceof CtField || typeMember instanceof CtMethod || typeMember instanceof CtConstructor) {
292 						elements.add(typeMember);
293 					}
294 				}
295 				addCommentToNear(comment, elements);
296 
297 				try {
298 					comment.getParent();
299 				} catch (ParentNotInitializedException ex) {
300 					e.addComment(comment);
301 				}
302 			}

304 			private <T> void visitInterfaceType(CtType<T> e) {
305 				final List<CtElement> elements = new ArrayList<>();
306 				for (CtTypeMember typeMember : e.getTypeMembers()) {
307 					if (typeMember instanceof CtField || typeMember instanceof CtMethod) {
308 						elements.add(typeMember);
309 					}
310 				}
311 				addCommentToNear(comment, elements);
312 
313 				try {
314 					comment.getParent();
315 				} catch (ParentNotInitializedException ex) {
316 					e.addComment(comment);
317 				}
318 			}

323 			}

328 			}

333 			}

350 			@Override
351 			public <E> void visitCtSwitch(CtSwitch<E> e) {
352 				List<CtCase<? super E>> cases = e.getCases();
353 				CtCase previous = null;
354 				for (CtCase<? super E> ctCase : cases) {
355 					if (previous == null) {
356 						if (comment.getPosition().getSourceStart() < ctCase.getPosition().getSourceStart()
357 								&& e.getPosition().getSourceStart() < comment.getPosition().getSourceStart()) {
358 							ctCase.addComment(comment);
359 							return;
360 						}
361 					} else {
362 						if (previous.getPosition().getSourceEnd() < comment.getPosition().getSourceStart()
363 								&& ctCase.getPosition().getSourceStart() > comment.getPosition().getSourceStart()) {
364 							addCommentToNear(comment, new ArrayList<>(previous.getStatements()));
365 							try {
366 								comment.getParent();
367 							} catch (ParentNotInitializedException ex) {
368 								previous.addStatement(comment);
369 							}
370 							return;
371 						}
372 					}
373 					previous = ctCase;
374 				}
375 				if (previous.getPosition().getSourceEnd() < comment.getPosition().getSourceStart()) {
376 					addCommentToNear(comment, new ArrayList<>(previous.getStatements()));
377 					try {
378 						comment.getParent();
379 					} catch (ParentNotInitializedException ex) {
380 						previous.addStatement(comment);
381 					}
382 					return;
383 				}
384 				try {
385 					comment.getParent();
386 				} catch (ParentNotInitializedException ex) {
387 					e.addComment(comment);
388 				}
389 			}

391 			@Override
392 			public void visitCtIf(CtIf e) {
393 				CtStatement thenStatement = e.getThenStatement();
394 				if (thenStatement != null) {
395 					if (!(thenStatement instanceof CtBlock)) {
396 						if (comment.getPosition().getSourceEnd() <= thenStatement.getPosition().getSourceStart()) {
397 							thenStatement.addComment(comment);
398 							return;
399 						}
400 					}
401 				}
402 				CtStatement elseStatement = e.getElseStatement();
403 				if (elseStatement != null && thenStatement != null) {
404 					SourcePosition thenPosition = thenStatement.getPosition();
405 					if (!thenPosition.isValidPosition() && thenStatement instanceof CtBlock) {
406 						CtStatement thenExpression = ((CtBlock) thenStatement).getStatement(0);
407 						thenPosition = thenExpression.getPosition();
408 					}
409 					SourcePosition elsePosition = elseStatement.getPosition();
410 					if (!elsePosition.isValidPosition() && elseStatement instanceof CtBlock) {
411 						CtStatement elseExpression = ((CtBlock) elseStatement).getStatement(0);
412 						elsePosition = elseExpression.getPosition();
413 					}
414 					if (comment.getPosition().getSourceStart() > thenPosition.getSourceEnd() && comment.getPosition().getSourceEnd() < elsePosition.getSourceStart()) {
415 						elseStatement.addComment(comment);
416 					}
417 				}
418 				try {
419 					comment.getParent();
420 				} catch (ParentNotInitializedException ex) {
421 					e.addComment(comment);
422 				}
423 			}
424 
425 			@Override
426 			public void scanCtStatement(CtStatement s) {
427 				if (!(s instanceof CtStatementList || s instanceof CtSwitch || s instanceof CtVariable)) {
428 					s.addComment(comment);
429 				}
430 			}

432 			@Override
433 			public void visitCtAnonymousExecutable(CtAnonymousExecutable e) {
434 				e.addComment(comment);
435 			}
436 
437 			@Override
438 			public <T> void visitCtLambda(CtLambda<T> e) {
439 				if (e.getExpression() != null) {
440 					CtParameter<?> lastParameter = e.getParameters().get(e.getParameters().size() - 1);
441 					if (comment.getPosition().getSourceStart() > lastParameter.getPosition().getSourceEnd()) {
442 						e.getExpression().addComment(comment);
443 					} else {
444 						e.addComment(comment);
445 					}
446 				} else if (e.getBody() != null) {
447 					e.addComment(comment);
448 				}
449 			}
450 
451 			@Override
452 			public <T> void visitCtNewArray(CtNewArray<T> e) {
453 				addCommentToNear(comment, new ArrayList<>(e.getElements()));
454 				try {
455 					comment.getParent();
456 				} catch (ParentNotInitializedException ex) {
457 					e.addComment(comment);
458 				}
459 			}
460 
461 			@Override
462 			public <T> void visitCtParameter(CtParameter<T> e) {
463 				e.addComment(comment);
464 			}
465 
466 			@Override
467 			public void visitCtCatch(CtCatch e) {
468 				if (comment.getPosition().getLine() <= e.getPosition().getLine()) {
469 					e.addComment(comment);
470 				} else {
471 					e.getBody().addComment(comment);
472 				}
473 			}
474 
475 			@Override
476 			public void visitCtModule(CtModule module) {
477 				addCommentToNear(comment, new ArrayList<>(module.getModuleDirectives()));
478 			}
479 
480 			@Override
481 			public <A extends Annotation> void visitCtAnnotation(CtAnnotation<A> e) {
482 				addCommentToNear(comment, new ArrayList<>(e.getValues().values()));
483 			}
484 		};
485 		insertionVisitor.scan(commentParent);
486 
487 		// postcondition
488 		// now we make sure that there is a parent
489 		// if there is no parent
490 		if (!comment.isParentInitialized()) {
491 			// that's a serious error, there is something to debug
492 			LOGGER.error("\"" + comment + "\" cannot be added into the AST, with parent " + commentParent.getClass()
493 					+ " at " + commentParent.getPosition().toString()
494 					+ ", please report the bug by posting on https://github.com/INRIA/spoon/issues/2482");
495 		}
496 	}

498 	/**
499 	 * Find the parent of a comment based on the position
500 	 * @param comment the comment
501 	 * @return the parent of the comment
502 	 */
503 	private CtElement findCommentParent(CtComment comment) {
504 		class FindCommentParentScanner extends EarlyTerminatingScanner<Void> {
505 			public CtElement commentParent;
506 
507 			private int start;
508 			private int end;
509 
510 			FindCommentParentScanner(int start, int end) {
511 				this.start = start;
512 				this.end = end;
513 				setVisitCompilationUnitContent(true);
514 			}
515 
516 			private boolean isCommentBetweenElementPosition(CtElement element) {
517 				return (element.getPosition().isValidPosition()
518 						&& element.getPosition().getSourceStart() <= this.start
519 						&& element.getPosition().getSourceEnd() >= this.end);
520 			}
521 
522 			@Override
523 			public void scan(CtElement element) {
524 				if (element == null) {
525 					return;
526 				}
527 				if (element.isImplicit() && !(element instanceof CtBlock)) {
528 					return;
529 				}
530 				CtElement body = getBody(element);
531 				if (body != null && body.getPosition().isValidPosition() == false) {
532 					body = null;
533 				}
534 
535 				boolean betweenElementPosition = this.isCommentBetweenElementPosition(element);
536 				boolean bodyBetweenElementPosition = (body != null) && this.isCommentBetweenElementPosition(body);
537 
538 				if (betweenElementPosition || bodyBetweenElementPosition) {
539 					commentParent = element;
540 					element.accept(this);
541 				}
542 			}
543 		}
544 		FindCommentParentScanner findCommentParentScanner = new FindCommentParentScanner(
545 				comment.getPosition().getSourceStart(),
546 				comment.getPosition().getSourceEnd());
547 
548 		findCommentParentScanner.scan(spoonUnit);
549 
550 		return findCommentParentScanner.commentParent;
551 	}

574 	public static String cleanComment(String comment) {
575 		if (comment == null) {
576 			return "";
577 		}
578 		return cleanComment(new StringReader(comment));
579 	}

581 	private static final Pattern startCommentRE = Pattern.compile("^/\\*{1,2} ?");
582 	private static final Pattern middleCommentRE = Pattern.compile("^[ \t]*\\*? ?");
583 	private static final Pattern endCommentRE = Pattern.compile("\\*/$");
584 
585 	private static String cleanComment(Reader comment) {
586 		StringBuilder ret = new StringBuilder();
587 		try (BufferedReader br = new BufferedReader(comment)) {
588 			String line = br.readLine();
589 			// nothing in the first line
590 			if (line == null) {
591 				return ret.toString();
592 			}
593 			boolean isLastLine = false;
594 			if (line.length() >= 2 && line.charAt(1) == '/') {
595 				//it is single line comment, which starts with "//"
596 				isLastLine = true;
597 				line = line.substring(2);
598 			} else {
599 				//it is potentially multiline comment, which starts with "/*" or "/**"
600 				//check end first
601 				if (line.endsWith("*/") && line.length() > 3) {
602 					//it is last line
603 					line = endCommentRE.matcher(line).replaceFirst("");
604 					isLastLine = true;
605 				}
606 				//skip beginning
607 				line = startCommentRE.matcher(line).replaceFirst("");
608 			}
609 			//append first line
610 			ret.append(line);
611 			while ((line = br.readLine()) != null) {
612 				if (isLastLine) {
613 					throw new SpoonException("Unexpected next line after last line");
614 				}
615 				if (line.endsWith("*/")) {
616 					//it is last line
617 					line = endCommentRE.matcher(line).replaceFirst("");
618 					isLastLine = true;
619 				}
620 				//always clean middle comment, but after end comment is detected
621 				line = middleCommentRE.matcher(line).replaceFirst("");
622 				//write next line - Note that Spoon model comment's lines are always separated by "\n"
623 				ret.append(CtComment.LINE_SEPARATOR);
624 				ret.append(line);
625 			}
626 			return ret.toString().trim();
627 		} catch (IOException e) {
628 			throw new SpoonException(e);
629 		}
630 	}
