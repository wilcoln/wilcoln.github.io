19  package org.apache.struts2.views.velocity;
20  
21  import org.apache.velocity.VelocityContext;
22  
23  import com.opensymphony.xwork2.util.ValueStack;
24  
25  import java.util.ArrayList;
26  import java.util.Arrays;
27  import java.util.List;
28  
29  public class StrutsVelocityContext extends VelocityContext {
30  
31      private ValueStack stack;
32      private List<VelocityContext> chainedContexts;
33  
34      /**
35       * Creates a content with link to the ValueStack and any other Velocity contexts
36       *
37       * @param chainedContexts Existing Velocity contexts to chain to
38       * @param stack Struts ValueStack
39       * @since 2.6
40       */
41      public StrutsVelocityContext(List<VelocityContext> chainedContexts, ValueStack stack) {
42          this.chainedContexts = chainedContexts;
43          this.stack = stack;
44      }
45  
46      /**
47       * @deprecated please use newly added {@link StrutsVelocityContext(List<VelocityContext)} constructor instead
48       * and pass {null} or empty list if no chained contexts were defined
49       */
50      @Deprecated
51      public StrutsVelocityContext(ValueStack stack) {
52          this((List<VelocityContext>) null, stack);
53      }
54  
55      /**
56       * @deprecated please use newly added {@link StrutsVelocityContext(List<VelocityContext)} constructor instead
57       */
58      @Deprecated()
59      public StrutsVelocityContext(VelocityContext[] chainedContexts, ValueStack stack) {
60          this(new ArrayList<>(Arrays.asList(chainedContexts)), stack);
61      }
62  
63      public boolean internalContainsKey(String key) {
64          boolean contains = super.internalContainsKey(key);
65  
66          // first let's check to see if we contain the requested key
67          if (contains) {
68              return true;
69          }
70  
71          // if not, let's search for the key in the ognl value stack
72          if (stack != null) {
73              Object o = stack.findValue(key);
74  
75              if (o != null) {
76                  return true;
77              }
78  
79              o = stack.getContext().get(key);
80              if (o != null) {
81                  return true;
82              }
83          }
84  
85          // if we still haven't found it, le's search through our chained contexts
86          if (chainedContexts != null) {
87              for (VelocityContext chainedContext : chainedContexts) {
88                  if (chainedContext.containsKey(key)) {
89                      return true;
90                  }
91              }
92          }
93  
94          // nope, i guess it's really not here
95          return false;
96      }
97  
98      public Object internalGet(String key) {
99          // first, let's check to see if have the requested value
100         if (super.internalContainsKey(key)) {
101             return super.internalGet(key);
102         }
103 
104         // still no luck?  let's look against the value stack
105         if (stack != null) {
106             Object object = stack.findValue(key);
107 
108             if (object != null) {
109                 return object;
110             }
111 
112             object = stack.getContext().get(key);
113             if (object != null) {
114                 return object;
115             }
116 
117         }
118 
119         // finally, if we're chained to other contexts, let's look in them
120         if (chainedContexts != null) {
121             for (VelocityContext chainedContext : chainedContexts) {
122                 if (chainedContext.containsKey(key)) {
123                     return chainedContext.internalGet(key);
124                 }
125             }
126         }
127 
128         // nope, i guess it's really not here
129         return null;
130     }
131 }
