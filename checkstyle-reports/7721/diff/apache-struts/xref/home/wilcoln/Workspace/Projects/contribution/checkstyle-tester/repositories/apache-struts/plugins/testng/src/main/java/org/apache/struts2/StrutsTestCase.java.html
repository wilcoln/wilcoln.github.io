19  package org.apache.struts2;
20  
21  import java.util.Map;
22  
23  import org.apache.struts2.dispatcher.Dispatcher;
24  import org.apache.struts2.util.StrutsTestCaseHelper;
25  import org.testng.annotations.AfterTest;
26  import org.testng.annotations.BeforeTest;
27  import org.springframework.mock.web.MockServletContext;
28  
29  import com.opensymphony.xwork2.TestNGXWorkTestCase;
30  
31  /**
32   * Base test class for TestNG unit tests.  Provides common Struts variables
33   * and performs Struts setup and teardown processes
34   */
35  public class StrutsTestCase extends TestNGXWorkTestCase {
36  
37      @BeforeTest
38      protected void setUp() throws Exception {
39          super.setUp();
40          initDispatcher(null);
41      }
42      
43      protected Dispatcher initDispatcher(Map<String,String> params) {
44          Dispatcher du = StrutsTestCaseHelper.initDispatcher(new MockServletContext(), params);
45          configurationManager = du.getConfigurationManager();
46          configuration = configurationManager.getConfiguration();
47          container = configuration.getContainer();
48          return du;
49      }
50  
51      /**
52       * A helper method which allows instantiate an action if this action extends
53       * {@link com.opensymphony.xwork2.ActionSupport} or any other action class
54       * that requires framework's dependencies injection.
55       */
56      protected <T> T createAction(Class<T> clazz) {
57          return container.inject(clazz);
58      }
59  
60      @AfterTest
61      protected void tearDown() throws Exception {
62          super.tearDown();
63          StrutsTestCaseHelper.tearDown();
64      }
65  }
