5   package net.sourceforge.pmd.lang.dfa.report;
6   
7   import java.io.BufferedWriter;
8   import java.io.File;
9   import java.io.IOException;
10  import java.nio.charset.StandardCharsets;
11  import java.nio.file.Files;
12  
13  import org.apache.commons.lang3.StringUtils;
14  
15  import net.sourceforge.pmd.PMD;
16  import net.sourceforge.pmd.RuleViolation;
17  
18  /**
19   * Uses the generated result tree instead of the result list. The
20   * visitor traverses the tree and creates several html files. The
21   * "package view" file (index.html) displays an overview of packages,
22   * classes and the number of * rule violations they contain. All the
23   * other html files represent a class * and show detailed information
24   * about the violations.
25   *
26   * @author raik
27   */
28  @Deprecated // will be removed with PMD 7.0.0 without replacement. See net.sourceforge.pmd.lang.dfa.report.ReportTree for details.
29  public class ReportHTMLPrintVisitor extends ReportVisitor {
30  
31      @SuppressWarnings("PMD.AvoidStringBufferField")
32      private StringBuilder packageBuf = new StringBuilder(30);
33      @SuppressWarnings("PMD.AvoidStringBufferField")
34      private StringBuilder classBuf = new StringBuilder(60);
35      private int length;
36      private String baseDir;
37  
38      private static final String FILE_SEPARATOR = System.getProperty("file.separator");
39  
40      public ReportHTMLPrintVisitor(String baseDir) {
41          this.baseDir = baseDir;
42      }
43  
44      /**
45       * Writes the buffer to file.
46       */
47      private void write(String filename, StringBuilder buf) throws IOException {
48          try (BufferedWriter bw = Files.newBufferedWriter(new File(baseDir + FILE_SEPARATOR + filename).toPath(),
49                  StandardCharsets.UTF_8)) {
50              bw.write(buf.toString(), 0, buf.length());
51          }
52      }
53  
54      /**
55       * Generates a html table with violation information.
56       */
57      private String displayRuleViolation(RuleViolation vio) {
58  
59          StringBuilder sb = new StringBuilder(200);
60          sb.append("<table border=\"0\">");
61          renderViolationRow(sb, "Rule:", vio.getRule().getName());
62          renderViolationRow(sb, "Description:", vio.getDescription());
63  
64          if (StringUtils.isNotBlank(vio.getVariableName())) {
65              renderViolationRow(sb, "Variable:", vio.getVariableName());
66          }
67  
68          if (vio.getEndLine() > 0) {
69              renderViolationRow(sb, "Line:", vio.getEndLine() + " and " + vio.getBeginLine());
70          } else {
71              renderViolationRow(sb, "Line:", Integer.toString(vio.getBeginLine()));
72          }
73  
74          sb.append("</table>");
75          return sb.toString();
76      }
77  
78      // TODO - join the 21st century, include CSS attributes :)
79      private void renderViolationRow(StringBuilder sb, String fieldName, String fieldData) {
80          sb.append("<tr><td><b>").append(fieldName).append("</b></td><td>").append(fieldData).append("</td></tr>");
81      }
82  
83      /**
84       * The visit method (Visitor Pattern). There are 3 types of ReportNodes:
85       * RuleViolation - contains a RuleViolation, Class - represents a class and
86       * contains the name of the class, Package - represents a package and
87       * contains the name(s) of the package.
88       */
89      @Override
90      public void visit(AbstractReportNode node) {
91  
92          /*
93           * The first node of result tree.
94           */
95          if (node.getParent() == null) {
96              packageBuf.insert(0,
97                      "<!DOCTYPE html>" + PMD.EOL
98                      + "<html>" + " <head>" + PMD.EOL + "    <meta charset=\"UTF-8\">" + PMD.EOL
99                              + "   <title>PMD</title>" + " </head>" + " <body>" + PMD.EOL
100                             + "<h2>Package View</h2>"
101                             + "<table border=\"1\" align=\"center\" cellspacing=\"0\" cellpadding=\"3\">" + " <tr>"
102                             + PMD.EOL + "<th>Package</th>" + "<th>Class</th>" + "<th>#</th>" + " </tr>" + PMD.EOL);
103 
104             length = packageBuf.length();
105         }
106 
107         super.visit(node);
108 
109         if (node instanceof ViolationNode) {
110             renderViolation((ViolationNode) node);
111         }
112         if (node instanceof ClassNode) {
113             renderClass((ClassNode) node);
114         }
115         if (node instanceof PackageNode) {
116             renderPackage((PackageNode) node);
117         }
118 
119         // The first node of result tree.
120         if (node.getParent() == null) {
121             packageBuf.append("</table> </body></html>");
122             try {
123                 write("index.html", packageBuf);
124             } catch (Exception e) {
125                 throw new RuntimeException("Error while writing HTML report: " + e.getMessage());
126             }
127         }
128     }
129 
130     private void renderViolation(ViolationNode vnode) {
131 
132         vnode.getParent().addNumberOfViolation(1);
133         RuleViolation vio = vnode.getRuleViolation();
134         classBuf.append("<tr>" + " <td>" + vio.getMethodName() + "</td>" + " <td>" + this.displayRuleViolation(vio)
135                 + "</td>" + "</tr>");
136     }
137 
138     private void renderPackage(PackageNode pnode) {
139 
140         String str;
141 
142         // rootNode
143         if (pnode.getParent() == null) {
144             str = "Aggregate";
145         } else { // all the other nodes
146             str = pnode.getPackageName();
147             pnode.getParent().addNumberOfViolation(pnode.getNumberOfViolations());
148         }
149 
150         packageBuf.insert(length, "<tr><td><b>" + str + "</b></td>" + " <td>-</td>" + " <td>"
151                 + pnode.getNumberOfViolations() + "</td>" + "</tr>" + PMD.EOL);
152     }
153 
154     private void renderClass(ClassNode cnode) {
155 
156         String str = cnode.getClassName();
157 
158         classBuf.insert(0,
159                 "<!DOCTYPE html>" + PMD.EOL
160                 + "<html><head><meta charset=\"UTF-8\"><title>PMD - " + str + "</title></head><body>" + PMD.EOL + "<h2>Class View</h2>"
161                         + "<h3 align=\"center\">Class: " + str + "</h3>"
162                         + "<table border=\"\" align=\"center\" cellspacing=\"0\" cellpadding=\"3\">" + " <tr>" + PMD.EOL
163                         + "<th>Method</th>" + "<th>Violation</th>" + " </tr>" + PMD.EOL);
164 
165         classBuf.append("</table>" + " </body>" + "</html>");
166 
167         try {
168             write(str + ".html", classBuf);
169         } catch (Exception e) {
170             throw new RuntimeException("Error while writing HTML report: " + e.getMessage());
171         }
172         classBuf = new StringBuilder();
173 
174         packageBuf.insert(this.length, "<tr>" + " <td>-</td>" + " <td><a href=\"" + str + ".html\">" + str + "</a></td>"
175                 + " <td>" + cnode.getNumberOfViolations() + "</td>" + "</tr>" + PMD.EOL);
176         cnode.getParent().addNumberOfViolation(cnode.getNumberOfViolations());
177     }
178 }
