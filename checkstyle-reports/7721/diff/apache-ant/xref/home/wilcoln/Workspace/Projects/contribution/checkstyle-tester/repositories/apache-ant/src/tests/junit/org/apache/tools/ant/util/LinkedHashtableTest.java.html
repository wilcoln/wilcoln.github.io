19  package org.apache.tools.ant.util;
20  
21  import java.util.Hashtable;
22  import java.util.Iterator;
23  import java.util.Map;
24  
25  import org.junit.Test;
26  
27  import static org.junit.Assert.assertFalse;
28  import static org.junit.Assert.assertSame;
29  import static org.junit.Assert.assertTrue;
30  import static org.junit.Assert.assertNull;
31  import static org.junit.Assert.assertEquals;
32  
33  public class LinkedHashtableTest {
34  
35      private static final Object K1 = new Object();
36      private static final Object K2 = new Object();
37      private static final Object V1 = new Object();
38      private static final Object V2 = new Object();
39      private Hashtable h = new LinkedHashtable();
40  
41      public void testClear() {
42          h.put(K1, V1);
43          h.clear();
44          assertTrue(h.isEmpty());
45      }
46  
47      public void testClone() {
48          h.put(K1, V1);
49          Hashtable h2 = (Hashtable) h.clone();
50          assertTrue(h2 instanceof LinkedHashtable);
51          assertTrue(h2.containsKey(K1));
52      }
53  
54      @Test
55      public void testContainsAndPut() {
56          h.put(K1, V1);
57          assertTrue(h.contains(K1));
58          assertTrue(h.containsKey(K1));
59          assertTrue(h.containsValue(V1));
60          assertFalse(h.containsKey(K2));
61      }
62  
63      @Test
64      public void testGet() {
65          assertNull(h.get(K1));
66          h.put(K1, V1);
67          assertSame(V1, h.get(K1));
68      }
69  
70      @Test
71      public void testIsEmpty() {
72          assertTrue(h.isEmpty());
73          h.put(K1, V1);
74          assertFalse(h.isEmpty());
75      }
76  
77      @Test
78      public void testPutReturnValue() {
79          assertNull(h.put(K1, V1));
80          assertSame(V1, h.put(K1, V2));
81      }
82  
83      @Test
84      public void testPutAll() {
85          LinkedHashtable h2 = new LinkedHashtable();
86          h.put(K1, V1);
87          h2.putAll(h);
88          assertTrue(h2.containsKey(K1));
89      }
90  
91      @Test
92      public void testRemove() {
93          h.put(K1, V1);
94          assertSame(V1, h.remove(K1));
95          assertTrue(h.isEmpty());
96          assertNull(h.remove(K1));
97      }
98  
99      @Test
100     public void testSize() {
101         assertEquals(0, h.size());
102         h.put(K1, V1);
103         assertEquals(1, h.size());
104     }
105 
106     @Test
107     public void testKeys() {
108         multiSetup();
109         assertKeys(CollectionUtils.asIterator(h.keys()));
110     }
111 
112     @Test
113     public void testKeySet() {
114         multiSetup();
115         assertKeys(h.keySet().iterator());
116     }
117 
118     @Test
119     public void testElements() {
120         multiSetup();
121         assertValues(CollectionUtils.asIterator(h.elements()));
122     }
123 
124     @Test
125     public void testValues() {
126         multiSetup();
127         assertValues(h.values().iterator());
128     }
129 
130     @Test
131     public void testEntrySet() {
132         multiSetup();
133         Iterator i = h.entrySet().iterator();
134         assertTrue(i.hasNext());
135         Map.Entry e = (Map.Entry) i.next();
136         assertSame(K1, e.getKey());
137         assertSame(V1, e.getValue());
138         assertTrue(i.hasNext());
139         e = (Map.Entry) i.next();
140         assertSame(K2, e.getKey());
141         assertSame(V2, e.getValue());
142         assertFalse(i.hasNext());
143     }
144 
145     private void multiSetup() {
146         h.put(K1, V1);
147         h.put(K2, V2);
148     }
149 
150     private static void assertKeys(Iterator i) {
151         assertTrue(i.hasNext());
152         assertSame(K1, i.next());
153         assertTrue(i.hasNext());
154         assertSame(K2, i.next());
155         assertFalse(i.hasNext());
156     }
157 
158     private static void assertValues(Iterator i) {
159         assertTrue(i.hasNext());
160         assertSame(V1, i.next());
161         assertTrue(i.hasNext());
162         assertSame(V2, i.next());
163         assertFalse(i.hasNext());
164     }
165 }
