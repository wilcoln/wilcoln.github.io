64  		@Override
65  		public Set<Entry<String, CtExpression>> entrySet() {
66  			Set<Entry<String, CtExpression>> result = new TreeSet<>(new Comparator<Entry<String, CtExpression>>() {
67  				final CtLineElementComparator comp = new CtLineElementComparator();
68  
69  				@Override
70  				public int compare(Entry<String, CtExpression> o1, Entry<String, CtExpression> o2) {
71  					return comp.compare(o1.getValue(), o2.getValue());
72  				}
73  			}
74  			);
75  			result.addAll(super.entrySet());
76  			return result;
77  		}

89  	public <T extends CtAnnotation<A>> T addValue(String elementName, Object value) {
90  		if (value instanceof CtExpression) {
91  			return addValueExpression(elementName, (CtExpression<?>) value);
92  		}
93  		return this.addValueExpression(elementName, convertValueToExpression(value));
94  	}

96  	private CtExpression convertValueToExpression(Object value) {
97  		CtExpression res;
98  		if (value.getClass().isArray()) {
99  			// Value should be converted to a CtNewArray.
100 			res = getFactory().Core().createNewArray();
101 			Object[] values = (Object[]) value;
102 
103 			res.setType(getFactory().Type().createArrayReference(getFactory().Type().createReference(value.getClass().getComponentType())));
104 			for (Object o : values) {
105 				((CtNewArray) res).addElement(convertValueToExpression(o));
106 			}
107 		} else if (value instanceof Collection) {
108 			// Value should be converted to a CtNewArray.
109 			res = getFactory().Core().createNewArray();
110 			Collection values = (Collection) value;
111 			res.setType(getFactory().Type().createArrayReference(getFactory().Type().createReference(values.toArray()[0].getClass())));
112 			for (Object o : values) {
113 				((CtNewArray) res).addElement(convertValueToExpression(o));
114 			}
115 		} else if (value instanceof Class) {
116 			// Value should be a field access to a .class.
117 			res = getFactory().Code().createClassAccess(getFactory().Type().createReference((Class) value));
118 		} else if (value instanceof Field) {
119 			// Value should be a field access to a field.
120 			CtFieldReference<Object> variable = getFactory().Field().createReference((Field) value);
121 			variable.setStatic(true);
122 			CtTypeAccess target = getFactory().Code().createTypeAccess(getFactory().Type().createReference(((Field) value).getDeclaringClass()));
123 			CtFieldRead fieldRead = getFactory().Core().createFieldRead();
124 			fieldRead.setVariable(variable);
125 			fieldRead.setTarget(target);
126 			fieldRead.setType(target.getAccessedType());
127 			res = fieldRead;
128 		} else if (isPrimitive(value.getClass()) || value instanceof String) {
129 			// Value should be a literal.
130 			res = getFactory().Code().createLiteral(value);
131 		} else if (value.getClass().isEnum()) {
132 			final CtTypeReference declaringClass = getFactory().Type().createReference(((Enum) value).getDeclaringClass());
133 			final CtFieldReference variableRef = getFactory().Field().createReference(declaringClass, declaringClass, ((Enum) value).name());
134 			CtTypeAccess target = getFactory().Code().createTypeAccess(declaringClass);
135 			CtFieldRead fieldRead = getFactory().Core().createFieldRead();
136 			fieldRead.setVariable(variableRef);
137 			fieldRead.setTarget(target);
138 			fieldRead.setType(declaringClass);
139 			res = fieldRead;
140 		} else {
141 			throw new SpoonException("Please, submit a valid value.");
142 		}
143 		return res;
144 	}

146 	private boolean isPrimitive(Class c) {
147 		return c.isPrimitive() || c == Byte.class || c == Short.class || c == Integer.class || c == Long.class || c == Float.class || c == Double.class || c == Boolean.class || c == Character.class;
148 	}

150 	private <T extends CtAnnotation<A>> T addValueExpression(String elementName, CtExpression<?> expression) {
151 		if (elementValues.containsKey(elementName)) {
152 			// Update value of the existing one.
153 			final CtExpression ctExpression = elementValues.get(elementName);
154 			if (ctExpression instanceof CtNewArray) {
155 				// Already an array, add the value inside it.
156 				if (expression instanceof CtNewArray) {
157 					List<CtExpression<?>> elements = ((CtNewArray) expression).getElements();
158 					for (CtExpression expInArray : elements) {
159 						((CtNewArray) ctExpression).addElement(expInArray);
160 					}
161 				} else {
162 					((CtNewArray) ctExpression).addElement(expression);
163 				}
164 			} else {
165 				// Switch the value to a CtNewArray.
166 				CtNewArray<Object> newArray = getFactory().Core().createNewArray();
167 				newArray.setType(ctExpression.getType());
168 				newArray.setParent(this);
169 				newArray.addElement(ctExpression);
170 				newArray.addElement(expression);
171 				elementValues.put(elementName, newArray);
172 			}
173 		} else {
174 			// Add the new value.
175 			expression.setParent(this);
176 			getFactory().getEnvironment().getModelChangeListener().onMapAdd(this, CtRole.VALUE, this.elementValues, elementName, expression);
177 			elementValues.put(elementName, expression);
178 		}
179 		return (T) this;
180 	}

202 	private Class<?> getElementType(String name) {
203 		// Try by CT reflection
204 		CtType<?> t = getAnnotationType().getDeclaration();
205 		if (t != null) {
206 			CtMethod<?> method = t.getMethod(name);
207 			return method.getType().getActualClass();
208 		}
209 		// Try with RT reflection
210 		Class<?> c = getAnnotationType().getActualClass();
211 		for (Method m : c.getMethods()) {
212 			if (m.getName().equals(name)) {
213 				return m.getReturnType();
214 			}
215 		}
216 		return null;
217 	}

224 	private CtExpression getDefaultExpression(String fieldName) {
225 		CtExpression ret = null;
226 		CtAnnotationType<?> at = (CtAnnotationType<?>) getAnnotationType().getDeclaration();
227 		if (at != null) {
228 			CtAnnotationMethod<?> f = (CtAnnotationMethod) at.getMethod(fieldName);
229 			ret = f.getDefaultExpression();
230 		}
231 		return ret;
232 	}

254 	@Override
255 	public Object getValueAsObject(String key) {
256 		CtExpression expr = getWrappedValue(key);
257 
258 		// no such value, per the contract of the method
259 		if (expr == null) {
260 			return null;
261 		}
262 
263 		Object ret = EvalHelper.convertElementToRuntimeObject(expr);
264 		Class<?> type = getElementType(key);
265 		return forceObjectToType(ret, type);
266 	}

268 	private Object forceObjectToType(Object ret, Class<?> type) {
269 		if (type.isPrimitive()) {
270 			if ((type == boolean.class) && (ret.getClass() != boolean.class)) {
271 				return Boolean.parseBoolean(ret.toString());
272 			} else if ((type == byte.class) && (ret.getClass() != byte.class)) {
273 				return Byte.parseByte(ret.toString());
274 			} else if ((type == char.class) && (ret.getClass() != char.class)) {
275 				return ret.toString().charAt(0);
276 			} else if ((type == double.class) && (ret.getClass() != double.class)) {
277 				return Double.parseDouble(ret.toString());
278 			} else if ((type == float.class) && (ret.getClass() != float.class)) {
279 				return Float.parseFloat(ret.toString());
280 			} else if ((type == int.class) && (ret.getClass() != int.class)) {
281 				return Integer.parseInt(ret.toString());
282 			} else if ((type == long.class) && (ret.getClass() != long.class)) {
283 				return Long.parseLong(ret.toString());
284 			} else if (type == short.class && ret.getClass() != short.class) {
285 				return Short.parseShort(ret.toString());
286 			}
287 		}
288 		return ret;
289 	}

292 	private CtExpression getValueAsExpression(String key) {
293 
294 		// get specified field in annotation directly
295 		CtExpression ret = this.elementValues.get(key);
296 		if (ret != null) {
297 			return ret;
298 		}
299 
300 		// get default value in annotation declaration in source code
301 		ret = getDefaultExpression(key);
302 		if (ret != null) {
303 			return ret;
304 		}
305 
306 		// get default value in annotation declaration in classpath
307 		Object value = getReflectValue(key);
308 		if (value != null) {
309 			return convertValueToExpression(value);
310 		}
311 
312 		return null;
313 	}

315 	@Override
316 	public <T extends CtExpression> T getWrappedValue(String key) {
317 		CtExpression ctExpression = this.getValue(key);
318 
319 			CtTypeReference typeReference = this.getAnnotationType();
320 			CtType type = typeReference.getTypeDeclaration();
321 			if (type != null) {
322 				CtMethod method = type.getMethod(key);
323 				if (method != null) {
324 					CtTypeReference returnType = method.getType();
325 					if (returnType instanceof CtArrayTypeReference && !(ctExpression instanceof CtNewArray)) {
326 						CtNewArray newArray = getFactory().Core().createNewArray();
327 						CtArrayTypeReference typeReference2 = this.getFactory().createArrayTypeReference();
328 						typeReference2.setComponentType(ctExpression.getType().clone());
329 						newArray.setType(typeReference2);
330 						newArray.addElement(ctExpression.clone());
331 						return (T) newArray;
332 					}
333 				}
334 		}
335 		return (T) ctExpression;
336 	}

339 		Map<String, Object> res = new TreeMap<>();
340 		for (Entry<String, CtExpression> elementValue : elementValues.entrySet()) {

342 		}

351 	@Override
352 	public Map<String, CtExpression> getAllValues() {
353 		Map<String, CtExpression> values = new TreeMap();
354 		// first, we put the default values
355 		CtAnnotationType<?> annotationType = (CtAnnotationType) getAnnotationType().getTypeDeclaration();
356 		for (CtAnnotationMethod m : annotationType.getAnnotationMethods()) {
357 			values.put(m.getSimpleName(), m.getDefaultExpression());
358 		}
359 
360 		// we override the values with ones of this expression
361 		values.putAll(elementValues);
362 		return Collections.unmodifiableMap(values);
363 	}

365 	private Object getReflectValue(String fieldname) {
366 		try {
367 			Class<?> c = getAnnotationType().getActualClass();
368 			Method m = c.getMethod(fieldname);
369 			return m.getDefaultValue();
370 		} catch (Exception e) {
371 			throw new RuntimeException(e);
372 		}
373 	}

418 	@Override
419 	@SuppressWarnings("unchecked")
420 	public A getActualAnnotation() {
421 		class AnnotationInvocationHandler implements InvocationHandler {
422 			CtAnnotation<? extends Annotation> annotation;
423 
424 			AnnotationInvocationHandler(CtAnnotation<? extends Annotation> annotation) {
425 				this.annotation = annotation;
426 			}
427 
428 			@Override
429 			public Object invoke(Object proxy, Method method, Object[] args) {
430 				String fieldname = method.getName();
431 				if ("toString".equals(fieldname)) {
432 					return CtAnnotationImpl.this.toString();
433 				} else if ("annotationType".equals(fieldname)) {
434 					return annotation.getAnnotationType().getActualClass();
435 				}
436 				return getValueAsObject(fieldname);
437 			}
438 		}
439 		return (A) Proxy.newProxyInstance(annotationType.getActualClass().getClassLoader(), new Class[] { annotationType.getActualClass() }, new AnnotationInvocationHandler(this));
440 	}
