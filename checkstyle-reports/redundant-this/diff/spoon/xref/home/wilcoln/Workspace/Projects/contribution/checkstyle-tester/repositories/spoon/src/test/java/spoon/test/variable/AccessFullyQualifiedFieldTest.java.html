43  	private String buildResourceAndReturnResult(String pathResource, String output) {
44  		Launcher spoon = new Launcher();
45  		//spoon.setArgs(new String[]{"--with-imports"});
46  		spoon.addInputResource(pathResource);
47  		spoon.setSourceOutputDirectory(output);
48  		spoon.run();
49  		PrettyPrinter prettyPrinter = spoon.createPrettyPrinter();
50  
51  		CtType element = spoon.getFactory().Class().getAll().get(0);
52  		List<CtType<?>> toPrint = new ArrayList<>();
53  		toPrint.add(element);
54  
55  		prettyPrinter.calculate(element.getPosition().getCompilationUnit(), toPrint);
56  		return prettyPrinter.getResult();
57  	}

59  	@Test
60  	public void testNoFQNWhenShadowedByField() {
61  		// contract: no fully qualified name if top package is shadowed by a field variable
62  
63  		String pathResource = "src/test/java/spoon/test/variable/testclasses/BurritosFielded.java";
64  		String output = "target/spooned-" + this.getClass().getSimpleName() + "-Field/";
65  		String result = this.buildResourceAndReturnResult(pathResource, output);
66  
67  		assertTrue("The java file should contain import for Launcher", result.contains("import spoon.Launcher;"));
68  		assertTrue("The xx variable is attributed with Launcher.SPOONED_CLASSES", result.contains("xx = Launcher.SPOONED_CLASSES"));
69  		canBeBuilt(output, 7);
70  	}

121 	@Test
122 	public void testStaticImportWithAutoImport() {
123 		String output = "target/spooned-" + this.getClass().getSimpleName() + "-MultiAutoImport/";
124 		String pathResource = "src/test/java/spoon/test/variable/testclasses/MultiBurritos.java";
125 
126 		Launcher spoon = new Launcher();
127 		spoon.setArgs(new String[]{"--with-imports"});
128 		spoon.addInputResource(pathResource);
129 		spoon.setSourceOutputDirectory(output);
130 		spoon.run();
131 		PrettyPrinter prettyPrinter = spoon.createPrettyPrinter();
132 
133 		CtType element = spoon.getFactory().Class().getAll().get(0);
134 		List<CtType<?>> toPrint = new ArrayList<>();
135 		toPrint.add(element);
136 
137 		prettyPrinter.calculate(element.getPosition().getCompilationUnit(), toPrint);
138 		String result = prettyPrinter.getResult();
139 		assertTrue("The result should contain a static import for spoon.Launcher.SPOONED_CLASSES", result.contains("import static spoon.Launcher.SPOONED_CLASSES;"));
140 		assertTrue("The variable x should be assigned with only SPOONED_CLASSES", result.contains("Object x = SPOONED_CLASSES;"));
141 		assertTrue("The result should not contain a static import for spoon.test.variable.testclasses.ForStaticVariables.foo as it is in the same package", !result.contains("import static spoon.test.variable.testclasses.ForStaticVariables.foo;"));
142 		assertTrue("The result should not contain a import static for spoon.test.variable.testclasses.MultiBurritos.toto as it is in the same class", !result.contains("import static spoon.test.variable.testclasses.MultiBurritos.toto;"));
143 		assertTrue("The result should not contain a FQN for toto", !result.contains("spoon.test.variable.testclasses.MultiBurritos.toto();"));
144 		assertTrue("The result should not contain a FQN for spoon access", !result.contains("spoon.test.variable.testclasses.MultiBurritos.spoon = \"truc\";"));
145 		assertTrue("The result should not contain a FQN for foo", !result.contains("spoon.test.variable.testclasses.ForStaticVariables.foo();"));
146 
147 		canBeBuilt(output, 7);
148 	}
