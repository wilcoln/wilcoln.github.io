6   package spoon.template;
7   
8   import spoon.reflect.code.CtBlock;
9   import spoon.reflect.declaration.CtClass;
10  import spoon.reflect.declaration.CtMethod;
11  import spoon.reflect.declaration.CtType;
12  
13  /**
14   * This class represents a template parameter that defines a block statement
15   * directly expressed in Java (must return an expression of type <code>R</code>
16   * ).
17   *
18   * <p>
19   * To define a new block template parameter, you must subclass this class and
20   * implement the {@link #block()} method, which actually defines the Java block.
21   * It corresponds to a {@link spoon.reflect.code.CtBlock}.
22   */
23  public abstract class TypedBlockTemplateParameter<R> implements TemplateParameter<R> {
24  
25  	/**
26  	 * Creates a new block template parameter.
27  	 */
28  	public TypedBlockTemplateParameter() {
29  	}
30  
31  	/**
32  	 * This method must be implemented to define the template block.
33  	 */
34  	public abstract R block();
35  
36  	@SuppressWarnings("unchecked")
37  	public CtBlock<R> getSubstitution(CtType<?> targetType) {
38  		CtClass<?> c;
39  		c = targetType.getFactory().Class().get(this.getClass());
40  		if (c == null) {
41  			c = targetType.getFactory().Class().get(this.getClass());
42  		}
43  		CtMethod m = c.getMethod("block");
44  		if (this instanceof Template) {
45  			return Substitution.substitute(targetType, (Template<?>) this, m.getBody());
46  		}
47  		return m.getBody().clone();
48  	}
49  
50  	@Override
51  	public R S() {
52  		return null;
53  	}
54  }
