19  package com.opensymphony.xwork2.config.entities;
20  
21  import com.opensymphony.xwork2.util.location.Located;
22  import com.opensymphony.xwork2.util.location.Location;
23  
24  import java.io.Serializable;
25  import java.util.Collections;
26  import java.util.LinkedHashMap;
27  import java.util.Map;
28  
29  
30  /**
31   * Configuration class for result types.
32   *
33   * <p>
34   * In the xml configuration file this is defined as the <code>result-type</code> tag.
35   * </p>
36   *
37   * @author Mike
38   * @author Rainer Hermanns
39   * @author Neo
40   */
41  public class ResultTypeConfig extends Located implements Serializable {
42  
43      protected String className;
44      protected String name;
45      protected String defaultResultParam;
46      protected Map<String,String> params;
47  
48      protected ResultTypeConfig(String name, String className) {
49          this.name = name;
50          this.className = className;
51          params = new LinkedHashMap<>();
52      }
53  
54      protected ResultTypeConfig(ResultTypeConfig orig) {
55          this.name = orig.name;
56          this.className = orig.className;
57          this.defaultResultParam = orig.defaultResultParam;
58          this.params = new LinkedHashMap<>(orig.params);
59          this.location = orig.location;
60      }
61  
62      public String getDefaultResultParam() {
63          return this.defaultResultParam;
64      }
65  
66      public String getClassName() {
67          return className;
68      }
69  
70      public String getName() {
71          return name;
72      }
73  
74      public Map<String,String> getParams() {
75          return Collections.unmodifiableMap(this.params);
76      }
77  
78      @Override
79      public boolean equals(Object o) {
80          if (this == o) return true;
81          if (o == null || getClass() != o.getClass()) return false;
82  
83          final ResultTypeConfig that = (ResultTypeConfig) o;
84  
85          if (className != null ? !className.equals(that.className) : that.className != null) return false;
86          if (name != null ? !name.equals(that.name) : that.name != null) return false;
87          if (params != null ? !params.equals(that.params) : that.params != null) return false;
88  
89          return true;
90      }
91  
92      @Override
93      public int hashCode() {
94          int result;
95          result = (className != null ? className.hashCode() : 0);
96          result = 29 * result + (name != null ? name.hashCode() : 0);
97          result = 29 * result + (params != null ? params.hashCode() : 0);
98          return result;
99      }
100 
101     @Override
102     public String toString() {
103         return "ResultTypeConfig: [" + name + "] => [" + className + "] " +
104                 "with defaultParam [" + defaultResultParam + "] with params " + params;
105     }
106 
107     /**
108      * The builder for this object.  An instance of this object is the only way to construct a new instance.  The
109      * purpose is to enforce the immutability of the object.  The methods are structured in a way to support chaining.
110      * After setting any values you need, call the {@link #build()} method to create the object.
111      */
112     public static final class Builder {
113         protected ResultTypeConfig target;
114 
115         public Builder(String name, String className) {
116             target = new ResultTypeConfig(name, className);
117         }
118 
119         public Builder(ResultTypeConfig orig) {
120             target = new ResultTypeConfig(orig);
121         }
122 
123         public Builder name(String name) {
124             target.name = name;
125             return this;
126         }
127 
128         public Builder className(String name) {
129             target.className = name;
130             return this;
131         }
132 
133          public Builder addParam(String name, String value) {
134             target.params.put(name, value);
135             return this;
136         }
137 
138         public Builder addParams(Map<String,String> params) {
139             target.params.putAll(params);
140             return this;
141         }
142 
143         public Builder defaultResultParam(String defaultResultParam) {
144             target.defaultResultParam = defaultResultParam;
145             return this;
146         }
147 
148         public Builder location(Location loc) {
149             target.location = loc;
150             return this;
151         }
152 
153         public ResultTypeConfig build() {
154             ResultTypeConfig result = target;
155             target = new ResultTypeConfig(target);
156             return result;
157         }
158     }
159 }
