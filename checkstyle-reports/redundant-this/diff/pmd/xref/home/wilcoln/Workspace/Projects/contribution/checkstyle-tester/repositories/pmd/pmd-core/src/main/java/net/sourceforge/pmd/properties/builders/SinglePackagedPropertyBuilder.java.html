16  @Deprecated
17  public abstract class SinglePackagedPropertyBuilder<V, T extends SinglePackagedPropertyBuilder<V, T>>
18      extends SingleValuePropertyBuilder<V, T> {
19  
20      protected String[] legalPackageNames;
21  
22  
23      public SinglePackagedPropertyBuilder(String name) {
24          super(name);
25      }
26  
27  
28      /**
29       * Specify the allowed package prefixes.
30       *
31       * @param packs The package prefixes
32       *
33       * @return The same builder
34       */
35      @SuppressWarnings("unchecked")
36      public T legalPackageNames(String... packs) {
37          if (packs != null) {
38              this.legalPackageNames = Arrays.copyOf(packs, packs.length);
39          }
40          return (T) this;
41      }
42  
43  
44      /**
45       * Specify the allowed package prefixes.
46       *
47       * @param packs The package prefixes
48       *
49       * @return The same builder
50       */
51      @SuppressWarnings("unchecked")
52      public T legalPackageNames(Collection<String> packs) {
53          if (packs != null) {
54              this.legalPackageNames = packs.toArray(new String[0]);
55          }
56          return (T) this;
57      }
