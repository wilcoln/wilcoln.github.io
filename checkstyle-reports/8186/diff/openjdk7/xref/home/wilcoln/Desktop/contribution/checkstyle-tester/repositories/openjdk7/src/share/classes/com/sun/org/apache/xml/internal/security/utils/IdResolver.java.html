34  /**
35   * Purpose of this class is to enable the XML Parser to keep track of ID
36   * attributes. This is done by 'registering' attributes of type ID at the
37   * IdResolver. This is necessary if we create a document from scratch and we
38   * sign some resources with a URI using a fragent identifier...
39   * <BR />
40   * The problem is that if you do not validate a document, you cannot use the
41   * <CODE>getElementByID</CODE> functionality. So this modules uses some implicit
42   * knowledge on selected Schemas and DTDs to pick the right Element for a given
43   * ID: We know that all <CODE>@Id</CODE> attributes in an Element from the XML
44   * Signature namespace are of type <CODE>ID</CODE>.
45   *
46   * @author $Author: mullan $
47   * @see <A HREF="http://www.xml.com/lpt/a/2001/11/07/id.html">"Identity Crisis" on xml.com</A>
48   */
49  public class IdResolver {
50  
51      /** {@link java.util.logging} logging facility */
52      private static java.util.logging.Logger log =
53          java.util.logging.Logger.getLogger(IdResolver.class.getName());
54  
55      private static WeakHashMap docMap = new WeakHashMap();
56  
57      /**
58       * Constructor IdResolver
59       *
60       */
61      private IdResolver() {
62         // we don't allow instantiation
63      }

65      /**
66       * Method registerElementById
67       *
68       * @param element the element to register
69       * @param idValue the value of the ID attribute
70       */
71      public static void registerElementById(Element element, String idValue) {
72          Document doc = element.getOwnerDocument();
73          WeakHashMap elementMap;
74          synchronized (docMap) {
75              elementMap = (WeakHashMap) docMap.get(doc);
76              if (elementMap == null) {
77                  elementMap = new WeakHashMap();
78                  docMap.put(doc, elementMap);
79              }
80          }
81          elementMap.put(idValue, new WeakReference(element));
82      }
83  
84      /**
85       * Method registerElementById
86       *
87       * @param element the element to register
88       * @param id the ID attribute
89       */
90      public static void registerElementById(Element element, Attr id) {
91          IdResolver.registerElementById(element, id.getNodeValue());
92      }
93  
94      /**
95       * Method getElementById
96       *
97       * @param doc the document
98       * @param id the value of the ID
99       * @return the element obtained by the id, or null if it is not found.
100      */
101     public static Element getElementById(Document doc, String id) {
102 
103         Element result = IdResolver.getElementByIdType(doc, id);
104 
105         if (result != null) {
106             log.log(java.util.logging.Level.FINE,
107             "I could find an Element using the simple getElementByIdType method: "
108             + result.getTagName());
109 
110             return result;
111         }
112 
113         result = IdResolver.getElementByIdUsingDOM(doc, id);
114 
115         if (result != null) {
116             log.log(java.util.logging.Level.FINE,
117              "I could find an Element using the simple getElementByIdUsingDOM method: "
118             + result.getTagName());
119 
120             return result;
121         }
122         // this must be done so that Xalan can catch ALL namespaces
123         //XMLUtils.circumventBug2650(doc);
124         result = IdResolver.getElementBySearching(doc, id);
125 
126         if (result != null) {
127             IdResolver.registerElementById(result, id);
128 
129             return result;
130         }
131 
132         return null;
133     }

136     /**
137      * Method getElementByIdUsingDOM
138      *
139      * @param doc the document
140      * @param id the value of the ID
141      * @return the element obtained by the id, or null if it is not found.
142      */
143     private static Element getElementByIdUsingDOM(Document doc, String id) {
144         if (log.isLoggable(java.util.logging.Level.FINE))
145             log.log(java.util.logging.Level.FINE, "getElementByIdUsingDOM() Search for ID " + id);
146         return doc.getElementById(id);
147     }
148 
149     /**
150      * Method getElementByIdType
151      *
152      * @param doc the document
153      * @param id the value of the ID
154      * @return the element obtained by the id, or null if it is not found.
155      */
156     private static Element getElementByIdType(Document doc, String id) {
157         if (log.isLoggable(java.util.logging.Level.FINE))
158             log.log(java.util.logging.Level.FINE, "getElementByIdType() Search for ID " + id);
159         WeakHashMap elementMap;
160         synchronized (docMap) {
161             elementMap = (WeakHashMap) docMap.get(doc);
162         }
163         if (elementMap != null) {
164             WeakReference weakReference = (WeakReference) elementMap.get(id);
165             if (weakReference != null) {
166                 return (Element) weakReference.get();
167             }
168         }
169         return null;
170     }

172     private static java.util.List names;
173     private static int namesLength;
174     static {
175         String namespaces[]={
176             Constants.SignatureSpecNS,
177             EncryptionConstants.EncryptionSpecNS,
178             "http://schemas.xmlsoap.org/soap/security/2000-12",
179             "http://www.w3.org/2002/03/xkms#",
180             "urn:oasis:names:tc:SAML:1.0:assertion",
181             "urn:oasis:names:tc:SAML:1.0:protocol"
182         };
183         names = Arrays.asList(namespaces);
184         namesLength = names.size();
185     }
186 
187 
188     private static Element getElementBySearching(Node root,String id) {
189         Element []els=new Element[namesLength + 1];
190         getEl(root,id,els);
191         for (int i=0;i<els.length;i++) {
192             if (els[i]!=null) {
193                 return els[i];
194             }
195         }
196         return null;
197     }
198 
199     private static int getEl(Node currentNode,String id,Element []els) {
200         Node sibling=null;
201         Node parentNode=null;
202         do {
203                 switch (currentNode.getNodeType()) {
204                 case Node.DOCUMENT_FRAGMENT_NODE :
205                 case Node.DOCUMENT_NODE :
206                         sibling= currentNode.getFirstChild();
207                         break;
208 
209 
210                 case Node.ELEMENT_NODE :
211                         Element currentElement = (Element) currentNode;
212                         if (isElement(currentElement, id, els)==1)
213                                 return 1;
214                         sibling= currentNode.getFirstChild();
215                         if (sibling==null) {
216                             if (parentNode != null) {
217                                         sibling= currentNode.getNextSibling();
218                                     }
219                         } else {
220                                 parentNode=currentElement;
221                         }
222                         break;
223         } while (sibling==null  && parentNode!=null) {
224                         sibling=parentNode.getNextSibling();
225                         parentNode=parentNode.getParentNode();
226                         if (!(parentNode instanceof Element)) {
227                                 parentNode=null;
228                         }
229                 }
230                 if (sibling==null)
231                         return 1;
232                 currentNode=sibling;
233                 sibling=currentNode.getNextSibling();
234         } while(true);
235 
236     }
237     public static int isElement(Element el, String id,Element[] els) {
238         if (!el.hasAttributes()) {
239                 return 0;
240         }
241         NamedNodeMap ns=el.getAttributes();
242         int elementIndex=names.indexOf(el.getNamespaceURI());
243             elementIndex=(elementIndex<0) ? namesLength : elementIndex;
244         for (int length=ns.getLength(), i=0; i<length; i++) {
245                 Attr n=(Attr)ns.item(i);
246                 String s=n.getNamespaceURI();
247 
248                     int index=s==null ? elementIndex : names.indexOf(n.getNamespaceURI());
249                     index=(index<0) ? namesLength : index;
250                     String name=n.getLocalName();
251                     if (name.length()>2)
252                         continue;
253                     String value=n.getNodeValue();
254                     if (name.charAt(0)=='I') {
255                         char ch=name.charAt(1);
256                         if (ch=='d' && value.equals(id)) {
257                                 els[index]=el;
258                                 if (index==0) {
259                                         return 1;
260                                 }
261                         } else if (ch=='D' &&value.endsWith(id)) {
262                                 if (index!=3) {
263                                     index=namesLength;
264                                 }
265                                 els[index]=el;
266                         }
267                     } else if ( "id".equals(name) && value.equals(id) ) {
268                         if (index!=2) {
269                                 index=namesLength;
270                         }
271                         els[index]=el;
272                     }
273         }
274         //For an element namespace search for importants
275         if ((elementIndex==3)&&(
276                     el.getAttribute("OriginalRequestID").equals(id) ||
277                     el.getAttribute("RequestID").equals(id) ||
278                     el.getAttribute("ResponseID").equals(id))) {
279                     els[3]=el;
280         } else if ((elementIndex==4)&&(
281                     el.getAttribute("AssertionID").equals(id))) {
282                     els[4]=el;
283         } else if ((elementIndex==5)&&(
284                     el.getAttribute("RequestID").equals(id) ||
285                     el.getAttribute("ResponseID").equals(id))) {
286                     els[5]=el;
287                  }
288         return 0;
289     }
