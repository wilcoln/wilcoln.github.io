17  package org.orekit.attitudes;
18  
19  import org.hipparchus.geometry.euclidean.threed.Vector3D;
20  import org.hipparchus.util.FastMath;
21  import org.hipparchus.util.MathArrays;
22  import org.orekit.bodies.Ellipsoid;
23  import org.orekit.errors.OrekitException;
24  import org.orekit.frames.Frame;
25  import org.orekit.time.AbsoluteDate;
26  import org.orekit.utils.PVCoordinatesProvider;
27  import org.orekit.utils.TimeStampedPVCoordinates;
28  
29  /**
30   * This class handles body center pointing attitude provider.
31  
32   * <p>
33   * This class represents the attitude provider where the satellite z axis is
34   * pointing to the body frame center.</p>
35   * <p>
36   * The object <code>BodyCenterPointing</code> is guaranteed to be immutable.
37   * </p>
38   * @see     GroundPointing
39   * @author V&eacute;ronique Pommier-Maurussane
40   */
41  public class BodyCenterPointing extends GroundPointing {
42  
43      /** Serializable UID. */
44      private static final long serialVersionUID = 20150529L;
45  
46      /** Body ellipsoid.  */
47      private final Ellipsoid ellipsoid;
48  
49      /** Creates new instance.
50       * @param inertialFrame frame in which orbital velocities are computed
51       * @param shape Body shape
52       * @exception OrekitException if the frame specified is not a pseudo-inertial frame
53       * @since 7.1
54       */
55      public BodyCenterPointing(final Frame inertialFrame, final Ellipsoid shape)
56          throws OrekitException {
57          super(inertialFrame, shape.getFrame());
58          this.ellipsoid = shape;
59      }
60  
61      /** {@inheritDoc} */
62      @Override
63      protected TimeStampedPVCoordinates getTargetPV(final PVCoordinatesProvider pvProv,
64                                                     final AbsoluteDate date, final Frame frame)
65          throws OrekitException {
66  
67          // spacecraft coordinates in body frame
68          final TimeStampedPVCoordinates scInBodyFrame = pvProv.getPVCoordinates(date, getBodyFrame());
69  
70          // central projection to ground (NOT the classical nadir point)
71          final double u     = scInBodyFrame.getPosition().getX() / ellipsoid.getA();
72          final double v     = scInBodyFrame.getPosition().getY() / ellipsoid.getB();
73          final double w     = scInBodyFrame.getPosition().getZ() / ellipsoid.getC();
74          final double d2    = u * u + v * v + w * w;
75          final double d     = FastMath.sqrt(d2);
76          final double ratio = 1.0 / d;
77          final Vector3D projectedP = new Vector3D(ratio, scInBodyFrame.getPosition());
78  
79          // velocity
80          final double uDot     = scInBodyFrame.getVelocity().getX() / ellipsoid.getA();
81          final double vDot     = scInBodyFrame.getVelocity().getY() / ellipsoid.getB();
82          final double wDot     = scInBodyFrame.getVelocity().getZ() / ellipsoid.getC();
83          final double dDot     = MathArrays.linearCombination(u, uDot, v, vDot, w, wDot) / d;
84          final double ratioDot = -dDot / d2;
85          final Vector3D projectedV = new Vector3D(ratio,    scInBodyFrame.getVelocity(),
86                                                   ratioDot, scInBodyFrame.getPosition());
87  
88          // acceleration
89          final double uDotDot      = scInBodyFrame.getAcceleration().getX() / ellipsoid.getA();
90          final double vDotDot      = scInBodyFrame.getAcceleration().getY() / ellipsoid.getB();
91          final double wDotDot      = scInBodyFrame.getAcceleration().getZ() / ellipsoid.getC();
92          final double dDotDot      = (MathArrays.linearCombination(u, uDotDot, v, vDotDot, w, wDotDot) +
93                                       uDot * uDot + vDot * vDot + wDot * wDot - dDot * dDot) / d;
94          final double ratioDotDot  = (2 * dDot * dDot - d * dDotDot) / (d * d2);
95          final Vector3D projectedA = new Vector3D(ratio,        scInBodyFrame.getAcceleration(),
96                                                   2 * ratioDot, scInBodyFrame.getVelocity(),
97                                                   ratioDotDot,  scInBodyFrame.getPosition());
98  
99          final TimeStampedPVCoordinates projected =
100                 new TimeStampedPVCoordinates(date, projectedP, projectedV, projectedA);
101         return getBodyFrame().getTransformTo(frame, date).transformPVCoordinates(projected);
102 
103     }
104 
105 }
