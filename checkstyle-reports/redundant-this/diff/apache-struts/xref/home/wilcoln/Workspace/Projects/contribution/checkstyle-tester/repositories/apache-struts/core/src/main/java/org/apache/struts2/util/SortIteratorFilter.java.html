19  package org.apache.struts2.util;
20  
21  import com.opensymphony.xwork2.Action;
22  import org.apache.logging.log4j.LogManager;
23  import org.apache.logging.log4j.Logger;
24  
25  import java.util.*;
26  
27  /**
28   * A bean that takes a source and comparator then attempt to sort the source
29   * utilizing the comparator. It is being used in SortIteratorTag.
30   *
31   * @see org.apache.struts2.views.jsp.iterator.SortIteratorTag
32   */
33  public class SortIteratorFilter extends IteratorFilterSupport implements Iterator, Action {
34      private static final Logger LOG = LogManager.getLogger(IteratorGenerator.class);
35  
36      Comparator comparator;
37      Iterator iterator;
38      List list;
39  
40      // Attributes ----------------------------------------------------
41      Object source;
42  
43  
44      public void setComparator(Comparator aComparator) {
45          this.comparator = aComparator;
46      }
47  
48      public List getList() {
49          return list;
50      }
51  
52      // Public --------------------------------------------------------
53      public void setSource(Object anIterator) {
54          source = anIterator;
55      }
56  
57      // Action implementation -----------------------------------------
58      public String execute() {
59          if (source == null) {
60              return ERROR;
61          } else {
62              try {
63                  if (!MakeIterator.isIterable(source)) {
64                      LOG.warn("Cannot create SortIterator for source: {}", source);
65                      return ERROR;
66                  }
67  
68                  list = new ArrayList();
69  
70                  Iterator i = MakeIterator.convert(source);
71  
72                  while (i.hasNext()) {
73                      list.add(i.next());
74                  }
75  
76                  // Sort it
77                  Collections.sort(list, comparator);
78                  iterator = list.iterator();
79  
80                  return SUCCESS;
81              } catch (Exception e) {
82                  LOG.warn("Error creating sort iterator.", e);
83                  return ERROR;
84              }
85          }
86      }
87  
88      // Iterator implementation ---------------------------------------
89      public boolean hasNext() {
90          return (source == null) ? false : iterator.hasNext();
91      }
92  
93      public Object next() {
94          return iterator.next();
95      }
96  
97      public void remove() {
98          throw new UnsupportedOperationException("Remove is not supported in SortIteratorFilter.");
99      }
100 }
