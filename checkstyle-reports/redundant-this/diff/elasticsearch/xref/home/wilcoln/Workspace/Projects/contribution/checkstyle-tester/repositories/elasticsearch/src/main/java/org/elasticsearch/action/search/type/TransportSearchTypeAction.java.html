109         protected BaseAsyncAction(SearchRequest request, ActionListener<SearchResponse> listener) {
110             this.request = request;
111             this.listener = listener;
112 
113             this.clusterState = clusterService.state();
114             nodes = clusterState.nodes();
115 
116             clusterState.blocks().globalBlockedRaiseException(ClusterBlockLevel.READ);
117 
118             String[] concreteIndices = clusterState.metaData().concreteIndices(request.indicesOptions(), request.indices());
119 
120             for (String index : concreteIndices) {
121                 clusterState.blocks().indexBlockedRaiseException(ClusterBlockLevel.READ, index);
122             }
123 
124             Map<String, Set<String>> routingMap = clusterState.metaData().resolveSearchRouting(request.routing(), request.indices());
125 
126             shardsIts = clusterService.operationRouting().searchShards(clusterState, request.indices(), concreteIndices, routingMap, request.preference());
127             expectedSuccessfulOps = shardsIts.size();
128             // we need to add 1 for non active partition, since we count it in the total!
129             expectedTotalOps = shardsIts.totalSizeWith1ForEmpty();
130 
131             firstResults = new AtomicArray<>(shardsIts.size());
132             // Not so nice, but we need to know if there're nodes below the supported version
133             // and if so fall back to classic scroll (based on from). We need to check every node
134             // because we don't to what nodes we end up sending the request (shard may fail or relocate)
135             boolean useSlowScroll = false;
136             if (request.scroll() != null) {
137                 for (DiscoveryNode discoveryNode : clusterState.nodes()) {
138                     if (discoveryNode.getVersion().before(ParsedScrollId.SCROLL_SEARCH_AFTER_MINIMUM_VERSION)) {
139                         useSlowScroll = true;
140                     }
141                 }
142             }
143             this.useSlowScroll = useSlowScroll;
144         }
145 
146         public void start() {
147             if (expectedSuccessfulOps == 0) {
148                 // no search shards to search on, bail with empty response (it happens with search across _all with no indices around and consistent with broadcast operations)
149                 listener.onResponse(new SearchResponse(InternalSearchResponse.empty(), null, 0, 0, buildTookInMillis(), ShardSearchFailure.EMPTY_ARRAY));
150                 return;
151             }
152             request.beforeStart();
153             int shardIndex = -1;
154             for (final ShardIterator shardIt : shardsIts) {
155                 shardIndex++;
156                 final ShardRouting shard = shardIt.nextOrNull();
157                 if (shard != null) {
158                     performFirstPhase(shardIndex, shardIt, shard);
159                 } else {
160                     // really, no shards active in this group
161                     onFirstPhaseResult(shardIndex, null, null, shardIt, new NoShardAvailableActionException(shardIt.shardId()));
162                 }
163             }
164         }
165 
166         void performFirstPhase(final int shardIndex, final ShardIterator shardIt, final ShardRouting shard) {
167             if (shard == null) {
168                 // no more active shards... (we should not really get here, but just for safety)
169                 onFirstPhaseResult(shardIndex, null, null, shardIt, new NoShardAvailableActionException(shardIt.shardId()));
170             } else {
171                 final DiscoveryNode node = nodes.get(shard.currentNodeId());
172                 if (node == null) {
173                     onFirstPhaseResult(shardIndex, shard, null, shardIt, new NoShardAvailableActionException(shardIt.shardId()));
174                 } else {
175                     String[] filteringAliases = clusterState.metaData().filteringAliases(shard.index(), request.indices());
176                     sendExecuteFirstPhase(node, internalSearchRequest(shard, shardsIts.size(), request, filteringAliases, startTime(), useSlowScroll), new SearchServiceListener<FirstResult>() {
177                         @Override
178                         public void onResult(FirstResult result) {
179                             onFirstPhaseResult(shardIndex, shard, result, shardIt);
180                         }
181 
182                         @Override
183                         public void onFailure(Throwable t) {
184                             onFirstPhaseResult(shardIndex, shard, node.id(), shardIt, t);
185                         }
186                     });
187                 }
188             }
189         }
190 
191         void onFirstPhaseResult(int shardIndex, ShardRouting shard, FirstResult result, ShardIterator shardIt) {
192             result.shardTarget(new SearchShardTarget(shard.currentNodeId(), shard.index(), shard.id()));
193             processFirstPhaseResult(shardIndex, shard, result);
194             // we need to increment successful ops first before we compare the exit condition otherwise if we
195             // are fast we could concurrently update totalOps but then preempt one of the threads which can
196             // cause the successor to read a wrong value from successfulOps if second phase is very fast ie. count etc.
197             successfulOps.incrementAndGet();
198             // increment all the "future" shards to update the total ops since we some may work and some may not...
199             // and when that happens, we break on total ops, so we must maintain them
200             final int xTotalOps = totalOps.addAndGet(shardIt.remaining() + 1);
201             if (xTotalOps == expectedTotalOps) {
202                 try {
203                     innerMoveToSecondPhase();
204                 } catch (Throwable e) {
205                     if (logger.isDebugEnabled()) {
206                         logger.debug(shardIt.shardId() + ": Failed to execute [" + request + "] while moving to second phase", e);
207                     }
208                     raiseEarlyFailure(new ReduceSearchPhaseException(firstPhaseName(), "", e, buildShardFailures()));
209                 }
210             } else if (xTotalOps > expectedTotalOps) {
211                 raiseEarlyFailure(new ElasticsearchIllegalStateException("unexpected higher total ops [" + xTotalOps + "] compared to expected [" + expectedTotalOps + "]"));
212             }
213         }
214 
215         void onFirstPhaseResult(final int shardIndex, @Nullable ShardRouting shard, @Nullable String nodeId, final ShardIterator shardIt, Throwable t) {
216             // we always add the shard failure for a specific shard instance
217             // we do make sure to clean it on a successful response from a shard
218             SearchShardTarget shardTarget = new SearchShardTarget(nodeId, shardIt.shardId().getIndex(), shardIt.shardId().getId());
219             addShardFailure(shardIndex, shardTarget, t);
220 
221             if (totalOps.incrementAndGet() == expectedTotalOps) {
222                 if (logger.isDebugEnabled()) {
223                     if (t != null && !TransportActions.isShardNotAvailableException(t)) {
224                         if (shard != null) {
225                             logger.debug(shard.shortSummary() + ": Failed to execute [" + request + "]", t);
226                         } else {
227                             logger.debug(shardIt.shardId() + ": Failed to execute [" + request + "]", t);
228                         }
229                     } else if (logger.isTraceEnabled()) {
230                         logger.trace("{}: Failed to execute [{}]", t, shard, request);
231                     }
232                 }
233                 if (successfulOps.get() == 0) {
234                     if (logger.isDebugEnabled()) {
235                         logger.debug("All shards failed for phase: [{}]", t, firstPhaseName());
236                     }
237                     // no successful ops, raise an exception
238                     raiseEarlyFailure(new SearchPhaseExecutionException(firstPhaseName(), "all shards failed", buildShardFailures()));
239                 } else {
240                     try {
241                         innerMoveToSecondPhase();
242                     } catch (Throwable e) {
243                         raiseEarlyFailure(new ReduceSearchPhaseException(firstPhaseName(), "", e, buildShardFailures()));
244                     }
245                 }
246             } else {
247                 final ShardRouting nextShard = shardIt.nextOrNull();
248                 final boolean lastShard = nextShard == null;
249                 // trace log this exception
250                 if (logger.isTraceEnabled()) {
251                     logger.trace(executionFailureMsg(shard, shardIt, request, lastShard), t);
252                 }
253                 if (!lastShard) {
254                     try {
255                         performFirstPhase(shardIndex, shardIt, nextShard);
256                     } catch (Throwable t1) {
257                         onFirstPhaseResult(shardIndex, shard, shard.currentNodeId(), shardIt, t1);
258                     }
259                 } else {
260                     // no more shards active, add a failure
261                     if (logger.isDebugEnabled() && !logger.isTraceEnabled()) { // do not double log this exception
262                         if (t != null && !TransportActions.isShardNotAvailableException(t)) {
263                             logger.debug(executionFailureMsg(shard, shardIt, request, lastShard), t);
264                         }
265                     }
266                 }
267             }
268         }
269 
270         private String executionFailureMsg(@Nullable ShardRouting shard, final ShardIterator shardIt, SearchRequest request, boolean lastShard) {
271             if (shard != null) {
272                 return shard.shortSummary() + ": Failed to execute [" + request + "] lastShard [" + lastShard + "]";
273             } else {
274                 return shardIt.shardId() + ": Failed to execute [" + request + "] lastShard [" + lastShard + "]";
275             }
276         }

278         protected final ShardSearchFailure[] buildShardFailures() {
279             AtomicArray<ShardSearchFailure> shardFailures = this.shardFailures;
280             if (shardFailures == null) {
281                 return ShardSearchFailure.EMPTY_ARRAY;
282             }
283             List<AtomicArray.Entry<ShardSearchFailure>> entries = shardFailures.asList();
284             ShardSearchFailure[] failures = new ShardSearchFailure[entries.size()];
285             for (int i = 0; i < failures.length; i++) {
286                 failures[i] = entries.get(i).value;
287             }
288             return failures;
289         }

291         protected final void addShardFailure(final int shardIndex, @Nullable SearchShardTarget shardTarget, Throwable t) {
292             // we don't aggregate shard failures on non active shards (but do keep the header counts right)
293             if (TransportActions.isShardNotAvailableException(t)) {
294                 return;
295             }
296 
297             // lazily create shard failures, so we can early build the empty shard failure list in most cases (no failures)
298             if (shardFailures == null) {
299                 synchronized (shardFailuresMutex) {
300                     if (shardFailures == null) {
301                         shardFailures = new AtomicArray<>(shardsIts.size());
302                     }
303                 }
304             }
305             ShardSearchFailure failure = shardFailures.get(shardIndex);
306             if (failure == null) {
307                 shardFailures.set(shardIndex, new ShardSearchFailure(t, shardTarget));
308             } else {
309                 // the failure is already present, try and not override it with an exception that is less meaningless
310                 // for example, getting illegal shard state
311                 if (TransportActions.isReadOverrideException(t)) {
312                     shardFailures.set(shardIndex, new ShardSearchFailure(t, shardTarget));
313                 }
314             }
315         }
316 
317         private void raiseEarlyFailure(Throwable t) {
318             for (AtomicArray.Entry<FirstResult> entry : firstResults.asList()) {
319                 try {
320                     DiscoveryNode node = nodes.get(entry.value.shardTarget().nodeId());
321                     if (node != null) { // should not happen (==null) but safeguard anyhow
322                         searchService.sendFreeContext(node, entry.value.id(), request);
323                     }
324                 } catch (Throwable t1) {
325                     logger.trace("failed to release context", t1);
326                 }
327             }
328             listener.onFailure(t);
329         }

331         /**
332          * Releases shard targets that are not used in the docsIdsToLoad.
333          */
334         protected void releaseIrrelevantSearchContexts(AtomicArray<? extends QuerySearchResultProvider> queryResults,
335                                                        AtomicArray<IntArrayList> docIdsToLoad) {
336             if (docIdsToLoad == null) {
337                 return;
338             }
339             // we only release search context that we did not fetch from if we are not scrolling
340             if (request.scroll() == null) {
341                 for (AtomicArray.Entry<? extends QuerySearchResultProvider> entry : queryResults.asList()) {
342                     if (docIdsToLoad.get(entry.index) == null) {
343                         try {
344                             DiscoveryNode node = nodes.get(entry.value.queryResult().shardTarget().nodeId());
345                             if (node != null) { // should not happen (==null) but safeguard anyhow
346                                 searchService.sendFreeContext(node, entry.value.queryResult().id(), request);
347                             }
348                         } catch (Throwable t1) {
349                             logger.trace("failed to release context", t1);
350                         }
351                     }
352                 }
353             }
354         }

356         protected ShardFetchSearchRequest createFetchRequest(QuerySearchResult queryResult, AtomicArray.Entry<IntArrayList> entry, ScoreDoc[] lastEmittedDocPerShard) {
357             if (lastEmittedDocPerShard != null) {
358                 ScoreDoc lastEmittedDoc = lastEmittedDocPerShard[entry.index];
359                 return new ShardFetchSearchRequest(request, queryResult.id(), entry.value, lastEmittedDoc);
360             } else {
361                 return new ShardFetchSearchRequest(request, queryResult.id(), entry.value);
362             }
363         }

367         protected final void processFirstPhaseResult(int shardIndex, ShardRouting shard, FirstResult result) {
368             firstResults.set(shardIndex, result);
369 
370             if (logger.isTraceEnabled()) {
371                 logger.trace("got first-phase result from {}", result != null ? result.shardTarget() : null);
372             }
373 
374             // clean a previous error on this shard group (note, this code will be serialized on the same shardIndex value level
375             // so its ok concurrency wise to miss potentially the shard failures being created because of another failure
376             // in the #addShardFailure, because by definition, it will happen on *another* shardIndex
377             AtomicArray<ShardSearchFailure> shardFailures = this.shardFailures;
378             if (shardFailures != null) {
379                 shardFailures.set(shardIndex, null);
380             }
381         }
382 
383         final void innerMoveToSecondPhase() throws Exception {
384             if (logger.isTraceEnabled()) {
385                 StringBuilder sb = new StringBuilder();
386                 boolean hadOne = false;
387                 for (int i = 0; i < firstResults.length(); i++) {
388                     FirstResult result = firstResults.get(i);
389                     if (result == null) {
390                         continue; // failure
391                     }
392                     if (hadOne) {
393                         sb.append(",");
394                     } else {
395                         hadOne = true;
396                     }
397                     sb.append(result.shardTarget());
398                 }
399 
400                 logger.trace("Moving to second phase, based on results from: {} (cluster state version: {})", sb, clusterState.version());
401             }
402             moveToSecondPhase();
403         }
