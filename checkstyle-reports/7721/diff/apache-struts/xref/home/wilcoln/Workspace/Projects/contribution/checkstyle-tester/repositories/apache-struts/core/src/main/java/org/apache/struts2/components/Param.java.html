19  package org.apache.struts2.components;
20  
21  import com.opensymphony.xwork2.util.ValueStack;
22  import org.apache.commons.lang3.StringUtils;
23  import org.apache.struts2.StrutsException;
24  import org.apache.struts2.views.annotations.StrutsTag;
25  import org.apache.struts2.views.annotations.StrutsTagAttribute;
26  
27  import java.io.Writer;
28  
29  /**
30   * <!-- START SNIPPET: javadoc -->
31   * <p>This tag can be used to parameterize other tags.<br>
32   * The include tag and bean tag are examples of such tags.
33   * </p>
34   *
35   * <p>
36   * The parameters can be added with or without a name as key.
37   * If the tag provides a name attribute the parameters are added using the
38   * {@link Component#addParameter(String, Object) addParamter} method.
39   * For unnamed parameters the Tag must implement the {@link UnnamedParametric} interface defined in
40   * this class (e.g. The TextTag does this).
41   * </p>
42   *
43   * <p>
44   * This tag has the following two paramters.
45   * </p>
46   * <!-- START SNIPPET: params -->
47   * <ul>
48   *      <li>name (String) - the name of the parameter</li>
49   *      <li>value (Object) - the value of the parameter</li>
50   *      <li>suppressEmptyParameters (boolean) - whether to suppress this parameter if empty</li>
51   * </ul>
52   * <!-- END SNIPPET: params -->
53   * <p>
54   * <b>Note:</b>
55   * When you declare the param tag, the value can be defined in either a <tt>value</tt> attribute or
56   * as text between the start and end tag. Struts behaves a bit different according to these two situations.
57   * This is best illustrated using an example:
58   * <br>&lt;param name="color"&gt;blue&lt;/param&gt; &lt;-- (A) --&gt;
59   * <br>&lt;param name="color" value="blue"/&gt; &lt;-- (B) --&gt;
60   * <br>In the first situation (A) the value would be evaluated to the stack as a <tt>java.lang.String</tt> object.
61   * And in situation (B) the value would be evaluated to the stack as a <tt>java.lang.Object</tt> object.
62   * <br>For more information see <a href="https://issues.apache.org/jira/browse/WW-808">WW-808</a>.
63   * </p>
64   * <!-- END SNIPPET: javadoc -->
65   *
66   * <p><b>Examples</b></p>
67   * <!-- START SNIPPET: example -->
68   * <pre>
69   * &lt;ui:component&gt;
70   *  &lt;ui:param name="key"     value="[0]"/&gt;
71   *  &lt;ui:param name="value"   value="[1]"/&gt;
72   *  &lt;ui:param name="context" value="[2]"/&gt;
73   * &lt;/ui:component&gt;
74   * </pre>
75   *
76   * <p>
77   * Whether to suppress empty parameters:
78   * </p>
79   *
80   * <pre>
81   * &lt;s:url action="eventAdd"&gt;
82   *   &lt;s:param name="bean.searchString" value="%{bean.searchString}" /&gt;
83   *   &lt;s:param name="bean.filter" value="%{bean.filter}" /&gt;
84   *   &lt;s:param name="bean.pageNum" value="%{pager.pageNumber}" suppressEmptyParameters="true" /&gt;
85   * &lt;/s:url&gt;
86   * </pre>
87   * <!-- END SNIPPET: example -->
88   * <p>
89   * <!-- START SNIPPET: exampledescription -->
90   * where the key will be the identifier and the value the result of an OGNL expression run against the current
91   * ValueStack.
92   * <!-- END SNIPPET: exampledescription -->
93   * </p>
94   * <p>
95   * This second example demonstrates how the text tag can use parameters from this param tag.
96   * </p>
97   * <!-- START SNIPPET: example2 -->
98   * <pre>
99   * &lt;s:text name="cart.total.cost"&gt;
100  *     &lt;s:param value="#session.cartTotal"/&gt;
101  * &lt;/s:text&gt;
102  * </pre>
103  * <!-- END SNIPPET: example2 -->
104  *
105  *
106  * @see Include
107  * @see Bean
108  * @see Text
109  *
110  */
111 @StrutsTag(name="param", tldTagClass="org.apache.struts2.views.jsp.ParamTag", description="Parametrize other tags")
112 public class Param extends Component {
113 
114     protected String name;
115     protected String value;
116     protected boolean suppressEmptyParameters;
117 
118     public Param(ValueStack stack) {
119         super(stack);
120     }
121 
122     public boolean end(Writer writer, String body) {
123         Component component = findAncestor(Component.class);
124         if (value != null) {
125             if (component instanceof UnnamedParametric) {
126                 ((UnnamedParametric) component).addParameter(findValue(value));
127             } else {
128                 String name = findString(this.name);
129 
130                 if (name == null) {
131                     throw new StrutsException("No name found for following expression: " + this.name);
132                 }
133 
134                 Object value = findValue(this.value);
135                 if (suppressEmptyParameters) {
136                     if (value != null && StringUtils.isNotBlank(value.toString())) {
137                         component.addParameter(name, value);
138                     } else {
139                         component.addParameter(name, null);
140                     }
141                 } else if (value == null || StringUtils.isBlank(value.toString())) {
142                     component.addParameter(name, "");
143                 } else {
144                     component.addParameter(name, value);
145                 }
146             }
147         } else {
148             if (component instanceof UnnamedParametric) {
149                 ((UnnamedParametric) component).addParameter(body);
150             } else {
151                 if (!(suppressEmptyParameters && StringUtils.isBlank(body))) {
152                     component.addParameter(findString(name), body);
153                 } else {
154                     component.addParameter(findString(name), null);
155                 }
156             }
157         }
158 
159         return super.end(writer, "");
160     }
161     
162     public boolean usesBody() {
163         return true;
164     }
165 
166     @StrutsTagAttribute(description="Name of Parameter to set")
167     public void setName(String name) {
168         this.name = name;
169     }
170 
171     @StrutsTagAttribute(description="Value expression for Parameter to set", defaultValue="The value of evaluating provided name against stack")
172     public void setValue(String value) {
173         this.value = value;
174     }
175     
176     @StrutsTagAttribute(description="Whether to suppress empty parameters", type="Boolean", defaultValue="false")
177     public void setSuppressEmptyParameters(boolean suppressEmptyParameters) {
178         this.suppressEmptyParameters = suppressEmptyParameters;
179     }
180     
181     /**
182      * <p>
183      * Tags can implement this to support nested param tags without the <tt>name</tt> attribute.
184      * </p>
185      * <p>
186      * The {@link Text TextTag} uses this approach. For unnamed parameters an example is given in the class
187      * javadoc for {@link Param ParamTag}.
188      * </p>
189      */
190     public interface UnnamedParametric {
191 
192         /**
193          * Adds the given value as a parameter to the outer tag.
194          * @param value  the value
195          */
196         public void addParameter(Object value);
197     }
198 
199 }
