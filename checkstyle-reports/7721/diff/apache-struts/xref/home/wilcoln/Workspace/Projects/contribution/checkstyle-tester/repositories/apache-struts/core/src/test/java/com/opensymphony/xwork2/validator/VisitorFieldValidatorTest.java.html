39      @Override
40      public void setUp() throws Exception {
41          super.setUp();
42  
43          action = container.inject(VisitorValidatorTestAction.class);
44  
45          TestBean bean = action.getBean();
46          Calendar cal = new GregorianCalendar(1900, 1, 1);
47          bean.setBirth(cal.getTime());
48          bean.setCount(-1);
49  
50          ActionConfig config = new ActionConfig.Builder("", "name", "").build();
51          ActionInvocation invocation = EasyMock.createNiceMock(ActionInvocation.class);
52          ActionProxy proxy = EasyMock.createNiceMock(ActionProxy.class);
53  
54          EasyMock.expect(invocation.getProxy()).andReturn(proxy).anyTimes();
55          EasyMock.expect(invocation.getAction()).andReturn(action).anyTimes();
56          EasyMock.expect(invocation.invoke()).andReturn(Action.SUCCESS).anyTimes();
57          EasyMock.expect(proxy.getMethod()).andReturn("execute").anyTimes();
58          EasyMock.expect(proxy.getConfig()).andReturn(config).anyTimes();
59          
60  
61          EasyMock.replay(invocation);
62          EasyMock.replay(proxy);
63  
64          ActionContext.getContext().setActionInvocation(invocation);
65      }

67      public void testArrayValidation() throws Exception {
68          TestBean[] beanArray = action.getTestBeanArray();
69          TestBean testBean = beanArray[0];
70          testBean.setName("foo");
71          validate("validateArray");
72  
73          assertTrue(action.hasFieldErrors());
74  
75          Map<String, List<String>> fieldErrors = action.getFieldErrors();
76  
77          //4 errors for the array, one for context
78          assertEquals(5, fieldErrors.size());
79          assertTrue(fieldErrors.containsKey("testBeanArray[1].name"));
80  
81          //the error from the action should be there too
82          assertTrue(fieldErrors.containsKey("context"));
83  
84          List<String> errors = fieldErrors.get("testBeanArray[1].name");
85          assertEquals(1, errors.size());
86          errors = fieldErrors.get("testBeanArray[2].name");
87          assertEquals(1, errors.size());
88          errors = fieldErrors.get("testBeanArray[3].name");
89          assertEquals(1, errors.size());
90          errors = fieldErrors.get("testBeanArray[4].name");
91          assertEquals(1, errors.size());
92      }

94      public void testBeanMessagesUseBeanResourceBundle() throws Exception {
95          validate("beanMessageBundle");
96          assertTrue(action.hasFieldErrors());
97  
98          Map<String, List<String>> fieldErrors = action.getFieldErrors();
99          assertTrue(fieldErrors.containsKey("bean.count"));
100 
101         List<String> beanCountMessages = fieldErrors.get("bean.count");
102         assertEquals(1, beanCountMessages.size());
103 
104         String beanCountMessage = beanCountMessages.get(0);
105         assertEquals("bean: TestBean model: Count must be between 1 and 100, current value is -1.", beanCountMessage);
106     }

180     public void testVisitorChildConversionValidation() throws Exception {
181         //add conversion error
182         Map<String, ConversionData> conversionErrors = new HashMap<>();
183         conversionErrors.put("bean.child.count", new ConversionData("bar", Integer.class));
184         ActionContext.getContext().setConversionErrors(conversionErrors);
185 
186         validate("visitorChildValidation");
187         assertTrue(action.hasFieldErrors());
188 
189         Map<String, List<String>> fieldErrors = action.getFieldErrors();
190         assertEquals(6, fieldErrors.size());
191         assertTrue(!fieldErrors.containsKey("bean.count"));
192         assertTrue(fieldErrors.containsKey("bean.name"));
193         assertTrue(fieldErrors.containsKey("bean.birth"));
194 
195         assertTrue(fieldErrors.containsKey("bean.child.name"));
196         assertTrue(fieldErrors.containsKey("bean.child.birth"));
197 
198         //the error from the action should be there too
199         assertTrue(fieldErrors.containsKey("context"));
200 
201         //nested visitor conversion error
202         assertTrue(fieldErrors.containsKey("bean.child.count"));
203     }
