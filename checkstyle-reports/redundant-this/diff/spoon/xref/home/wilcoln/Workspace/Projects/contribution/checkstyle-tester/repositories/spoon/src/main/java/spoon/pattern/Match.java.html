6   package spoon.pattern;
7   
8   import java.util.List;
9   import java.util.Map;
10  
11  import spoon.SpoonException;
12  import spoon.reflect.declaration.CtElement;
13  import spoon.support.util.ImmutableMap;
14  
15  /**
16   * Represents a single match of {@link Pattern}
17   */
18  public class Match {
19  	private final List<?> matchingElements;
20  	private final ImmutableMap parameters;
21  
22  	public Match(List<?> matches, ImmutableMap parameters) {
23  		this.parameters = parameters;
24  		this.matchingElements = matches;
25  	}
26  	/**
27  	 * @return {@link List} of elements, which match to the Pattern.
28  	 * Use {@link #getMatchingElement()} if the {@link Pattern} matches single root element.
29  	 * But when {@link Pattern} contains sequence of root elements, then this is the right way how to get them all
30  	 */
31  	public List<CtElement> getMatchingElements() {
32  		return getMatchingElements(CtElement.class);
33  	}
34  	/**
35  	 * Same like {@link #getMatchingElement()} but additionally it checks that each matching element is instance of `clazz`
36  	 * @param clazz the required type of all elements.
37  	 * @return a {@link List} typed to `clazz` or throws {@link SpoonException} if Pattern matched different elements
38  	 */
39  	@SuppressWarnings("unchecked")
40  	public <T> List<T> getMatchingElements(Class<T> clazz) {
41  		for (Object object : matchingElements) {
42  			if (object != null && clazz.isInstance(object) == false) {
43  				throw new SpoonException("Match contains a " + object.getClass() + " which cannot be cast to " + clazz);
44  			}
45  		}
46  		return (List<T>) matchingElements;
47  	}
48  
49  	/**
50  	 * @return a matching element of a {@link Pattern}
51  	 * It fails if {@link Pattern} is designed to match sequence of elements. In such case use {@link #getMatchingElements()}
52  	 */
53  	public CtElement getMatchingElement() {
54  		return getMatchingElement(CtElement.class, true);
55  	}
56  	/**
57  	 * Same like {@link #getMatchingElement()}, but checks that matching element is expected class and casts returned value to that type
58  	 * @param clazz required type
59  	 * @return matched element cast to `clazz`
60  	 */
61  	public <T> T getMatchingElement(Class<T> clazz) {
62  		return getMatchingElement(clazz, true);
63  	}
64  
65  	/**
66  	 * @param clazz the Class of returned element. throws SpoonException if matching value is not assignable to `clazz`
67  	 * @param failIfMany if there is more then one matching element and `failIfMany` == true, then it throws SpoonException.
68  	 * @return first matching element
69  	 */
70  	private <T> T getMatchingElement(Class<T> clazz, boolean failIfMany) {
71  		if (matchingElements.isEmpty()) {
72  			return null;
73  		}
74  		if (failIfMany && matchingElements.size() != 1) {
75  			throw new SpoonException("There is more then one match");
76  		}
77  		Object object = matchingElements.get(0);
78  		if (object != null && clazz.isInstance(object) == false) {
79  			throw new SpoonException("Match contains a " + object.getClass() + " which cannot be cast to " + clazz);
80  		}
81  		return clazz.cast(object);
82  	}
83  
84  	/**
85  	 * @return {@link ImmutableMap} with values of {@link Pattern} parameters, which fits to current match
86  	 */
87  	public ImmutableMap getParameters() {
88  		return parameters;
89  	}
90  	/**
91  	 * @return {@link Map} with values of {@link Pattern} parameters, which fits to current match
92  	 */
93  	public Map<String, Object> getParametersMap() {
94  		return parameters.asMap();
95  	}
96  
97  	@Override
98  	public String toString() {
99  		StringBuilder sb = new StringBuilder();
100 		sb.append("{\n");
101 		sb.append(parameters.toString());
102 		sb.append("\n}\n----------");
103 		for (int i = 0; i < matchingElements.size(); i++) {
104 			sb.append("\n").append(i + 1).append(") ").append(matchingElements.get(i));
105 		}
106 		return sb.toString();
107 	}
108 }
