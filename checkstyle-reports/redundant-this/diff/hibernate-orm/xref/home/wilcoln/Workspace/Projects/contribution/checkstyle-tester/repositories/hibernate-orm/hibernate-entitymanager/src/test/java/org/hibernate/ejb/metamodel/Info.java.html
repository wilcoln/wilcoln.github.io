24  package org.hibernate.ejb.metamodel;
25  import javax.persistence.Column;
26  import javax.persistence.Entity;
27  import javax.persistence.Id;
28  import javax.persistence.JoinTable;
29  import javax.persistence.OneToOne;
30  import javax.persistence.Table;
31  
32  /**
33   * TODO : javadoc
34   *
35   * @author Steve Ebersole
36   */
37  @Entity
38  @Table(name = "INFO_TABLE")
39  public class Info implements java.io.Serializable {
40  	private String id;
41  	private String street;
42  	private String city;
43  	private String state;
44  	private String zip;
45  	private Spouse spouse;
46  
47  	public Info() {
48  	}
49  
50  	public Info(String v1, String v2, String v3, String v4, String v5) {
51  		id = v1;
52  		street = v2;
53  		city = v3;
54  		state = v4;
55  		zip = v5;
56  	}
57  
58  	public Info(
59  			String v1, String v2, String v3, String v4,
60  			String v5, Spouse v6) {
61  		id = v1;
62  		street = v2;
63  		city = v3;
64  		state = v4;
65  		zip = v5;
66  		spouse = v6;
67  	}
68  
69  	@Id
70  	@Column(name = "ID")
71  	public String getId() {
72  		return id;
73  	}
74  
75  	public void setId(String v) {
76  		id = v;
77  	}
78  
79  	@Column(name = "INFOSTREET")
80  	public String getStreet() {
81  		return street;
82  	}
83  
84  	public void setStreet(String v) {
85  		street = v;
86  	}
87  
88  	@Column(name = "INFOSTATE")
89  	public String getState() {
90  		return state;
91  	}
92  
93  	public void setState(String v) {
94  		state = v;
95  	}
96  
97  	@Column(name = "INFOCITY")
98  	public String getCity() {
99  		return city;
100 	}
101 
102 	public void setCity(String v) {
103 		city = v;
104 	}
105 
106 	@Column(name = "INFOZIP")
107 	public String getZip() {
108 		return zip;
109 	}
110 
111 	public void setZip(String v) {
112 		zip = v;
113 	}
114 
115 	@OneToOne(mappedBy = "info") @JoinTable( name = "INFO_SPOUSE_TABLE" )
116 	public Spouse getSpouse() {
117 		return spouse;
118 	}
119 
120 	public void setSpouse(Spouse v) {
121 		this.spouse = v;
122 	}
123 
124 }
