34  	public void testFinder() {
35  		ClassPathFinder finder = new ClassPathFinder();
36  		finder.setPattern("**/xwork-test-wildcard-*.xml");
37  		Vector<String> found = finder.findMatches();
38  		assertEquals(found.contains("com/opensymphony/xwork2/config/providers/xwork-test-wildcard-1.xml"), true );
39  		assertEquals(found.contains("com/opensymphony/xwork2/config/providers/xwork-test-wildcard-2.xml"), true );
40  		assertEquals(found.contains("com/opensymphony/xwork2/config/providers/xwork-test-wildcard-include.xml"), true );
41  		assertEquals(found.contains("com/opensymphony/xwork2/config/providers/xwork-test-results.xml"), false);
42  		
43  		ClassPathFinder finder2 = new ClassPathFinder();
44  		finder2.setPattern("com/*/xwork2/config/providers/xwork-test-wildcard-1.xml");
45  		Vector<String> found2 = finder2.findMatches();
46  		assertEquals(found2.contains("com/opensymphony/xwork2/config/providers/xwork-test-wildcard-1.xml"), true);
47  		assertEquals(found2.contains("com/opensymphony/xwork2/config/providers/xwork-test-wildcard-2.xml"), false);
48  		
49  		ClassPathFinder finder3 = new ClassPathFinder();
50  		finder3.setPattern("com/opensymphony/xwork2/config/providers/xwork-test-wildcard-1.xml");
51  		Vector<String> found3 = finder3.findMatches();
52  		assertEquals(found3.contains("com/opensymphony/xwork2/config/providers/xwork-test-wildcard-1.xml"), true);
53  		assertEquals(found3.contains("com/opensymphony/xwork2/config/providers/xwork-test-wildcard-2.xml"), false);
54  		
55  		ClassPathFinder finder4 = new ClassPathFinder();
56  		finder4.setPattern("no/matches/*");
57  		Vector<String> found4 = finder4.findMatches();
58  		assertEquals(found4.isEmpty(), true);
59  		
60  	}

62      public void testFinderNotURLClassLoader() throws Exception {
63          NotURLClassLoader loader = new NotURLClassLoader(Thread.currentThread().getContextClassLoader());
64          Thread.currentThread().setContextClassLoader(loader);
65  
66          try {
67              Class<?> clazz = loader.loadClass(ClassPathFinderTest.class.getName());
68              Object test = clazz.getConstructor().newInstance();
69  
70              clazz.getMethod("testFinder").invoke(test);
71          } finally {
72              Thread.currentThread().setContextClassLoader(loader.parentClassLoader);
73          }
74      }
75  
76  
77  	private class NotURLClassLoader extends ClassLoader {
78          private Map<String, Class<?>> loadedClasses = new HashMap<>();
79          private ClassLoader parentClassLoader;
80  
81          NotURLClassLoader(ClassLoader parentClassLoader) {
82              super(null);
83              this.parentClassLoader = parentClassLoader;
84          }
85  
86          @Override
87          protected Class<?> findClass(String name) throws ClassNotFoundException {
88              if (!loadedClasses.containsKey(name)) {
89                  try {
90                      byte[] classBits = IOUtils.toByteArray(ClassLoader.getSystemResourceAsStream(
91                              name.replace('.', '/') + ".class"));
92                      loadedClasses.put(name, defineClass(name, classBits, 0, classBits.length));
93                  } catch (IOException e) {
94                      throw new ClassNotFoundException("class " + name + " is not findable", e);
95                  } catch (Exception e) {
96                      loadedClasses.put(name, parentClassLoader.loadClass(name));
97                  }
98              }
99  
100             return loadedClasses.get(name);
101         }
102 
103         @Override
104         protected Enumeration<URL> findResources(String name) throws IOException {
105             return parentClassLoader.getResources(name);
106         }
107     }
