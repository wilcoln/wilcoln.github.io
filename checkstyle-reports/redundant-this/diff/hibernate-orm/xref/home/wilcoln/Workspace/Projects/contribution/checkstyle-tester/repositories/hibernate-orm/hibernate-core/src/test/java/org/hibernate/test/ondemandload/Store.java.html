24  package org.hibernate.test.ondemandload;
25  
26  import javax.persistence.CascadeType;
27  import javax.persistence.Column;
28  import javax.persistence.Entity;
29  import javax.persistence.FetchType;
30  import javax.persistence.Id;
31  import javax.persistence.OneToMany;
32  import javax.persistence.Version;
33  import java.io.Serializable;
34  import java.util.ArrayList;
35  import java.util.List;
36  
37  @Entity
38  public class Store implements Serializable {
39  	private int id;
40  	private String name;
41  	private List<Inventory> inventories = new ArrayList<Inventory>();
42  	private int version;
43  
44  	protected Store() {
45  	}
46  
47  	public Store(int id) {
48  		this.id = id;
49  	}
50  
51  	@Id
52  	@Column(name = "ID")
53  	public Integer getId() {
54  		return id;
55  	}
56  
57  	private void setId(int id) {
58  		this.id = id;
59  	}
60  
61  	@Column(name = "NAME")
62  	public String getName() {
63  		return name;
64  	}
65  
66  	public Store setName(String name) {
67  		this.name = name;
68  		return this;
69  	}
70  
71  	@OneToMany(mappedBy = "store", cascade = CascadeType.ALL, fetch = FetchType.LAZY)
72  	public List<Inventory> getInventories() {
73  		return inventories;
74  	}
75  
76  	public void setInventories(List<Inventory> inventories) {
77  		this.inventories = inventories;
78  	}
79  
80  	public Inventory addInventoryProduct(Product product) {
81  		final Inventory inventory = new Inventory( this, product );
82  		this.inventories.add( inventory );
83  		return inventory;
84  	}
85  
86  	@Version
87  	public int getVersion() {
88  		return version;
89  	}
90  
91  	private void setVersion(int version) {
92  		this.version = version;
93  	}
94  }
