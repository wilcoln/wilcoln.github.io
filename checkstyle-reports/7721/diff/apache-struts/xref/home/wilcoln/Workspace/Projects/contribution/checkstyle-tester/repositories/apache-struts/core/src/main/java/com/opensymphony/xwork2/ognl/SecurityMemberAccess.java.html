19  package com.opensymphony.xwork2.ognl;
20  
21  import com.opensymphony.xwork2.util.ProxyUtil;
22  import ognl.MemberAccess;
23  import org.apache.logging.log4j.LogManager;
24  import org.apache.logging.log4j.Logger;
25  
26  import java.lang.reflect.AccessibleObject;
27  import java.lang.reflect.Field;
28  import java.lang.reflect.Member;
29  import java.lang.reflect.Modifier;
30  import java.util.Collections;
31  import java.util.Map;
32  import java.util.Set;
33  import java.util.regex.Matcher;
34  import java.util.regex.Pattern;
35  
36  /**
37   * Allows access decisions to be made on the basis of whether a member is static or not.
38   * Also blocks or allows access to properties.
39   */
40  public class SecurityMemberAccess implements MemberAccess {
41  
42      private static final Logger LOG = LogManager.getLogger(SecurityMemberAccess.class);
43  
44      private final boolean allowStaticFieldAccess;
45      private final boolean allowStaticMethodAccess;
46      private Set<Pattern> excludeProperties = Collections.emptySet();
47      private Set<Pattern> acceptProperties = Collections.emptySet();
48      private Set<Class<?>> excludedClasses = Collections.emptySet();
49      private Set<Pattern> excludedPackageNamePatterns = Collections.emptySet();
50      private Set<String> excludedPackageNames = Collections.emptySet();
51      private boolean disallowProxyMemberAccess;
52  
53      /**
54       * SecurityMemberAccess
55       *   - access decisions based on whether member is static (or not)
56       *   - block or allow access to properties (configurable-after-construction)
57       * 
58       * @param allowStaticMethodAccess
59       * @param allowStaticFieldAccess
60       */
61      public SecurityMemberAccess(boolean allowStaticMethodAccess, boolean allowStaticFieldAccess) {
62          this.allowStaticMethodAccess = allowStaticMethodAccess;
63          this.allowStaticFieldAccess = allowStaticFieldAccess;
64      }
65  
66      public final boolean getAllowStaticMethodAccess() {
67          return allowStaticMethodAccess;
68      }
69  
70      public final boolean getAllowStaticFieldAccess() {
71          return allowStaticFieldAccess;
72      }
73  
74      @Override
75      public Object setup(Map context, Object target, Member member, String propertyName) {
76          Object result = null;
77  
78          if (isAccessible(context, target, member, propertyName)) {
79              final AccessibleObject accessible = (AccessibleObject) member;
80  
81              if (!accessible.isAccessible()) {
82                  result = Boolean.FALSE;
83                  accessible.setAccessible(true);
84              }
85          }
86          return result;
87      }
88  
89      @Override
90      public void restore(Map context, Object target, Member member, String propertyName, Object state) {
91          if (state != null) {
92              final AccessibleObject accessible = (AccessibleObject) member;
93              final boolean stateboolean = ((Boolean) state).booleanValue();  // Using twice (avoid unboxing)
94              if (!stateboolean) {
95                  accessible.setAccessible(stateboolean);
96              }
97              else {
98                  throw new IllegalArgumentException("Improper restore state [" + stateboolean + "] for target [" + target +
99                                                     "], member [" + member + "], propertyName [" + propertyName + "]");
100             }
101         }
102     }
103 
104     @Override
105     public boolean isAccessible(Map context, Object target, Member member, String propertyName) {
106         LOG.debug("Checking access for [target: {}, member: {}, property: {}]", target, member, propertyName);
107 
108         final int memberModifiers = member.getModifiers();
109 
110         if (!checkPublicMemberAccess(memberModifiers)) {
111             LOG.warn("Access to non-public [{}] is blocked!", member);
112             return false;
113         }
114 
115         if (!checkStaticFieldAccess(member, memberModifiers)) {
116             LOG.warn("Access to static field [{}] is blocked!", member);
117             return false;
118         }
119 
120         if (checkEnumAccess(target, member)) {
121             LOG.trace("Allowing access to enum: target [{}], member [{}]", target, member);
122             return true;
123         }
124 
125         if (!checkStaticMethodAccess(member, memberModifiers)) {
126             LOG.warn("Access to static method [{}] is blocked!", member);
127             return false;
128         }
129 
130         final Class memberClass = member.getDeclaringClass();
131 
132         if (isClassExcluded(memberClass)) {
133             LOG.warn("Declaring class of member type [{}] is excluded!", member);
134             return false;
135         }
136 
137         // target can be null in case of accessing static fields, since OGNL 3.2.8
138         final Class targetClass = Modifier.isStatic(memberModifiers) ? memberClass : target.getClass();
139 
140         if (isPackageExcluded(targetClass.getPackage(), memberClass.getPackage())) {
141             LOG.warn("Package [{}] of target class [{}] of target [{}] or package [{}] of member [{}] are excluded!", targetClass.getPackage(), targetClass,
142                     target, memberClass.getPackage(), member);
143             return false;
144         }
145 
146         if (isClassExcluded(targetClass)) {
147             LOG.warn("Target class [{}] of target [{}] is excluded!", targetClass, target);
148             return false;
149         }
150 
151         if (disallowProxyMemberAccess && ProxyUtil.isProxyMember(member, target)) {
152             LOG.warn("Access to proxy is blocked! Target class [{}] of target [{}], member [{}]", targetClass, target, member);
153             return false;
154         }
155 
156         return isAcceptableProperty(propertyName);
157     }
158 
159     /**
160      * Check access for static method (via modifiers).
161      * 
162      * Note: For non-static members, the result is always true.
163      * 
164      * @param member
165      * @param memberModifiers
166      * 
167      * @return
168      */
169     protected boolean checkStaticMethodAccess(Member member, int memberModifiers) {
170         if (Modifier.isStatic(memberModifiers) && !(member instanceof Field)) {
171             if (allowStaticMethodAccess) {
172                 LOG.debug("Support for accessing static methods [member: {}] is deprecated!", member);
173             }
174             return allowStaticMethodAccess;
175         } else {
176             return true;
177         }
178     }
179 
180     /**
181      * Check access for static field (via modifiers).
182      * 
183      * Note: For non-static members, the result is always true.
184      * 
185      * @param member
186      * @param memberModifiers
187      * 
188      * @return
189      */
190     protected boolean checkStaticFieldAccess(Member member, int memberModifiers) {
191         if (Modifier.isStatic(memberModifiers) && member instanceof Field) {
192             return allowStaticFieldAccess;
193         } else {
194             return true;
195         }
196     }
197 
198    /**
199      * Check access for public members (via modifiers)
200      * 
201      * Returns true if-and-only-if the member is public.
202      * 
203      * @param memberModifiers
204      * 
205      * @return
206      */
207     protected boolean checkPublicMemberAccess(int memberModifiers) {
208         return Modifier.isPublic(memberModifiers);
209     }
210 
211     protected boolean checkEnumAccess(Object target, Member member) {
212         if (target instanceof Class) {
213             final Class clazz = (Class) target;
214             if (Enum.class.isAssignableFrom(clazz) && member.getName().equals("values")) {
215                 return true;
216             }
217         }
218         return false;
219     }
220 
221     protected boolean isPackageExcluded(Package targetPackage, Package memberPackage) {
222         if (targetPackage == null || memberPackage == null) {
223             LOG.warn("The use of the default (unnamed) package is discouraged!");
224         }
225         
226         String targetPackageName = targetPackage == null ? "" : targetPackage.getName();
227         String memberPackageName = memberPackage == null ? "" : memberPackage.getName();
228 
229         for (Pattern pattern : excludedPackageNamePatterns) {
230             if (pattern.matcher(targetPackageName).matches() || pattern.matcher(memberPackageName).matches()) {
231                 return true;
232             }
233         }
234 
235         targetPackageName = targetPackageName + ".";
236         memberPackageName = memberPackageName + ".";
237 
238         for (String packageName: excludedPackageNames) {
239             if (targetPackageName.startsWith(packageName) || memberPackageName.startsWith(packageName)) {
240                 return true;
241             }
242         }
243 
244         return false;
245     }
246 
247     protected boolean isClassExcluded(Class<?> clazz) {
248         if (clazz == Object.class || (clazz == Class.class && !allowStaticMethodAccess)) {
249             return true;
250         }
251         for (Class<?> excludedClass : excludedClasses) {
252             if (clazz.isAssignableFrom(excludedClass)) {
253                 return true;
254             }
255         }
256         return false;
257     }
258 
259     protected boolean isAcceptableProperty(String name) {
260         return name == null || ((!isExcluded(name)) && isAccepted(name));
261     }
262 
263     protected boolean isAccepted(String paramName) {
264         if (!this.acceptProperties.isEmpty()) {
265             for (Pattern pattern : acceptProperties) {
266                 Matcher matcher = pattern.matcher(paramName);
267                 if (matcher.matches()) {
268                     return true;
269                 }
270             }
271 
272             //no match, but acceptedParams is not empty
273             return false;
274         }
275 
276         //empty acceptedParams
277         return true;
278     }
279 
280     protected boolean isExcluded(String paramName) {
281         if (!this.excludeProperties.isEmpty()) {
282             for (Pattern pattern : excludeProperties) {
283                 Matcher matcher = pattern.matcher(paramName);
284                 if (matcher.matches()) {
285                     return true;
286                 }
287             }
288         }
289         return false;
290     }
291 
292     public void setExcludeProperties(Set<Pattern> excludeProperties) {
293         this.excludeProperties = excludeProperties;
294     }
295 
296     public void setAcceptProperties(Set<Pattern> acceptedProperties) {
297         this.acceptProperties = acceptedProperties;
298     }
299 
300     public void setExcludedClasses(Set<Class<?>> excludedClasses) {
301         this.excludedClasses = excludedClasses;
302     }
303 
304     public void setExcludedPackageNamePatterns(Set<Pattern> excludedPackageNamePatterns) {
305         this.excludedPackageNamePatterns = excludedPackageNamePatterns;
306     }
307 
308     public void setExcludedPackageNames(Set<String> excludedPackageNames) {
309         this.excludedPackageNames = excludedPackageNames;
310     }
311 
312     public void setDisallowProxyMemberAccess(boolean disallowProxyMemberAccess) {
313         this.disallowProxyMemberAccess = disallowProxyMemberAccess;
314     }
