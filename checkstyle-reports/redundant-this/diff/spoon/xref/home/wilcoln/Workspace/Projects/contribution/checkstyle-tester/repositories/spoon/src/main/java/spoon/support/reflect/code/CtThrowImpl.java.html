6   package spoon.support.reflect.code;
7   
8   import spoon.reflect.annotations.MetamodelPropertyField;
9   import spoon.reflect.code.CtCodeElement;
10  import spoon.reflect.code.CtExpression;
11  import spoon.reflect.code.CtThrow;
12  import spoon.reflect.declaration.CtType;
13  import spoon.reflect.visitor.CtVisitor;
14  
15  import static spoon.reflect.path.CtRole.EXPRESSION;
16  
17  public class CtThrowImpl extends CtStatementImpl implements CtThrow {
18  	private static final long serialVersionUID = 1L;
19  
20  	@MetamodelPropertyField(role = EXPRESSION)
21  	CtExpression<? extends Throwable> throwExpression;
22  
23  	@Override
24  	public void accept(CtVisitor visitor) {
25  		visitor.visitCtThrow(this);
26  	}
27  
28  	@Override
29  	public CtExpression<? extends Throwable> getThrownExpression() {
30  		return throwExpression;
31  	}
32  
33  	@Override
34  	public <T extends CtThrow> T setThrownExpression(CtExpression<? extends Throwable> expression) {
35  		if (expression != null) {
36  			expression.setParent(this);
37  		}
38  		getFactory().getEnvironment().getModelChangeListener().onObjectUpdate(this, EXPRESSION, expression, this.throwExpression);
39  		this.throwExpression = expression;
40  		return (T) this;
41  	}
42  
43  	@Override
44  	public CtThrow clone() {
45  		return (CtThrow) super.clone();
46  	}
47  
48  	@Override
49  	public Void S() {
50  		return null;
51  	}
52  
53  	public CtCodeElement getSubstitution(CtType<?> targetType) {
54  		return clone();
55  	}
56  
57  }
