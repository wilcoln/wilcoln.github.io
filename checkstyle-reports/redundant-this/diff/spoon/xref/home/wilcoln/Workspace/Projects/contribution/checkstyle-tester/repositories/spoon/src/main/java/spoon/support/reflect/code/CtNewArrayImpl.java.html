6   package spoon.support.reflect.code;
7   
8   import spoon.reflect.annotations.MetamodelPropertyField;
9   import spoon.reflect.code.CtExpression;
10  import spoon.reflect.code.CtNewArray;
11  import spoon.reflect.visitor.CtVisitor;
12  import spoon.support.reflect.declaration.CtElementImpl;
13  
14  import java.util.ArrayList;
15  import java.util.List;
16  
17  import static spoon.reflect.ModelElementContainerDefaultCapacities.NEW_ARRAY_DEFAULT_EXPRESSIONS_CONTAINER_DEFAULT_CAPACITY;
18  import static spoon.reflect.path.CtRole.DIMENSION;
19  import static spoon.reflect.path.CtRole.EXPRESSION;
20  
21  public class CtNewArrayImpl<T> extends CtExpressionImpl<T> implements CtNewArray<T> {
22  	private static final long serialVersionUID = 1L;
23  
24  	@MetamodelPropertyField(role = DIMENSION)
25  	List<CtExpression<Integer>> dimensionExpressions = emptyList();
26  
27  	@MetamodelPropertyField(role = EXPRESSION)
28  	List<CtExpression<?>> expressions = emptyList();
29  
30  	@Override
31  	public void accept(CtVisitor visitor) {
32  		visitor.visitCtNewArray(this);
33  	}
34  
35  	@Override
36  	public List<CtExpression<Integer>> getDimensionExpressions() {
37  		return dimensionExpressions;
38  	}
39  
40  	@Override
41  	public List<CtExpression<?>> getElements() {
42  		return expressions;
43  	}
44  
45  	@Override
46  	public <C extends CtNewArray<T>> C setDimensionExpressions(List<CtExpression<Integer>> dimensionExpressions) {
47  		if (dimensionExpressions == null || dimensionExpressions.isEmpty()) {
48  			this.dimensionExpressions = CtElementImpl.emptyList();
49  			return (C) this;
50  		}
51  		getFactory().getEnvironment().getModelChangeListener().onListDeleteAll(this, DIMENSION, this.dimensionExpressions, new ArrayList<>(this.dimensionExpressions));
52  		this.dimensionExpressions.clear();
53  		for (CtExpression<Integer> expr : dimensionExpressions) {
54  			addDimensionExpression(expr);
55  		}
56  		return (C) this;
57  	}
58  
59  	@Override
60  	public <C extends CtNewArray<T>> C addDimensionExpression(CtExpression<Integer> dimension) {
61  		if (dimension == null) {
62  			return (C) this;
63  		}
64  		if (dimensionExpressions == CtElementImpl.<CtExpression<Integer>>emptyList()) {
65  			dimensionExpressions = new ArrayList<>(NEW_ARRAY_DEFAULT_EXPRESSIONS_CONTAINER_DEFAULT_CAPACITY);
66  		}
67  		dimension.setParent(this);
68  		getFactory().getEnvironment().getModelChangeListener().onListAdd(this, DIMENSION, this.dimensionExpressions, dimension);
69  		dimensionExpressions.add(dimension);
70  		return (C) this;
71  	}
72  
73  	@Override
74  	public boolean removeDimensionExpression(CtExpression<Integer> dimension) {
75  		if (dimensionExpressions == CtElementImpl.<CtExpression<Integer>>emptyList()) {
76  			return false;
77  		}
78  		getFactory().getEnvironment().getModelChangeListener().onListDelete(this, DIMENSION, dimensionExpressions, dimensionExpressions.indexOf(dimension), dimension);
79  		return dimensionExpressions.remove(dimension);
80  	}
81  
82  	@Override
83  	public <C extends CtNewArray<T>> C setElements(List<CtExpression<?>> expressions) {
84  		if (expressions == null || expressions.isEmpty()) {
85  			this.expressions = CtElementImpl.emptyList();
86  			return (C) this;
87  		}
88  		getFactory().getEnvironment().getModelChangeListener().onListDeleteAll(this, EXPRESSION, this.expressions, new ArrayList<>(this.expressions));
89  		this.expressions.clear();
90  		for (CtExpression<?> expr : expressions) {
91  			addElement(expr);
92  		}
93  		return (C) this;
94  	}
95  
96  	@Override
97  	public <C extends CtNewArray<T>> C addElement(CtExpression<?> expression) {
98  		if (expression == null) {
99  			return (C) this;
100 		}
101 		if (expressions == CtElementImpl.<CtExpression<?>>emptyList()) {
102 			this.expressions = new ArrayList<>();
103 		}
104 		expression.setParent(this);
105 		getFactory().getEnvironment().getModelChangeListener().onListAdd(this, EXPRESSION, this.expressions, expression);
106 		expressions.add(expression);
107 		return (C) this;
108 	}
109 
110 	@Override
111 	public boolean removeElement(CtExpression<?> expression) {
112 		if (expressions == CtElementImpl.<CtExpression<?>>emptyList()) {
113 			return false;
114 		}
115 		getFactory().getEnvironment().getModelChangeListener().onListDelete(this, EXPRESSION, expressions, expressions.indexOf(expression), expression);
116 		return expressions.remove(expression);
117 	}
118 
119 	@Override
120 	public CtNewArray<T> clone() {
121 		return (CtNewArray<T>) super.clone();
122 	}
123 }
