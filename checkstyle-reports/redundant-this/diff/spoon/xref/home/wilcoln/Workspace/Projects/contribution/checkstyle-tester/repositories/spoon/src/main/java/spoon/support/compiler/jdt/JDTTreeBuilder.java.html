240 	private LiteralBase getBase(NumberLiteral numberLiteral) {
241 		String sourceString = new String(numberLiteral.source());
242 
243 		if (sourceString.startsWith("0x") || sourceString.startsWith("0X")) {
244 			return LiteralBase.HEXADECIMAL;
245 		}
246 
247 		if (sourceString.startsWith("0b") || sourceString.startsWith("0B")) {
248 			return LiteralBase.BINARY;
249 		}
250 
251 		if (sourceString.startsWith("0")) {
252 			if ((numberLiteral instanceof IntLiteral && sourceString.length() > 1)
253 				|| (numberLiteral instanceof LongLiteral && sourceString.length() > 2)) {
254 				return LiteralBase.OCTAL;
255 			}
256 		}
257 
258 		return LiteralBase.DECIMAL;
259 	}

450 	@Override
451 	public void endVisit(LabeledStatement labeledStatement, BlockScope scope) {
452 		ASTPair pair = context.stack.peek();
453 		CtBlock<?> block = (CtBlock<?>) pair.element;
454 		if (block.getStatements().size() == 1) {
455 			CtStatement childStmt = block.getStatement(0);
456 			if (childStmt.getLabel() == null) {
457 				//the child statement has no label, so we can move label from `block` to child statement and to remove this `block`
458 				//example code:
459 				//label: while(true);
460 				childStmt.setLabel(block.getLabel());
461 				SourcePosition oldPos = childStmt.getPosition();
462 				int newSourceStart = Math.min(oldPos.getSourceStart(), block.getPosition().getSourceStart());
463 				if (newSourceStart != oldPos.getSourceStart()) {
464 					childStmt.setPosition(block.getFactory().Core().createSourcePosition(
465 							oldPos.getCompilationUnit(),
466 							newSourceStart, oldPos.getSourceEnd(),
467 							oldPos.getCompilationUnit().getLineSeparatorPositions()));
468 				}
469 				//call exit with origin labeled statement
470 				//because some listeners needs origin one
471 				//we cannot call exit on unexpected child
472 				context.exit(labeledStatement);
473 				//use childStmt instead of helper block
474 				CtElement parent = block.getParent();
475 				//remember whether parent was implicit
476 				boolean parentIsImplicit = parent.isImplicit();
477 				//because replace resets CtBlock#isImplicit to false
478 				block.replace(childStmt);
479 				//but we need to keep it as it was before
480 				parent.setImplicit(parentIsImplicit);
481 				return;
482 			}
483 		}
484 		//else example code:
485 		//label:;
486 		//label1: label2: while(true);
487 		//needs to keep an implicit helper CtBlock as holder of `label1`
488 		context.exit(labeledStatement);
489 	}

793 		CompilationUnit compilationUnitSpoon = factory.CompilationUnit().getOrCreate(new String(compilationUnitDeclaration.getFileName()));

795 			compilationUnitSpoon.setLineSeparatorPositions(compilationUnitDeclaration.compilationResult.lineSeparatorPositions);
796 		} else if (compilationUnitSpoon.getLineSeparatorPositions() != compilationUnitDeclaration.compilationResult.lineSeparatorPositions) {

802 	@Override
803 	public boolean visit(CompilationUnitDeclaration compilationUnitDeclaration, CompilationUnitScope scope) {
804 		context.compilationunitdeclaration = scope.referenceContext;
805 		context.compilationUnitSpoon = getOrCreateCompilationUnit(context.compilationunitdeclaration, getFactory());
806 		ModuleBinding enclosingModule = scope.fPackage.enclosingModule;
807 
808 		CtModule module;
809 		if (!enclosingModule.isUnnamed() && enclosingModule.shortReadableName() != null && enclosingModule.shortReadableName().length > 0) {
810 			module = getFactory().Module().getOrCreate(String.valueOf(enclosingModule.shortReadableName()));
811 		} else {
812 			module = getFactory().Module().getUnnamedModule();
813 		}
814 
815 		context.compilationUnitSpoon.setDeclaredPackage(getFactory().Package().getOrCreate(CharOperation.toString(scope.currentPackageName), module));
816 		CtPackageDeclaration packageDeclaration = context.compilationUnitSpoon.getPackageDeclaration();
817 		if (packageDeclaration != null) {
818 			ImportReference packageRef = compilationUnitDeclaration.currentPackage;
819 			if (packageRef != null) {
820 				char[] content = context.getCompilationUnitContents();
821 				int declStart = packageRef.declarationSourceStart;
822 				//look for first comment
823 				int firstComment = PositionBuilder.findNextNonWhitespace(false, content, packageRef.sourceStart(), 0);
824 				if (firstComment < packageRef.sourceStart() && content[firstComment] == '/' && content[firstComment + 1] == '*') {
825 					//there is a `/*` or `/**`comment before package reference;
826 					//such comment is understood as compilation unit comment
827 					//all next comments belong to package declaration
828 					int commentEnd = PositionBuilder.getEndOfComment(content, packageRef.sourceStart(), firstComment);
829 					declStart = PositionBuilder.findNextNonWhitespace(false, content, packageRef.sourceStart(), commentEnd + 1);
830 				} else {
831 					declStart = firstComment;
832 				}
833 				packageDeclaration.setPosition(factory.Core().createCompoundSourcePosition(
834 						context.compilationUnitSpoon, packageRef.sourceStart(), packageRef.sourceEnd(), declStart, packageRef.declarationEnd, context.compilationUnitSpoon.getLineSeparatorPositions()));
835 				packageDeclaration.getReference().setPosition(factory.Core().createSourcePosition(context.compilationUnitSpoon, packageRef.sourceStart(), packageRef.sourceEnd(), context.compilationUnitSpoon.getLineSeparatorPositions()));
836 			}
837 		}
838 		return true;
839 	}
