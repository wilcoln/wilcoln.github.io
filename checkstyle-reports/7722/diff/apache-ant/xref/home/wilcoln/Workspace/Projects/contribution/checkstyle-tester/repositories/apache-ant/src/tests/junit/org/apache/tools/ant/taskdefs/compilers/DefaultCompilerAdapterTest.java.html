37              return sb.toString();
38          }

41      private static class SourceTargetHelper extends DefaultCompilerAdapter {
42          /**
43           * Overridden to have no effect.
44           */
45          protected Commandline setupJavacCommandlineSwitches(Commandline cmd,
46                                                              boolean debug) {
47              return cmd;
48          }
49  
50          public boolean execute() { return false; }

52          /**
53           * public to avoid classloader issues.
54           */
55          public Commandline setupModernJavacCommandlineSwitches(Commandline cmd) {
56              return super.setupModernJavacCommandlineSwitches(cmd);
57          }
58      }

181     private void commonSourceDowngrades(String javaVersion) {
182         testSource("1.3", javaVersion,
183                    "If you specify -target 1.1 you now must also specify"
184                    + " -source 1.3", "1.1");
185         testSource("1.3", javaVersion,
186                    "If you specify -target 1.2 you now must also specify"
187                    + " -source 1.3", "1.2");
188         testSource("1.3", javaVersion,
189                    "If you specify -target 1.3 you now must also specify"
190                    + " -source 1.3", "1.3");
191         testSource("1.4", javaVersion,
192                    "If you specify -target 1.4 you now must also specify"
193                    + " -source 1.4", "1.4");
194     }
195 
196     private void testSource(String expectedSource, String javaVersion,
197                             String expectedLog, String configuredTarget) {
198         testSource(expectedSource, javaVersion, expectedLog, configuredTarget,
199                    null);
200     }
201 
202     private void testSource(String expectedSource, String javaVersion,
203                             String expectedLog, String configuredTarget,
204                             String configuredSource) {
205         LogCapturingJavac javac = new LogCapturingJavac();
206         javac.setProject(new Project());
207         javac.setCompiler(javaVersion);
208         javac.setSource(configuredSource);
209         javac.setTarget(configuredTarget);
210         SourceTargetHelper sth = new SourceTargetHelper();
211         sth.setJavac(javac);
212         Commandline cmd = new Commandline();
213         sth.setupModernJavacCommandlineSwitches(cmd);
214         if ("".equals(expectedLog)) {
215             assertEquals("", javac.getLog());
216         } else {
217             String l = javac.getLog();
218             assertContains(expectedLog, l);
219         }
220         String[] args = cmd.getCommandline();
221         assertEquals(expectedSource == null ? 0 : 2, args.length);
222         if (expectedSource != null) {
223             assertEquals("-source", args[0]);
224             assertEquals(expectedSource, args[1]);
225         }
226     }
