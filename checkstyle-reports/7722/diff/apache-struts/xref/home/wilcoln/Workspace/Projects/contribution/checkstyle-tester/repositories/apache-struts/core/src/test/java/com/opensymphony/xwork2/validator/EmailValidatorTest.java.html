19  package com.opensymphony.xwork2.validator;
20  
21  import com.opensymphony.xwork2.ActionContext;
22  import com.opensymphony.xwork2.ActionSupport;
23  import com.opensymphony.xwork2.TextProviderFactory;
24  import com.opensymphony.xwork2.XWorkTestCase;
25  import com.opensymphony.xwork2.util.ValueStack;
26  import com.opensymphony.xwork2.validator.validators.EmailValidator;

28  /**
29   * Test case for Email Validator
30   *
31   * @author tm_jee
32   * @version $Date$ $Id$
33   */
34  public class EmailValidatorTest extends XWorkTestCase {
35  
36      private TextProviderFactory tpf;
37  
38      public void testEmailValidity() throws Exception {
39          assertTrue(verifyEmailValidity("TmJee@Yahoo.com"));
40          assertTrue(verifyEmailValidity("tmjee@yahoo.com"));
41          assertTrue(verifyEmailValidityWithExpression("tmjee@yahoo.com", "\\b^[a-z]+@[a-z]+(\\.[a-z]+)*\\.com$\\b"));
42          assertTrue(verifyEmailValidity("tm_jee@yahoo.co"));
43          assertTrue(verifyEmailValidityWithExpression("tm_jee@yahoo.co", "\\b^[a-z_]+@[a-z]+(\\.[a-z]+)*\\.co$\\b"));
44          assertTrue(verifyEmailValidity("tm.jee@yahoo.co.uk"));
45          assertTrue(verifyEmailValidity("tm.jee@yahoo.co.biz"));
46          assertTrue(verifyEmailValidity("tm_jee@yahoo.com"));
47          assertTrue(verifyEmailValidity("tm_jee@yahoo.net"));
48          assertTrue(verifyEmailValidity(" user@subname1.subname2.subname3.domainname.co.uk "));
49          assertTrue(verifyEmailValidity("tm.j'ee@yahoo.co.uk"));
50          assertTrue(verifyEmailValidity("tm.j'e.e'@yahoo.co.uk"));
51          assertTrue(verifyEmailValidity("tmj'ee@yahoo.com"));
52          assertTrue(verifyEmailValidity("ferda+mravenec@yahoo.com"));
53          assertTrue(verifyEmailValidity("Ferda+Mravenec@yaHoo.CoM"));
54          assertTrue(verifyEmailValidity("user@domainname.tech"));
55          assertTrue(verifyEmailValidity("Ferda+Mravenec@yaHoo.cat"));
56          assertTrue(verifyEmailValidity("user@domainname.swiss"));
57  
58          assertFalse(verifyEmailValidity("tm_jee#marry@yahoo.co.uk"));
59          assertFalse(verifyEmailValidity("tm_jee@ yahoo.co.uk"));
60          assertFalse(verifyEmailValidity("tm_jee  @yahoo.co.uk"));
61          assertFalse(verifyEmailValidity("tm_j ee  @yah oo.co.uk"));
62          assertFalse(verifyEmailValidity("tm_jee  @yah oo.co.uk"));
63          assertFalse(verifyEmailValidity("tm_jee @ yahoo.com"));
64          assertFalse(verifyEmailValidity(" user@subname1.subname2.subname3.domainn#ame.co.uk "));
65          assertFalse(verifyEmailValidity("aaa@aa.aaaaaaa"));
66          assertFalse(verifyEmailValidity("+ferdamravenec@yahoo.com"));
67  
68          assertTrue(verifyEmailValidityWithExpression("tmjee@yahoo.co", "\\b^[a-z]+@[a-z]+(\\.[a-z]+)*\\.com$\\b"));
69      }
70  
71      protected boolean verifyEmailValidity(final String email) throws Exception {
72          ActionSupport action = new ActionSupport() {
73              public String getMyEmail() {
74                  return email;
75              }
76          };
77  
78          EmailValidator validator = new EmailValidator();
79          validator.setValidatorContext(new DelegatingValidatorContext(action, tpf));
80          validator.setFieldName("myEmail");
81          validator.setDefaultMessage("invalid email");
82          validator.setValueStack(ActionContext.getContext().getValueStack());
83          validator.validate(action);
84  
85          return (action.getFieldErrors().size() == 0);
86      }
87  
88      public boolean verifyEmailValidityWithExpression(final String email, final String expression) throws Exception {
89          ActionSupport action = new ActionSupport() {
90              public String getMyEmail() {
91                  return email;
92              }
93  
94              public String getEmailExpression() {
95                  return expression;
96              }
97          };
98  
99          EmailValidator validator = new EmailValidator();
100         ValueStack valueStack = ActionContext.getContext().getValueStack();
101         valueStack.push(action);
102         validator.setValueStack(valueStack);
103 
104         validator.setValidatorContext(new DelegatingValidatorContext(action, tpf));
105         validator.setFieldName("myEmail");
106         validator.setDefaultMessage("invalid email");
107         validator.setRegexExpression("${emailExpression}");
108 
109         validator.validate(action);
110 
111         return (action.getFieldErrors().size() == 0);
112     }
113 
114     public void testCaseSensitiveViaExpression() throws Exception {
115         EmailValidator validator = verifyCaseSensitive(true);
116         assertTrue(validator.isCaseSensitive());
117 
118         validator = verifyCaseSensitive(false);
119         assertFalse(validator.isCaseSensitive());
120     }
121 
122     private EmailValidator verifyCaseSensitive(final boolean caseSensitive) {
123         ActionSupport action = new ActionSupport() {
124             public boolean getEmailCaseSensitive() {
125                 return caseSensitive;
126             }
127         };
128 
129         EmailValidator validator = new EmailValidator();
130         ValueStack valueStack = ActionContext.getContext().getValueStack();
131         valueStack.push(action);
132         validator.setValueStack(valueStack);
133 
134         validator.setCaseSensitiveExpression("${emailCaseSensitive}");
135 
136         return validator;
137     }
138 
139     public void testTrimViaExpression() throws Exception {
140         EmailValidator validator = verifyTrim(true);
141         assertTrue(validator.isTrimed());
142 
143         validator = verifyTrim(false);
144         assertFalse(validator.isTrimed());
145     }
146 
147     private EmailValidator verifyTrim(final boolean trim) {
148         ActionSupport action = new ActionSupport() {
149             public boolean getTrimEmail() {
150                 return trim;
151             }
152         };
153 
154         EmailValidator validator = new EmailValidator();
155         ValueStack valueStack = ActionContext.getContext().getValueStack();
156         valueStack.push(action);
157         validator.setValueStack(valueStack);
158 
159         validator.setTrimExpression("${trimEmail}");
160 
161         return validator;
162     }
