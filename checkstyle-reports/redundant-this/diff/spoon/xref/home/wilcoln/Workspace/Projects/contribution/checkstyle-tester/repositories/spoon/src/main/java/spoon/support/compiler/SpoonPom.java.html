69  	/**
70  	 * Extract the information from the pom
71  	 * @param path the path to the pom
72  	 * @param parent the parent pom
73  	 * @throws IOException when the file does not exist
74  	 * @throws XmlPullParserException when the file is corrupted
75  	 */
76  	public SpoonPom(String path, SpoonPom parent, MavenLauncher.SOURCE_TYPE sourceType, Environment environment) throws IOException, XmlPullParserException {
77  		this.parent = parent;
78  		this.sourceType = sourceType;
79  		this.environment = environment;
80  		if (!path.endsWith(".xml") && !path.endsWith(".pom")) {
81  			path = Paths.get(path, "pom.xml").toString();
82  		}
83  		this.pomFile = new File(path);
84  		if (!pomFile.exists()) {
85  			throw new IOException("Pom does not exists.");
86  		}
87  		this.directory = pomFile.getParentFile();
88  		MavenXpp3Reader pomReader = new MavenXpp3Reader();
89  		try (FileReader reader = new FileReader(pomFile)) {
90  			this.model = pomReader.read(reader);
91  			for (String module : model.getModules()) {
92  				addModule(new SpoonPom(Paths.get(pomFile.getParent(), module).toString(), this, sourceType, environment));
93  			}
94  		} catch (FileNotFoundException e) {
95  			throw new IOException("Pom does not exists.");
96  		}
97  	}

112 	 * Get the list of source directories of the project
113 	 * @return the list of source directories
114 	 */
115 	public List<File> getSourceDirectories() {
116 		List<File> output = new ArrayList<>();
117 		String sourcePath = null;
118 
119 		Build build = model.getBuild();
120 		if (build != null) {
121 			sourcePath = build.getSourceDirectory();
122 		}
123 		if (sourcePath == null) {
124 			sourcePath = Paths.get(directory.getAbsolutePath(), "src", "main", "java").toString();
125 		}
126 		File source = new File(sourcePath);
127 		if (source.exists()) {
128 			output.add(source);
129 		}
130 		File generatedSource = Paths.get(directory.getAbsolutePath(), "target", "generated-sources").toFile();
131 		if (generatedSource.exists()) {
132 			output.add(generatedSource);
133 		}
134 		for (SpoonPom module : modules) {
135 			output.addAll(module.getSourceDirectories());
136 		}
137 		return output;
138 	}

185 	// Pattern corresponding to maven properties ${propertyName}
186 	private static Pattern mavenProperty = Pattern.compile("\\$\\{.*\\}");

188 	/**
189 	 * Extract the variable from a string
190 	 */
191 	private String extractVariable(String value) {
192 		String val = value;
193 		if (value != null && value.contains("$")) {
194 			Matcher matcher = mavenProperty.matcher(value);
195 			while (matcher.find()) {
196 				String var = matcher.group();
197 				val = val.replace(var, getProperty(var.substring(2, var.length() - 1)));
198 			}
199 		}
200 		return val;
201 	}

203 	/**
204 	 * Get the value of a property
205 	 * @param key the key of the property
206 	 * @return the property value if key exists or null
207 	 */
208 	private String getProperty(String key) {
209 		if ("project.version".equals(key)  || "pom.version".equals(key)) {
210 			if (model.getVersion() != null) {
211 				return model.getVersion();
212 			} else if (model.getParent() != null) {
213 				return model.getParent().getVersion();
214 			}
215 		} else if ("project.groupId".equals(key) || "pom.groupId".equals(key)) {
216 			if (model.getGroupId() != null) {
217 				return model.getGroupId();
218 			} else if (model.getParent() != null) {
219 				return model.getParent().getGroupId();
220 			}
221 		} else if ("project.artifactId".equals(key)  || "pom.artifactId".equals(key)) {
222 			if (model.getArtifactId() != null) {
223 				return model.getArtifactId();
224 			} else if (model.getParent() != null) {
225 				return model.getParent().getArtifactId();
226 			}
227 		}
228 		String value = extractVariable(model.getProperties().getProperty(key));
229 		if (value == null) {
230 			if (parent == null) {
231 				return null;
232 			}
233 			return parent.getProperty(key);
234 		}
235 		return value;
236 	}

238 	/**
239 	 * Get the source version of the project
240 	 * @return the source version of the project
241 	 */
242 	public int getSourceVersion() {
243 		String javaVersion = null;
244 		if (model.getBuild() != null) {
245 			javaVersion = getSourceVersion(model.getBuild());
246 		}
247 		if (javaVersion != null) {
248 			return correctJavaVersion(javaVersion);
249 		}
250 		for (Profile profile: model.getProfiles()) {
251 			if (profile.getActivation() != null && profile.getActivation().isActiveByDefault()) {
252 				javaVersion = getSourceVersion(profile.getBuild());
253 			}
254 		}
255 		if (javaVersion != null) {
256 			return correctJavaVersion(javaVersion);
257 		}
258 		javaVersion = getProperty("java.version");
259 		if (javaVersion != null) {
260 			return correctJavaVersion(javaVersion);
261 		}
262 		javaVersion = getProperty("java.src.version");
263 		if (javaVersion != null) {
264 			return correctJavaVersion(javaVersion);
265 		}
266 		javaVersion = getProperty("maven.compiler.source");
267 		if (javaVersion != null) {
268 			return correctJavaVersion(javaVersion);
269 		}
270 		javaVersion = getProperty("maven.compile.source");
271 		if (javaVersion != null) {
272 			return correctJavaVersion(javaVersion);
273 		}
274 		// return the current compliance level of spoon
275 		return environment.getComplianceLevel();
276 	}
277 
278 	private int correctJavaVersion(String javaVersion) {
279 		String version = extractVariable(javaVersion);
280 		return Integer.parseInt((version.contains(".") ? version.substring(2) : version));
281 	}

283 	private String getSourceVersion(BuildBase build) {
284 		for (Plugin plugin : build.getPlugins()) {
285 			if (!"maven-compiler-plugin".equals(plugin.getArtifactId())) {
286 				continue;
287 			}
288 			Xpp3Dom configuration = (Xpp3Dom) plugin.getConfiguration();
289 			if (configuration != null) {
290 				Xpp3Dom source = configuration.getChild("source");
291 				if (source != null) {
292 					return source.getValue();
293 				}
294 			}
295 			break;
296 		}
297 		return null;
298 	}

300 	@Override
301 	public String toString() {
302 		StringBuilder sb = new StringBuilder();
303 		sb.append(model.getGroupId());
304 		sb.append(":");
305 		sb.append(model.getArtifactId());
306 		sb.append(":");
307 		sb.append(model.getVersion());
308 		if (modules.isEmpty()) {
309 			return sb.toString();
310 		}
311 		sb.append(" {\n");
312 		for (SpoonPom spoonPom : modules) {
313 			String child = spoonPom.toString();
314 			for (String s : child.split("\n")) {
315 				sb.append("\t");
316 				sb.append(s);
317 				sb.append("\n");
318 			}
319 		}
320 		sb.append("}");
321 		return sb.toString();
322 	}

324 	private void generateClassPathFile(File mvnHome, MavenLauncher.SOURCE_TYPE sourceType, Logger LOGGER, boolean forceRefresh) {
325 		// Check if classpath file already exist and is recent enough (1h)
326 		File classpathFile = new File(directory, getSpoonClasspathTmpFileName(sourceType));
327 		Date date = new Date();
328 		long time = date.getTime();
329 		if (forceRefresh || !classpathFile.exists() || ((time - classpathFile.lastModified()) > classpathTmpFilesTTL)) {
330 			//Run mvn dependency:build-classpath -Dmdep.outputFile="spoon.classpath.tmp"
331 			//This should write the classpath used by maven in spoon.classpath.tmp
332 			InvocationRequest request = new DefaultInvocationRequest();
333 			request.setBatchMode(true);
334 			request.setPomFile(pomFile);
335 			request.setGoals(Collections.singletonList("dependency:build-classpath"));
336 			Properties properties = new Properties();
337 			if (sourceType == MavenLauncher.SOURCE_TYPE.APP_SOURCE) {
338 				properties.setProperty("includeScope", "runtime");
339 			}
340 			properties.setProperty("mdep.outputFile", getSpoonClasspathTmpFileName(sourceType));
341 			request.setProperties(properties);
342 
343 			if (LOGGER != null) {
344 				request.getOutputHandler(s -> LOGGER.debug(s));
345 				request.getErrorHandler(s -> LOGGER.debug(s));
346 			}
347 
348 			Invoker invoker = new DefaultInvoker();
349 			invoker.setMavenHome(mvnHome);
350 			invoker.setWorkingDirectory(directory);
351 			invoker.setErrorHandler(s -> LOGGER.debug(s));
352 			invoker.setOutputHandler(s -> LOGGER.debug(s));
353 			try {
354 				invoker.execute(request);
355 			} catch (MavenInvocationException e) {
356 				throw new SpoonException("Maven invocation failed to build a classpath.");
357 			}
358 			classpathFile.setLastModified(time);
359 		}
360 	}

362 	/**
363 	 *
364 	 * @param classPathFiles File[] containing the classpath elements separated with ':'
365 	 *                       It can be an array of file instead of an unique one for multi module projects.
366 	 */
367 	private static String[] readClassPath(File... classPathFiles) throws IOException {
368 		List<String> classpathElements = new ArrayList<>();
369 
370 		//Read the content of spoon.classpath.tmp
371 		for (File classPathFile: classPathFiles) {
372 			try (BufferedReader br = new BufferedReader(new FileReader(classPathFile))) {
373 				StringBuilder sb = new StringBuilder();
374 				String line = br.readLine();
375 				while (line != null) {
376 					sb.append(line);
377 					line = br.readLine();
378 				}
379 				if (!"".equals(sb.toString())) {
380 					String[] classpath = sb.toString().split(File.pathSeparator);
381 					for (String cpe : classpath) {
382 						if (!classpathElements.contains(cpe)) {
383 							classpathElements.add(cpe);
384 						}
385 					}
386 				}
387 			}
388 		}
389 
390 		return classpathElements.toArray(new String[0]);
391 	}

393 	private static String guessMavenHome() {
394 		String mvnHome = null;
395 		try {
396 			String[] cmd;
397 			if (System.getProperty("os.name").contains("Windows")) {
398 				cmd = new String[]{"mvn.cmd", "-version"};
399 			} else {
400 				cmd = new String[]{"mvn", "-version"};
401 			}
402 			Process p = Runtime.getRuntime().exec(cmd);
403 			try (BufferedReader output = new BufferedReader(new InputStreamReader(p.getInputStream()))) {
404 				String line;
405 
406 				while ((line = output.readLine()) != null) {
407 					if (line.contains(mavenVersionParsing)) {
408 						return line.replace(mavenVersionParsing, "");
409 					}
410 				}
411 			}
412 
413 			p.waitFor();
414 		} catch (IOException e) {
415 			throw new SpoonException("Maven home detection has failed.");
416 		} catch (InterruptedException e) {
417 			throw new SpoonException("Maven home detection was interrupted.");
418 		}
419 		return mvnHome;
420 	}

422 	/**
423 	 * Call maven invoker to generate the classpath. Either M2_HOME must be
424 	 * initialized, or the command mvn must be in PATH.
425 	 *
426 	 * @param mvnHome the path to the m2repository
427 	 * @param sourceType the source type (App, test, or all)
428 	 * @param LOGGER Logger used for maven output
429 	 * @param forceRefresh if true forces the invocation of maven to regenerate classpath
430 	 */
431 	public String[] buildClassPath(String mvnHome, MavenLauncher.SOURCE_TYPE sourceType, Logger LOGGER, boolean forceRefresh) {
432 		if (mvnHome == null) {
433 			mvnHome = guessMavenHome();
434 			if (mvnHome == null) {
435 				throw new SpoonException("M2_HOME must be initialized to use this MavenLauncher constructor.");
436 			}
437 		}
438 		generateClassPathFile(new File(mvnHome), sourceType, LOGGER, forceRefresh);
439 
440 		List<File> classPathPrints;
441 		String[] classpath;
442 		try {
443 			classPathPrints = getClasspathTmpFiles(getSpoonClasspathTmpFileName(sourceType));
444 			File[] classPathPrintFiles = new File[classPathPrints.size()];
445 			classPathPrintFiles = classPathPrints.toArray(classPathPrintFiles);
446 			classpath = readClassPath(classPathPrintFiles);
447 		} catch (IOException e) {
448 			throw new SpoonException("Failed to generate class path for " + pomFile.getAbsolutePath() + ".");
449 		}
450 		return classpath;
451 	}

453 	private static String getSpoonClasspathTmpFileName(MavenLauncher.SOURCE_TYPE sourceType) {
454 		// As the temporary file containing the classpath is re-generated only
455 		// once per hour, we need a different file for different dependency
456 		// resolution scopes.
457 		if (MavenLauncher.SOURCE_TYPE.TEST_SOURCE == sourceType) {
458 			return spoonClasspathTmpFileNameTest;
459 		} else if (MavenLauncher.SOURCE_TYPE.APP_SOURCE == sourceType) {
460 			return spoonClasspathTmpFileNameApp;
461 		} else {
462 			return spoonClasspathTmpFileName;
463 		}
464 	}
