94      protected void addAutoStartBundles(Properties configProps) {
95          //starts system bundles in level 1
96          List<String> bundleJarsLevel1 = new ArrayList<String>();
97          bundleJarsLevel1.add(getJarUrl(ShellService.class));
98          bundleJarsLevel1.add(getJarUrl(ServiceTracker.class));
99  
100         configProps.put(AutoProcessor.AUTO_START_PROP + ".1", StringUtils.join(bundleJarsLevel1, " "));
101 
102         //get a list of directories under /bundles with numeric names (the runlevel)
103         Map<String, String> runLevels = getRunLevelDirs("bundles");
104         if (runLevels.isEmpty()) {
105             //there are no run level dirs, search for bundles in that dir
106             List<String> bundles = getBundlesInDir("bundles");
107             if (!bundles.isEmpty()) {
108                 configProps.put(AutoProcessor.AUTO_START_PROP + ".2", StringUtils.join(bundles, " "));
109             }
110         } else {
111             for (Map.Entry<String, String> runLevel : runLevels.entrySet()) {
112                 String runLevelKey = runLevel.getKey();
113                 if ("1".endsWith(runLevelKey)) {
114                     throw new StrutsException("Run level dirs must be greater than 1. Run level 1 is reserved for the Felix bundles");
115                 }
116                 List<String> bundles = getBundlesInDir(runLevel.getValue());
117                 configProps.put(AutoProcessor.AUTO_START_PROP + "." + runLevelKey, StringUtils.join(bundles, " "));
118             }
119         }
120     }

122     /**
123      * @param dir directory
124      * @return  a list of directories under a directory whose name is a number
125      */
126     protected Map<String, String> getRunLevelDirs(String dir) {
127         Map<String, String> dirs = new HashMap<String, String>();
128         try {
129             ResourceFinder finder = new ResourceFinder();
130             URL url = finder.find("bundles");
131             if (url != null) {
132                 if ("file".equals(url.getProtocol())) {
133                     File bundlesDir = new File(url.toURI());
134                     String[] runLevelDirs = bundlesDir.list(new FilenameFilter() {
135                         public boolean accept(File file, String name) {
136                             try {
137                                 return file.isDirectory() && Integer.valueOf(name) > 0;
138                             } catch (NumberFormatException ex) {
139                                 //the name is not a number
140                                 return false;
141                             }
142                         }
143                     });
144 
145                     if (runLevelDirs != null && runLevelDirs.length > 0) {
146                         //add all the dirs to the list
147                         for (String runLevel : runLevelDirs) {
148                             dirs.put(runLevel, StringUtils.removeEnd(dir,  "/") + "/" + runLevel);
149                         }
150                     } else {
151                         LOG.debug("No run level directories found under the [{}] directory", dir);
152                     }
153                 } else {
154                     LOG.warn("Unable to read [{}] directory", dir);
155                 }
156             } else {
157                 LOG.warn("The [{}] directory was not found", dir);
158             }
159         } catch (Exception e) {
160             LOG.warn("Unable load bundles from the [{}] directory", dir, e);
161         }
162         return dirs;
163     }
164 
165     protected List<String> getBundlesInDir(String dir) {
166         List<String> bundleJars = new ArrayList<String>();
167         try {
168             ResourceFinder finder = new ResourceFinder();
169             URL url = finder.find(dir);
170             if (url != null) {
171                 if ("file".equals(url.getProtocol())) {
172                     File bundlesDir = new File(url.toURI());
173                     File[] bundles = bundlesDir.listFiles(new FilenameFilter() {
174                         public boolean accept(File file, String name) {
175                             return StringUtils.endsWith(name, ".jar");
176                         }
177                     });
178 
179                     if (bundles != null && bundles.length > 0) {
180                         //add all the bundles to the list
181                         for (File bundle : bundles) {
182                             String externalForm = bundle.toURI().toURL().toExternalForm();
183                             LOG.debug("Adding bundle [{}]", externalForm);
184                             bundleJars.add(externalForm);
185                         }
186 
187                     } else {
188                         LOG.debug("No bundles found under the [{}] directory", dir);
189                     }
190                 } else {
191                     LOG.warn("Unable to read [{}] directory", dir);
192                 }
193             } else {
194                 LOG.warn("The [{}] directory was not found", dir);
195             }
196         } catch (Exception e) {
197             LOG.warn("Unable load bundles from the [{}] directory", dir, e);
198         }
199         return bundleJars;
200     }

202     protected String getJarUrl(Class clazz) {
203         ProtectionDomain protectionDomain = clazz.getProtectionDomain();
204         CodeSource codeSource = protectionDomain.getCodeSource();
205         URL loc = codeSource.getLocation();
206         return loc.toString();
207     }

209     protected void replaceSystemPackages(Properties properties) {
210         //Felix has a way to load the config file and substitution expressions
211         //but the method does not have a way to specify the file (other than in an env variable)
212 
213         //${jre-${java.specification.version}}
214         String systemPackages = (String) properties.get(Constants.FRAMEWORK_SYSTEMPACKAGES);
215         String jreVersion = "jre-" + System.getProperty("java.version").substring(0, 3);
216         systemPackages = systemPackages.replace("${jre-${java.specification.version}}", (String) properties.get(jreVersion));
217         properties.put(Constants.FRAMEWORK_SYSTEMPACKAGES, systemPackages);
218     }

220     /*
221         Find subpackages of the packages defined in the property file and export them
222      */
223     protected void addExportedPackages(Properties strutsConfigProps, Properties configProps) {
224         String[] rootPackages = StringUtils.split((String) strutsConfigProps.get("scanning.package.includes"), ",");
225         ResourceFinder finder = new ResourceFinder(StringUtils.EMPTY);
226         List<String> exportedPackages = new ArrayList<String>();
227         //build a list of subpackages
228         for (String rootPackage : rootPackages) {
229             try {
230                 String version = null;
231                 if (rootPackage.indexOf(";") > 0) {
232                     String[] splitted = rootPackage.split(";");
233                     rootPackage = splitted[0];
234                     version = splitted[1];
235                 }
236                 Map<URL, Set<String>> subpackagesMap = finder.findPackagesMap(StringUtils.replace(rootPackage.trim(), ".", "/"));
237                 for (Map.Entry<URL, Set<String>> entry : subpackagesMap.entrySet()) {
238                     URL url = entry.getKey();
239                     Set<String> packages = entry.getValue();
240 
241                     //get version if not set
242                     if (StringUtils.isBlank(version)) {
243                         version = getVersion(url);
244                     }
245 
246                     if (packages != null) {
247                         for (String subpackage : packages) {
248                             exportedPackages.add(subpackage + "; version=" + version);
249                         }
250                     }
251                 }
252             } catch (IOException e) {
253                 LOG.error("Unable to find subpackages of [{}]", rootPackage, e);
254             }
255         }
256 
257         //make a string with the exported packages and add it to the system properties
258         if (!exportedPackages.isEmpty()) {
259             String systemPackages = (String) configProps.get(Constants.FRAMEWORK_SYSTEMPACKAGES);
260             systemPackages = StringUtils.removeEnd(systemPackages, ",") + "," + StringUtils.join(exportedPackages, ",");
261             configProps.put(Constants.FRAMEWORK_SYSTEMPACKAGES, systemPackages);
262         }
263     }

265     /**
266      * @param url URL for package
267      * @return  the version used to export the packages. it tries to get it from MANIFEST.MF, or the file name
268      */
269     protected String getVersion(URL url) {
270         if ("jar".equals(url.getProtocol())) {
271             try {
272                 FileManager fileManager = ServletActionContext.getContext().getInstance(FileManagerFactory.class).getFileManager();
273                 try (JarFile jarFile = new JarFile(new File(fileManager.normalizeToFileProtocol(url).toURI()))) {
274                     Manifest manifest = jarFile.getManifest();
275                     if (manifest != null) {
276                         String version = manifest.getMainAttributes().getValue("Bundle-Version");
277                         if (StringUtils.isNotBlank(version)) {
278                             return getVersionFromString(version);
279                         }
280                     } else {
281                         // try to get the version from the file name
282                         return getVersionFromString(jarFile.getName());
283                     }
284                 }
285             } catch (Exception e) {
286                 LOG.error("Unable to extract version from [{}], defaulting to '1.0.0'", url.toExternalForm());
287             }
288         }
289         return "1.0.0";
290     }

292     /**
293      * @param str string for extract version
294      * @return Extracts numbers followed by "." or "-" from the string and joins them with "."
295      */
296     protected String getVersionFromString(String str) {
297         Matcher matcher = versionPattern.matcher(str);
298         List<String> parts = new ArrayList<String>();
299         while (matcher.find()) {
300             parts.add(matcher.group(1));
301         }
302         //default
303         if (parts.size() == 0) {
304             return "1.0.0";
305         }
306         while (parts.size() < 3) {
307             parts.add("0");
308         }
309         return StringUtils.join(parts, ".");
310     }

312     protected Properties getProperties(String fileName) {
313         ResourceFinder finder = new ResourceFinder("");
314         try {
315             return finder.findProperties(fileName);
316         } catch (IOException e) {
317             if (LOG.isErrorEnabled()) {
318                 LOG.error("Unable to read property file [#]", fileName);
319             }
320             return new Properties();
321         }
322     }
