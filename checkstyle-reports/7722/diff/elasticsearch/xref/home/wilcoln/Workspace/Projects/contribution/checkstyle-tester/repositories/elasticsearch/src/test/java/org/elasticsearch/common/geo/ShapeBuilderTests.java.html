183     public void testGeoCircle() {
184         double earthCircumference = 40075016.69;
185         Circle circle = ShapeBuilder.newCircleBuilder().center(0, 0).radius("100m").build();
186         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
187         assertEquals((Point) new PointImpl(0, 0, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
188         circle = ShapeBuilder.newCircleBuilder().center(+180, 0).radius("100m").build();
189         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
190         assertEquals((Point) new PointImpl(180, 0, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
191         circle = ShapeBuilder.newCircleBuilder().center(-180, 0).radius("100m").build();
192         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
193         assertEquals((Point) new PointImpl(-180, 0, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
194         circle = ShapeBuilder.newCircleBuilder().center(0, 90).radius("100m").build();
195         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
196         assertEquals((Point) new PointImpl(0, 90, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
197         circle = ShapeBuilder.newCircleBuilder().center(0, -90).radius("100m").build();
198         assertEquals((360 * 100) / earthCircumference, circle.getRadius(), 0.00000001);
199         assertEquals((Point) new PointImpl(0, -90, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
200         double randomLat = (randomDouble() * 180) - 90;
201         double randomLon = (randomDouble() * 360) - 180;
202         double randomRadius = randomIntBetween(1, (int) earthCircumference / 4);
203         circle = ShapeBuilder.newCircleBuilder().center(randomLon, randomLat).radius(randomRadius + "m").build();
204         assertEquals((360 * randomRadius) / earthCircumference, circle.getRadius(), 0.00000001);
205         assertEquals((Point) new PointImpl(randomLon, randomLat, ShapeBuilder.SPATIAL_CONTEXT), circle.getCenter());
206     }

230         // tests that the following shape (defined in counterclockwise OGC order)
231         // https://gist.github.com/anonymous/7f1bb6d7e9cd72f5977c crosses the dateline
232         // expected results: 3 polygons, 1 with a hole
233 
234         // a giant c shape
235         PolygonBuilder builder = ShapeBuilder.newPolygon()
236             .point(174,0)
237             .point(-176,0)
238             .point(-176,3)
239             .point(177,3)
240             .point(177,5)
241             .point(-176,5)
242             .point(-176,8)
243             .point(174,8)
244             .point(174,0);
245 
246         // 3/4 of an embedded 'c', crossing dateline once
247         builder.hole()
248             .point(175, 1)
249             .point(175, 7)
250             .point(-178, 7)
251             .point(-178, 6)
252             .point(176, 6)
253             .point(176, 2)
254             .point(179, 2)
255             .point(179,1)
256             .point(175, 1);
257 
258         // embedded hole right of the dateline
259         builder.hole()
260             .point(-179, 1)

262             .point(-177, 2)
