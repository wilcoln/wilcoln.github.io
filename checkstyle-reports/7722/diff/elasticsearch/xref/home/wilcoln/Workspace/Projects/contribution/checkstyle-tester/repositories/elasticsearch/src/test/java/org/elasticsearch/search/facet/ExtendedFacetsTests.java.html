226     private void addControlValues(Map<String, Map<String, Integer>> queryValToFacetFieldEntries, String fieldVal, String queryVal) {
227         Map<String, Integer> controlFieldFacets = queryValToFacetFieldEntries.get(queryVal);
228         if (controlFieldFacets == null) {
229             controlFieldFacets = new HashMap<>();
230             queryValToFacetFieldEntries.put(queryVal, controlFieldFacets);
231         }
232         Integer controlCount = controlFieldFacets.get(fieldVal);
233         if (controlCount == null) {
234             controlCount = 0;
235         }
236         controlFieldFacets.put(fieldVal, ++controlCount);
237     }
238 
239     private void addControlValues(Map<String, Map<String, Integer>> queryValToFacetFieldEntries, Set<String> fieldValues, String queryVal) {
240         for (String fieldValue : fieldValues) {
241             addControlValues(queryValToFacetFieldEntries, fieldValue, queryVal);
242         }
243     }

245     private List<Tuple<Text, Integer>> getExpectedFacetEntries(Set<String> fieldValues,
246                                                                Map<String, Integer> controlFacetsField,
247                                                                int size,
248                                                                TermsFacet.ComparatorType sort,
249                                                                List<String> excludes,
250                                                                String regex,
251                                                                boolean allTerms) {
252         Pattern pattern = null;
253         if (regex != null) {
254             pattern = Regex.compile(regex, null);
255         }
256 
257         List<Tuple<Text, Integer>> entries = new ArrayList<>();
258         for (Map.Entry<String, Integer> e : controlFacetsField.entrySet()) {
259             if (excludes.contains(e.getKey())) {
260                 continue;
261             }
262             if (pattern != null && !pattern.matcher(e.getKey()).matches()) {
263                 continue;
264             }
265 
266             entries.add(new Tuple<Text, Integer>(new StringText(e.getKey()), e.getValue()));
267         }
268 
269         if (allTerms) {
270             for (String fieldValue : fieldValues) {
271                 if (!controlFacetsField.containsKey(fieldValue)) {
272                     if (excludes.contains(fieldValue)) {
273                         continue;
274                     }
275                     if (pattern != null && !pattern.matcher(fieldValue).matches()) {
276                         continue;
277                     }
278 
279                     entries.add(new Tuple<Text, Integer>(new StringText(fieldValue), 0));
280                 }
281             }
282         }
283 
284         switch (sort) {
285             case COUNT:
286                 Collections.sort(entries, count);
287                 break;
288             case REVERSE_COUNT:
289                 Collections.sort(entries, count_reverse);
290                 break;
291             case TERM:
292                 Collections.sort(entries, term);
293                 break;
294             case REVERSE_TERM:
295                 Collections.sort(entries, term_reverse);
296                 break;
297         }
298         return size >= entries.size() ? entries : entries.subList(0, size);
299     }

301     private final static Count count = new Count();
302     private final static CountReverse count_reverse = new CountReverse();
303     private final static Term term = new Term();
304     private final static TermReverse term_reverse = new TermReverse();
305 
306     private static class Count implements Comparator<Tuple<Text, Integer>> {
307 
308         @Override
309         public int compare(Tuple<Text, Integer> o1, Tuple<Text, Integer> o2) {
310             int cmp = o2.v2() - o1.v2();
311             if (cmp != 0) {
312                 return cmp;
313             }
314             cmp = o2.v1().compareTo(o1.v1());
315             if (cmp != 0) {
316                 return cmp;
317             }
318             return System.identityHashCode(o2) - System.identityHashCode(o1);
319         }
320 
321     }
322 
323     private static class CountReverse implements Comparator<Tuple<Text, Integer>> {
324 
325         @Override
326         public int compare(Tuple<Text, Integer> o1, Tuple<Text, Integer> o2) {
327             return -count.compare(o1, o2);
328         }
329 
330     }
331 
332     private static class Term implements Comparator<Tuple<Text, Integer>> {
333 
334         @Override
335         public int compare(Tuple<Text, Integer> o1, Tuple<Text, Integer> o2) {
336             return o1.v1().compareTo(o2.v1());
337         }
338 
339     }
340 
341     private static class TermReverse implements Comparator<Tuple<Text, Integer>> {
342 
343         @Override
344         public int compare(Tuple<Text, Integer> o1, Tuple<Text, Integer> o2) {
345             return -term.compare(o1, o2);
346         }
