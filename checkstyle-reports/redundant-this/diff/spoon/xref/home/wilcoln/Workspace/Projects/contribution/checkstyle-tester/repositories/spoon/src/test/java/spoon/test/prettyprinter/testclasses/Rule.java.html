19  public class Rule {
20      public enum Language {
21          DTD, WSDL, OTHER
22      }
23      public static final class Phoneme implements PhonemeExpr {
24          public interface Bidule {}
25  
26          public static final Comparator<Phoneme> COMPARATOR = new Comparator<Phoneme>() {
27              @Override
28              public int compare(final Phoneme o1, final Phoneme o2) {
29                  for (int i = 0; i < o1.phonemeText.length(); i++) {
30                      if (i >= o2.phonemeText.length()) {
31                          return +1;
32                      }
33                      final int c = o1.phonemeText.charAt(i) - o2.phonemeText.charAt(i);
34                      if (c != 0) {
35                          return c;
36                      }
37                  }
38  
39                  if (o1.phonemeText.length() < o2.phonemeText.length()) {
40                      return -1;
41                  }
42  
43                  return 0;
44              }
45          };
46  
47          private final StringBuilder phonemeText;
48          private final Language language;
49  
50          public Phoneme(final CharSequence phonemeText, final Language language) {
51              this.phonemeText = new StringBuilder(phonemeText);
52              this.language = language;
53          }
54  
55          @Override
56          public Iterable<Phoneme> getPhonemes() {
57              return Collections.singleton(this);
58          }
59      }
60  
61      public interface PhonemeExpr {
62          Iterable<Phoneme> getPhonemes();
63      }
64  
65      public static final class PhonemeList implements PhonemeExpr {
66          private final List<Phoneme> phonemes;
67  
68          public PhonemeList(final List<Phoneme> phonemes) {
69              this.phonemes = phonemes;
70          }
71  
72          @Override
73          public List<Phoneme> getPhonemes() {
74              return this.phonemes;
75          }
76      }
77  
78      public static final String ALL = "ALL";
79  
80      private static final String DOUBLE_QUOTE = "\"";
81  
82      private static final String HASH_INCLUDE = "#include";
83  
84      private static Phoneme parsePhoneme(final String ph) {
85          final int open = ph.indexOf("[");
86          if (open >= 0) {
87              if (!ph.endsWith("]")) {
88                  throw new IllegalArgumentException("Phoneme expression contains a '[' but does not end in ']'");
89              }
90              final String before = ph.substring(0, open);
91              final String in = ph.substring(open + 1, ph.length() - 1);
92              final Set<String> langs = new HashSet<String>(Arrays.asList(in.split("[+]")));
93  
94              return new Phoneme(before, Language.DTD);
95          }
96          return new Phoneme(ph, Language.WSDL);
97      }
