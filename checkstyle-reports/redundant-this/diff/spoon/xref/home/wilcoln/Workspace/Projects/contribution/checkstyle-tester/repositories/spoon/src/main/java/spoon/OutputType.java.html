6   package spoon;
7   
8   import java.util.Locale;
9   
10  /**
11   * Types of output.
12   */
13  public enum OutputType {
14  	/**
15  	 * Analysis only, models are not pretty-printed to disk.
16  	 */
17  	NO_OUTPUT,
18  
19  	/**
20  	 * One file per top-level class.
21  	 */
22  	CLASSES,
23  
24  	/**
25  	 * Follows the compilation units given by the input.
26  	 */
27  	COMPILATION_UNITS;
28  
29  	@Override
30  	public String toString() {
31  		return this.name().toLowerCase(Locale.US).replaceAll("_", "");
32  	}
33  
34  	/**
35  	 * Gets the output type from an option string.
36  	 *
37  	 * @param string the string, as given in the launcher's options
38  	 *
39  	 * @return the corresponding output type, null if no match is found
40  	 *
41  	 * @see Launcher#printUsage()
42  	 */
43  	public static OutputType fromString(String string) {
44  		for (OutputType outputType : OutputType.values()) {
45  			if (outputType.toString().equals(string)) {
46  				return outputType;
47  			}
48  		}
49  		return null;
50  	}
51  }
