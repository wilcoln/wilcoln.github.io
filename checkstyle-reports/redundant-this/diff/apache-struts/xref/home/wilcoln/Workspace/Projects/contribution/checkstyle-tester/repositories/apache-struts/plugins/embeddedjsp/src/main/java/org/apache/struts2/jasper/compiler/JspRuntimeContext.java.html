47  /**
48   * <p>
49   * Class for tracking JSP compile time file dependencies when the
50   * &#060;%{@literal @}include file="..."%&#062; directive is used.
51   * </p>
52   *
53   * <p>
54   * A background thread periodically checks the files a JSP page
55   * is dependent upon.  If a dependent file changes the JSP page
56   * which included it is recompiled.
57   * </p>
58   *
59   * <p>
60   * Only used if a web application context is a directory.
61   * </p>
62   *
63   * @author Glenn L. Nielsen
64   * @version $Revision: 505593 $
65   */
66  public final class JspRuntimeContext {
67  
68      // Logger
69      private Log log = LogFactory.getLog(JspRuntimeContext.class);
70  
71      /*
72       * Counts how many times the webapp's JSPs have been reloaded.
73       */
74      private int jspReloadCount;
75  
76      /**
77       * Preload classes required at runtime by a JSP servlet so that
78       * we don't get a defineClassInPackage security exception.
79       */
80      static {
81          JspFactoryImpl factory = new JspFactoryImpl();
82          SecurityClassLoad.securityClassLoad(factory.getClass().getClassLoader());
83          if (System.getSecurityManager() != null) {
84              String basePackage = "org.apache.struts2.jasper.";
85              try {
86                  factory.getClass().getClassLoader().loadClass(basePackage +
87                          "runtime.JspFactoryImpl$PrivilegedGetPageContext");
88                  factory.getClass().getClassLoader().loadClass(basePackage +
89                          "runtime.JspFactoryImpl$PrivilegedReleasePageContext");
90                  factory.getClass().getClassLoader().loadClass(basePackage +
91                          "runtime.JspRuntimeLibrary");
92                  factory.getClass().getClassLoader().loadClass(basePackage +
93                          "runtime.JspRuntimeLibrary$PrivilegedIntrospectHelper");
94                  factory.getClass().getClassLoader().loadClass(basePackage +
95                          "runtime.ServletResponseWrapperInclude");
96                  factory.getClass().getClassLoader().loadClass(basePackage +
97                          "servlet.JspServletWrapper");
98              } catch (ClassNotFoundException ex) {
99                  throw new IllegalStateException(ex);
100             }
101         }
102 
103         JspFactory.setDefaultFactory(factory);
104     }

108     /**
109      * <p>
110      * Create a JspRuntimeContext for a web application context.
111      * </p>
112      *
113      * <p>
114      * Loads in any previously generated dependencies from file.
115      * </p>
116      *
117      * @param context ServletContext for web application
118      * @param options options
119      */
120     public JspRuntimeContext(ServletContext context, Options options) {
121 
122         this.context = context;
123         this.options = options;
124 
125         // Get the parent class loader
126         parentClassLoader = Thread.currentThread().getContextClassLoader();
127         if (parentClassLoader == null) {
128             parentClassLoader = this.getClass().getClassLoader();
129         }
130 
131         if (log.isDebugEnabled()) {
132             if (parentClassLoader != null) {
133                 log.debug(Localizer.getMessage("jsp.message.parent_class_loader_is",
134                         parentClassLoader.toString()));
135             } else {
136                 log.debug(Localizer.getMessage("jsp.message.parent_class_loader_is",
137                         "<none>"));
138             }
139         }
140 
141         try {
142             initClassPath();
143         } catch (IOException e) {
144             context.log("ClassPath Init for context failed", e);
145         }
146 
147         if (context instanceof JspCServletContext) {
148             return;
149         }
150 
151         if (Constants.IS_SECURITY_ENABLED) {
152             initSecurity();
153         }
154 
155         // If this web application context is running from a
156         // directory, start the background compilation thread
157         String appBase = context.getRealPath("/");
158         if (!options.getDevelopment()
159                 && appBase != null
160                 && options.getCheckInterval() > 0) {
161             lastCheck = System.currentTimeMillis();
162         }
163     }

289     /**
290      * Method used by background thread to check the JSP dependencies
291      * registered with this class for JSP's.
292      */
293     public void checkCompile() {
294 
295         if (lastCheck < 0) {
296             // Checking was disabled
297             return;
298         }
299         long now = System.currentTimeMillis();
300         if (now > (lastCheck + (options.getCheckInterval() * 1000L))) {
301             lastCheck = now;
302         } else {
303             return;
304         }
305 
306         Object[] wrappers = jsps.values().toArray();
307         for (Object wrapper : wrappers) {
308             JspServletWrapper jsw = (JspServletWrapper) wrapper;
309             JspCompilationContext ctxt = jsw.getJspEngineContext();
310             // JspServletWrapper also synchronizes on this when
311             // it detects it has to do a reload
312             synchronized (jsw) {
313                 try {
314                     ctxt.compile();
315                 } catch (FileNotFoundException ex) {
316                     ctxt.incrementRemoved();
317                 } catch (Throwable t) {
318                     jsw.getServletContext().log("Background compile failed",
319                             t);
320                 }
321             }
322         }
323 
324     }

337     /**
338      * Method used to initialize classpath for compiles.
339      */
340     private void initClassPath() throws IOException {
341 
342         URL[] urls;
343         if (parentClassLoader instanceof URLClassLoader) {
344             urls = ((URLClassLoader) parentClassLoader).getURLs();
345         } else {    //jdk9 or later
346             urls = Collections.list(parentClassLoader.getResources("")).toArray(new URL[0]);
347         }
348         StringBuffer cpath = new StringBuffer();
349         String sep = System.getProperty("path.separator");
350 
351         for (int i = 0; i < urls.length; i++) {
352             // Tomcat 4 can use URL's other than file URL's,
353             // a protocol other than file: will generate a
354             // bad file system path, so only add file:
355             // protocol URL's to the classpath.
356 
357             if (urls[i].getProtocol().equals("file")) {
358                 cpath.append((String) urls[i].getFile() + sep);
359             }
360         }
361 
362         cpath.append(options.getScratchDir() + sep);
363 
364         String cp = (String) context.getAttribute(Constants.SERVLET_CLASSPATH);
365         if (cp == null || cp.equals("")) {
366             cp = options.getClassPath();
367         }
368 
369         classpath = cpath.toString() + cp;
370 
371         if (log.isDebugEnabled()) {
372             log.debug("Compilation classpath initialized: " + getClassPath());
373         }
374     }

376     /**
377      * Method used to initialize SecurityManager data.
378      */
379     private void initSecurity() {
380 
381         // Setup the PermissionCollection for this web app context
382         // based on the permissions configured for the root of the
383         // web app context directory, then add a file read permission
384         // for that directory.
385         Policy policy = Policy.getPolicy();
386         if (policy != null) {
387             try {
388                 // Get the permissions for the web app context
389                 String docBase = context.getRealPath("/");
390                 if (docBase == null) {
391                     docBase = options.getScratchDir().toString();
392                 }
393                 String codeBase = docBase;
394                 if (!codeBase.endsWith(File.separator)) {
395                     codeBase = codeBase + File.separator;
396                 }
397                 File contextDir = new File(codeBase);
398                 URL url = contextDir.getCanonicalFile().toURL();
399                 codeSource = new CodeSource(url, (Certificate[]) null);
400                 permissionCollection = policy.getPermissions(codeSource);
401 
402                 // Create a file read permission for web app context directory
403                 if (!docBase.endsWith(File.separator)) {
404                     permissionCollection.add
405                             (new FilePermission(docBase, "read"));
406                     docBase = docBase + File.separator;
407                 } else {
408                     permissionCollection.add
409                             (new FilePermission
410                                     (docBase.substring(0, docBase.length() - 1), "read"));
411                 }
412                 docBase = docBase + "-";
413                 permissionCollection.add(new FilePermission(docBase, "read"));
414 
415                 // Create a file read permission for web app tempdir (work)
416                 // directory
417                 String workDir = options.getScratchDir().toString();
418                 if (!workDir.endsWith(File.separator)) {
419                     permissionCollection.add
420                             (new FilePermission(workDir, "read"));
421                     workDir = workDir + File.separator;
422                 }
423                 workDir = workDir + "-";
424                 permissionCollection.add(new FilePermission(workDir, "read"));
425 
426                 // Allow the JSP to access org.apache.struts2.jasper.runtime.HttpJspBase
427                 permissionCollection.add(new RuntimePermission(
428                         "accessClassInPackage.org.apache.struts2.jasper.runtime"));
429 
430                 URL[] urls;
431                 if (parentClassLoader instanceof URLClassLoader) {
432                     urls = ((URLClassLoader) parentClassLoader).getURLs();
433                 } else {    //jdk9 or later
434                     urls = Collections.list(parentClassLoader.getResources("")).toArray(new URL[0]);
435                 }
436                 String jarUrl = null;
437                 String jndiUrl = null;
438                 for (URL url1 : urls) {
439                     if (jndiUrl == null
440                             && url1.toString().startsWith("jndi:")) {
441                         jndiUrl = url1.toString() + "-";
442                     }
443                     if (jarUrl == null
444                             && url1.toString().startsWith("jar:jndi:")
445                             ) {
446                         jarUrl = url1.toString();
447                         jarUrl = jarUrl.substring(0, jarUrl.length() - 2);
448                         jarUrl = jarUrl.substring(0,
449                                 jarUrl.lastIndexOf('/')) + "/-";
450                     }
451                 }
452                 if (jarUrl != null) {
453                     permissionCollection.add(
454                             new FilePermission(jarUrl, "read"));
455                     permissionCollection.add(
456                             new FilePermission(jarUrl.substring(4), "read"));
457                 }
458                 if (jndiUrl != null)
459                     permissionCollection.add(
460                             new FilePermission(jndiUrl, "read"));
461             } catch (Exception e) {
462                 context.log("Security Init for context failed", e);
463             }
464         }
465     }
