17  		return id;
18  	}
19

22  	}
23

26  	}
27  
28  	public void setUsername(String username) {
29  		this.username = username;
30  	}
31  
32  	public Set getMemberships() {
33  		return memberships;
34  	}
35  
36  	public void setMemberships(Set memberships) {
37  		this.memberships = memberships;
38  	}
39  
40  	public void addMembership(Group membership) {
41  		if (membership == null) {
42  			throw new IllegalArgumentException("Membership to add cannot be null");
43  		}
44  
45  		this.memberships.add(membership);
46  		membership.getMembers().add(this);
47  	}
