6   package spoon.support.compiler.jdt;
7   
8   import org.eclipse.jdt.core.compiler.CategorizedProblem;
9   import org.eclipse.jdt.internal.compiler.CompilationResult;
10  import org.eclipse.jdt.internal.compiler.ICompilerRequestor;
11  
12  public class TreeBuilderRequestor implements ICompilerRequestor {
13  
14  	private final JDTBasedSpoonCompiler jdtCompiler;
15  
16  	/**
17  	 * @param jdtCompiler
18  	 */
19  	TreeBuilderRequestor(JDTBasedSpoonCompiler jdtCompiler) {
20  		this.jdtCompiler = jdtCompiler;
21  	}
22  
23  	@Override
24  	public void acceptResult(CompilationResult result) {
25  		if (result.hasErrors()) {
26  			for (CategorizedProblem problem : result.problems) {
27  				this.jdtCompiler.reportProblem(problem);
28  			}
29  		}
30  	}
31  }
